182   /** This method uses the JDom structure for our XSchema needs. What it does is
183     * to add semantics to the XML elements defined in XSchema. In the end we need
184     * to construct an AnnotationSchema object form an XSchema file.
185     *
186     * @param jDom the JDOM structure containing the XSchema document. It must not
187     * be <b>null<b>
188     */
189   private void workWithJDom(org.jdom.Document jDom){
190     // Use the jDom structure the way we want
191     org.jdom.Element rootElement = jDom.getRootElement();
192     namespace = rootElement.getNamespace();
193     // get all children elements from the rootElement
194     List rootElementChildrenList = rootElement.getChildren("element", namespace);
195     Iterator rootElementChildrenIterator = rootElementChildrenList.iterator();
196     while (rootElementChildrenIterator.hasNext()){
197       org.jdom.Element childElement =
198                         (org.jdom.Element) rootElementChildrenIterator.next();
199       createAnnotationSchemaObject(childElement);
200     }//end while
201   } // workWithJdom
202 
203   /** This method creates an AnnotationSchema object fom an org.jdom.Element
204     * @param anElement is an XSchema element element
205     */
206   private void createAnnotationSchemaObject(org.jdom.Element anElement){
207     // Get the value of the name attribute. If this attribute doesn't exists
208     // then it will receive a default one.
209     annotationName = anElement.getAttributeValue("name");
210     if (annotationName == null)
211         annotationName = "UnknownElement";
212     // See if this element has a complexType element inside it
213     org.jdom.Element complexTypeElement = anElement.getChild("complexType",
214                                                              namespace);
215     if (complexTypeElement != null){
216       List complexTypeCildrenList = complexTypeElement.getChildren("attribute",
217                                                                    namespace);
218       Iterator complexTypeCildrenIterator = complexTypeCildrenList.iterator();
219       if (complexTypeCildrenIterator.hasNext())
220         featureSchemaSet = new LinkedHashSet<FeatureSchema>();
221       while (complexTypeCildrenIterator.hasNext()) {
222         org.jdom.Element childElement =
223                     (org.jdom.Element) complexTypeCildrenIterator.next();
224         createAndAddFeatureSchemaObject(childElement);
225       }// end while
226     }// end if
227   } // createAnnoatationSchemaObject
228 
229   /** This method creates and adds a FeatureSchema object to the current
230     * AnnotationSchema one.
231     * @param anAttributeElement is an XSchema attribute element
232     */
233   public void createAndAddFeatureSchemaObject(org.jdom.Element
234                                                           anAttributeElement) {
235     String featureName = null;
236     Class<?> featureType = null;
237     String featureUse  = null;
238     String featureValue = null;
239     Set    featurePermittedValuesSet = null;
240 
241     // Get the value of the name attribute. If this attribute doesn't exists
242     // then it will receive a default one.
243     featureName = anAttributeElement.getAttributeValue("name");
244     if (featureName == null)
245       featureName = "UnknownFeature";
246 
247     // See if it has a type attribute associated
248     String featureTypeName = anAttributeElement.getAttributeValue("type");
249     if (featureTypeName != null)
250       // Set it to the corresponding Java type
251       featureType = xSchema2JavaMap.get(featureTypeName);
252 
253     // Get the value of use attribute
254     featureUse = anAttributeElement.getAttributeValue("use");
255     if (featureUse == null)
256       // Set it to the default value
257       featureUse = "optional";
258 
259     // Get the value of value attribute
260     featureValue = anAttributeElement.getAttributeValue("value");
261     if (featureValue == null)
262       featureValue = "";
263 
264     // Let's check if it has a simpleType element inside
265     org.jdom.Element simpleTypeElement  =
266                                   anAttributeElement.getChild("simpleType",
267                                                               namespace);
268 
269     // If it has (!= null) then check to see if it has a restrictionElement
270     if (simpleTypeElement != null) {
271       org.jdom.Element restrictionElement =
272                               simpleTypeElement.getChild("restriction",
273                                                          namespace);
274       if (restrictionElement != null) {
275         // Get the type attribute for restriction element
276         featureTypeName = restrictionElement.getAttributeValue("base");
277         if (featureTypeName == null) featureTypeName = "string";
278         // Set it to the corresponding Java type
279         featureType =  xSchema2JavaMap.get(featureTypeName);
280 
281         // Check to see if there are any enumeration elements inside
282         List enumerationElementChildrenList =
283                                  restrictionElement.getChildren("enumeration",
284                                                                 namespace);
285         Iterator enumerationChildrenIterator =
286                                 enumerationElementChildrenList.iterator();
287 
288         // Check if there is any enumeration element in the list
289         if (enumerationChildrenIterator.hasNext())
290             featurePermittedValuesSet = new HashSet();
291         while (enumerationChildrenIterator.hasNext()) {
292           org.jdom.Element enumerationElement =
293                         (org.jdom.Element) enumerationChildrenIterator.next();
294           String permissibleValue =
295                             enumerationElement.getAttributeValue("value");
296           // Add that value to the featureSchema possible values set.
297           featurePermittedValuesSet.add(permissibleValue);
298         }// end while
299       }// end if( restrictionElement != null)
300     }// end if (simpleTypeElement != null)
301 
302     // If it doesn't have a simpleTypeElement inside and featureType is null or
303     // it wasn't recognised, then we set the default type to string.
304     if (simpleTypeElement == null && featureType == null )
305       featureType = xSchema2JavaMap.get("string");
306 
307     // Create an add a featureSchema object
308     FeatureSchema featureSchema = new FeatureSchema(
309                                                    featureName,
310                                                    featureType,
311                                                    featureValue,
312                                                    featureUse,
313                                                    featurePermittedValuesSet);
314     featureSchemaSet.add(featureSchema);
315   } // createAndAddFeatureSchemaObject

317   /** @return a String containing the XSchema document representing
318     *  an AnnotationSchema object.
319     */
320   public String toXSchema(){
321     StringBuffer schemaString = new StringBuffer();
322     schemaString.append("<?xml version=\"1.0\"?>\n" +
323                    "<schema xmlns=\"http://www.w3.org/2000/10/XMLSchema\">\n"+
324                    " <element name=\"" + annotationName + "\"");
325 
326     if (featureSchemaSet == null)
327       schemaString.append("/>\n");
328     else {
329       schemaString.append(">\n  <complexType>\n");
330       for(FeatureSchema fs : featureSchemaSet){
331         schemaString.append(fs.toXSchema(java2xSchemaMap));
332       }
333       schemaString.append("  </complexType>\n");
334       schemaString.append(" </element>\n");
335     }// end if else
336     schemaString.append("</schema>\n");
337     return schemaString.toString();
338   }// toXSchema
