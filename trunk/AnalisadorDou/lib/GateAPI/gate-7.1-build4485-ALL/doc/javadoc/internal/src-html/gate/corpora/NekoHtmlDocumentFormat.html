042 /**
043  * <p>
044  * DocumentFormat that uses Andy Clark's <a
045  * href="http://people.apache.org/~andyc/neko/doc/html/">NekoHTML</a>
046  * parser to parse HTML documents. It tries to render HTML in a similar
047  * way to a web browser, i.e. whitespace is normalized, paragraphs are
048  * separated by a blank line, etc. By default the text content of style
049  * and script tags is ignored completely, though the set of tags treated
050  * in this way is configurable via a CREOLE parameter.
051  * </p>
052  */
053 @CreoleResource(name = "GATE HTML Document Format", isPrivate = true,
054     autoinstances = {@AutoInstance(hidden = true)})
055 public class NekoHtmlDocumentFormat extends TextualDocumentFormat {
056   /** Debug flag */
057   private static final boolean DEBUG = false;
058

093   /**
094    * Unpack the markup in the document. This converts markup from the
095    * native format into annotations in GATE format. If the document was
096    * created from a String, then is recomandable to set the doc's
097    * sourceUrl to <b>null</b>. So, if the document has a valid URL,
098    * then the parser will try to parse the XML document pointed by the
099    * URL.If the URL is not valid, or is null, then the doc's content
100    * will be parsed. If the doc's content is not a valid XML then the
101    * parser might crash.
102    *
103    * @param doc The gate document you want to parse. If
104    *          <code>doc.getSourceUrl()</code> returns <b>null</b>
105    *          then the content of doc will be parsed. Using a URL is
106    *          recomended because the parser will report errors corectlly
107    *          if the document is not well formed.
108    */
109   public void unpackMarkup(Document doc, RepositioningInfo repInfo,
110           RepositioningInfo ampCodingInfo) throws DocumentFormatException {
111     if((doc == null)
112             || (doc.getSourceUrl() == null && doc.getContent() == null)) {
113 
114       throw new DocumentFormatException(
115               "GATE document is null or no content found. Nothing to parse!");
116     }// End if
117 
118     // Create a status listener
119     StatusListener statusListener = new StatusListener() {
120       public void statusChanged(String text) {
121         // This is implemented in DocumentFormat.java and inherited here
122         fireStatusChanged(text);
123       }
124     };
125 
126     boolean docHasContentButNoValidURL = hasContentButNoValidUrl(doc);
127 
128     NekoHtmlDocumentHandler handler = null;
129     try {
130       org.cyberneko.html.HTMLConfiguration parser = new HTMLConfiguration();
131 
132       // convert element and attribute names to lower case
133       parser.setProperty("http://cyberneko.org/html/properties/names/elems",
134               "lower");
135       parser.setProperty("http://cyberneko.org/html/properties/names/attrs",
136               "lower");
137       // make parser augment infoset with location information
138       parser.setFeature(NekoHtmlDocumentHandler.AUGMENTATIONS, true);
139 
140       // Create a new Xml document handler
141       handler = new NekoHtmlDocumentHandler(doc, null, ignorableTags);
142       // Register a status listener with it
143       handler.addStatusListener(statusListener);
144       // set repositioning object
145       handler.setRepositioningInfo(repInfo);
146       // set the object with ampersand coding positions
147       handler.setAmpCodingInfo(ampCodingInfo);
148       // construct the list of offsets for each line of the document
149       int[] lineOffsets = buildLineOffsets(doc.getContent().toString());
150       handler.setLineOffsets(lineOffsets);
151 
152       // set the handlers
153       parser.setDocumentHandler(handler);
154       parser.setErrorHandler(handler);
155 
156       // Parse the XML Document with the appropriate encoding
157       XMLInputSource is;
158 
159       if(docHasContentButNoValidURL) {
160         // no URL, so parse from string
161         is =
162                 new XMLInputSource(null, null, null, new StringReader(doc
163                         .getContent().toString()), null);
164       }
165       else if(doc instanceof TextualDocument) {
166         // textual document - load with user specified encoding
167         String docEncoding = ((TextualDocument)doc).getEncoding();
168         // XML, so no BOM stripping.
169         Reader docReader =
170                 new InputStreamReader(doc.getSourceUrl().openStream(),
171                         docEncoding);
172         is =
173                 new XMLInputSource(null, doc.getSourceUrl().toString(), doc
174                         .getSourceUrl().toString(), docReader, docEncoding);
175 
176         // since we control the encoding, tell the parser to ignore any
177         // meta http-equiv hints
178         parser
179                 .setFeature(
180                         "http://cyberneko.org/html/features/scanner/ignore-specified-charset",
181                         true);
182       }
183       else {
184         // let the parser decide the encoding
185         is =
186                 new XMLInputSource(null, doc.getSourceUrl().toString(), doc
187                         .getSourceUrl().toString());
188       }
189 
190       /* The following line can forward an
191        * ArrayIndexOutOfBoundsException from
192        * org.cyberneko.html.HTMLConfiguration.parse and crash GATE.    */
193       parser.parse(is);
194       // Angel - end
195       ((DocumentImpl)doc).setNextAnnotationId(handler.getCustomObjectsId());
196     }
197 
198     /* Handle IOException specially.      */
199     catch(IOException e) {
200       throw new DocumentFormatException("I/O exception for "
201               + doc.getSourceUrl().toString(), e);
202     }
203 
204     /* Handle XNIException and ArrayIndexOutOfBoundsException:
205      * flag the parsing error and keep going.     */
206     catch(Exception e) {
207       doc.getFeatures().put("parsingError", Boolean.TRUE);
208 
209       Boolean bThrow =
210               (Boolean)doc.getFeatures().get(
211                       GateConstants.THROWEX_FORMAT_PROPERTY_NAME);
212 
213       if(bThrow != null && bThrow.booleanValue()) {
214         // the next line is commented to avoid Document creation fail on
215         // error
216         throw new DocumentFormatException(e);
217       }
218       else {
219         Out.println("Warning: Document remains unparsed. \n"
220                 + "\n  Stack Dump: ");
221         e.printStackTrace(Out.getPrintWriter());
222       } // if
223 
224     }
225     finally {
226       if(handler != null) handler.removeStatusListener(statusListener);
227     }// End if else try
228 
229   }
230 
231   /**
232    * Pattern that matches the beginning of every line in a multi-line
233    * string. The regular expression engine handles the different types
234    * of newline characters (\n, \r\n or \r) automatically.
235    */
236   private static Pattern afterNewlinePattern =
237           Pattern.compile("^", Pattern.MULTILINE);
238 
239   /**
240    * Build an array giving the starting character offset of each line in
241    * the document. The HTML parser only reports event positions as line
242    * and column numbers, so we need this information to be able to
243    * correctly infer the repositioning information.
244    *
245    * @param docContent
246    * @return
247    */
248   private int[] buildLineOffsets(String docContent) {
249     Matcher m = afterNewlinePattern.matcher(docContent);
250     // we have to scan the text twice, first to determine how many lines
251     // there are (i.e. how long the array needs to be)...
252     int numMatches = 0;
253     while(m.find()) {
254       if(DEBUG) {
255         System.out.println("found line starting at offset " + m.start());
256       }
257       numMatches++;
258     }
259 
260     int[] lineOffsets = new int[numMatches];
261 
262     // ... and then again to populate the array with values.
263     m.reset();
264     for(int i = 0; i < lineOffsets.length; i++) {
265       m.find();
266       lineOffsets[i] = m.start();
267     }
268 
269     return lineOffsets;
270   }

272   /** Initialise this resource, and return it. */
273   public Resource init() throws ResourceInstantiationException {
274     // Register HTML mime type
275     MimeType mime = new MimeType("text", "html");
276     // Register the class handler for this mime type
277     mimeString2ClassHandlerMap.put(mime.getType() + "/" + mime.getSubtype(),
278             this);
279     // Register the mime type with mine string
280     mimeString2mimeTypeMap.put(mime.getType() + "/" + mime.getSubtype(), mime);
281     // Register file sufixes for this mime type
282     suffixes2mimeTypeMap.put("html", mime);
283     suffixes2mimeTypeMap.put("htm", mime);
284     // Register magic numbers for this mime type
285     magic2mimeTypeMap.put("<html", mime);
286     // Set the mimeType for this language resource
287     setMimeType(mime);
288     return this;
289   }// init()
