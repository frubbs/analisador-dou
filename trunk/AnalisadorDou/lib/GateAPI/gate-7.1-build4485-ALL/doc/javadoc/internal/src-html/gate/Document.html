017 package gate;
018 
019 import java.net.URL;
020 import java.util.Map;
021 import java.util.Set;
022 
023 import gate.corpora.DocumentStaxUtils;
024 import gate.event.DocumentListener;
025 import gate.util.InvalidOffsetException;

028 /** Represents the commonalities between all sorts of documents.
029  */
030 public interface Document extends SimpleDocument {
031 
032   /**
033   * The parameter name that determines whether or not a document is markup aware
034   */
035   public static final String
036     DOCUMENT_MARKUP_AWARE_PARAMETER_NAME = "markupAware";
037 
038   public static final String
039     DOCUMENT_ENCODING_PARAMETER_NAME = "encoding";
040 
041   public static final String
042     DOCUMENT_PRESERVE_CONTENT_PARAMETER_NAME = "preserveOriginalContent";
043 
044   public static final String
045     DOCUMENT_STRING_CONTENT_PARAMETER_NAME = "stringContent";
046 
047   public static final String
048     DOCUMENT_MIME_TYPE_PARAMETER_NAME = "mimeType";
049   
050   public static final String
051     DOCUMENT_REPOSITIONING_PARAMETER_NAME = "collectRepositioningInfo";
052 
053   public static final String
054     DOCUMENT_START_OFFSET_PARAMETER_NAME = "sourceUrlStartOffset";
055 
056   public static final String
057     DOCUMENT_END_OFFSET_PARAMETER_NAME = "sourceUrlEndOffset";
058  
059   /* parameter to store additional info about the document type, 
060    * e.g. publication, javadoc, etc. */
061   public static final String
062   DOCUMENT_TYPE_PARAMETER_NAME = "documentType";
063   
064   /** Documents may be packed within files; in this case an optional pair of
065    *  offsets refer to the location of the document.
066    */
067   public Long[] getSourceUrlOffsets();
068 
069   /** Documents may be packed within files; in this case an optional pair of
070    *  offsets refer to the location of the document. This method gets the
071    *  start offset.
072    */
073   public Long getSourceUrlStartOffset();
074 
075   /** Documents may be packed within files; in this case an optional pair of
076    *  offsets refer to the location of the document. This method gets the
077    *  end offset.
078    */
079   public Long getSourceUrlEndOffset();
080 
081   /** Returns a map with the named annotation sets
082     */
083   public Map<String, AnnotationSet> getNamedAnnotationSets();
084 
085   /** Make the document markup-aware. This will trigger the creation
086    *  of a DocumentFormat object at Document initialisation time; the
087    *  DocumentFormat object will unpack the markup in the Document and
088    *  add it as annotations. Documents are <B>not</B> markup-aware by default.
089    *
090    *  @param b markup awareness status.
091    */
092   public void setMarkupAware(Boolean b);
093 
094   /** Get the markup awareness status of the Document.
095    *
096    *  @return whether the Document is markup aware.
097    */
098   public Boolean getMarkupAware();
099 
100   /**
101    * Allow/disallow preserving of the original document content.
102    * If is <B>true</B> the original content will be retrieved from
103    * the DocumentContent object and preserved as document feature.
104    */
105   public void setPreserveOriginalContent(Boolean b);
106 
107   /** Get the preserving of content status of the Document.
108    *
109    *  @return whether the Document should preserve it's original content.
110    */
111   public Boolean getPreserveOriginalContent();
112 
113   /**
114    *  Allow/disallow collecting of repositioning information.
115    *  If is <B>true</B> information will be retrieved and preserved
116    *  as document feature.<BR>
117    *  Preserving of repositioning information give the possibilities
118    *  for converting of coordinates between the original document content and
119    *  extracted from the document text.
120    */
121   public void setCollectRepositioningInfo(Boolean b);
122 
123   /** Get the collectiong and preserving of repositioning information
124    *  for the Document. <BR>
125    *  Preserving of repositioning information give the possibilities
126    *  for converting of coordinates between the original document content and
127    *  extracted from the document text.
128    *
129    *  @return whether the Document should collect and preserve information.
130    */
131   public Boolean getCollectRepositioningInfo();
132 
133   /** Returns a GateXml document. This document is actually a serialization of
134    *  a Gate Document in XML. The <code>writeDocument</code> methods of
135    *  {@link DocumentStaxUtils} provide the standard implementation of this
136    *  serialization format which will work for any Document implementation.
137    *  Implementations of <code>toXml</code> will typically delegate to
138    *  <code>DocumentStaxUtils</code>, and in many cases it will be more
139    *  efficient for callers to use that directly rather than calling
140    *  <code>toXml</code>.
141    *  @see DocumentStaxUtils
142     * @return a string representing a Gate Xml document
143     */
144   public String toXml();
145 
146   /** Returns an XML document aming to preserve the original markups(
147     * the original markup will be in the same place and format as it was
148     * before processing the document) and include (if possible)
149     * the annotations specified in the aSourceAnnotationSet.
150     * <b>Warning:</b> Annotations from the aSourceAnnotationSet will be lost
151     * if they will cause a crosed over situation.
152     * @param aSourceAnnotationSet is an annotation set containing all the
153     * annotations that will be combined with the original marup set.
154     * @param includeFeatures determines whether or not features and gate IDs
155     * of the annotations should be included as attributes on the tags or not.
156     * If false, then only the annotation types are exported as tags, with no
157     * attributes.
158     * @return a string representing an XML document containing the original
159     * markup + dumped annotations form the aSourceAnnotationSet
160     */
161   public String toXml(Set aSourceAnnotationSet, boolean includeFeatures);
162 
163   /**
164    * Equivalent to toXml(aSourceAnnotationSet, true).
165    */
166   public String toXml(Set aSourceAnnotationSet);
167 
168   /** Make changes to the content.
169    */
170   public void edit(Long start, Long end, DocumentContent replacement)
171     throws InvalidOffsetException;
172 
173   /**
174    * Adds a {@link gate.event.DocumentListener} to this document.
175    * All the registered listeners will be notified of changes occured to the
176    * document.
177    */
178   public void addDocumentListener(DocumentListener l);
179 
180   /**
181    * Removes one of the previously registered document listeners.
182    */
183   public void removeDocumentListener(DocumentListener l);
184 
185 
186   /** Documents may be packed within files; in this case an optional pair of
187     * offsets refer to the location of the document. This method sets the
188     * end offset.
189     */
190   public void setSourceUrlEndOffset(Long sourceUrlEndOffset);
191 
192 
193   /** Documents may be packed within files; in this case an optional pair of
194     * offsets refer to the location of the document. This method sets the
195     * start offset.
196     */
197   public void setSourceUrlStartOffset(Long sourceUrlStartOffset);
198 
199 } // interface Document
