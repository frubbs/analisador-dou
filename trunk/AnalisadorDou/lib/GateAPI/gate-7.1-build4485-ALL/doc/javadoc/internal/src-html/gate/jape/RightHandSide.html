191   public void addBlock(String name, String namedBlock) {
192     // is it really a named block?

200       actionClassString.append(

203       );
204 
205     actionClassString.append(
206       "    if(" + name + "Annots != null && " + name +
207       "Annots.size() != 0) { " + nl);
208       
209     actionClassString.append(sourceInfo.addBlock(actionClassString.toString(), namedBlock));
210       
211    actionClassString.append(
212       nl + "    }" + nl
213     );
214   } // addBlock(name, block)

253   /** Remove class files created for actions. */
254   public static void cleanUp() {
255     if(tempFiles.size() == 0) return;
256 
257     // traverse the list in reverse order, coz any directories we
258     // created were done first
259     for(ListIterator i = tempFiles.listIterator(tempFiles.size()-1);
260         i.hasPrevious();
261        ) {
262       File tempFile = (File) i.previous();
263       tempFile.delete();
264     } // for each tempFile
265 
266     tempFiles.clear();
267   } // cleanUp

282   private void readObject(java.io.ObjectInputStream in)
283   throws IOException, ClassNotFoundException{
284     in.defaultReadObject();
285     //now read the class
286   String className = getActionClassName();
287   if(Gate.getClassLoader().findExistingClass(className) == null) {
288     try{
289       Map<String, String> actionClasses = new HashMap<String, String>();
290       actionClasses.put(className, getActionClassString());
291       gate.util.Javac.loadClasses(actionClasses, Gate.getClassLoader().getDisposableClassLoader(in.toString()));
292     }catch(Exception e1){
293       throw new GateRuntimeException (e1);
294     }
295   }

298   /** Makes changes to the document, using LHS bindings. */
299   public void transduce(Document doc, java.util.Map<String, AnnotationSet> bindings,
300                         AnnotationSet inputAS, final AnnotationSet outputAS,
301                         Ontology ontology,
302                         ActionContext actionContext)
303                         throws JapeException {
304     if(theActionObject == null) {
305       instantiateActionClass();
306     }
307 
308     // the 'annotations' parameter of the RhSAction.doIt method has been
309     // deprecated. As there is no way in Java to deprecate a parameter
310     // we will have to be a little clever/sneaky! We will create a proxy
311     // around the outputAS and pass the proxy to the Jape RHS. If the RHS
312     // code calls any method on the annotations proxy the following
313     // handler will be called instead so we can warn about the
314     // deprecation and then forward the method onwards to the outputAS
315     // so that the JAPE code will still work.
316     AnnotationSet annotations = (AnnotationSet)Proxy.newProxyInstance(
317             getClass().getClassLoader(), new Class[] {AnnotationSet.class},
318             new InvocationHandler() {
319 
320               public Object invoke(Object proxy, Method method, Object[] args)
321                       throws Throwable {
322 
323                 StackTraceElement japeSTE = null;
324                 int lineNumber = -1;
325 
326                 // find the stack trace element corresponding to the
327                 // call on the annotations proxy. This should always be
328                 // the third element but just to be on the safe side we
329                 // will find it by looping
330                 for(StackTraceElement ste : (new Throwable()).getStackTrace()) {
331                   if(ste.getClassName().equals(actionClassQualifiedName)) {
332 
333                     if(ste.getLineNumber() >= 0 && sourceInfo != null) {
334                       japeSTE = sourceInfo.getStackTraceElement(ste.getLineNumber());
335                       lineNumber = ste.getLineNumber();
336                     }
337                     else {
338                       // this will happen if we are running from a
339                       // serialised jape grammar as we don't keep the
340                       // source info
341                       japeSTE = new StackTraceElement(getPhaseName(),
342                               getRuleName(), null, -1);
343                     }
344 
345                     break;
346                   }
347                 }                
348 
349                 if(!warnings.contains(japeSTE)) {
350                   // we only want to warn about each use once per
351                   // invocation of GATE so we keep a cache of the stack
352                   // trace elements we have already warned about
353                   Err.println(nl + "WARNING: the JAPE 'annotations' parameter has been deprecated. Please use 'inputAS' or 'outputAS' instead.");
354                   Err.println(japeSTE);
355                   if(lineNumber >= 0)
356                     Err.println("\t" + sourceInfo.getSource(getActionClassString(), lineNumber).trim());
357 
358                   warnings.add(japeSTE);
359                 }
360 
361                 //pass the method on so that the JAPE code still works
362                 return method.invoke(outputAS, args);
363               }
364             });
365     
366     // run the action class
367     try {
368       ((RhsAction) theActionObject).setActionContext(actionContext);
369       ((RhsAction) theActionObject).doit(doc, bindings, annotations,
370                                          inputAS, outputAS, ontology);
371     } catch (NonFatalJapeException e) {
372       // if the action class throws a non-fatal exception then respond by
373       // dumping a whole bunch of useful debug information but then allow
374       // processing to continue on as if nothing happened.
375       Throwable t = e.getCause();
376       Err.println("A non-fatal JAPE exception occurred while processing document '"+doc.getName()+"'.");
377       Err.println("The issue occurred during execution of rule '"+getRuleName()+"' in phase '"+getPhaseName()+"':");
378       if (t != null) {
379         sourceInfo.enhanceTheThrowable(t);
380         t.printStackTrace(Err.getPrintWriter());
381       } else {
382         Err.println("Line number and exception details are not available!");
383       }
384     } catch (Throwable e) {
385       // if the action class throws an exception, re-throw it with a
386       // full description of the problem, inc. stack trace and the RHS
387       // action class code
388       if (sourceInfo != null) sourceInfo.enhanceTheThrowable(e);
389       if(e instanceof Error) {
390         throw (Error)e;
391       }
392       if(e instanceof JapeException) {
393         throw (JapeException)e;
394       }      
395       if(e instanceof RuntimeException) {
396         throw (RuntimeException)e;
397       }
398       
399       // shouldn't happen...
400         throw new JapeException(
401           "Couldn't run RHS action", e);
402     }
403   } // transduce
