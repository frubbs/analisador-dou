071   /**
072    * Given a query, this method parses it to convert it into one or more
073    * lucene queries.
074    * @throws gate.creole.ir.SearchException
075    */
076   public Query[] parse(String field, String query,
077           String baseTokenAnnotationType, String corpusID, String annotationSetToSearchIn)
078           throws gate.creole.ir.SearchException {
079     this.field = field;
080     this.baseTokenAnnotationType = baseTokenAnnotationType;
081     this.position = 0;
082     // at the moment this supports only | operator
083     // it also support klene operators * and +
084     // implicit operator is &
085     // It supports simple String queries
086     // it supports eight kinds of tokens
087     // 1. String (without quotes)
088     // 2. "String" (with quotes)
089     // 3. {AnnotationType}
090     // 4. {AnnotationType==String}
091     // 5. {AnnotationType=="String"}
092     // 7. {AnnotationType.feature==string}
093     // 8. {AnnotationType.feature=="string"}
094 
095     // Steps
096     // The query would we searched from left to right order
097 
098     // returned arraylist contains queries where each query is required
099     // to
100     // be converted into the Phrase query
101     queries = SubQueryParser.parseQuery(query);
102     Query[] q = new Query[queries.size()];
103     for(int i = 0; i < queries.size(); i++) {
104       Query phraseQuery = createPhraseQuery((String)queries.get(i));
105       // if the corpusID is not provided we donot want to create a
106       // boolean query
107       if(corpusID == null && annotationSetToSearchIn == null) {
108         BooleanQuery booleanQuery = new BooleanQuery();
109         Term t = new Term(Constants.ANNOTATION_SET_ID, Constants.COMBINED_SET);
110         TermQuery tQuery = new TermQuery(t);
111         booleanQuery.add(tQuery, false, true);
112         booleanQuery.add(phraseQuery, true, false);
113         q[i] = booleanQuery;
114       }
115       else {
116         BooleanQuery booleanQuery = new BooleanQuery();
117         booleanQuery.add(phraseQuery, true, false);
118         if(corpusID != null) {
119           Term t = new Term(Constants.CORPUS_ID, corpusID);
120           TermQuery tQuery = new TermQuery(t);
121           booleanQuery.add(tQuery, true, false);
122         }
123         
124         if(annotationSetToSearchIn != null) {
125           Term t = new Term(Constants.ANNOTATION_SET_ID, annotationSetToSearchIn);
126           TermQuery tQuery = new TermQuery(t);
127           booleanQuery.add(tQuery, true, false);
128         } else {
129           Term t = new Term(Constants.ANNOTATION_SET_ID, Constants.COMBINED_SET);
130           TermQuery tQuery = new TermQuery(t);
131           booleanQuery.add(tQuery, false, true);
132         }
133         
134 
135         q[i] = booleanQuery;
136       }
137     }
138     return q;
139   }

150   /**
151    * This method will create each normalized query into a Phrase or Term
152    * query If the query has only one term to search, it will be returned
153    * as a TermQuery otherwise, it will be returned as the PhraseQuery
154    */
155   private Query createPhraseQuery(String query)
156           throws gate.creole.ir.SearchException {
157     // Here we play the actual trick with lucene
158     // For a query like {Lookup}{Token}{Person.gender=="male"}
159     // internally this query is converted into the following PhraseQuery
160     // (Lookup Token Person male)
161     // these are the four terms which will be searched and they should
162     // occur
163     // in this order only
164     // but what we need is
165     // a pattern where
166     // Lookup -> the first annotation is of type Lookup
167     // Token -> the second annotation type is Token
168     // Person male -> and the third annotation must have a type person
169     // and a
170     // feature gender with male
171     // that means Person and male should be considered at the same
172     // location
173     // By default lucene doesn't do this and look for a position that is
174     // 1
175     // step more than the previous one
176     // so it will search for the first position of Lookup
177     // let say it is 19 (i.e. 19th annotation in the document)
178     // then it would consider 20th location for Token
179     // 21st for Person
180     // 22nd for male
181     // but we need, 19th for Lookup, 20th for Token and 21st for both
182     // Person
183     // and Male
184     // so from here itself we send our choice for the Location of
185     // annotations in this termPositions array :-).
186     // isn't it a great crack?
187     position = 0;
188 
189     PhraseQuery phQuery = new PhraseQuery();
190     // we will tokenize this query to convert it into different tokens
191     // query is like {Person}"said" "Hello" {Person.gender=="male"}
192     // we need to convert this into different tokens
193     // {Person}
194     // "said"
195     // "Hello"
196     // {Person.gender=="male"}
197     List<String> tokens = findTokens(query);
198 
199     // and then convert each token into separate terms
200     if(tokens.size() == 1) {
201       ArrayList[] termsPos = (createTerms(tokens.get(0)));
202       ArrayList terms = termsPos[0];
203       if(terms.size() == 1) {
204         if(areAllTermsTokens)
205           needValidation = false;
206         else needValidation = true;
207         return new TermQuery((Term)terms.get(0));
208       }
209       else {
210         position = 0;
211       }
212     }
213 
214     int totalTerms = 0;
215     boolean hadPreviousTermsAToken = true;
216 
217     needValidation = false;
218 
219     // and now for each token we need to create Term(s)
220     outer: for(int i = 0; i < tokens.size(); i++) {
221       ArrayList[] termpositions = createTerms((String)tokens.get(i));
222       ArrayList terms = termpositions[0];
223       ArrayList pos = termpositions[1];
224       ArrayList consider = termpositions[2];
225 
226       boolean allTermsTokens = true;
227       // lets first find out if there's any token in this terms
228       for(int k = 0; k < terms.size(); k++) {
229         Term t = (Term)terms.get(k);
230 
231         if(allTermsTokens) allTermsTokens = isBaseTokenTerm(t);
232       }
233 
234       if(!hadPreviousTermsAToken) {
235         needValidation = true;
236         break;
237       }
238 
239       if(!allTermsTokens) {
240         // we want to break here
241         needValidation = true;
242         if(i > 0)
243           break outer;
244       }
245 
246       for(int k = 0; k < terms.size(); k++) {
247         Term t = (Term)terms.get(k);
248         boolean considerValue = ((Boolean)consider.get(k)).booleanValue();
249         phQuery.add(t, (Integer)pos.get(k), considerValue);
250         if(considerValue) totalTerms++;
251       }
252 
253       hadPreviousTermsAToken = allTermsTokens;
254     }
255     phQuery.setTotalTerms(totalTerms);
256     return phQuery;
257   }

259   /**
260    * Returns true if the provided Term is a based token term. To be a
261    * base token term it has to satisify the following terms: 1. If its
262    * text is baseTokenAnnotationType and the type is "*" or 2. If its
263    * type = "baseTokenAnnotationType.feature"
264    */
265   private boolean isBaseTokenTerm(Term t) {
266     // the term refers to the base token
267     // only if it satisfies the following conditions
268     // 1. If its text is baseTokenAnnotationType and the type is "*"
269     // or 2. If its type = "baseTokenAnnotationType.feature"
270 
271     // condition 1
272     if(t.text().equals(baseTokenAnnotationType) && t.type().equals("*"))
273       return true;
274 
275     // condition 2
276     if(t.type().startsWith(baseTokenAnnotationType + ".")) return true;
277 
278     return false;
279   }

283   /**
284    * Given a query this method returns tokens. Here token is an object
285    * of string.
286    * @throws gate.creole.ir.SearchException
287    */
288   public List<String> findTokens(String query)
289           throws gate.creole.ir.SearchException {
290     List<String> tokens = new ArrayList<String>();
291     String token = "";
292     char ch = ' ';
293     char prev = ' ';
294     int balance = 0;
295     for(int i = 0; i < query.length(); i++) {
296       prev = ch;
297       ch = query.charAt(i);
298       if(isOpeneningBrace(ch, prev)) {
299         if(balance != 0) {
300           throw new SearchException("unbalanced braces",
301             "a closing brace (}) is missing before this opening brace", query, i);
302         }
303 
304         if(!token.trim().equals("")) {
305           tokens.add(token.trim());
306         }
307 
308         balance++;
309         token = "{";
310         continue;
311       }
312 
313       if(isClosingBrace(ch, prev)) {
314         balance--;
315         if(balance != 0) {
316           throw new SearchException("unbalanced braces",
317             "an opening brace ({) is missing before this closing brace", query, i);
318         }
319 
320         token += "}";
321         tokens.add(token.trim());
322         token = "";
323         continue;
324       }
325 
326       token += (char)ch;
327     }
328 
329     if(balance != 0) {
330       if (balance > 0) {
331         throw new SearchException("unbalanced braces",
332                 "One closing brace (}) is missing in this expression", query);
333       } else {
334         throw new SearchException("unbalanced braces",
335                 "One opening brace ({) is missing in this expression", query);
336       }
337     }
338 
339     if(!token.trim().equals("")) tokens.add(token);
340 
341     return tokens;
342   }

365   private ArrayList splitString(String string, char with, boolean normalize) {
366     // here we want to split the string
367     // but also make sure the with character is not escaped
368     ArrayList strings = new ArrayList();
369     StringBuffer newString = new StringBuffer();
370     for(int i = 0; i < string.length(); i++) {
371       if(i == 0) {
372         newString.append(string.charAt(0));
373         continue;
374       }
375 
376       if(string.charAt(i) == with) {
377         // need to check the previous character
378         if(string.charAt(i - 1) == '\\') {
379           newString.append(with);
380           continue;
381         }
382         else {
383           if(normalize)
384             strings.add(norm(newString.toString()));
385           else strings.add(newString.toString());
386 
387           newString = new StringBuffer();
388           continue;
389         }
390       }
391 
392       newString.append(string.charAt(i));
393     }
394     if(newString.length() > 0) {
395       if(normalize)
396         strings.add(norm(newString.toString()).trim());
397       else strings.add(newString.toString().trim());
398     }
399     return strings;
400   }

402   private int findIndexOf(String element, char ch) {
403     int index1 = -1;
404     int start = -1;
405     while(true) {
406       index1 = element.indexOf(ch, start);
407       if(isEscapeSequence(element, index1)) {
408         start = index1 + 1;
409       }
410       else {
411         break;
412       }
413     }
414     return index1;
415   }

417   private String norm(String string) {
418     StringBuffer sb = new StringBuffer();
419     for(int i = 0; i < string.length(); i++) {
420       if(string.charAt(i) == '\\') {
421         if(i + 1 <= string.length() - 1) {
422           char ch = string.charAt(i + 1);
423           if(ch == ',' || ch == '.' || ch == '(' || ch == ')' || ch == '{'
424                   || ch == '}' || ch == '"')
425           ;
426           continue;
427         }
428       }
429       sb.append(string.charAt(i));
430     }
431     return sb.toString();
432   }

434   public ArrayList[] createTerms(String elem)
435           throws gate.creole.ir.SearchException {
436     areAllTermsTokens = true;
437     ArrayList terms = new ArrayList();
438     ArrayList pos = new ArrayList();
439     ArrayList consider = new ArrayList();
440 
441     elem = elem.trim();
442     if(elem.charAt(0) == '{' && elem.charAt(elem.length() - 1) == '}') {
443       // possible
444       elem = elem.substring(1, elem.length() - 1);
445       int index = elem.indexOf("==");
446       int index1 = findIndexOf(elem, '.');
447 
448       if(index == -1 && index1 == -1) {
449         // 3. {AnnotationType}
450         // this can be {AnnotationType, AnnotationType...}
451         ArrayList fields = splitString(elem, ',', true);
452 
453         for(int p = 0; p < fields.size(); p++) {
454           if(areAllTermsTokens
455                   && !((String)fields.get(p)).equals(baseTokenAnnotationType))
456             areAllTermsTokens = false;
457 
458           terms.add(new Term(field, norm(((String)fields.get(p))), "*"));
459           pos.add(new Integer(position));
460           if(p == 0)
461             consider.add(new Boolean(true));
462           else consider.add(new Boolean(false));
463 
464         }
465         position++;
466       }
467       else if(index != -1 && index1 == -1) {
468         // 4. {AnnotationType==String}
469         // 5. {AnnotationType=="String"}
470 
471         ArrayList fields = splitString(elem, ',', false);
472         for(int p = 0; p < fields.size(); p++) {
473           index = ((String)fields.get(p)).indexOf("==");
474           // here this is also posible
475           // {AnnotationType, AnnotationType=="String"}
476           if(index != -1) {
477             String annotType = norm(((String)fields.get(p)).substring(0, index)
478                     .trim());
479             String annotText = norm(((String)fields.get(p)).substring(
480                     index + 2, ((String)fields.get(p)).length()).trim());
481             if(annotText.length() > 2 && annotText.charAt(0) == '\"'
482                     && annotText.charAt(annotText.length() - 1) == '\"') {
483               annotText = annotText.substring(1, annotText.length() - 1);
484             }
485             if(!annotType.trim().equals(baseTokenAnnotationType))
486               areAllTermsTokens = false;
487             
488             terms.add(new Term(field, annotText, annotType + ".string"));
489             pos.add(new Integer(position));
490             if(p == 0)
491               consider.add(new Boolean(true));
492             else consider.add(new Boolean(false));
493 
494           }
495           else {
496             if(!(norm((String)fields.get(p))).equals(baseTokenAnnotationType))
497               areAllTermsTokens = false;
498             
499             terms.add(new Term(field, norm(((String)fields.get(p))), "*"));
500             pos.add(new Integer(position));
501             if(p == 0)
502               consider.add(new Boolean(true));
503             else consider.add(new Boolean(false));
504           }
505         }
506 
507         position++;
508 
509       }
510       else if(index == -1 && index1 != -1) {
511         throw new SearchException("missing operator",
512                 "an equal operator (==) is missing",
513                 elem, (elem.indexOf("=", index1)!=-1)?
514                        elem.indexOf("=", index1):elem.length());
515       }
516       else if(index != -1 && index1 != -1) {
517 
518         // it can be {AT, AT.f==S, AT=="S"}
519         int index2 = findIndexOf(elem, ',');
520         String[] subElems = null;
521         if(index2 == -1) {
522           subElems = new String[] {elem};
523         }
524         else {
525           ArrayList list = splitString(elem, ',', false);
526           subElems = new String[list.size()];
527           for(int k = 0; k < list.size(); k++) {
528             subElems[k] = (String)list.get(k);
529           }
530         }
531 
532         int lengthTravelledSoFar = 0;
533         for(int j = 0; j < subElems.length; j++) {
534           // 7. {AnnotationType.feature==string}
535           // 8. {AnnotationType.feature=="string"}
536           index = subElems[j].indexOf("==");
537           index1 = findIndexOf(subElems[j], '.');
538           if(index == -1 && index1 == -1) {
539             // this is {AT}
540             if(!norm(subElems[j].trim()).equals(baseTokenAnnotationType))
541               areAllTermsTokens = false;

546             else consider.add(new Boolean(false));
547 
548           }
549           else if(index != -1 && index1 == -1) {

561             pos.add(new Integer(position));

564             else consider.add(new Boolean(false));

596       }
