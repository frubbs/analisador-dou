028 /** A convenience implementation of ProcessingResource with some default
029   * code.
030   */
031 abstract public class AbstractProcessingResource
032 extends AbstractResource implements ProcessingResource, ANNIEConstants
033 {
034   /** Initialise this resource, and return it. */
035   public Resource init() throws ResourceInstantiationException {
036     return this;
037   } // init()
038 
039   /** Run the resource. It doesn't make sense not to override
040    *  this in subclasses so the default implementation signals an
041    *  exception.
042    */
043   public void execute() throws ExecutionException{
044     throw new ExecutionException(
045       "Resource " + getClass() + " hasn't overriden the execute() method"
046     );
047   } // execute()
048 
049   /**
050    * Reinitialises the processing resource. After calling this method the
051    * resource should be in the state it is after calling init.
052    * If the resource depends on external resources (such as rules files) then
053    * the resource will re-read those resources. If the data used to create
054    * the resource has changed since the resource has been created then the
055    * resource will change too after calling reInit().
056    * The implementation in this class simply calls {@link #init()}. This
057    * functionality must be overriden by derived classes as necessary.
058    */
059   public void reInit() throws ResourceInstantiationException{
060     init();
061   } // reInit()
062 
063   /** should clear all internal data of the resource. Does nothing now */
064   public void cleanup() {
065   }
066 
067   /**
068    * Checks whether this PR has been interrupted since the last time its
069    * {@link #execute()} method was called.
070    */
071   public boolean isInterrupted(){
072     return interrupted;
073   }
074 
075   /**
076    * Notifies this PR that it should stop its execution as soon as possible.
077    */
078   public void interrupt(){
079     interrupted = true;
080   }

083   /**
084    * Removes a {@link gate.event.StatusListener} from the list of listeners for
085    * this processing resource
086    */
087   public synchronized void removeStatusListener(StatusListener l) {
088     if (statusListeners != null && statusListeners.contains(l)) {
089       Vector v = (Vector) statusListeners.clone();
090       v.removeElement(l);
091       statusListeners = v;
092     }
093   }
094 
095   /**
096    * Adds a {@link gate.event.StatusListener} to the list of listeners for
097    * this processing resource
098    */
099   public synchronized void addStatusListener(StatusListener l) {
100     Vector v = statusListeners == null ? new Vector(2) : (Vector) statusListeners.clone();
101     if (!v.contains(l)) {
102       v.addElement(l);
103       statusListeners = v;
104     }
105   }
106 
107   /**
108    * Notifies all the {@link gate.event.StatusListener}s of a change of status.
109    * @param e the message describing the status change
110    */
111   protected void fireStatusChanged(String e) {
112     if (statusListeners != null) {
113       Vector listeners = statusListeners;
114       int count = listeners.size();
115       for (int i = 0; i < count; i++) {
116         ((StatusListener) listeners.elementAt(i)).statusChanged(e);
117       }
118     }
119   }
120 
121   /**
122    * Adds a {@link gate.event.ProgressListener} to the list of listeners for
123    * this processing resource.
124    */
125   public synchronized void addProgressListener(ProgressListener l) {
126     Vector v = progressListeners == null ? new Vector(2) : (Vector) progressListeners.clone();
127     if (!v.contains(l)) {
128       v.addElement(l);
129       progressListeners = v;
130     }
131   }
132 
133   /**
134    * Removes a {@link gate.event.ProgressListener} from the list of listeners
135    * for this processing resource.
136    */
137   public synchronized void removeProgressListener(ProgressListener l) {
138     if (progressListeners != null && progressListeners.contains(l)) {
139       Vector v = (Vector) progressListeners.clone();
140       v.removeElement(l);
141       progressListeners = v;
142     }
143   }
144 
145   /**
146    * Notifies all the {@link gate.event.ProgressListener}s of a progress change
147    * event.
148    * @param e the new value of execution completion
149    */
150   protected void fireProgressChanged(int e) {
151     if (progressListeners != null) {
152       Vector listeners = progressListeners;
153       int count = listeners.size();
154       for (int i = 0; i < count; i++) {
155         ((ProgressListener) listeners.elementAt(i)).progressChanged(e);
156       }
157     }
158   }
159 
160   /**
161    * Notifies all the {@link gate.event.ProgressListener}s of a progress
162    * finished.
163    */
164   protected void fireProcessFinished() {
165     if (progressListeners != null) {
166       Vector listeners = progressListeners;
167       int count = listeners.size();
168       for (int i = 0; i < count; i++) {
169         ((ProgressListener) listeners.elementAt(i)).processFinished();
170       }
171     }
172   }

174   /**
175    * Get the current values for all of a specified resource's
176    * registered runtime parameters.
177    */
178   public static FeatureMap getRuntimeParameterValues(Resource res)
179               throws ResourceInstantiationException {
180     ResourceData rData = (ResourceData)Gate.getCreoleRegister().get(
181             res.getClass().getName());
182     if(rData == null)
183       throw new ResourceInstantiationException(
184               "Could not find CREOLE data for " + res.getClass().getName());
185 
186     ParameterList params = rData.getParameterList();
187 
188     return AbstractResource.getParameterValues(res,
189             params.getRuntimeParameters());
190   }
191   
192   /**
193    * Get the current values for all this resource's registered
194    * init-time parameters.
195    */
196   public FeatureMap getRuntimeParameterValues()
197               throws ResourceInstantiationException {
198     return getRuntimeParameterValues(this);
199   }

201   /**
202    * A progress listener used to convert a 0..100 interval into a smaller one
203    */
204   protected class IntervalProgressListener implements ProgressListener{
205     public IntervalProgressListener(int start, int end){
206       this.start = start;
207       this.end = end;
208     }
209     public void progressChanged(int i){
210       fireProgressChanged(start + (end - start) * i / 100);
211     }
212 
213     public void processFinished(){
214       fireProgressChanged(end);
215     }
216 
217     int start;
218     int end;
219   }//IntervalProgressListener

222    * A simple status listener used to forward the events upstream.
223    */
224   protected class InternalStatusListener implements StatusListener{
225     public void statusChanged(String message){
226       fireStatusChanged(message);
227     }
228   }//InternalStatusListener

240   private transient Vector progressListeners;
