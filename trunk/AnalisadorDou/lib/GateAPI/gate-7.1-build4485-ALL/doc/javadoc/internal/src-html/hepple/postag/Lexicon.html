028 import gate.util.BomStrippingInputStreamReader;
029 
030 import java.util.*;
031 import java.io.*;
032 import java.net.URL;
033 
034 import org.apache.commons.io.IOUtils;
035 
036 /**
037  * A {@link java.util.HashMap} that maps from lexical entry
038  * ({@link java.lang.String}) to possible POS categories
039  * ({@link java.util.List}
040  */
041 class Lexicon extends HashMap {
042 
043   private String encoding;
044 
045 
046   /**
047    * @deprecated The lexicon file is read at construction time, so setting the
048    * encoding later will have no effect.  Use the two argument constructor to
049    * set the encoding.
050    */
051   public void setEncoding(String encoding) {
052     throw new IllegalStateException("Cannot change encoding once POS tagger "
053                                   + "has been constructed.  Use the three "
054                                   + "argument constructor to specify "
055                                   + "encoding.");
056   }
057 
058   /**
059    * Constructor.
060    * @param lexiconURL an URL for the file contianing the lexicon.
061    */
062   public Lexicon(URL lexiconURL) throws IOException{
063     this(lexiconURL, null);
064   }
065 
066   /**
067    * Constructor.
068    * @param lexiconURL an URL for the file contianing the lexicon.
069    * @param encoding the character encoding to use for reading the lexicon.
070    */
071   public Lexicon(URL lexiconURL, String encoding) throws IOException{
072     this.encoding = encoding;
073     String line;
074     BufferedReader lexiconReader = null;
075     InputStream lexiconStream = null;
076     
077     try {
078       lexiconStream = lexiconURL.openStream();
079       
080       if(encoding == null) {
081         lexiconReader = new BomStrippingInputStreamReader(lexiconStream);
082       } else {
083         lexiconReader = new BomStrippingInputStreamReader(lexiconStream,encoding);
084       }
085   
086       line = lexiconReader.readLine();
087       String entry;
088       List<String> categories;
089       while(line != null){
090         StringTokenizer tokens = new StringTokenizer(line);
091         entry = tokens.nextToken();
092         categories = new ArrayList<String>();
093         while(tokens.hasMoreTokens()) categories.add(tokens.nextToken());
094         put(entry, categories);
095   
096         line = lexiconReader.readLine();
097       }//while(line != null)
098     }
099     finally {
100       IOUtils.closeQuietly(lexiconReader);
101       IOUtils.closeQuietly(lexiconStream);
102     }
103   }//public Lexicon(URL lexiconURL) throws IOException
104 
105 }//class Lexicon
