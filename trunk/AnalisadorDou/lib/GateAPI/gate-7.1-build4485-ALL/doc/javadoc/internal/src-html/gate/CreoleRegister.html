029 /** The CREOLE register records the set of resources that are currently
030   * known to the system. Each member of the register is a
031   * <A HREF=creole/ResourceData.html>ResourceData</A> object, indexed by
032   * the class name of the resource.
033   * <P>
034   * The register is accessible from the static method
035   * <A HREF=Gate.html#getCreoleRegister()>gate.Gate.getCreoleRegister
036   * </A>;
037   * there is only one per application of the GATE framework.
038   * <P>
039   * Clients use the register by adding URLs (using the
040   * <A HREF=#addDirectory(java.net.URL)>addDirectory</A> method)
041   * pointing to CREOLE directories. A <B>CREOLE directory</B> is a URL at
042   * which resides a file called <CODE>creole.xml</CODE> describing
043   * the resources present, and one or more Jar files implementing
044   * those resources. E.g., the CREOLE resources at
045   * <A HREF=http://gate.ac.uk/>gate.ac.uk</A> are registered by Gate.init()
046   * by registering the directory URL
047   * <A HREF=http://gate.ac.uk/creole/>http://gate.ac.uk/creole/</A>, under
048   * which lives a file called creole.xml.
049   * <P>
050   * To register resources clients use the <CODE>registerDirectories</CODE>
051   * methods. When resources have been registered they can be accessed via
052   * their <CODE>ResourceData</CODE> objects. So a typical use of the register
053   * is to: add the set of URLs containing CREOLE directories; register
054   * all resources found at those URLs; browse the set of registered
055   * resources.
056   * <P>
057   * In all cases, where a resource or a directory is added which is
058   * already present in the register, the new silently overwrites the old.
059   *
060   * The CreoleRegister notifies all registered listeners of all
061   * {@link gate.event.CreoleEvent}s that occur in the system regardless of
062   * whether they were initially fired by the {@link Factory}, the
063   * {@link DataStoreRegister} or the {@link CreoleRegister} itself.
064   *
065   * @see gate.Gate
066   * @see gate.creole.ResourceData
067   */
068 public interface CreoleRegister extends Map<String, ResourceData>, Serializable, CreoleListener
069 {
070   /** Add a CREOLE directory URL. The directory is <B>not</B> registered. */
071   public void addDirectory(URL directoryUrl);
072 
073   /** Get the list of CREOLE directory URLs. */
074   public Set getDirectories();
075 
076   /** Register all the CREOLE directories that we know of.
077     * The <CODE>creole.xml</CODE> files
078     * at the URLs are parsed, and <CODE>CreoleData</CODE> objects added
079     * to the register.
080     */
081   public void registerDirectories() throws GateException;
082 
083   /**
084    * Given the class object for a class with {@link CreoleResource}
085    * annotations, register that class is if it was found in a scanned jar
086    * file with no additional creole.xml information.
087    *
088    * This API is intended for use in embedded GATE applications where
089    * the 'application' is created via the API. Components registered with this
090    *  API won't work in saved applications, but they can be added
091    *  to saved applications at runtime.
092    *
093    * @param clazz Class object for class with CreoleResource annotations.
094    */
095   public void registerComponent(Class<? extends Resource> clazz) throws GateException;
096 
097   /** Register a single CREOLE directory. The <CODE>creole.xml</CODE>
098     * file at the URL is parsed, and <CODE>CreoleData</CODE> objects added
099     * to the register. If the directory URL has not yet been added it
100     * is now added.
101     */
102   public void registerDirectories(URL directoryUrl) throws GateException;
103 
104   /**
105    * Removes a CREOLE directory from the set of loaded directories.
106    * @param directory
107    */
108   public void removeDirectory(URL directory);
109 
110   /** Register resources that are built in to the GATE distribution.
111     * These resources are described by the <TT>creole.xml</TT> file in
112     * <TT>resources/creole</TT>.
113     */
114   public void registerBuiltins() throws GateException;
115 
116   /** This is a utility method for creating CREOLE directory files
117     * (typically called <CODE>creole.xml</CODE>) from a list of Jar
118     * files that contain resources. The method concatenates the
119     * <CODE>resource.xml</CODE> files that the Jars contain.
120     * <P>
121     * If Java allowed class methods in interfaces this would be static.
122     */
123   public File createCreoleDirectoryFile(File directoryFile, Set jarFileNames);
124 
125   /** Get the list of types of LR in the register. */
126   public Set<String> getLrTypes();
127 
128   /** Get the list of types of PR in the register. */
129   public Set<String> getPrTypes();
130 
131   /** Get the list of types of VR in the register. */
132   public Set<String> getVrTypes();
133 
134   /** Get the list of types of VR in the register. */
135   public Set<String> getControllerTypes();
136   
137   /** Get the list of packaged application types in the register. */
138   public Set<String> getApplicationTypes();
139 
140   /** Get the list of types of tool in the register. */
141   public Set<String> getToolTypes();
142 
143   /** Get a list of all instantiations of LR in the register. */
144   public List<LanguageResource> getLrInstances();
145 
146   /** Get a list of all instantiations of PR in the register. */
147   public List<ProcessingResource> getPrInstances();
148 
149   /** Get a list of all instantiations of VR in the register. */
150   public List<VisualResource> getVrInstances();
151 
152   /**
153    * Get a list of all the known Language Resource instances in the register
154    * whose class is <code>resourceTypeName</code>.  This is only direct
155    * instances of the specified class, not its sub-classes, so if
156    * <code>resourceTypeName</code> refers to an interface or abstract class an
157    * empty list will be returned.
158    */
159   public List<LanguageResource> getLrInstances(String resourceTypeName);
160 
161   /**
162    * Get a list of all the known Processing Resource instances in the register
163    * whose class is <code>resourceTypeName</code>.  This is only direct
164    * instances of the specified class, not its sub-classes, so if
165    * <code>resourceTypeName</code> refers to an interface or abstract class an
166    * empty list will be returned.
167    */
168   public List<ProcessingResource> getPrInstances(String resourceTypeName);
169 
170   /**
171    * Get a list of all the known Visual Resource instances in the register
172    * whose class is <code>resourceTypeName</code>.  This is only direct
173    * instances of the specified class, not its sub-classes, so if
174    * <code>resourceTypeName</code> refers to an interface or abstract class an
175    * empty list will be returned.
176    */
177   public List<VisualResource> getVrInstances(String resourceTypeName);
178 
179   /** Get a list of all non-private instantiations of LR in the register. */
180   public List<LanguageResource> getPublicLrInstances();
181 
182   /** Get a list of all non-private instantiations of PR in the register. */
183   public List<ProcessingResource> getPublicPrInstances();
184 
185   /** Get a list of all non-private instantiations of VR in the register. */
186   public List<VisualResource> getPublicVrInstances();
187 
188   /** Get a list of all non-private types of LR in the register. */
189   public List<String> getPublicLrTypes();
190 
191   /** Get a list of all non-private types of PR in the register. */
192   public List<String> getPublicPrTypes();
193 
194   /** Get a list of all non-private types of VR in the register. */
195   public List<String> getPublicVrTypes();
196 
197   /** Get a list of all non-private types of Controller in the register. */
198   public List<String> getPublicControllerTypes();
199 
200   /**
201    * Get a list of all the known Resource instances in the register that are of
202    * the specified (class or interface) type or one of its sub-types.
203    * Specifically, all known Resources <code>r</code> such that
204    * <pre>
205    * Gate.getClassLoader().loadClass(type).isInstance(r)
206    * </pre>
207    *
208    * Instances that are hidden (<code>Gate.getHiddenAttribute(r.getFeatures())
209    * == true</code>) are not included in the returned list.
210    */
211   public List<Resource> getAllInstances(String type) throws GateException;
212 
213   /**
214    * Returns a list of strings representing class names for large VRs valid
215    * for a given type of language/processing resource.
216    * The default VR will be the first in the returned list.
217    */
218   public List<String> getLargeVRsForResource(String resourceClassName);
219 
220   /**
221    * Returns a list of strings representing class names for small VRs valid
222    * for a given type of language/processing resource
223    * The default VR will be the first in the returned list.
224    */
225   public List<String> getSmallVRsForResource(String resourceClassName);
226 
227   /**
228     * Returns a list of strings representing class names for annotation VRs
229     * that are able to display/edit all types of annotations.
230     * The default VR will be the first in the returned list.
231     */
232    public List<String> getAnnotationVRs();
233 
234   /**
235     * Returns a list of strings representing class names for annotation VRs
236     * that are able to display/edit a given annotation type
237     * The default VR will be the first in the returned list.
238     */
239    public List<String> getAnnotationVRs(String annotationType);
240 
241 
242   /**
243     * Returns a list of strings representing annotations types for which
244     * there are custom viewers/editor registered.
245     */
246    public List<String> getVREnabledAnnotationTypes();
247 
248    /**
249     * Renames an existing resource.
250     */
251    public void setResourceName(Resource res, String newName);
252 
253   /**
254    * Registers a {@link gate.event.CreoleListener}with this CreoleRegister.
255    * The register will fire events every time a resource is added to or removed
256    * from the system and when a datastore is created, opened or closed.
257    */
258   public void addCreoleListener(CreoleListener l);
259 
260   /**
261    * Removes a {@link gate.event.CreoleListener} previously registered with this
262    * CreoleRegister.
263    * @see #addCreoleListener(CreoleListener)
264    */
265   public void removeCreoleListener(CreoleListener l);
266 
267 } // interface CreoleRegister
