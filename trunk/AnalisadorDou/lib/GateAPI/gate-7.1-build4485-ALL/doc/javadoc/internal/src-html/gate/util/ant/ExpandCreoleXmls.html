030 /**
031  * Ant task to take a bunch of creole.xml files, process the
032  * {@link CreoleResource} annotations on their resources, and write the
033  * augmented XML to a target directory.  If the "classesOnly" attribute is
034  * set to "true" (or "1", "yes" or "on" in the normal Ant way) then the task
035  * performs only the first JAR scanning step, adding
036  * <pre>
037  * &lt;RESOURCE&gt;
038  *   &lt;CLASS&gt;com.example.ClassName&lt;/CLASS&gt;
039  * &lt;/RESOURCE&gt;
040  * </pre>
041  * for each annotated class in the JAR, but does not process the annotation
042  * fully.  This limited expansion is useful in cases where you will be using
043  * the plugin in a way that does not allow JAR scanning to take place at
044  * runtime, for example if you will be loading the plugin directly from a
045  * <code>jar:<code> URL (with the plugin's JAR files placed on your
046  * application's classpath).
047  */
048 public class ExpandCreoleXmls extends Task {
049 
050   private List<FileSet> srcFiles = new ArrayList<FileSet>();
051   
052   private File toDir;
053   
054   private File gateHome = null;
055   
056   private File pluginsHome = null;
057 
058   private boolean classesOnly = false;
059   
060   private SAXBuilder builder;
061   
062   private XMLOutputter outputter = new XMLOutputter();
063 
064   public ExpandCreoleXmls() {
065     builder = new SAXBuilder(false);
066     builder.setXMLFilter(new CreoleXmlUpperCaseFilter());
067   }
068   
069   public void addFileset(FileSet fs) {
070     srcFiles.add(fs);
071   }
072   
073   public void setTodir(File toDir) {
074     this.toDir = toDir;
075   }
076 
077   public void setClassesOnly(boolean classesOnly) {
078     this.classesOnly = classesOnly;
079   }
080   
081   @Override
082   public void execute() throws BuildException {
083     if(toDir == null) {
084       throw new BuildException("Please specify a destination directory using todir", getLocation());
085     }
086     if(toDir.isFile()) {
087       throw new BuildException("Destination already exists and is not a directory", getLocation());
088     }
089 
090     if(Gate.isInitialised()) {
091       log("GATE already initialised, gatehome and pluginshome attributes ignored", Project.MSG_VERBOSE);
092     } else {
093       try {
094         Gate.setGateHome(gateHome);
095         Gate.setPluginsHome(pluginsHome);
096         Gate.init();
097       }
098       catch(GateException e) {
099         throw new BuildException("Error initialising GATE", e, getLocation());
100       }
101     }
102     for(FileSet fs : srcFiles) {
103       DirectoryScanner ds = fs.getDirectoryScanner(getProject());
104       for(String f : ds.getIncludedFiles()) {
105         File creoleFile = new File(ds.getBasedir(), f);
106         try {
107           File plugin = creoleFile.getParentFile();
108           File destFile = new File(toDir, f);
109           File destPlugin = destFile.getParentFile();
110 
111           log("Expanding " + creoleFile + " to " + destFile, Project.MSG_VERBOSE);
112           Gate.addKnownPlugin(plugin.toURI().toURL());
113           CreoleAnnotationHandler annotationHandler = new CreoleAnnotationHandler(creoleFile.toURI().toURL());
114           Document creoleDoc = builder.build(creoleFile);
115           annotationHandler.createResourceElementsForDirInfo(creoleDoc);
116           if(!classesOnly) {
117             annotationHandler.addJarsToClassLoader(Gate.getClassLoader().getDisposableClassLoader(plugin.toURI().toURL().toString()), creoleDoc);
118             annotationHandler.processAnnotations(creoleDoc);
119           }
120           
121           // strip out SCAN="true" attributes as scanning has now been done
122           Iterator scannedJars = creoleDoc.getDescendants(new ElementFilter("JAR").and(new Filter() {
123             public boolean matches(Object o) {
124               return "true".equalsIgnoreCase(((Element)o).getAttributeValue("SCAN"));
125             }
126           }));
127           while(scannedJars.hasNext()) {
128             ((Element)scannedJars.next()).removeAttribute("SCAN");
129           }
130           
131           destPlugin.mkdirs();
132           FileOutputStream fos = new FileOutputStream(destFile);
133           try {
134             outputter.output(creoleDoc, fos);
135           }
136           finally {
137             fos.close();
138           }
139         }
140         catch(Exception e) {
141           log("Error processing " + creoleFile + ", skipped", Project.MSG_WARN);
142           StringWriter sw = new StringWriter();
143           PrintWriter pw = new PrintWriter(sw);
144           e.printStackTrace(pw);
145           log(sw.toString(), Project.MSG_VERBOSE);
146         }
147       }
148     }
149   }
150 
151   public void setGateHome(File gateHome) {
152     this.gateHome = gateHome;
153   }
154   
155   public void setPluginsHome(File pluginsHome) {
156     this.pluginsHome = pluginsHome;
157   }
