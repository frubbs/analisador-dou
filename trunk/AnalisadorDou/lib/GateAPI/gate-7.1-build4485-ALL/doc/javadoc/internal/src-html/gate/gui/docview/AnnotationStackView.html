146   class PreviousAnnotationAction extends AbstractAction {
147     public PreviousAnnotationAction() {
148       super("Previous boundary");
149       putValue(SHORT_DESCRIPTION,
150         "<html>Move to the previous annotation boundary" +
151         "&nbsp;&nbsp;<font color=#667799><small>Alt-Left" +

155     public void actionPerformed(ActionEvent e) {
156       nextAnnotationAction.setEnabled(true);
157       List<Annotation> list = new ArrayList<Annotation>();
158       for(SetHandler setHandler : annotationSetsView.setHandlers) {
159         for(TypeHandler typeHandler: setHandler.typeHandlers) {
160           if (typeHandler.isSelected()) {
161             // adds all the annotations from the selected type contained
162             // between the beginning of the document and the caret position - 1
163             list.addAll(setHandler.set.get(typeHandler.name).getContained(
164               0l, (long)textView.getTextView().getCaretPosition()-1));
165           }
166         }
167       }
168       boolean enabled = false;
169       if (list.size() > 0) {
170         if (overlappingCheckBox.isSelected()) {
171           Collections.sort(list, new OffsetComparator());
172           boolean found = false;
173           for (int i = list.size()-1; i > 0; i--) {
174             if (list.get(i).overlaps(list.get(i-1))) {
175               if (found) { enabled = true; break; }
176               // set the caret on the previous overlapping annotation found
177               textView.getTextView().setCaretPosition(
178                 list.get(i).getEndNode().getOffset().intValue());
179               found = true;
180             }
181           }
182         } else {
183           SortedSet<Annotation> set =
184             new TreeSet<Annotation>(new OffsetComparator());
185           set.addAll(list); // remove same offsets annotations
186           list = new ArrayList<Annotation>(set);
187           // set the caret on the previous annotation found
188           textView.getTextView().setCaretPosition(
189             list.get(list.size()-1).getEndNode().getOffset().intValue());
190           enabled = (list.size() > 1);
191         }
192         try {
193           textView.getTextView().scrollRectToVisible(
194           textView.getTextView().modelToView(
195             textView.getTextView().getCaretPosition()));
196         } catch (BadLocationException exception) {
197           exception.printStackTrace();
198         }
199       }
200       setEnabled(enabled);
201       textView.getTextView().requestFocusInWindow();
202     }
203   }

215       previousAnnotationAction.setEnabled(true);
216       List<Annotation> list = new ArrayList<Annotation>();
217       for(SetHandler setHandler : annotationSetsView.setHandlers) {
218         for(TypeHandler typeHandler: setHandler.typeHandlers) {
219           if (typeHandler.isSelected()) {
220             // adds all the annotations from the selected type contained
221             // between the caret position and the end of the document
222             list.addAll(setHandler.set.get(typeHandler.name).getContained(
223               (long)textView.getTextView().getCaretPosition(),
224               document.getContent().size()-1));
225           }
226         }
227       }
228       boolean enabled = false;
229       if (list.size() > 0) {
230         if (overlappingCheckBox.isSelected()) {
231           Collections.sort(list, new OffsetComparator());
232           boolean found = false;
233           for (int i = 0; i < list.size()-1; i++) {
234             if (list.get(i).overlaps(list.get(i+1))) {
235               if (found) { enabled = true; break; }
236               // set the caret on the next overlapping annotation found
237               textView.getTextView().setCaretPosition(
238                 list.get(i).getEndNode().getOffset().intValue());
239               found = true;
240             }
241           }
242         } else {
243           SortedSet<Annotation> set =
244             new TreeSet<Annotation>(new OffsetComparator());
245           set.addAll(list); // remove same offsets annotations
246           list = new ArrayList<Annotation>(set);
247           // set the caret on the next annotation found
248           textView.getTextView().setCaretPosition(
249             list.get(0).getEndNode().getOffset().intValue());
250           enabled = (list.size() > 1);
251         }
252         try {
253           textView.getTextView().scrollRectToVisible(
254           textView.getTextView().modelToView(
255             textView.getTextView().getCaretPosition()));
256         } catch (BadLocationException exception) {
257           exception.printStackTrace();
258         }
259       }
260       setEnabled(enabled);
261       textView.getTextView().requestFocusInWindow();
262     }

277   public void updateStackView() {
278     if (textView == null) { return; }
279 
280     int caretPosition = textView.getTextView().getCaretPosition();
281 
282     // get the context around the annotation
283     int context = 40;
284     String text = "";
285     try {
286       text = document.getContent().getContent(
287         Math.max(0l, caretPosition - context),
288         Math.min(document.getContent().size(),
289                  caretPosition + 1 + context)).toString();
290     } catch (InvalidOffsetException e) {
291       e.printStackTrace();
292     }
293 
294     // initialise the annotation stack
295     stackPanel.setText(text);
296     stackPanel.setExpressionStartOffset(caretPosition);
297     stackPanel.setExpressionEndOffset(caretPosition + 1);
298     stackPanel.setContextBeforeSize(Math.min(caretPosition, context));
299     stackPanel.setContextAfterSize(Math.min(
300       document.getContent().size().intValue()-1-caretPosition, context));
301     stackPanel.setAnnotationMouseListener(annotationMouseListener);
302     stackPanel.setHeaderMouseListener(headerMouseListener);
303     stackPanel.clearAllRows();
304 
305     // add stack rows and annotations for each selected annotation set
306     // in the annotation sets view
307     for(SetHandler setHandler : annotationSetsView.setHandlers) {
308       for(TypeHandler typeHandler: setHandler.typeHandlers) {
309         if (typeHandler.isSelected()) {
310           stackPanel.addRow(setHandler.set.getName(), typeHandler.name,
311             typesFeatures.get(typeHandler.name),
312             null, null, AnnotationStack.CROP_MIDDLE);
313           Set<Annotation> annotations = setHandler.set.get(typeHandler.name)
314             .get(Math.max(0l, caretPosition - context), Math.min(
315               document.getContent().size(), caretPosition + 1 + context));
316           for (Annotation annotation : annotations) {
317             stackPanel.addAnnotation(annotation);
318           }
319         }
320       }
321     }
322 
323     stackPanel.drawStack();
324   }

326   /** @return true if the user input a valid annotation set */
327   boolean askTargetSet() {
328     Object[] messageObjects;
329     final JTextField setsTextField = new JTextField("consensus", 15);
330     if (document.getAnnotationSetNames() != null
331      && !document.getAnnotationSetNames().isEmpty()) {
332       Collator collator = Collator.getInstance(Locale.ENGLISH);
333       collator.setStrength(Collator.TERTIARY);
334       SortedSet<String> setNames = new TreeSet<String>(collator);
335       setNames.addAll(document.getAnnotationSetNames());
336       JList list = new JList(setNames.toArray());
337       list.setVisibleRowCount(Math.min(10, setNames.size()));
338       list.setSelectionMode(ListSelectionModel.SINGLE_SELECTION);
339       list.setSelectedValue(targetSetName, true);
340       JScrollPane scroll = new JScrollPane(list);
341       JPanel vspace = new JPanel();
342       vspace.setSize(0, 5);
343       list.addListSelectionListener(new ListSelectionListener() {
344         public void valueChanged(ListSelectionEvent e) {
345           JList list = (JList) e.getSource();
346           if (list.getSelectedValue() != null) {
347             setsTextField.setText((String) list.getSelectedValue());
348           }
349         }
350       });
351       messageObjects = new Object[] { "Existing annotation sets:",
352         scroll, vspace, "Target set:", setsTextField };
353     } else {
354       messageObjects = new Object[] { "Target set:", setsTextField };
355     }
356     String options[] = { "Use this target set", "Cancel" };
357     JOptionPane optionPane = new JOptionPane(
358       messageObjects, JOptionPane.QUESTION_MESSAGE,
359       JOptionPane.YES_NO_OPTION, null, options, "Cancel");
360     JDialog optionDialog = optionPane.createDialog(
361       owner, "Copy annotation to another set");
362     setsTextField.requestFocus();
363     optionDialog.setVisible(true);
364     Object selectedValue = optionPane.getValue();
365     if (selectedValue == null
366      || selectedValue.equals("Cancel")
367      || setsTextField.getText().trim().length() == 0) {
368       textView.getTextView().requestFocusInWindow();
369       return false;
370     }
371     targetSetName = setsTextField.getText();
372     targetSetLabel.setText("Target set: " + targetSetName);
373     textView.getTextView().requestFocusInWindow();
374     return true;
375   }

411     public void processMouseEvent(MouseEvent me) {
412 
413       if(me.isPopupTrigger()) { // context menu
414         // add annotation editors to the context menu
415         JPopupMenu popup = new JPopupMenu();
416         List<Action> specificEditorActions =
417           annotationListView.getSpecificEditorActions(
418           annotationData.getAnnotationSet(), annotationData.getAnnotation());
419         for (Action action : specificEditorActions) {
420           popup.add(action);
421         }
422         List<Action> genericEditorActions =
423           annotationListView.getGenericEditorActions(
424             annotationData.getAnnotationSet(), annotationData.getAnnotation());
425         for (Action action : genericEditorActions) {
426           if (specificEditorActions.contains(action)) { continue; }
427           popup.add(action);
428         }
429         if (specificEditorActions.size() + genericEditorActions.size() > 1) {
430           popup.show(me.getComponent(), me.getX(), me.getY());
431         } else { // only one choice so use it directly
432           if (specificEditorActions.size() == 1) {
433             specificEditorActions.get(0).actionPerformed(null);
434           } else {
435             genericEditorActions.get(0).actionPerformed(null);
436           }
437         }
438 
439       } else if (me.getID() == MouseEvent.MOUSE_CLICKED
440               && me.getButton() == MouseEvent.BUTTON1
441               && me.getClickCount() == 1
442               && (me.getModifiersEx() & MouseEvent.CTRL_DOWN_MASK) != 0
443               && (me.getModifiersEx() & MouseEvent.SHIFT_DOWN_MASK) != 0) {
444               // control + shift + click -> delete the annotation
445         annotationData.getAnnotationSet().remove(annotation);
446 
447       } else if (me.getID() == MouseEvent.MOUSE_CLICKED
448               && me.getButton() == MouseEvent.BUTTON1
449               && me.getClickCount() == 1
450               && (me.getModifiersEx() & MouseEvent.CTRL_DOWN_MASK) != 0) {
451                // control + click
452         String featureDisplayed = typesFeatures.get(annotation.getType());
453         Object[] features;
454         if (featureDisplayed != null) {
455           features = new Object[]{featureDisplayed};
456         } else {
457           features = annotation.getFeatures().keySet().toArray();
458         }
459         for (Object feature : features) {
460           String value = (String) annotation.getFeatures().get(feature);
461           if (value == null) { continue; }
462           Pattern pattern = Pattern.compile("(https?://[^\\s,;]+)");
463           Matcher matcher = pattern.matcher(value);
464           if (matcher.find()) {
465             // if the feature value contains an url display it in a browser
466             MainFrame.getInstance().showHelpFrame(
467               matcher.group(), "Annotation Stack View");
468           }
469         }
470 
471       } else if (me.getID() == MouseEvent.MOUSE_CLICKED
472               && me.getButton() == MouseEvent.BUTTON1
473               && me.getClickCount() == 2) { // double click
474         if (targetSetName == null) {
475           if (!askTargetSet()) { return; }
476         }
477         // copy the annotation to the target annotation set
478         try {
479           document.getAnnotations(targetSetName).add(
480             annotation.getStartNode().getOffset(),
481             annotation.getEndNode().getOffset(),
482             annotation.getType(),
483             annotation.getFeatures());
484         } catch (InvalidOffsetException e) {
485           e.printStackTrace();
486         }
487         // wait some time
488         Date timeToRun = new Date(System.currentTimeMillis() + 500);
489         Timer timer = new Timer("Annotation stack view select type", true);
490         timer.schedule(new TimerTask() {
491           public void run() {
492             SwingUtilities.invokeLater(new Runnable() { public void run() {
493               // select the new annotation and update the stack view
494               annotationSetsView.setTypeSelected(targetSetName,
495                 annotation.getType(), true);
496             }});
497           }
498         }, timeToRun);
499       }
500       textView.getTextView().requestFocusInWindow();
501     }

503     public void mouseEntered(MouseEvent e) {
504       dismissDelay = toolTipManager.getDismissDelay();
505       initialDelay = toolTipManager.getInitialDelay();
506       reshowDelay = toolTipManager.getReshowDelay();
507       enabled = toolTipManager.isEnabled();
508       Component component = e.getComponent();
509       if (component instanceof JLabel
510       && !((JLabel)component).getToolTipText().contains("Ctr-Sh-click")) {
511         JLabel label = (JLabel) component;
512         String toolTip = (label.getToolTipText() == null) ?

580     // when double clicked shows a list of features for this annotation type
581     public void mouseClicked(MouseEvent e) {
582       if (popupWindow != null && popupWindow.isVisible()) {
583         popupWindow.dispose();
584         return;
585       }
586       if (e.getButton() != MouseEvent.BUTTON1
587        || e.getClickCount() != 2) {
588         return;
589       }
590       // get a list of features for the current annotation type
591       TreeSet<String> features = new TreeSet<String>();
592       Set<String> setNames = new HashSet<String>();
593       if (document.getAnnotationSetNames() != null) {
594         setNames.addAll(document.getAnnotationSetNames());
595       }
596       setNames.add(null); // default set name
597       for (String setName : setNames) {
598         int count = 0;
599         for (Annotation annotation :
600           document.getAnnotations(setName).get(type)) {
601           for (Object feature : annotation.getFeatures().keySet()) {
602             features.add((String) feature);
603           }
604           count++; // checks only the 50 first annotations per set
605           if (count == 50) { break; } // to avoid slowing down
606         }
607       }
608       features.add("          ");
609       // create the list component
610       final JList list = new JList(features.toArray());
611       list.setVisibleRowCount(Math.min(8, features.size()));
612       list.setSelectionMode(ListSelectionModel.SINGLE_SELECTION);
613       list.setBackground(Color.WHITE);
614       list.addMouseListener(new MouseAdapter() {
615         public void mouseClicked(MouseEvent e) {
616           if (e.getClickCount() == 1) {
617             String feature = (String) list.getSelectedValue();
618             if (feature.equals("          ")) {
619               typesFeatures.remove(type);
620             } else {
621               typesFeatures.put(type, feature);
622             }
623             popupWindow.setVisible(false);
624             popupWindow.dispose();
625             updateStackView();
626             textView.getTextView().requestFocusInWindow();
627           }
628         }
629       });
630       // create the window that will contain the list
631       popupWindow = new JWindow();
632       popupWindow.addKeyListener(new KeyAdapter() {
633         public void keyPressed(KeyEvent e) {
634           if (e.getKeyCode() == KeyEvent.VK_ESCAPE) {
635             popupWindow.setVisible(false);
636             popupWindow.dispose();
637           }
638         }
639       });
640       popupWindow.add(new JScrollPane(list));
641       Component component = e.getComponent();
642       popupWindow.setBounds(
643         component.getLocationOnScreen().x,
644         component.getLocationOnScreen().y + component.getHeight(),
645         component.getWidth(),
646         Math.min(8*component.getHeight(),
647           features.size()*component.getHeight()));
648       popupWindow.pack();
649       popupWindow.setVisible(true);
650       SwingUtilities.invokeLater(new Runnable() {
651       public void run() {
652         String newFeature = typesFeatures.get(type);
653         if (newFeature == null) { newFeature = "          "; }
654         list.setSelectedValue(newFeature, true);
655         popupWindow.requestFocusInWindow();
656       }});
657     }
