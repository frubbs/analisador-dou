031 /**
032  * Action to create a new annotation property.
033  */
034 public class AnnotationPropertyAction extends AbstractAction {
035   private static final long serialVersionUID = 3546358452780544048L;
036 
037   /**
038    * Constructor
039    * 
040    * @param s
041    *          - Label assigned to the Button
042    * @param icon
043    *          - Icon assigned to the Button
044    */
045   public AnnotationPropertyAction(String s, Icon icon) {
046     super(s, icon);
047     mainPanel = new JPanel(new GridBagLayout());
048     GridBagConstraints gbc = new GridBagConstraints();
049     gbc.insets = new Insets(3, 3, 3, 3);
050     gbc.anchor = GridBagConstraints.WEST;
051 
052     mainPanel.add(new JLabel("Name Space:"), gbc);
053     mainPanel.add(nameSpace = new JTextField(30), gbc);
054 
055     gbc.gridy = 1;
056     mainPanel.add(new JLabel("Property Name:"), gbc);
057     mainPanel.add(propertyName = new JTextField(30), gbc);
058   }
059 
060   public void actionPerformed(ActionEvent actionevent) {
061     nameSpace.setText(ontology.getDefaultNameSpace() == null
062         ? "http://gate.ac.uk/example#"
063         : ontology.getDefaultNameSpace());
064     JOptionPane pane =
065         new JOptionPane(mainPanel, JOptionPane.QUESTION_MESSAGE,
066             JOptionPane.OK_CANCEL_OPTION,
067             MainFrame.getIcon("ontology-annotation-property")) {
068           public void selectInitialValue() {
069             propertyName.requestFocusInWindow();
070             propertyName.selectAll();
071           }
072         };
073     pane.createDialog(MainFrame.getInstance(), "New Annotation Property")
074         .setVisible(true);
075     Object selectedValue = pane.getValue();
076     if(selectedValue != null && selectedValue instanceof Integer
077         && (Integer)selectedValue == JOptionPane.OK_OPTION) {
078       String s = nameSpace.getText();
079       if(!Utils.isValidNameSpace(s)) {
080         JOptionPane.showMessageDialog(MainFrame.getInstance(),
081             "Invalid Name Space: " + s
082                 + "\nExample: http://gate.ac.uk/example#");
083         return;
084       }
085       if(!Utils.isValidOntologyResourceName(propertyName.getText())) {
086         JOptionPane.showMessageDialog(MainFrame.getInstance(),
087             "Invalid Property Name: " + propertyName.getText());
088         return;
089       }
090       if(Utils.getOResourceFromMap(ontology, s + propertyName.getText()) != null) {
091         JOptionPane.showMessageDialog(MainFrame.getInstance(), "<html>"
092             + "Resource <b>" + s + propertyName.getText()
093             + "</b> already exists.");
094         return;
095       }
096 
097       ontology.addAnnotationProperty(ontology.createOURI(nameSpace.getText()
098           + propertyName.getText()));
099     }
100   }
101 
102   /**
103    * @return the associated ontology
104    */
105   public Ontology getOntology() {
106     return ontology;
107   }
108 
109   /**
110    * Specifies the ontology that should be used to add/remove resource to/from.
111    */
112   public void setOntology(Ontology ontology) {
113     this.ontology = ontology;
114   }

116   protected JPanel mainPanel;
117 
118   protected JTextField nameSpace;
119 
120   protected JTextField propertyName;
121 
122   protected Ontology ontology;
123 }
