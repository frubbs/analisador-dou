161   /**
162    * Called internally whenever the data represented changes.
163    * Get feature names from targetFeatures and schema then sort them
164    * and add them to featureList.
165    * Fire a table data changed event for the feature table whith featureList
166    * used as data model.
167    */
168   protected void populate(){
169     featureList.clear();
170     //get all the existing features
171     Set fNames = new HashSet();
172     
173     if(targetFeatures != null){
174       //add all the schema features
175       fNames.addAll(targetFeatures.keySet());
176       if(schema != null && schema.getFeatureSchemaSet() != null){
177         for(FeatureSchema featureSchema : schema.getFeatureSchemaSet()) {
178           //        if(featureSchema.isRequired())
179           fNames.add(featureSchema.getFeatureName());
180         }
181       }
182       List featureNames = new ArrayList(fNames);
183       Collections.sort(featureNames);
184       for(Object featureName : featureNames) {
185         String name = (String) featureName;
186         Object value = targetFeatures.get(name);
187         featureList.add(new Feature(name, value));
188       }
189     }
190     if (!featureList.contains(emptyFeature)) {
191       featureList.add(emptyFeature);
192     }
193     featuresModel.fireTableDataChanged();
194 //    mainTable.setSize(mainTable.getPreferredScrollableViewportSize());
195   }

215   protected class Feature{
216     String name;
217     Object value;
218 
219     public Feature(String name, Object value){
220       this.name = name;
221       this.value = value;
222     }
223     boolean isSchemaFeature(){
224       return schema != null && schema.getFeatureSchema(name) != null;
225     }
226     boolean isCorrect(){
227       if(schema == null) return true;
228       FeatureSchema fSchema = schema.getFeatureSchema(name);
229       return fSchema == null || fSchema.getPermittedValues() == null||
230              fSchema.getPermittedValues().contains(value);
231     }
232     boolean isRequired(){
233       if(schema == null) return false;
234       FeatureSchema fSchema = schema.getFeatureSchema(name);
235       return fSchema != null && fSchema.isRequired();
236     }
237     Object getDefaultValue(){
238       if(schema == null) return null;
239       FeatureSchema fSchema = schema.getFeatureSchema(name);
240       return fSchema == null ? null : fSchema.getFeatureValue();
241     }
242   }

271     public void setValueAt(Object aValue, int rowIndex,  int columnIndex){
272       Feature feature = featureList.get(rowIndex);
273       if (feature == null) { return; }
274       if(targetFeatures == null){
275         targetFeatures = Factory.newFeatureMap();
276         target.setFeatures(targetFeatures);
277         setTargetFeatures(targetFeatures);
278       }
279       switch(columnIndex){
280         case VALUE_COL:
281           if (feature.value != null
282            && feature.value.equals(aValue)) { return; }
283           feature.value = aValue;
284           if(feature.name != null && feature.name.length() > 0){
285             targetFeatures.removeFeatureMapListener(instance);
286             targetFeatures.put(feature.name, aValue);
287             targetFeatures.addFeatureMapListener(instance);
288             SwingUtilities.invokeLater(new Runnable() {
289               public void run() {
290                 // edit the last row that is empty
291                 FeaturesSchemaEditor.this.editCellAt(FeaturesSchemaEditor.this.getRowCount() - 1, NAME_COL);
292               }
293             });
294           }
295           break;
296         case NAME_COL:
297           if (feature.name.equals(aValue)) {
298             return;
299           }
300           targetFeatures.remove(feature.name);
301           feature.name = (String)aValue;
302           targetFeatures.put(feature.name, feature.value);
303           if(feature == emptyFeature) emptyFeature = new Feature("", null);
304           populate();
305           int newRow;
306           for (newRow = 0; newRow < FeaturesSchemaEditor.this.getRowCount(); newRow++) {
307             if (FeaturesSchemaEditor.this.getValueAt(newRow, NAME_COL).equals(feature.name)) {
308               break; // find the previously selected row in the new table
309             }
310           }
311           final int newRowF = newRow;
312           SwingUtilities.invokeLater(new Runnable() {
313             public void run() {
314               // edit the cell containing the value associated with this name
315               FeaturesSchemaEditor.this.editCellAt(newRowF, VALUE_COL);
316             }
317           });
318           break;
319         case DELETE_COL:
320           //nothing
321           break;
322         default:
323           throw new GateRuntimeException("Non editable cell!");
324       }
325       
326     }

492     protected void prepareCombo(JComboBox combo, int row, int column){
493       Feature feature = featureList.get(row);
494       DefaultComboBoxModel comboModel = (DefaultComboBoxModel)combo.getModel(); 
495       comboModel.removeAllElements();
496       switch(column){
497         case NAME_COL:
498           List fNames = new ArrayList();
499           if(schema != null && schema.getFeatureSchemaSet() != null){
500             Iterator fSchemaIter = schema.getFeatureSchemaSet().iterator();
501             while(fSchemaIter.hasNext())
502               fNames.add(((FeatureSchema)fSchemaIter.next()).getFeatureName());
503           }
504           if(!fNames.contains(feature.name))fNames.add(feature.name);
505           Collections.sort(fNames);
506           for(Iterator nameIter = fNames.iterator(); 
507               nameIter.hasNext(); 
508               comboModel.addElement(nameIter.next()));
509           combo.getEditor().getEditorComponent().setBackground(defaultBackground);          
510           combo.setSelectedItem(feature.name);
511           break;
512         case VALUE_COL:
513           List fValues = new ArrayList();
514           if(feature.isSchemaFeature()){
515             Set permValues = schema.getFeatureSchema(feature.name).
516               getPermittedValues();
517             if(permValues != null) fValues.addAll(permValues);
518           }
519           if(!fValues.contains(feature.value)) fValues.add(feature.value);
520           Collections.sort(fValues, defaultComparator);
521           for(Iterator valIter = fValues.iterator(); 
522               valIter.hasNext(); 
523               comboModel.addElement(valIter.next()));
524           combo.getEditor().getEditorComponent().setBackground(feature.isCorrect() ?
525               defaultBackground :
526               (feature.isRequired() ? REQUIRED_WRONG : OPTIONAL_WRONG));
527           combo.setSelectedItem(feature.value);
528           break;
529         default: ;
530       }
