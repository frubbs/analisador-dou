026 /**
027  * A set of predicates/expressions that apply to a single
028  * {@link Annotation} type. It doesn't extend PatternElement, even
029  * though it has to "match", because a set of Constraint must be applied
030  * together in order to avoid doing separate selectAnnotations calls for
031  * each one.
032  * <p>
033  * <b>Matching Logic:</b>The matching function of a non-negated
034  * constraint can be defined as, "There exists an annotation for which
035  * all of the predicates are true." Thus, given a collection of
036  * annotations, only a single annotation must meet all of the predicates
037  * of the constraint in order for the match to be successful.
038  * <p>
039  * <b>Negation:</b> A negated constraint means strictly the opposite -
040  * "There does not exist an annotation for which any of these predicates
041  * are true." Negation does <b>not</b> mean "There exists an annotation
042  * for which all of these predicates are false." Thus, negation makes
043  * more intuitive sense when thought of as applying to a set of
044  * annotations rather than to an individual annotation.
045  */
046 public class Constraint implements JapeConstants, gate.creole.ANNIEConstants,
047                        java.io.Serializable, Cloneable {

124   /** Add an attribute. */
125   public void addAttribute(ConstraintPredicate attr) {
126     predicates.add(attr);
127 
128     // check for LOOKUP_CLASS_FEATURE_NAME and
129     // LOOKUP_ONTOLOGY_FEATURE_NAME
130     // predicates and store some information about them once so we don't
131     // have to look it up on every matches call.
132     if(attr.getAccessor() instanceof AnnotationFeatureAccessor) {
133       String featureName = (String)attr.getAccessor().getKey();
134       if(featureName.equals(LOOKUP_CLASS_FEATURE_NAME)) {
135         ontLookupClassPred = attr;
136         getOntFeatureMap().put(LOOKUP_CLASS_FEATURE_NAME,
137                 ontLookupClassPred.getValue());
138         if(!attr.getOperator().equals(ConstraintPredicate.EQUAL))
139           gate.util.Err
140                   .println("Warning: If an ontology is specified at runtime, "
141                          + "Ontology class feature will be compared using "
142                          + "ontology subsumption, not " + attr.getOperator());
143       }
144       else if(featureName.equals(LOOKUP_ONTOLOGY_FEATURE_NAME)) {
145         getOntFeatureMap().put(LOOKUP_ONTOLOGY_FEATURE_NAME, attr.getValue());
146       }
147     }
148 
149   } // addAttribute

289   /**
290    * Invoke {@link #matches(Annotation, Ontology, AnnotationSet)} on all provided
291    * annotations.
292    *
293    * @param annots collection of Annotations to test
294    * @param ontology optional Ontology to compare ont-specific features
295    * @return Collection of annotations which matches successfully
296    *         against predicates.
297    */
298   public List<Annotation> matches(Collection<Annotation> annots,
299           Ontology ontology, AnnotationSet context) {
300     List<Annotation> retVal = new ArrayList<Annotation>();
301 
302     if(annots == null) return retVal;
303 
304     for(Annotation annot : annots) {
305       if(matches(annot, ontology, context)) {
306         retVal.add(annot);
307         // small optimization - if constraint is negated and there is a
308         // match,
309         // don't bother checking all the other annots since a negated
310         // constraint
311         // fails if even a single annotation matches it.
312         if(negated) break;
313       }
314     }
315 
316     return retVal;
317   }

319   /**
320    * Test if an annotation is of the proper type for this constraint and
321    * if it complies with the {@link ConstraintPredicate}s of this
322    * constraint.
323    *
324    * @param annot an Annotation
325    * @param ontologyLR optional ontology to use when comparing
326    *          ont-related features
327    * @return <code>true</code> if the annotation is of the proper type
328    *         and matches all predicates. If the constraint is negated,
329    *         an annotation need only match a single predicate to return
330    *         true.
331    */
332   public final boolean matches(Annotation annot, Ontology ontologyLR, 
333           AnnotationSet context) {
334     if(annot == null) return false;
335     if(!annot.getType().equals(getAnnotType())) return false;
336 
337     // if no predicates, then we have a match based solely on the annot
338     // type.
339     if(predicates == null || predicates.isEmpty()) return true;
340 
341     for(ConstraintPredicate predicate : predicates) {
342       boolean successful = false;
343       try {
344         // do some special checking if this predicate deals with the
345         // ontology.  Note that we assume the operator for the predicate
346         // was ==. We issue a warning when the predicate is first set
347         // if it's anything else.
348         if(predicate == ontLookupClassPred && ontologyLR != null)
349           successful = annot.getFeatures().subsumes(ontologyLR, ontFeatureMap);
350         else successful = predicate.matches(annot, context);
351       } catch(JapeException je) {
352         gate.util.Err.println(je.getMessage());
353       }
354       if(!successful) return false;
355       // else, keep checking the rest of the predicates
356     } // checked all predicates
357     return true;
358   }

361    * Test if an annotation is of the proper type for this constraint and
362    * if it complies with the {@link ConstraintPredicate}s of this
363    * constraint.
364    *
365    * @param annot a Annotation
366    * @return <code>true</code> if the annotation is of the proper type
367    *         and matches all predicates. If the constraint is negated,
368    *         an annotation need only match a single predicate to return
369    *         true.
370    */
371   public boolean matches(Annotation annot, AnnotationSet context) {
372     return matches(annot, null, context);
373   }
