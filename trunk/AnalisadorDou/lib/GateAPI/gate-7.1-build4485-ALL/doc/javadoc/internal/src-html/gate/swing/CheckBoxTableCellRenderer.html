15 package gate.swing;
16 
17 import java.awt.Color;
18 import java.awt.Component;
19 
20 import javax.swing.BorderFactory;
21 import javax.swing.JCheckBox;
22 import javax.swing.JTable;
23 import javax.swing.UIManager;
24 import javax.swing.border.Border;
25 import javax.swing.table.TableCellRenderer;
26 
27 /**
28  * A TableCellRenderer for JCheckBox that disables the checkbox when the cell
29  * isn't editable to make it clear that you can't click on it
30  * 
31  * @author Mark A. Greenwood
32  */
33 @SuppressWarnings("serial")
34 public class CheckBoxTableCellRenderer extends JCheckBox implements
35                                                         TableCellRenderer {
36 
37   /**
38    * An empty border we use when the cell doesn't have focus
39    */
40   private static final Border NO_FOCUS = BorderFactory.createEmptyBorder(1, 1,
41           1, 1);
42 
43   public CheckBoxTableCellRenderer() {
44     super();
45 
46     // centre the checkbox within the cell
47     setHorizontalAlignment(JCheckBox.CENTER);
48 
49     // make sure we always paint the cell border
50     setBorderPainted(true);
51 
52     // make sure we always paint the background
53     setOpaque(true);
54   }
55 
56   public Component getTableCellRendererComponent(JTable table, Object value,
57           boolean isSelected, boolean hasFocus, int row, int column) {
58 
59     // this is needed for Nimbus which has alternative rows in different colours
60     // hopefully other L&Fs that also do this use the same UIManager key
61     Color alternate = UIManager.getColor("Table.alternateRowColor");
62 
63     // strangely the background color from Nimbus doesn't render properly unless
64     // we convert it in this way. I'm guessing the problem is to do with the
65     // DerivedColor class that nimbus uses
66     Color normal = new Color(table.getBackground().getRGB());
67 
68     if(isSelected) {
69       // if the cell is selected then set it's colors from the table
70       setForeground(table.getSelectionForeground());
71       setBackground(table.getSelectionBackground());
72     } else {
73       // if the cell isn't selected set it's colors taking into account the
74       // possibility of alternating colors that Nimbus throws into the mix
75       setForeground(table.getForeground());
76       setBackground(alternate != null && row % 2 == 0 ? alternate : normal);
77     }
78 
79     // if the cell isn't editable then disable the checkbox to give some visual
80     // feedback to the user
81     setEnabled(table.isCellEditable(row, column));
82 
83     // make the checkbox reflect the underlying boolean data
84     setSelected(value != null && (Boolean)value);
85 
86     if(hasFocus) {
87       // if the cell has focus then draw the border set by the L&F
88       setBorder(UIManager.getBorder("Table.focusCellHighlightBorder"));
89     } else {
90       // if the cell doesn't have the focus then draw the empty border
91       setBorder(NO_FOCUS);
92     }
93 
94     // now return the checkbox for rendering into the table
95     return this;
96   }
97 }
