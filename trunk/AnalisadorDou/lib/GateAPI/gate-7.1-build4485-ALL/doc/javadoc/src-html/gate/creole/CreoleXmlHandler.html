067   /**
068    * The current parameter disjunction.  This is a map where each key
069    * is a "priority" and the value is a list of parameters tagged with
070    * that priority.  This map is flattened into a single list when
071    * all the parameters in a single disjunction have been processed,
072    * such that parameters with a smaller priority value are listed
073    * ahead of those with a larger value, and those with no explicit
074    * priority are listed last of all.  Parameters at the same priority
075    * are listed in document order.  This is not so useful when writing
076    * creole.xml files by hand but is necessary to ensure a predictable
077    * order when using CreoleParameter annotations.  The GATE developer
078    * GUI offers the first listed (i.e. highest priority) parameter for
079    * each disjunction as the default option in the resource parameters
080    * dialog box. 
081    */
082   private SortedMap<Integer, List<Parameter>> currentParamDisjunction =
083     new TreeMap<Integer, List<Parameter>>();

152   /** A verboase method for Attributes*/
153   private String attributes2String(Attributes atts){
154     StringBuffer strBuf = new StringBuffer("");
155     if (atts == null) return strBuf.toString();
156     for (int i = 0; i < atts.getLength(); i++) {
157      String attName  = atts.getQName(i);
158      String attValue = atts.getValue(i);
159      strBuf.append(" ");
160      strBuf.append(attName);
161      strBuf.append("=");
162      strBuf.append(attValue);
163     }// End for
164     return strBuf.toString();
165   }// attributes2String()

167   /** Called when the SAX parser encounts the beginning of an XML element */
168   public void startElement (String uri, String qName, String elementName,
169                                                              Attributes atts) throws SAXException {
170 
171     // call characterActions
172     if(readCharacterStatus) {
173       readCharacterStatus = false;
174       charactersAction(new String(contentBuffer).toCharArray(),0,contentBuffer.length());
175     }
176 
177     if(DEBUG) {
178       Out.pr("startElement: ");
179       Out.println(
180         elementName + " " +
181         attributes2String(atts)
182       );
183     }
184 
185     // create a new ResourceData when it's a RESOURCE element
186     if(elementName.toUpperCase().equals("RESOURCE")) {
187       resourceData = new ResourceData();
188       resourceData.setFeatures(Factory.newFeatureMap());
189       currentAutoinstances = new ArrayList<FeatureMap>();
190     }// End if RESOURCE
191 
192     // record the attributes of this element
193     currentAttributes = atts;
194 
195     // When an AUTOINSTANCE element is found a params FeatureMap will
196     // be prepared in order for the resource to be instantiated
197     if (elementName.toUpperCase().equals("AUTOINSTANCE")){
198       currentAutoinstanceParams = Factory.newFeatureMap();
199     }// End if AUTOINSTANCE
200 
201     //When an HIDDEN-AUTOINSTANCE element is found a params FeatureMap will
202     // be prepared in order for the resource to be instantiated
203     if (elementName.toUpperCase().equals("HIDDEN-AUTOINSTANCE")){
204       currentAutoinstanceParams = Factory.newFeatureMap();
205       Gate.setHiddenAttribute(currentAutoinstanceParams, true);
206     }// End if AUTOINSTANCE
207     
208     // When a PARAN start element is found, the parameter would be instantiated
209     // with a value and added to the autoinstanceParams
210     if (elementName.toUpperCase().equals("PARAM")){
211       // The autoinstanceParams should always be != null because of the fact
212       // that PARAM is incuded into an AUTOINSTANCE element.
213       // IF a AUTOINSTANCE starting element would be missing then the
214       // parser would signal this later....
215       if (currentAutoinstanceParams == null)
216         currentAutoinstanceParams = Factory.newFeatureMap();
217       // Take the param's name and value
218       String paramName = currentAttributes.getValue("NAME");
219       String paramStrValue = currentAttributes.getValue("VALUE");
220       if (paramName == null)
221         throw new GateRuntimeException ("Found in creole.xml a PARAM element" +
222         " for resource "+ resourceData.getClassName()+ " without a NAME"+
223         " attribute. Check the file and try again.");
224       if (paramStrValue == null)
225         throw new GateRuntimeException("Found in creole.xml a PARAM element"+
226         " for resource "+ resourceData.getClassName()+ " without a VALUE"+
227         " attribute. Check the file and try again.");
228       // Add the paramname and its value to the autoinstanceParams
229       currentAutoinstanceParams.put(paramName,paramStrValue);
230     }// End if PARAM
231 
232     // process attributes of parameter and GUI elements
233     if(elementName.toUpperCase().equals("PARAMETER")) {
234       if(DEBUG) {
235         for(int i=0, len=currentAttributes.getLength(); i<len; i++) {
236           Out.prln(currentAttributes.getLocalName(i));
237           Out.prln(currentAttributes.getValue(i));
238         }// End for
239       }// End if
240       currentParam.comment = currentAttributes.getValue("COMMENT");
241       currentParam.helpURL = currentAttributes.getValue("HELPURL");
242       currentParam.defaultValueString = currentAttributes.getValue("DEFAULT");
243       currentParam.optional =
244         Boolean.valueOf(currentAttributes.getValue("OPTIONAL")).booleanValue();
245       currentParam.name = currentAttributes.getValue("NAME");
246       currentParam.runtime =
247         Boolean.valueOf(currentAttributes.getValue("RUNTIME")).booleanValue();
248       currentParam.itemClassName =
249                                 currentAttributes.getValue("ITEM_CLASS_NAME");
250       // read the suffixes and transform them to a Set of Strings
251       String suffixes = currentAttributes.getValue("SUFFIXES");
252       Set<String> suffiexesSet = null;
253       if (suffixes != null){
254         suffiexesSet = new HashSet<String>();
255         StringTokenizer strTokenizer = new StringTokenizer(suffixes,";");
256         while(strTokenizer.hasMoreTokens()){
257            suffiexesSet.add(strTokenizer.nextToken());
258         }// End while
259       }// End if
260       currentParam.suffixes = suffiexesSet;
261     }else if(elementName.toUpperCase().equals("GUI")){
262       String typeValue = currentAttributes.getValue("TYPE");
263       if (typeValue != null){
264         if (typeValue.toUpperCase().equals("LARGE"))
265           resourceData.setGuiType(ResourceData.LARGE_GUI);
266         if (typeValue.toUpperCase().equals("SMALL"))
267           resourceData.setGuiType(ResourceData.SMALL_GUI);
268       }// End if
269     }// End if
270 
271     // if there are any parameters awaiting addition to the list, add them
272     // (note that they're not disjunctive or previous "/OR" would have got 'em)
273     if(elementName.toUpperCase().equals("OR")) {
274       if(! currentParamDisjunction.isEmpty()) {
275         currentParamList.addAll(currentFlattenedDisjunction());
276         currentParamDisjunction.clear();
277       }// End if
278     }// End if
279   } // startElement()

288   } // checkStack

290   /** Called when the SAX parser encounts the end of an XML element.
291     * This is where ResourceData objects get values set, and where
292     * they are added to the CreoleRegister when we parsed their complete
293     * metadata entries.
294     */
295   public void endElement (String uri, String qName, String elementName)
296                                                     throws GateSaxException, SAXException {
297     // call characterActions
298     if(readCharacterStatus) {
299       readCharacterStatus = false;
300       charactersAction(new String(contentBuffer).toCharArray(),0,contentBuffer.length());
301     }
302 
303     if(DEBUG) Out.prln("endElement: " + elementName);
304 
305     //////////////////////////////////////////////////////////////////
306     if(elementName.toUpperCase().equals("RESOURCE")) {
307       // check for validity of the resource data
308       if(! resourceData.isValid())
309         throw new GateSaxException(
310           "Invalid resource data: " + resourceData.getValidityMessage()
311         );
312 
313       //set the URL to the creole.xml file on the resource data object
314       resourceData.setXmlFileUrl(creoleFileUrl);
315       // add the new resource data object to the creole register
316       register.put(resourceData.getClassName(), resourceData);
317       // if the resource is auto-loading, try and load it
318       if(resourceData.isAutoLoading())
319         try {
320           @SuppressWarnings("unused")
321           Class<?> resClass = resourceData.getResourceClass();
322 //          Resource res = Factory.createResource(
323 //              resourceData.getClassName(), Factory.newFeatureMap()
324 //          );
325 //          resourceData.makeInstantiationPersistant(res);
326         } catch(ClassNotFoundException e) {
327           throw new GateSaxException(
328             "Couldn't load autoloading resource: " +
329             resourceData.getName() + "; problem was: " + e
330           );
331         }// End try
332 
333       // if there are any parameters awaiting addition to the list, add them
334       // (note that they're not disjunctive or the "/OR" would have got them)
335       if(! currentParamDisjunction.isEmpty()) {
336         currentParamList.addAll(currentFlattenedDisjunction());
337         currentParamDisjunction.clear();
338       }// End if
339 
340       // add the parameter list to the resource (and reinitialise it)
341       resourceData.setParameterList(currentParamList);
342       currentParamList = new ParameterList();
343       
344       // final initialization of the ResourceData
345       try {
346         resourceData.init();
347       } catch(Exception ex) {
348         throw new GateSaxException(
349             "Couldn't initialize ResourceData for "
350             + resourceData.getName(), ex);
351       }
352 
353       if(DEBUG) Out.println("added: " + resourceData);
354       // Iterate through autoinstances and try to instanciate them
355       if ( currentAutoinstances != null && !currentAutoinstances.isEmpty()){
356         ResourceData rd = Gate.getCreoleRegister().get(resourceData.getClassName());
357         ParameterList existingParameters = null;
358         if (rd.getReferenceCount() > 1) {
359           // we aren't going to redefine a resource but we do need to use the
360           // parameters from the new instance so we get the right base URL and
361           // default values etc.
362           existingParameters = rd.getParameterList();
363           rd.setParameterList(resourceData.getParameterList());
364         }
365         
366         try {
367           Iterator<FeatureMap> iter = currentAutoinstances.iterator();
368           while (iter.hasNext()){
369             FeatureMap autoinstanceParams = iter.next();
370             iter.remove();
371             FeatureMap autoinstanceFeatures = null;
372             //if the hidden attribute was set in the parameters, create a feature 
373             //map and move the hidden attribute there.
374             if(Gate.getHiddenAttribute(autoinstanceParams)){
375               autoinstanceFeatures = Factory.newFeatureMap();
376               Gate.setHiddenAttribute(autoinstanceFeatures, true);
377               autoinstanceParams.remove(GateConstants.HIDDEN_FEATURE_KEY);
378             }
379             
380             // Try to create the resource.
381             try {
382               // Resource res = 
383               Factory.createResource(
384                                 resourceData.getClassName(), autoinstanceParams, 
385                                 autoinstanceFeatures);
386               //resourceData.makeInstantiationPersistant(res);
387               // all resource instantiations are persistent
388             } catch(ResourceInstantiationException e) {
389               throw new GateSaxException(
390                 "Couldn't auto-instantiate resource: " +
391                 resourceData.getName() + "; problem was: " + e
392               );
393             }// End try
394           }// End while
395         }
396         finally {
397           // if we played around with the parameters of an already loaded
398           // resource then put them back before we break something
399           if (existingParameters != null)
400             rd.setParameterList(existingParameters);
401         }
402       }// End if
403       currentAutoinstances = null;
404     // End RESOURCE processing
405     //////////////////////////////////////////////////////////////////
406     } else if(elementName.toUpperCase().equals("AUTOINSTANCE") ||
407             elementName.toUpperCase().equals("HIDDEN-AUTOINSTANCE")) {
408       //checkStack("endElement", "AUTOINSTANCE");
409       // Cache the auto-instance into the autoins
410       if (currentAutoinstanceParams != null)
411         currentAutoinstances.add(currentAutoinstanceParams);
412     // End AUTOINSTANCE processing
413     //////////////////////////////////////////////////////////////////
414     } else if(elementName.toUpperCase().equals("PARAM")) {
415     // End PARAM processing
416     //////////////////////////////////////////////////////////////////
417     } else if(elementName.toUpperCase().equals("NAME")) {
418       checkStack("endElement", "NAME");
419       resourceData.setName((String) contentStack.pop());
420     // End NAME processing
421     //////////////////////////////////////////////////////////////////
422     } else if (elementName.toUpperCase().equals("IVY")) {
423       if (!contentStack.isEmpty()) contentStack.pop();
424     // End IVY processing
425     //////////////////////////////////////////////////////////////////
426     } else if(elementName.toUpperCase().equals("JAR")) {
427       checkStack("endElement", "JAR");
428 
429       // add jar file name
430       String jarFileName = (String) contentStack.pop();
431       if(resourceData != null) {
432         resourceData.setJarFileName(jarFileName);
433       }
434 
435       // add jar file URL if there is one
436       if(sourceUrl != null) {
437         String sourceUrlName = sourceUrl.toExternalForm();
438         String separator = "/";
439 
440         if(sourceUrlName.endsWith(separator))
441           separator = "";
442         URL jarFileUrl = null;
443 
444         try {
445           jarFileUrl = new URL(sourceUrlName + separator + jarFileName);
446           if(resourceData != null) {
447             resourceData.setJarFileUrl(jarFileUrl);
448           }
449 
450           // We no longer need to add the jar URL to the class loader, as this
451           // is done before the SAX parse
452         } catch(MalformedURLException e) {
453           throw new GateSaxException("bad URL " + jarFileUrl + e);
454         }// End try
455       }// End if
456     // End JAR processing
457     //////////////////////////////////////////////////////////////////
458     } else if(elementName.toUpperCase().equals("CLASS")) {
459       checkStack("endElement", "CLASS");
460       resourceData.setClassName((String) contentStack.pop());
461     // End CLASS processing
462     //////////////////////////////////////////////////////////////////
463     } else if(elementName.toUpperCase().equals("COMMENT")) {
464       checkStack("endElement", "COMMENT");
465       resourceData.setComment((String) contentStack.pop());
466     // End COMMENT processing
467     //////////////////////////////////////////////////////////////////
468     } else if(elementName.toUpperCase().equals("HELPURL")) {
469       checkStack("endElement", "HELPURL");
470       resourceData.setHelpURL((String) contentStack.pop());
471     // End HELPURL processing
472     //////////////////////////////////////////////////////////////////
473     } else if(elementName.toUpperCase().equals("INTERFACE")) {
474       checkStack("endElement", "INTERFACE");
475       resourceData.setInterfaceName((String) contentStack.pop());
476     // End INTERFACE processing
477     //////////////////////////////////////////////////////////////////
478     } else if(elementName.toUpperCase().equals("ICON")) {
479       checkStack("endElement", "ICON");
480       resourceData.setIcon((String) contentStack.pop());
