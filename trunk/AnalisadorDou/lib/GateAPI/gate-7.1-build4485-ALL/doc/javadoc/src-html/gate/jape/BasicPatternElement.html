027 /**
028   * A pattern element within curly braces. Has a set of Constraint,
029   * which all must be satisfied at whatever position the element is being
030   * matched at.
031   */
032 public class BasicPatternElement
033 extends PatternElement implements JapeConstants, java.io.Serializable
034 {
035   /** Debug flag */
036   private static final boolean DEBUG = false;
037 
038   /** A set of Constraint. Used during parsing. */
039   private ArrayList<Constraint> constraints1;
040 
041   /** A set of Constraint. Used during matching. */
042   private Constraint[] constraints2;
043 
044   /** A map of constraint annot type to constraint. Used during parsing. */
045   private HashMap<Object, Constraint> constraintsMap;
046 
047   /** Cache of the last position we failed at (-1 when none). */
048   private int lastFailurePoint = -1;
049 
050   /** The position of the next available annotation of the type required
051     * by the first constraint.
052     */
053   //private MutableInteger nextAvailable = new MutableInteger();
054 
055   /** The set of annotations we have matched. */
056   private AnnotationSet matchedAnnots;
057 
058   private final boolean negationCompatMode;
059   
060   /** Construction. */
061   public BasicPatternElement(SinglePhaseTransducer spt) {
062     constraintsMap = new HashMap<Object, Constraint>();
063     constraints1 = new ArrayList<Constraint>();
064     lastFailurePoint = -1;
065     //nextAvailable = new MutableInteger();
066     matchedAnnots = new AnnotationSetImpl((Document) null);
067     negationCompatMode = spt.isNegationCompatMode();
068   } // construction
069 
070   /** Need cloning for processing of macro references. See comments on
071     * <CODE>PatternElement.clone()</CODE>
072     */
073   public Object clone() {
074     BasicPatternElement newPE = (BasicPatternElement) super.clone();
075     newPE.constraintsMap = (HashMap<Object, Constraint>) constraintsMap.clone();
076     newPE.constraints1 = new ArrayList<Constraint>();
077     int consLen = constraints1.size();
078     for(int i = 0; i < consLen; i++)
079       newPE.constraints1.add(
080         (Constraint)constraints1.get(i).clone()
081       );
082 //    newPE.matchedAnnots = new AnnotationSetImpl((Document) null);
083 //    newPE.matchedAnnots.addAll(matchedAnnots);
084     return newPE;
085   } // clone
086 
087   /** Add a constraint. Ensures that only one constraint of any given
088     * annotation type and negation state exists.
089     */
090   public void addConstraint(Constraint newConstraint) {
091     // find if we need to merge this new constraint with an existing one
092     Constraint existingConstraint = null;
093     String annotType = newConstraint.getAnnotType();
094     Pair typeNegKey = new Pair(annotType, newConstraint.isNegated());
095     if(negationCompatMode && newConstraint.isNegated()) {
096       // compatibility mode (pre GATE 7.0) where multiple negative constraints
097       // on the same annotation type do NOT get grouped together and AND'ed 
098       // before the negation is applied
099       existingConstraint = null;
100     } else {
101       // positive constraint OR negated, but in default mode:
102       // if a constraint with the same negation state as this constraint is
103       // already mapped, put its attributes on the existing constraint, else
104       // add it
105       existingConstraint = constraintsMap.get(typeNegKey);
106     }      
107 
108     if(existingConstraint != null) {
109       existingConstraint.addAttributes(newConstraint.getAttributeSeq());
110     } else {
111       constraintsMap.put(typeNegKey, newConstraint);
112       constraints1.add(newConstraint);      
113     }
114   } // addConstraint

117   /**
118    * Indicates whether this constraint deals with only one type of annotation or
119    * multiple types.
120    */
121   public boolean isMultiType() {
122       return constraints2 != null ? constraints2.length > 1 :
123              constraints1 != null ? constraints1.size() > 1 :
124              false;
125   }
126 
127   /** Finish: replace dynamic data structures with Java arrays; called
128     * after parsing.
129     */
130   public void finish() {
131     int j=0;
132     constraints2 = new Constraint[constraints1.size()];
133     for(Constraint c : constraints1 ) {
134       constraints2[j] = c;
135       constraints2[j++].finish();
136     }
137     constraints1 = null;
138   } // finish

142   /** Create a string representation of the object. */
143   public String toString() {
144     StringBuffer result = new StringBuffer("{");
145     Constraint[] constraints = getConstraints();
146     for(int i = 0; i<constraints.length; i++){
147       result.append(constraints[i].shortDesc() + ",");
148     }
149     result.setCharAt(result.length() -1, '}');
150     return result.toString();
151   }

153   /** Create a string representation of the object. */
154   public String toString(String pad) {
155     String newline = Strings.getNl();
156     String newPad = Strings.addPadding(pad, INDENT_PADDING);
157 
158     StringBuffer buf = new StringBuffer(pad +
159       "BPE: lastFailurePoint(" + lastFailurePoint + "); constraints("
160     );
161 
162     // constraints
163     if(constraints1 != null) {
164       for(int len = constraints1.size(), i = 0; i < len; i++)
165         buf.append(
166           newline + constraints1.get(i).getDisplayString(newPad)
167         );
168     } else {
169       for(int len = constraints2.length, i = 0; i < len; i++)
170         buf.append(newline + constraints2[i].getDisplayString(newPad));
171     }
172 
173     // matched annots
174 //    buf.append(
175 //      newline + pad + "matchedAnnots: " + matchedAnnots +
176 //      newline + pad + ") BPE."
177 //    );
178 
179     return buf.toString();
180   } // toString

182   /**
183     * Returns a short description.
184     */
185   public String shortDesc() {
186     String res = "";
187     if(constraints1 != null) {
188       for(int len = constraints1.size(), i = 0; i < len; i++)
189         res += constraints1.get(i).toString();
190     } else {
191       for(int len = constraints2.length, i = 0; i < len; i++)
192         res += constraints2[i].shortDesc();
193     }
194     return res;
195   }

201    * the finish() method has been used.
202    */
203   public ArrayList<Constraint> getUnfinishedConstraints() {
204     return constraints1;
205   }

207   /**
208    * Get the finished Constraint objects. Can only be used after the

210    * @return an array of constraint objects. Will be null before the
211    * finish() method has been used.
212    */
213   public Constraint[] getConstraints(){
214     return constraints2;
215   }
216 } // class BasicPatternElement
