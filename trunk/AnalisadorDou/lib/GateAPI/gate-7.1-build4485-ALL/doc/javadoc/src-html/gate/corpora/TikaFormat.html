081   @Override
082   public void unpackMarkup(Document doc, RepositioningInfo repInfo,
083           RepositioningInfo ampCodingInfo) throws DocumentFormatException {
084     if(doc == null || doc.getSourceUrl() == null) {
085 
086       throw new DocumentFormatException(
087       "GATE document is null or no content found. Nothing to parse!");
088     }// End if
089 
090     // Create a status listener
091     StatusListener statusListener = new StatusListener() {
092       public void statusChanged(String text) {
093         // This is implemented in DocumentFormat.java and inherited here
094         fireStatusChanged(text);
095       }
096     };
097 
098     XmlDocumentHandler ch = new XmlDocumentHandler(doc, this.markupElementsMap,
099             this.element2StringMap);
100     Metadata metadata = extractParserTips(doc);
101 
102     ch.addStatusListener(statusListener);
103     ch.setRepositioningInfo(repInfo);
104     // set the object with ampersand coding positions
105     ch.setAmpCodingInfo(ampCodingInfo);
106     InputStream input = null;     
107     try {
108       Parser tikaParser = new TikaConfig().getParser();      
109       input = doc.getSourceUrl().openStream();
110       tikaParser.parse(input, ch, metadata, new ParseContext());
111       setDocumentFeatures(metadata, doc);
112     } catch (IOException e) {
113       throw new DocumentFormatException(e);
114     } catch (SAXException e) {
115       throw new DocumentFormatException(e);
116     } catch (TikaException e) {
117       throw new DocumentFormatException(e);
118     }
119     finally {
120       IOUtils.closeQuietly(input); // null safe
121       ch.removeStatusListener(statusListener);
122     }
123 
124     if (doc instanceof DocumentImpl) {
125       ((DocumentImpl)doc).setNextAnnotationId(ch.getCustomObjectsId());
126     }
127   }

129   private void setDocumentFeatures(Metadata metadata, Document doc) {
130     Map fmap = doc.getFeatures();
131     setTikaFeature(metadata, Metadata.TITLE, fmap);
132     setTikaFeature(metadata, Metadata.AUTHOR, fmap);
133     setTikaFeature(metadata, Metadata.COMMENTS, fmap);
134     setTikaFeature(metadata, Metadata.CREATOR, fmap);
135     if (fmap.get("AUTHORS") == null && fmap.get("AUTHOR") != null)
136       fmap.put("AUTHORS", fmap.get(Metadata.AUTHOR));
137     fmap.put("MimeType", metadata.get(Metadata.CONTENT_TYPE));
138   }

140   private void setTikaFeature(Metadata metadata, String key, Map fmap) {
141     String value = metadata.get(key);
142     if (value == null) {
143       return;
144     }
145 
146     value = value.trim();
147     if (value.length() == 0) {
148       return;
149     }
150     key = key.toUpperCase();
151     if (fmap.containsKey(key)) {
152       fmap.put("TIKA_" + key, value);
153     }
154     else {
155       fmap.put(key, value);
156       fmap.put("TIKA_" + key, value);
157     }    
158   }

160   /**
161    * Tries to extract tips for the parser as specified here -
162    * http://tika.apache.org/0.7/parser.html . The tips are not critical
163    * for successful parsing.
164    * 
165    * @param doc
166    * @return metadata, not null but may be empty
167    */
168   private Metadata extractParserTips(Document doc) {
169     Metadata metadata = new Metadata();
170     Object inputMime = doc.getFeatures().get("MimeType");
171     if (inputMime instanceof String) {  
172       if (!"application/tika".equals(inputMime)) {
173         metadata.add(Metadata.CONTENT_TYPE, (String) doc.getFeatures().get("MimeType"));
174       }
175     }
176     if (doc instanceof DocumentImpl) {
177       if (((DocumentImpl)doc).getMimeType() != null) {
178         metadata.add(Metadata.CONTENT_TYPE, ((DocumentImpl)doc).getMimeType());
179       }
180     }
181     if (doc.getSourceUrl() != null && doc.getSourceUrl().getProtocol().startsWith("file")) {
182       try {
183         File fn =new File(doc.getSourceUrl().toURI());
184         metadata.add(Metadata.RESOURCE_NAME_KEY, fn.getName());
185       } catch (URISyntaxException e) {
186         log.debug("Could not extract filename from uri: " + doc.getSourceUrl(), e);
187       } catch (IllegalArgumentException e) {
188         log.debug("Could not extract filename from uri: " + doc.getSourceUrl(), e);
189       }
190     }
191     return metadata;
192   }
