196         TreePath path = e.getPath();
197         
198         DefaultMutableTreeNode node = (DefaultMutableTreeNode)path.getLastPathComponent();
199 
200         if(node.getChildCount() == 0 && node.getUserObject() instanceof DSType) {
201           DSType dsType = (DSType)node.getUserObject();
202           if (dsType.expanded) return;
203           node.removeAllChildren();
204           try {
205             Iterator lrIDsIter = datastore.getLrIds(dsType.type).iterator();
206             while(lrIDsIter.hasNext()) {
207               String id = (String)lrIDsIter.next();
208               DSEntry entry =
209                   new DSEntry(datastore.getLrName(id), id, dsType.type);
210               DefaultMutableTreeNode lrNode =
211                   new DefaultMutableTreeNode(entry, false);
212               treeModel.insertNodeInto(lrNode, node, node.getChildCount());
213               node.add(lrNode);
214             }
215             dsType.expanded = true;
216           } catch(PersistenceException pe) {
217             throw new GateRuntimeException(pe.toString());
218           }
219         }
220       }
221     });
222     
223     try {
224       Iterator lrTypesIter = datastore.getLrTypes().iterator();

359     public void actionPerformed(ActionEvent e) {
360       Runnable runner = new Runnable(){
361         public void run(){
362           // load all selected resources
363           TreePath[] selectedPaths = mainTree.getSelectionPaths();
364           if(ignoreSelection){
365             ignoreSelection = false;
366            selectedPaths = null;
367           }
368           // if no selection -> load path under cursor
369           if(selectedPaths == null && location != null) {
370             selectedPaths = new TreePath[] {location};
371             location = null;
372           }
373           if(selectedPaths != null) {
374             for(TreePath aPath : selectedPaths) {
375               Object value = ((DefaultMutableTreeNode)aPath.getLastPathComponent())
376                       .getUserObject();
377               if(value instanceof DSEntry) {
378                 DSEntry entry = (DSEntry)value;
379                 try {
380                   MainFrame.lockGUI("Loading " + entry.name);
381                   long start = System.currentTimeMillis();
382                   fireStatusChanged("Loading " + entry.name);
383                   fireProgressChanged(0);
384                   FeatureMap params = Factory.newFeatureMap();
385                   params.put(DataStore.DATASTORE_FEATURE_NAME, datastore);
386                   params.put(DataStore.LR_ID_FEATURE_NAME, entry.id);
387                   FeatureMap features = Factory.newFeatureMap();
388                   Resource res = Factory.createResource(entry.type, params, features,
389                           entry.name);
390                   // project.frame.resourcesTreeModel.treeChanged();
391                   fireProgressChanged(0);
392                   fireProcessFinished();
393                   long end = System.currentTimeMillis();
394                   fireStatusChanged(entry.name
395                           + " loaded in "
396                           + NumberFormat.getInstance().format(
397                                   (double)(end - start) / 1000) + " seconds");
398                 }
399                 catch(ResourceInstantiationException rie) {
400                   MainFrame.unlockGUI();
401                   JOptionPane.showMessageDialog(SerialDatastoreViewer.this,
402                           "Error!\n" + rie.toString(), "GATE",
403                           JOptionPane.ERROR_MESSAGE);
404                   rie.printStackTrace(Err.getPrintWriter());
405                   fireProgressChanged(0);
406                   fireProcessFinished();
407                 }
408                 finally {
409                   MainFrame.unlockGUI();
410                 }            
411               }
412             }
413           }  
414         }
415       };
416       Thread thread = new Thread(runner, 
417               SerialDatastoreViewer.this.getClass().getCanonicalName() +  
418               " DS Loader");
419       thread.setPriority(Thread.MIN_PRIORITY);
420       thread.start();
421     }

509   public void resourceDeleted(DatastoreEvent e) {
510     String resID = (String)e.getResourceID();
511     DefaultMutableTreeNode node = null;
512     Enumeration nodesEnum = treeRoot.depthFirstEnumeration();
513     boolean found = false;
514     while(nodesEnum.hasMoreElements() && !found) {
515       node = (DefaultMutableTreeNode)nodesEnum.nextElement();
516       Object userObject = node.getUserObject();
517       found = userObject instanceof DSEntry
518               && ((DSEntry)userObject).id.equals(resID);
519     }
520     if(found) {
521       DefaultMutableTreeNode parent = (DefaultMutableTreeNode)node.getParent();
522       treeModel.removeNodeFromParent(node);
523       if(parent.getChildCount() == 0) treeModel.removeNodeFromParent(parent);
524     }
525   }

527   public void resourceWritten(DatastoreEvent e) {
528     Resource res = e.getResource();
529     String resID = (String)e.getResourceID();
530     String resType = ((ResourceData)Gate.getCreoleRegister().get(
531             res.getClass().getName())).getName();
532     DefaultMutableTreeNode parent = treeRoot;
533     DefaultMutableTreeNode node = null;
534     // first look for the type node
535     Enumeration childrenEnum = parent.children();
536     boolean found = false;
537     while(childrenEnum.hasMoreElements() && !found) {
538       node = (DefaultMutableTreeNode)childrenEnum.nextElement();
539       if (node.getUserObject() instanceof DSType) {
540         found = ((DSType)node.getUserObject()).name.equals(resType);
541       }
542     }
543     if(!found) {
544       // exhausted the children without finding the node -> new type
545       node = new DefaultMutableTreeNode(resType);
546       treeModel.insertNodeInto(node, parent, parent.getChildCount());
547     }
548     //mainTree.expandPath(new TreePath(new Object[] {parent, node}));
549 
550     if (node.getUserObject() instanceof DSType) {
551       if (!((DSType)node.getUserObject()).expanded) return;
552     }
553     
554     // now look for the resource node
555     parent = node;
556     childrenEnum = parent.children();
557     found = false;
558     while(childrenEnum.hasMoreElements() && !found) {
559       node = (DefaultMutableTreeNode)childrenEnum.nextElement();
560       found = ((DSEntry)node.getUserObject()).id.equals(resID);
561     }
562     if(!found) {
563       // exhausted the children without finding the node -> new resource
564       try {
565         DSEntry entry = new DSEntry(datastore.getLrName(resID), resID, res
566                 .getClass().getName());
567         node = new DefaultMutableTreeNode(entry, false);
568         treeModel.insertNodeInto(node, parent, parent.getChildCount());
569       }
570       catch(PersistenceException pe) {
571         pe.printStackTrace(Err.getPrintWriter());
572       }
573     }
574   }// public void resourceWritten(DatastoreEvent e)
