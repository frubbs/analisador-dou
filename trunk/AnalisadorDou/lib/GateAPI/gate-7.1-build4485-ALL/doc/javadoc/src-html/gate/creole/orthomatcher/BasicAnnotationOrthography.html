079   public boolean fuzzyMatch(String s1, String s2) {
080     String s1Lower = s1.toLowerCase();
081     String s2Lower = s2.toLowerCase();
082     if(s1Lower.equals(s2Lower)) { return true; }
083     // System.out.println("Now comparing " + s1 + " | " + s2) ;
084     Set<String> formalNameSet = nicknameMap.get(s1Lower);
085     if(formalNameSet != null) {
086       if(formalNameSet.contains(s2Lower)) { return true; }
087     }
088     formalNameSet = nicknameMap.get(s2Lower);
089     if(formalNameSet != null) {
090       if(formalNameSet.contains(s1Lower)) { return true; }
091     }
092     return false;
093   }

095   /**
096    * @return true if all of the tokens in firstName are either found in second
097    *         name or are stop words
098    */
099   public boolean allNonStopTokensInOtherAnnot(ArrayList<Annotation> firstName,
100       ArrayList<Annotation> secondName, String TOKEN_STRING_FEATURE_NAME,
101       boolean caseSensitive) {
102     for(Annotation a : firstName) {
103       if(!a.getFeatures().containsKey("ortho_stop")) {
104         String aString = (String)a.getFeatures().get(TOKEN_STRING_FEATURE_NAME);
105         boolean foundAMatchInSecond = false;
106         for(Annotation b : secondName) {
107           if(OrthoMatcherHelper.straightCompare(aString, (String)b
108               .getFeatures().get(TOKEN_STRING_FEATURE_NAME), caseSensitive)) {
109             foundAMatchInSecond = true;
110             break;
111           }
112         }
113         if(!foundAMatchInSecond) { return false; }
114       }
115     }
116     return true;
117   }

119   /**
120    * Return a person name without a title. Also remove title from global
121    * variable tokensMap
122    */
123   public String stripPersonTitle(String annotString, Annotation annot,
124       Document doc, Map<Integer, List<Annotation>> tokensMap,
125       HashMap normalizedTokensMap, AnnotationSet nameAllAnnots)
126       throws ExecutionException {
127     FeatureMap queryFM = Factory.newFeatureMap();
128     // get the offsets
129     Long startAnnot = annot.getStartNode().getOffset();
130     Long endAnnot = annot.getEndNode().getOffset();
131     // determine "Lookup" annotation set
132     queryFM.clear();
133     queryFM.put("majorType", "title");
134     AnnotationSet as1 = nameAllAnnots.getContained(startAnnot, endAnnot);
135     if(as1 == null || as1.isEmpty()) return annotString;
136     AnnotationSet as = as1.get("Lookup", queryFM);
137     if(as != null && !as.isEmpty()) {
138       List<Annotation> titles = new ArrayList<Annotation>(as);
139       Collections.sort(titles, new gate.util.OffsetComparator());
140       Iterator<Annotation> iter = titles.iterator();
141       while(iter.hasNext()) {
142         Annotation titleAnn = iter.next();
143         // we've not found a title at the start offset,
144         // there's no point in looking further
145         // coz titles come first
146         if(titleAnn.getStartNode().getOffset().compareTo(startAnnot) != 0)
147           return annotString;
148         try {
149           // the title from the current annotation
150           String annotTitle =
151               doc.getContent()
152                   .getContent(titleAnn.getStartNode().getOffset(),
153                       titleAnn.getEndNode().getOffset()).toString();
154           // eliminate the title from annotation string and return the result
155           if(annotTitle.length() < annotString.length()) {
156             // remove from the array of tokens, so then we can compare properly
157             // the remaining tokens
158             // log.debug("Removing title from: " + annot + " with string " +
159             // annotString);
160             // log.debug("Tokens are " + tokensMap.get(annot.getId()));
161             // log.debug("Title is " + annotTitle);
162             ((ArrayList)tokensMap.get(annot.getId())).remove(0);
163             ((ArrayList)normalizedTokensMap.get(annot.getId())).remove(0);
164             return annotString.substring(annotTitle.length() + 1,
165                 annotString.length());
166           }
167         } catch(InvalidOffsetException ioe) {
168           throw new ExecutionException("Invalid offset of the annotation");
169         }// try
170       }// while
171     }// if
172     return annotString;
173   }

176       List matchesDocFeature, AnnotationSet nameAllAnnots) {
177     // the two annotations are already matched if the matches list of the first
178     // contains the id of the second
179     List matchesList =

184     return false;

187   public Annotation updateMatches(Annotation newAnnot, String annotString,
188       HashMap processedAnnots, AnnotationSet nameAllAnnots,
189       List matchesDocFeature) {
190     Annotation matchedAnnot = null;
191     Integer id;
192     // first find a processed annotation with the same string
193     // TODO: Andrew Borthwick 7/26/08: The below is very inefficient. We should
194     // be doing a lookup into a hash
195     // which is indexed on string rather than testing every id. Need to have the
196     // index be String + Type
197     // for safety
198     Iterator iter = processedAnnots.keySet().iterator();
199     // System.out.println("ID's examined: ");
200     while(iter.hasNext()) {
201       id = (Integer)iter.next();
202       String oldString = (String)processedAnnots.get(id);
203       // System.out.print(id + " ");
204       if(annotString.equals(oldString)) {
205         Annotation tempAnnot = nameAllAnnots.get(id);
206         if(tempAnnot == null) {
207           log.warn("Orthomatcher: TempAnnot is null when looking at "
208               + annotString + " | " + oldString + " | old id: " + id);
209           return null;
210         }
211         // Below is a new Spock addition to prevent unpredictable behavior when
212         // the same string is given more than one type. We want to return null
213         // if there is no match on name + type (other than Unknown)
214         if(newAnnot.getType().equals(unknownType)
215             || tempAnnot.getType().equals(newAnnot.getType())) {
216           matchedAnnot = tempAnnot;
217           break;
218         }
219       }
220     }// while
221      // System.out.println();
222     if(matchedAnnot == null) return null;
223     List matchesList =
224         (List)matchedAnnot.getFeatures().get(ANNOTATION_COREF_FEATURE_NAME);
225     if((matchesList == null) || matchesList.isEmpty()) {
226       // no previous matches, so need to add
227       if(matchesList == null) {
228         matchesList = new ArrayList();
229         matchedAnnot.getFeatures().put(ANNOTATION_COREF_FEATURE_NAME,
230             matchesList);
231         matchesDocFeature.add(matchesList);
232       }// if
233       matchesList.add(matchedAnnot.getId());
234       matchesList.add(newAnnot.getId());
235     } else {
236       // just add the new annotation
237       matchesList.add(newAnnot.getId());
238     }// if
239      // add the matches list to the new annotation
240     newAnnot.getFeatures().put(OrthoMatcher.ANNOTATION_COREF_FEATURE_NAME,
241         matchesList);
242     return matchedAnnot;
243   }

245   public void updateMatches(Annotation newAnnot, Annotation prevAnnot,
246       List matchesDocFeature, AnnotationSet nameAllAnnots) {
247     List matchesList =
248         (List)prevAnnot.getFeatures().get(
249             OrthoMatcher.ANNOTATION_COREF_FEATURE_NAME);
250     if((matchesList == null) || matchesList.isEmpty()) {
251       // no previous matches, so need to add
252       if(matchesList == null) {
253         matchesList = new ArrayList();
254         prevAnnot.getFeatures().put(OrthoMatcher.ANNOTATION_COREF_FEATURE_NAME,
255             matchesList);
256         matchesDocFeature.add(matchesList);
257       }// if
258       matchesList.add(prevAnnot.getId());
259       matchesList.add(newAnnot.getId());
260     } else {
261       // just add the new annotation
262       matchesList.add(newAnnot.getId());
263     }// if
264      // add the matches list to the new annotation
265     newAnnot.getFeatures().put(OrthoMatcher.ANNOTATION_COREF_FEATURE_NAME,
266         matchesList);
267     // propagate the gender if two persons are matched
268     if(prevAnnot.getType().equals(this.personType)) {
269       String prevGender =
270           (String)prevAnnot.getFeatures().get(
271               OrthoMatcher.PERSON_GENDER_FEATURE_NAME);
272       String newGender =
273           (String)newAnnot.getFeatures().get(
274               OrthoMatcher.PERSON_GENDER_FEATURE_NAME);
275       boolean unknownPrevGender = isUnknownGender(prevGender);
276       boolean unknownNewGender = isUnknownGender(newGender);
277       if(unknownPrevGender && !unknownNewGender)
278         prevAnnot.getFeatures().put(OrthoMatcher.PERSON_GENDER_FEATURE_NAME,
279             newGender);
280       else if(unknownNewGender && !unknownPrevGender)
281         newAnnot.getFeatures().put(OrthoMatcher.PERSON_GENDER_FEATURE_NAME,
282             prevGender);
283     }// if
284   }

286   /**
287    * Tables for namematch info (used by the namematch rules)
288    */
289   public HashSet buildTables(AnnotationSet nameAllAnnots) {
290     FeatureMap tempMap = Factory.newFeatureMap();
291     // reset the tables first
292     HashSet cdg = new HashSet();
293     if(!extLists) {
294       // i.e. get cdg from Lookup annotations
295       // get all Lookup annotations
296       tempMap.clear();
297       tempMap.put(gate.creole.ANNIEConstants.LOOKUP_MAJOR_TYPE_FEATURE_NAME,
298           "cdg");
299       // now get all lookup annotations which are cdg
300       AnnotationSet nameAnnots =
301           nameAllAnnots.get(LOOKUP_ANNOTATION_TYPE, tempMap);
302       if((nameAnnots == null) || nameAnnots.isEmpty()) return cdg;
303       Iterator<Annotation> iter = nameAnnots.iterator();
304       while(iter.hasNext()) {
305         Annotation annot = iter.next();
306         // get the actual string
307         Long offsetStartAnnot = annot.getStartNode().getOffset();
308         Long offsetEndAnnot = annot.getEndNode().getOffset();
309         try {
310           gate.Document doc = nameAllAnnots.getDocument();
311           String annotString =
312               doc.getContent().getContent(offsetStartAnnot, offsetEndAnnot)
313                   .toString();
314           cdg.add(annotString);
315         } catch(InvalidOffsetException ioe) {
316           ioe.printStackTrace(Err.getPrintWriter());
317         }
318       }// while
319     }// if
320     return cdg;
321   }// buildTables

330   protected Map<String, HashSet<String>> initNicknames(
331       String nicknameFileEncoding, java.net.URL fileURL) throws IOException {
332     Pattern spacePat = Pattern.compile("(\\s+)");
333     nicknameMap = new HashMap<String, HashSet<String>>();
334     // create the relative URL
335     BufferedReader reader =
336         new BomStrippingInputStreamReader(fileURL.openStream(),
337             nicknameFileEncoding);
338     String lineRead = null;
339     int ctr = 0;
340     while((lineRead = reader.readLine()) != null) {
341       if(lineRead.length() == 0 || lineRead.charAt(0) == '#') {
342         continue;
343       }
344       ArrayList<String> nickNameLine =
345           new ArrayList<String>(Arrays.asList(spacePat.split(lineRead
346               .toLowerCase().trim())));
347       if(nickNameLine.size() != 3
348           && (nickNameLine.size() != 4 && ((nickNameLine.get(3) != "M") || nickNameLine
349               .get(3) != "F"))) {
350         continue;
351       }
352       if(round2Places(Double.valueOf(nickNameLine.get(2))) < OrthoMatcherHelper
353           .round2Places(minimumNicknameLikelihood)) {
354         continue;
355       }
356       if(nicknameMap.containsKey(nickNameLine.get(0))) {
357         /*
358          * System.out.println("Adding to existing nickname of " +
359          * nickNameLine.get(0) + " " + nickNameLine.get(1));
360          */
361         nicknameMap.get(nickNameLine.get(0)).add(nickNameLine.get(1));
362       } else {
363         /*
364          * System.out.println("Adding new nickname of " + nickNameLine.get(0) +
365          * " " + nickNameLine.get(1));
366          */
367         nicknameMap.put(nickNameLine.get(0),
368             new HashSet<String>(Collections.singleton(nickNameLine.get(1))));
369       }
370     }
371     return nicknameMap;
372   }
