001 package gate.util;
002 
003 import java.io.*;
004 import gate.util.GateRuntimeException;
005 
006 /**
007  * Class that supports running an external process and either silently
008  * consuming its standard output and error streams, or copying them to Java's
009  * stdout and stderr.
010  *
011  * This implementation reads the output and error streams in separate threads,
012  * but tries to reuse these threads from one external call to the next, unlike
013  * other approaches I've seen (which all spawn a pair of new threads for every
014  * external call).  As a result, instances of this class are <b>not thread
015  * safe</b>.  You must use a different instance of ProcessManager in each
016  * thread that you use to run external processes.
017  */
018 public class ProcessManager {
019   /**
020    * Debug flag.
021    */
022   private static final boolean DEBUG = false;
023 
024   /**
025    * StreamGobbler thread for standard output.
026    */
027   private StreamGobbler stdout;
028 
029   /**
030    * StreamGobbler thread for standard error.
031    */
032   private StreamGobbler stderr;
033 
034   /**
035    * Construct a ProcessManager object and start the gobbler threads.
036    */
037   public ProcessManager() {
038     stdout = new StreamGobbler();
039     Thread t = new Thread(stdout);
040     t.setDaemon(true);
041     t.start();
042 
043     stderr = new StreamGobbler();
044     t = new Thread(stderr);
045     t.setDaemon(true);
046     t.start();
047   }
048 
049   /**
050    * Run the given external process.  If an exception results from starting the
051    * process, or while reading the output from the process, it will be thrown.
052    * Otherwise, the exit value from the process is returned.
053    *
054    * @param argv the process command line, suitable for passing to
055    * <code>Runtime.exec</code>.
056    * @param dumpOutput should we copy the process output and error streams to
057    * the Java output and error streams or just consume them silently?
058    */
059   public synchronized int runProcess(String[] argv, boolean dumpOutput)
060                           throws IOException {
061     return runProcess(argv, null, (dumpOutput ? System.out : null), (dumpOutput ? System.err : null));
062   }
063   
064   public synchronized int runProcess(String[] argv, OutputStream out, OutputStream err)
065   throws IOException {
066     return runProcess(argv, null, out, err);
067   }
068   
069   /**
070    * Run the given external process.  If an exception results from starting the
071    * process, or while reading the output from the process, it will be thrown.
072    * Otherwise, the exit value from the process is returned.
073    *
074    * @param argv the process command line, suitable for passing to
075    * <code>Runtime.exec</code>.
076    */
077   public synchronized int runProcess(String[] argv, File dir, OutputStream out, OutputStream err)
078                           throws IOException {
079     // Start the process.  This may throw an exception
080     if(DEBUG) {
081       System.err.println("Starting process");
082     }
083     
084     Process proc = Runtime.getRuntime().exec(argv, null, dir);
085     
086     // set up the stream gobblers for stdout and stderr
087     if(DEBUG) {
088       System.err.println("Configuring gobblers");
089     }
090     stdout.setInputStream(proc.getInputStream());
091     stdout.setOutputStream(out);
092 
093     stderr.setInputStream(proc.getErrorStream());
094     stderr.setOutputStream(err);
095 
096     // start the gobblers
097     if(DEBUG) {
098       System.err.println("Waking up gobblers");
099     }
100     this.notifyAll();
101 
102     // wait for the gobblers to finish their jobs
103     while(!stderr.isDone() || !stdout.isDone()) {
104       try {
105         if(DEBUG) {
106           System.err.println("Gobblers not done, waiting...");
107         }
108         this.wait();
109       }
110       catch(InterruptedException ie) {
111         // if interrupted, try waiting again
112       }
113     }
114 
115     // get the return code from the process
116     Integer returnCode = null;
117     while(returnCode == null) {
118       try {
119         returnCode = new Integer(proc.waitFor());
120       }
121       catch(InterruptedException ie) {
122         // if interrupted, just try waiting again
123       }
124     }
125     
126     // reset the gobblers
127     if(DEBUG) {
128       System.err.println("Gobblers done - resetting");
129     }
130     stdout.reset();
131     stderr.reset();
132     
133     // if there was an exception during running, throw that
134     Exception ex = null;
135     if(stdout.hasException()) {
136       ex = stdout.getException();
137       stderr.getException(); // to reset exception cache
138     }
139     else if(stderr.hasException()) {
140       ex = stderr.getException();
141     }
142 
143     if(ex != null) {
144       if(DEBUG) {
145         System.err.println("Rethrowing exception");
146       }
147       if(ex instanceof IOException) {
148         throw (IOException)ex;
149       }
150       else if(ex instanceof RuntimeException) {
151         throw (RuntimeException)ex;
152       }
153       else throw new GateRuntimeException(ex);
154     }
155     // otherwise return the exit code
156     else {
157       return returnCode.intValue();
158     }
159   }

161   /**
162    * Thread body that takes a stream and either consumes it silently or echoes
163    * it to another stream.
164    */
165   private class StreamGobbler implements Runnable {
166     /**
167      * The input stream to gobble.  If this is null, the thread is idle.
168      */
169     private InputStream inputStream = null;
170 
171     /**
172      * The output stream to echo to.  If this is null the gobbler silently
173      * discards the input stream contents.
174      */
175     private OutputStream outputStream = null;;
176 
177     /**
178      * Buffer used for reading and writing.
179      */
180     private byte[] buf = new byte[4096];
181 
182     /**
183      * Are we finished?  This is set to true once the stream has been emptied.
184      */
185     private boolean done = false;
186 
187     /**
188      * If an exception is thrown during gobbling, it is stored here.
189      */
190     private Exception exception = null;
191 
192     /**
193      * Set the stream to gobble.  This should not be called while the thread is
194      * active.
195      */
196     void setInputStream(InputStream is) {
197       inputStream = is;
198     }
199 
200     /**
201      * Set the output stream to redirect output to.  A value of
202      * <code>null</code> indicates that we should discard the output without
203      * echoing it.
204      */
205     void setOutputStream(OutputStream os) {
206       outputStream = os;
207     }
208 
209     /**
210      * Has an exception been thrown since {@link #getException()} was last
211      * called?
212      */
213     boolean hasException() {
214       return (exception != null);
215     }
216 
217     /**
218      * Return the last exception thrown.  This also resets the cached exception
219      * to <code>null</code>.
220      */
221     Exception getException() {
222       Exception ex = exception;
223       exception = null;
224       return ex;
225     }
226 
227     boolean isDone() {
228       return done;
229     }
230 
231     /**
232      * Reset state.
233      */
234     void reset() {
235       done = false;
236       exception = null;
237       inputStream = null;
238     }
239 
240     /**
241      * Main body of the thread.  Waits until we have been given a stream to
242      * gobble, then reads it until there is no more input available.
243      */
244     public void run() {
245       if(DEBUG) {
246         System.err.println("StreamGobbler starting");
247       }
248       // wait until we have a stream to gobble
249       synchronized(ProcessManager.this) {
250         while(inputStream == null) {
251           try {
252             if(DEBUG) {
253               System.err.println("Waiting for stream...");
254             }
255             ProcessManager.this.wait();
256           }
257           catch(InterruptedException ie) {
258           }
259         }
260       }
261 
262       while(true) {
263         // read the stream until end of file or an exception is thrown.
264         BufferedInputStream bis = new BufferedInputStream(inputStream);
265         int bytesRead = -1;
266         try {
267           if(DEBUG) {
268             System.err.println("Gobbling stream");
269           }
270           while((bytesRead = bis.read(buf)) != -1) {
271             // echo to outputStream if necessary
272             if(outputStream != null) {
273               outputStream.write(buf, 0, bytesRead);
274             }
275           }
276         }
277         catch(Exception ex) {
278           // any exception is stored to be retrieved by the ProcessManager
279           exception = ex;
280           if(DEBUG) {
281             System.err.println("Exception thrown");
282           }
283         }
284         
285         try {
286           bis.close();
287         }
288         catch(IOException ioe) {
289           // oh well, it's not the end of the world
290         }
291 
292         done = true;
293         inputStream = null;
294         outputStream = null;
295 
296         // wake up ProcessManager to say we've finished, and start waiting for
297         // the next round.
298         synchronized(ProcessManager.this) {
299           if(DEBUG) {
300             System.err.println("Waking process manager");
301           }
302           ProcessManager.this.notifyAll();
303           while(inputStream == null) {
304             try {
305               if(DEBUG) {
306                 System.err.println("Waiting for stream (2)");
307               }
308               ProcessManager.this.wait();
309             }
310             catch(InterruptedException ie) {
311             }
312           }
313         }
314       }
315     }
316   }
