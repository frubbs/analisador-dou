114   public void execute() throws ExecutionException{
115     //check the parameters
116     if(document == null) throw new ExecutionException(
117       "No document to process!");
118     if(inputASName != null && inputASName.equals("")) inputASName = null;

120                             document.getAnnotations() :
121                             document.getAnnotations(inputASName);
122 
123                            
124     if(baseTokenAnnotationType == null || baseTokenAnnotationType.trim().length()==0) {
125         throw new ExecutionException("No base Token Annotation Type provided!");
126     }

139     }
140 
141     AnnotationSet sentencesAS = inputAS.get(baseSentenceAnnotationType);
142     AnnotationSet tokensAS = inputAS.get(baseTokenAnnotationType);
143     if(sentencesAS != null && sentencesAS.size() > 0
144        && tokensAS != null && tokensAS.size() > 0){
145       long startTime = System.currentTimeMillis();
146       fireStatusChanged("POS tagging " + document.getName());
147       fireProgressChanged(0);
148       //prepare the input for HepTag
149       List sentenceForTagger = new ArrayList();
150       List sentencesForTagger = new ArrayList(1);
151       sentencesForTagger.add(sentenceForTagger);
152 
153       //define a comparator for annotations by start offset
154       Comparator offsetComparator = new OffsetComparator();
155 
156       //read all the tokens and all the sentences
157       List<Annotation> sentencesList = new ArrayList(sentencesAS);
158       Collections.sort(sentencesList, offsetComparator);
159       List<Annotation> tokensList = new ArrayList(tokensAS);
160       Collections.sort(tokensList, offsetComparator);
161 
162       Iterator<Annotation> sentencesIter = sentencesList.iterator();
163       ListIterator<Annotation> tokensIter = tokensList.listIterator();
164 
165       List<Annotation> tokensInCurrentSentence = new ArrayList();
166       Annotation currentToken = tokensIter.next();
167       int sentIndex = 0;
168       int sentCnt = sentencesAS.size();
169       while(sentencesIter.hasNext()){
170         Annotation currentSentence = sentencesIter.next();
171         tokensInCurrentSentence.clear();
172         sentenceForTagger.clear();
173         while(currentToken != null
174               &&
175               currentToken.getEndNode().getOffset().compareTo(
176               currentSentence.getEndNode().getOffset()) <= 0){
177           // If we're only POS tagging Tokens within baseSentenceAnnotationType, don't add the sentence if the Tokens aren't within the span of baseSentenceAnnotationType
178           if (posTagAllTokens || currentToken.withinSpanOf(currentSentence)) {
179             tokensInCurrentSentence.add(currentToken);
180             sentenceForTagger.add(currentToken.getFeatures().
181                                 get(TOKEN_STRING_FEATURE_NAME));
182           }
183           currentToken = (tokensIter.hasNext() ?
184                                      tokensIter.next() : null);
185         }
186         //run the POS tagger
187         List taggerList = tagger.runTagger(sentencesForTagger);
188         if(taggerList != null && taggerList.size() > 0){
189           List taggerResults = (List) taggerList.get(0);
190           //add the results
191           //make sure no malfunction occurred
192           if(taggerResults.size() != tokensInCurrentSentence.size())
193             throw new ExecutionException(
194                 "POS Tagger malfunction: the output size (" +
195                 taggerResults.size() +
196                 ") is different from the input size (" +
197                 tokensInCurrentSentence.size() + ")!");
198           Iterator resIter = taggerResults.iterator();
199           Iterator<Annotation> tokIter = tokensInCurrentSentence.iterator();
200           while(resIter.hasNext()){
201               Annotation annot = tokIter.next();
202               addFeatures(annot, TOKEN_CATEGORY_FEATURE_NAME, ((String[])resIter.next())[1]);
203           }
204         }
205         fireProgressChanged(sentIndex++ * 100 / sentCnt);
206       }//while(sentencesIter.hasNext())
207 
208       if(currentToken != null && posTagAllTokens){ // Tag remaining Tokens if we are not considering those only within baseSentenceAnnotationType
209         //we have remaining tokens after the last sentence
210         tokensInCurrentSentence.clear();
211         sentenceForTagger.clear();
212         while(currentToken != null){
213           tokensInCurrentSentence.add(currentToken);
214           sentenceForTagger.add(currentToken.getFeatures().
215                                 get(TOKEN_STRING_FEATURE_NAME));
216           currentToken = (tokensIter.hasNext() ?
217                                       tokensIter.next() : null);
218         }
219         //run the POS tagger
220         List taggerResults = (List)tagger.runTagger(sentencesForTagger).get(0);
221         //add the results
222         //make sure no malfunction occurred
223         if(taggerResults.size() != tokensInCurrentSentence.size())
224           throw new ExecutionException(
225               "POS Tagger malfunction: the output size (" +
226               taggerResults.size() +
227               ") is different from the input size (" +
228               tokensInCurrentSentence.size() + ")!");
229         Iterator resIter = taggerResults.iterator();
230         Iterator<Annotation> tokIter = tokensInCurrentSentence.iterator();
231         while(resIter.hasNext()){
232             Annotation annot = tokIter.next();
233             addFeatures(annot, TOKEN_CATEGORY_FEATURE_NAME, ((String[])resIter.next())[1]);
234         }
235       }//if(currentToken != null)
236       fireProcessFinished();
237       fireStatusChanged(
238         document.getName() + " tagged in " +
239         NumberFormat.getInstance().format(
240         (double)(System.currentTimeMillis() - startTime) / 1000) +
241         " seconds!");
242     }else{
243       if(failOnMissingInputAnnotations) {
244         throw new ExecutionException("No sentences or tokens to process in document "+document.getName()+"\n" +
245                                      "Please run a sentence splitter "+
246                                      "and tokeniser first!");
247       } else {
248         Utils.logOnce(logger,Level.INFO,"POS tagger: no sentence or token annotations in input document - see debug log for details.");
249         logger.debug("No input annotations in document "+document.getName());
250       }
251     }

332   protected void addFeatures(Annotation annot, String featureName, String featureValue) throws GateRuntimeException {
333       String tempIASN = inputASName == null ? "" : inputASName;
334       String tempOASN = outputASName == null ? "" : outputASName;
335       if(outputAnnotationType.equals(baseTokenAnnotationType) && tempIASN.equals(tempOASN)) {
336           annot.getFeatures().put(featureName, featureValue);
337           return;
338       } else {
339           int start = annot.getStartNode().getOffset().intValue();
340           int end = annot.getEndNode().getOffset().intValue();
341           
342           // get the annotations of type outputAnnotationType
343           AnnotationSet outputAS = (outputASName == null) ?
344                   document.getAnnotations() :
345                   document.getAnnotations(outputASName);
346           AnnotationSet annotations = outputAS.get(outputAnnotationType);
347           if(annotations == null || annotations.size() == 0) {
348               // add new annotation
349               FeatureMap features = Factory.newFeatureMap();
350               features.put(featureName, featureValue);
351               try {
352                   outputAS.add(new Long(start), new Long(end), outputAnnotationType, features);
353               } catch(Exception e) {
354                   throw new GateRuntimeException("Invalid Offsets");
355               }
356           } else {
357               // search for the annotation if there is one with the same start and end offsets
358               ArrayList tempList = new ArrayList(annotations.get());
359               boolean found = false;
360               for(int i=0;i<tempList.size();i++) {
361                   Annotation annotation = (Annotation) tempList.get(i);
362                   if(annotation.getStartNode().getOffset().intValue() == start && annotation.getEndNode().getOffset().intValue() == end) {
363                       // this is the one
364                       annotation.getFeatures().put(featureName, featureValue);
365                       found = true;
366                       break;
367                   }
368               }
369               
370               if(!found) {
371                   // add new annotation
372                   FeatureMap features = Factory.newFeatureMap();
373                   features.put(featureName, featureValue);
374                   try {
375                       outputAS.add(new Long(start), new Long(end), outputAnnotationType, features);
376                   } catch(Exception e) {
377                       throw new GateRuntimeException("Invalid Offsets");
378                   }
379               }
380           }
381       }
