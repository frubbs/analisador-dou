034 /**
035  * <p>
036  * Title: Flexible Gazetteer
037  * </p>
038  * <p>
039  * The Flexible Gazetteer provides users with the flexibility to choose
040  * their own customised input and an external Gazetteer. For example,
041  * the user might want to replace words in the text with their base
042  * forms (which is an output of the Morphological Analyser).
043  * </p>
044  * <p>
045  * The Flexible Gazetteer performs lookup over a document based on the
046  * values of an arbitrary feature of an arbitrary annotation type, by
047  * using an externally provided gazetteer. It is important to use an
048  * external gazetteer as this allows the use of any type of gazetteer
049  * (e.g. an Ontological gazetteer).
050  * </p>
051  * 
052  * @author niraj aswani
053  * @version 1.0
054  */
055 public class FlexibleGazetteer extends AbstractLanguageAnalyser 
056   implements ProcessingResource {
057   
058   private static final long serialVersionUID = -1023682327651886920L;
059   private static final String wrappedOutputASName = "Output";
060   private static final String wrappedInputASName = "Input";
061   
062   // SET TO false BEFORE CHECKING IN
063   private static final boolean DEBUG = false;
064 
065   /**
066    * Does the actual loading and parsing of the lists. This method must be
067    * called before the gazetteer can be used
068    */
069   public Resource init() throws ResourceInstantiationException {
070     if(gazetteerInst == null) { throw new ResourceInstantiationException(
071         "No Gazetteer Provided!"); }
072     return this;
073   }
074 
075   /**
076    * This method runs the gazetteer. It assumes that all the needed parameters
077    * are set. If they are not, an exception will be fired.
078    */
079   public void execute() throws ExecutionException {
080     fireProgressChanged(0);
081     fireStatusChanged("Checking Document...");
082     if(document == null) { throw new ExecutionException(
083         "No document to process!"); }
084     // obtain the inputAS
085     AnnotationSet inputAS = document.getAnnotations(inputASName);
086     // anything in the inputFeatureNames?
087     if(inputFeatureNames == null || inputFeatureNames.size() == 0) { throw new ExecutionException(
088         "No input feature names provided!"); }
089     // for each input feature, create a temporary document and run the
090     // gazetteer
091     for(String aFeature : inputFeatureNames) {
092       // find out the feature name user wants us to use
093       String[] keyVal = aFeature.split("\\.");
094       // if invalid feature name
095       if(keyVal.length != 2) {
096         System.err.println("Invalid input feature name:" + aFeature);
097         continue;
098       }
099       // keyVal[0] = annotation type
100       // keyVal[1] = feature name
101       // holds mapping for newly created annotations
102       FlexGazMappingTable mappingTable = new FlexGazMappingTable();
103       fireStatusChanged("Creating temporary Document for feature " + aFeature);
104       StringBuilder newdocString =
105           new StringBuilder(document.getContent().toString());
106       // sort annotations
107       List<Annotation> annotations =
108           Utils.inDocumentOrder(inputAS.get(keyVal[0]));
109 
110       // remove duplicate annotations
111       // (this makes the reverse mapping much easier)
112       removeOverlappingAnnotations(annotations);
113       // initially no space is deducted
114       int totalDeductedSpaces = 0;
115       // now replace the document content with the value of the feature that
116       // user has provided
117       for(Annotation currentAnnotation : annotations) {
118         // if there's no such feature, continue
119         if(!currentAnnotation.getFeatures().containsKey(keyVal[1])) continue;
120         String newTokenValue =
121             currentAnnotation.getFeatures().get(keyVal[1]).toString();
122         // if no value found for this feature
123         if(newTokenValue == null) continue;
124         // feature value found so we need to replace it
125         // find the start and end offsets for this token
126         long startOffset = Utils.start(currentAnnotation);
127         long endOffset = Utils.end(currentAnnotation);
128         // let us find the difference between the lengths of the
129         // actual string and the newTokenValue
130         long actualLength = endOffset - startOffset;
131         long lengthDifference = actualLength - newTokenValue.length();
132         // so lets find out the new startOffset and endOffset
133         long newStartOffset = startOffset - totalDeductedSpaces;
134         long newEndOffset = newStartOffset + newTokenValue.length();
135         totalDeductedSpaces += lengthDifference;
136 
137         mappingTable.add(startOffset, endOffset, newStartOffset, newEndOffset);
138         
139         // and finally replace the actual string in the document
140         // with the new document
141         newdocString.replace((int)newStartOffset, (int)newStartOffset
142             + (int)actualLength, newTokenValue);
143       }
144 
145       // proceed only if there was any replacement Map
146       if(mappingTable.isEmpty()) continue;
147       
148       /* All the binary search stuff is done inside FlexGazMappingTable
149        * now, so it's guaranteed to return valid original annotation start
150        * and end offsets.       */
151       
152       // otherwise create a temporary document for the new text
153       Document tempDoc = null;
154       // update the status
155       fireStatusChanged("Processing document with Gazetteer...");
156       try {
157         FeatureMap params = Factory.newFeatureMap();
158         params.put("stringContent", newdocString.toString());
159         // set the appropriate encoding
160         if(document instanceof DocumentImpl) {
161           params.put("encoding", ((DocumentImpl)document).getEncoding());
162           params.put("markupAware", ((DocumentImpl)document).getMarkupAware());
163         }
164         FeatureMap features = Factory.newFeatureMap();
165         Gate.setHiddenAttribute(features, true);
166         tempDoc =
167             (Document)Factory.createResource("gate.corpora.DocumentImpl",
168                 params, features);
169 
170         /* Mark the temp document with the locations of the input annotations so
171          * that we can later eliminate Lookups that are out of scope.       */
172         for (NodePosition mapping : mappingTable.getMappings()) {
173           tempDoc.getAnnotations(wrappedInputASName).add(mapping.getTempStartOffset(), 
174               mapping.getTempEndOffset(), "Input", Factory.newFeatureMap());
175         }
176       } 
177       catch(ResourceInstantiationException rie) {
178         throw new ExecutionException("Temporary document cannot be created", rie);
179       } 
180       catch(InvalidOffsetException e) {
181         throw new ExecutionException("Error duplicating Input annotations", e);
182       }
183       try {
184         // lets create the gazetteer based on the provided gazetteer name
185         gazetteerInst.setDocument(tempDoc);
186         gazetteerInst.setAnnotationSetName(wrappedOutputASName);
187         fireStatusChanged("Executing Gazetteer...");
188         gazetteerInst.execute();
189         // now the tempDoc has been looked up, we need to shift the annotations
190         // from this temp document to the original document
191         fireStatusChanged("Transfering new annotations to the original one...");
192         AnnotationSet originalDocOutput = document.getAnnotations(outputASName);
193         
194         if (DEBUG) {
195           mappingTable.dump();
196         }
197         
198         // Now iterate over the new annotations and transfer them from the 
199         // temp document back to the real one
200         for(Annotation currentLookup : tempDoc.getAnnotations(wrappedOutputASName)) {
201           long tempStartOffset = Utils.start(currentLookup);
202           long tempEndOffset = Utils.end(currentLookup);
203 
204           /* Ignore annotations that fall entirely outside the input annotations,
205            * so that we don't get dodgy Lookups outside the area covered by
206            * Tokens copied into a restricted working set by the AST PR
207            * (for example)           */
208           if (coveredByInput(tempStartOffset, tempEndOffset, tempDoc.getAnnotations(wrappedInputASName)))  {
209             long destinationStart = mappingTable.getBestOriginalStart(tempStartOffset);
210             long destinationEnd = mappingTable.getBestOriginalEnd(tempEndOffset);
211 
212             boolean valid = (destinationStart >= 0) && (destinationEnd >= 0);  
213 
214             if (valid) {
215               // Now make sure there is no other annotation like this
216               AnnotationSet testSet = originalDocOutput.getContained(destinationStart, destinationEnd).get(
217                   currentLookup.getType(), currentLookup.getFeatures());
218               for(Annotation annot : testSet) {
219                 if(Utils.start(annot) == destinationStart
220                     && Utils.end(annot) == destinationEnd
221                     && annot.getFeatures().size() == currentLookup.getFeatures().size()) {
222                   valid = false;
223                   break;
224                 }
225               }
226             }
227             
228             if(valid) {
229               addToOriginal(originalDocOutput, destinationStart, destinationEnd, 
230                   tempStartOffset, tempEndOffset, currentLookup, tempDoc);
231             }
232           } // END if coveredByInput(...)
233         } // END for OVER ALL THE Lookups
234       } 
235       finally {
236         gazetteerInst.setDocument(null);
237         if(tempDoc != null) {
238           // now remove the newDoc
239           Factory.deleteResource(tempDoc);
240         }
241       }
242     } // for
243     fireProcessFinished();
244   } // END execute METHOD

247   /**
248    * Removes the overlapping annotations. preserves the one that appears first
249    * in the list.  This assumes the list has been sorted already.
250    * 
251    * @param annotations
252    */
253   private void removeOverlappingAnnotations(List<Annotation> annotations) {
254     for(int i = 0; i < annotations.size() - 1; i++) {
255       Annotation annot1 = annotations.get(i);
256       Annotation annot2 = annotations.get(i + 1);
257       long annot2Start = Utils.start(annot2);
258       if(annot2Start >= Utils.start(annot1) && annot2Start < Utils.end(annot1)) {
259         annotations.remove(annot2);
260         i--;
261         continue;
262       }
263     }
264   }

267   /* We try hard not to cause InvalidOffsetExceptions, but let's have
268    * some better debugging info in case they happen.
269    */
270   private void addToOriginal(AnnotationSet original, long originalStart, long originalEnd, 
271       long tempStart, long tempEnd, Annotation tempLookup, Document tempDoc) throws ExecutionException {
272     try {
273       original.add(originalStart, originalEnd, tempLookup.getType(), tempLookup.getFeatures());
274     }
275     catch(InvalidOffsetException ioe) {
276       String errorDetails = String.format("temp %d, %d [%s]-> original %d, %d  ", tempStart, tempEnd, Utils.stringFor(tempDoc, tempLookup), 
277           originalStart, originalEnd);
278       throw new ExecutionException(errorDetails, ioe);
279     }
280   }

284   /* Is this Lookup within the scope of the input annotations?  It might not be, if Token annotations
285    * have been copied by AST only over the significant sections of the document.
286    */
287   private boolean coveredByInput(long tempStart, long tempEnd, AnnotationSet tempInputAS) {
288     if (tempInputAS.getCovering(wrappedInputASName, tempStart, tempStart).isEmpty()) {
289       return false;
290     }
291     // implied else
292     if (tempInputAS.getCovering(wrappedInputASName, tempEnd, tempEnd).isEmpty()) {
293       return false;
294     }
295     // implied else
296     return true;
297   }
