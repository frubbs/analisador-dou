089   /**
090    * Extract all the IVY elements from the given JDOM document and then add all
091    * the jars resulting from ivy's dependency resolution to the GateClassLoader.
092    * 
093    * @param creoleDoc
094    *          JDOM document representing a parsed creole.xml file.
095    */
096   private void addIvyDependencies(GateClassLoader gcl, Document creoleDoc) throws IOException {
097 
098     try {
099       List<Element> ivyElts = ExpandIvy.getIvyElements(creoleDoc);
100 
101       if(ivyElts.size() > 0) {
102 
103         Ivy ivy = ExpandIvy.getIvy(ExpandIvy.getSettingsURL());
104 
105         ResolveOptions resolveOptions = new ResolveOptions();
106         resolveOptions.setArtifactFilter(FilterHelper
107             .getArtifactTypeFilter(new String[]{"jar"}));
108         resolveOptions.setLog(LogOptions.LOG_QUIET);
109 
110         for(Element e : ivyElts) {
111           URL url = new URL(creoleFileUrl, ExpandIvy.getIvyPath(e));
112 
113           ResolveReport report = ivy.resolve(url, resolveOptions);
114           if(report.getAllProblemMessages().size() > 0)
115             throw new Exception("Unable to resolve all IVY dependencies");
116 
117           for(ArtifactDownloadReport dlReport : report.getAllArtifactsReports()) {
118             gcl.addURL(
119                 dlReport.getLocalFile().toURI().toURL());
120           }
121 
122         }
123       }
124     } catch(Exception ex) {
125       throw new IOException("Error using Ivy to add required dependencies", ex);
126     }
127   }

129   /**
130    * Recursively search the given element for JAR entries and add these
131    * jars to the GateClassLoader
132    *
133    * @param jdomElt JDOM element representing a creole.xml file
134    */
135   @SuppressWarnings("unchecked")
136   private void addJarsToClassLoader(GateClassLoader gcl, Element jdomElt)
137           throws MalformedURLException {
138     if("JAR".equals(jdomElt.getName())) {
139       URL url = new URL(creoleFileUrl, jdomElt.getTextTrim());
140       try {
141         java.io.InputStream s = url.openStream();
142         s.close();
143         gcl.addURL(url);
144       } catch(IOException e) {
145         log.debug("Unable to add JAR " + url + " specified in creole file " + creoleFileUrl +
146             " to class loader, hopefully the required classes are already on the classpath.");
147       }
148     }
149     else {
150       for(Element child : (List<Element>)jdomElt.getChildren()) {
151         addJarsToClassLoader(gcl, child);
152       }
153     }
154   }

156   /**
157    * Fetches the directory information for this handler's creole plugin
158    * and adds additional RESOURCE elements to the given JDOM document so
159    * that it contains a RESOURCE for every resource type defined in the
160    * plugin's directory info.
161    *
162    * @param jdomDoc JDOM document which should be the parsed creole.xml
163    *          that this handler was configured for.
164    */
165   public void createResourceElementsForDirInfo(Document jdomDoc)
166           throws MalformedURLException {
167     Element jdomElt = jdomDoc.getRootElement();
168     URL directoryUrl = new URL(creoleFileUrl, ".");
169     DirectoryInfo dirInfo = Gate.getDirectoryInfo(directoryUrl);
170     if(dirInfo != null) {
171       Map<String, Element> resourceElements = new HashMap<String, Element>();
172       findResourceElements(resourceElements, jdomElt);
173       for(ResourceInfo resInfo : (List<ResourceInfo>)dirInfo
174               .getResourceInfoList()) {
175         if(!resourceElements.containsKey(resInfo.getResourceClassName())) {
176           // no existing RESOURCE element for this resource type (so it
177           // was
178           // auto-discovered from a <JAR SCAN="true">), so add a minimal
179           // RESOURCE element which will be filled in by the annotation
180           // processor.
181           jdomElt.addContent(new Element("RESOURCE").addContent(new Element(
182                   "CLASS").setText(resInfo.getResourceClassName())));
183         }
184       }
185     }

189   @SuppressWarnings("unchecked")
190   private void findResourceElements(Map<String, Element> map, Element elt) {
191     if(elt.getName().equals("RESOURCE")) {
192       String className = elt.getChildTextTrim("CLASS");
193       if(className != null) {
194         map.put(className, elt);
195       }
196     }
197     else {
198       for(Element child : (List<Element>)elt.getChildren()) {
199         findResourceElements(map, child);
200       }
201     }

216    * Process annotations for the given element. If the element is a

218    * recursively for all the children of the given element.
219    *

222   @SuppressWarnings("unchecked")
223   private void processAnnotations(Element element) throws GateException {
224     if("RESOURCE".equals(element.getName())) {
225       processAnnotationsForResource(element);
226     }

229         processAnnotations(child);
230       }
231     }

234   /**
235    * Process the given RESOURCE element, adding extra elements to it
236    * based on the annotations on the resource class.
237    *
238    * @param element the RESOURCE element to process.
239    */
240   private void processAnnotationsForResource(Element element)
241           throws GateException {
242     String className = element.getChildTextTrim("CLASS");
243     if(className == null) {
244       throw new GateException("\"CLASS\" element not found for resource in "
245               + creoleFileUrl);
246     }
247     Class<?> resourceClass = null;
248     try {
249       resourceClass = Gate.getClassLoader().loadClass(className);
250     }
251     catch(ClassNotFoundException e) {
252       log.debug("Couldn't load class " + className + " for resource in "
253           + creoleFileUrl, e);
254       throw new GateException("Couldn't load class " + className
255               + " for resource in " + creoleFileUrl);
256     }
257 
258     processCreoleResourceAnnotations(element, resourceClass);
259   }

261   @SuppressWarnings("unchecked")
262   public void processCreoleResourceAnnotations(Element element,
263           Class<?> resourceClass) throws GateException {
264     CreoleResource creoleResourceAnnot = resourceClass
265             .getAnnotation(CreoleResource.class);
266     if(creoleResourceAnnot != null) {
267       // first process the top-level data and autoinstances
268       processResourceData(resourceClass, element);
269 
270       // now deal with parameters
271       // build a map holding the element corresponding to each parameter
272       Map<String, Element> parameterMap = new HashMap<String, Element>();
273       Map<String, Element> disjunctionMap = new HashMap<String, Element>();
274       for(Element paramElt : (List<Element>)element.getChildren("PARAMETER")) {
275         parameterMap.put(paramElt.getAttributeValue("NAME"), paramElt);
276       }
277       for(Element disjunctionElt : (List<Element>)element.getChildren("OR")) {
278         String disjunctionId = disjunctionElt.getAttributeValue("ID");
279         if(disjunctionId != null) {
280           disjunctionMap.put(disjunctionId, disjunctionElt);
281         }
282         for(Element paramElt : (List<Element>)disjunctionElt
283                 .getChildren("PARAMETER")) {
284           parameterMap.put(paramElt.getAttributeValue("NAME"), paramElt);
285         }
286       }
287 
288       processParameters(resourceClass, element, parameterMap, disjunctionMap);
289     }
290   }

292   /**
293    * Process the {@link CreoleResource} data for this class. This method
294    * first extracts the non-inheritable data (PRIVATE, MAIN_VIEWER,
295    * NAME and TOOL), then calls {@link #processInheritableResourceData}
296    * to process the inheritable data, then deals with any specified
297    * {@link AutoInstance}s.
298    *
299    * @param resourceClass the resource class to process, which must be
300    *          annotated with {@link CreoleResource}.
301    * @param element the RESOURCE element to which data should be added.
302    */
303   private void processResourceData(Class<?> resourceClass, Element element) {
304     CreoleResource cr = resourceClass.getAnnotation(CreoleResource.class);
305     if(cr.isPrivate() && element.getChild("PRIVATE") == null) {
306       element.addContent(new Element("PRIVATE"));
307     }
308     if(cr.mainViewer() && element.getChild("MAIN_VIEWER") == null) {
309       element.addContent(new Element("MAIN_VIEWER"));
310     }
311     if(cr.tool() && element.getChild("TOOL") == null) {
312       element.addContent(new Element("TOOL"));
313     }
314     // NAME is the name given in the annotation, or the simple name of
315     // the class if omitted
316     addElement(element, ("".equals(cr.name()))
317             ? resourceClass.getSimpleName()
318             : cr.name(), "NAME");
319     processInheritableResourceData(resourceClass, element);
320     processAutoInstances(cr, element);
321   }

323   /**
324    * Recursive method to process the {@link CreoleResource} elements
325    * that can be inherited from superclasses and interfaces (everything
326    * except the PRIVATE and MAIN_VIEWER flags, the NAME and the
327    * AUTOINSTANCEs). Once data has been extracted from the current class
328    * the method calls itself recursively for the superclass and any
329    * implemented interfaces. For any given attribute, the first value
330    * specified wins (i.e. the one on the most specific class).
331    *
332    * @param clazz the class to process
333    * @param element the RESOURCE element to which data should be added.
334    */
335   private void processInheritableResourceData(Class<?> clazz, Element element) {
336     CreoleResource cr = clazz.getAnnotation(CreoleResource.class);
337     if(cr != null) {
338       addElement(element, cr.comment(), "COMMENT");
339       addElement(element, cr.helpURL(), "HELPURL");
340       addElement(element, cr.interfaceName(), "INTERFACE");
341       addElement(element, cr.icon(), "ICON");
342       if(cr.guiType() != GuiType.NONE && element.getChild("GUI") == null) {
343         Element guiElement = new Element("GUI").setAttribute("TYPE", cr
344                 .guiType().toString());
345         element.addContent(guiElement);
346         addElement(guiElement, cr.resourceDisplayed(), "RESOURCE_DISPLAYED");
347       }
348       addElement(element, cr.annotationTypeDisplayed(),
349               "ANNOTATION_TYPE_DISPLAYED");
350     }
351 
352     Class<?> superclass = clazz.getSuperclass();
353     if(superclass != null) {
354       processInheritableResourceData(superclass, element);
355     }
356 
357     for(Class<?> intf : clazz.getInterfaces()) {
358       processInheritableResourceData(intf, element);
359     }
360   }

362   /**
363    * Adds an element with the given name and text value to the parent
364    * element, but only if no such child element already exists and the
365    * value is not the empty string.
366    *
367    * @param parent the parent element
368    * @param value the text value for the new child
369    * @param elementName the name of the new child element
370    */
371   private void addElement(Element parent, String value, String elementName) {
372     if(!"".equals(value) && parent.getChild(elementName) == null) {
373       parent.addContent(new Element(elementName).setText(value));
374     }
375   }

377   /**
378    * Process the {@link AutoInstance} annotations contained in the given
379    * {@link CreoleResource} and add the corresponding
380    * AUTOINSTANCE/HIDDEN-AUTOINSTANCE elements to the given parent.
381    *
382    * @param cr the {@link CreoleResource} annotation
383    * @param element the parent element
384    */
385   private void processAutoInstances(CreoleResource cr, Element element) {
386     for(AutoInstance ai : cr.autoinstances()) {
387       Element aiElt = null;
388       if(ai.hidden()) {
389         aiElt = new Element("HIDDEN-AUTOINSTANCE");
390       }
391       else {
392         aiElt = new Element("AUTOINSTANCE");
393       }
394       element.addContent(aiElt);
395       for(AutoInstanceParam param : ai.parameters()) {
396         aiElt.addContent(new Element("PARAM")
397                 .setAttribute("NAME", param.name()).setAttribute("VALUE",
398                         param.value()));
399       }
400     }

403   /**
404    * Process any {@link CreoleParameter} and
405    * {@link HiddenCreoleParameter} annotations on set methods of the
406    * given class and set up the corresponding PARAMETER elements.
407    *
408    * @param resourceClass the resource class to process
409    * @param resourceElement the RESOURCE element to which the PARAMETERs
410    *          are to be added
411    * @param parameterMap a map from parameter names to the PARAMETER
412    *          elements that define them. This is used as we combine
413    *          information from the original creole.xml, the parameter
414    *          annotation on the target method and the annotations on the
415    *          same method of its superclasses and interfaces. Parameter
416    *          names that have been hidden by a
417    *          {@link HiddenCreoleParameter} annotation are explicitly
418    *          mapped to <code>null</code> in this map.
419    * @param disjunctionMap a map from disjunction IDs to the OR elements
420    *          that define them. Disjunctive parameters are handled by
421    *          specifying a disjunction ID on the {@link CreoleParameter}
422    *          annotations - parameters with the same disjunction ID are
423    *          grouped under the same OR element.
424    */
425   private void processParameters(Class<?> resourceClass,
426           Element resourceElement, Map<String, Element> parameterMap,
427           Map<String, Element> disjunctionMap) throws GateException {
428     for(Method method : resourceClass.getDeclaredMethods()) {
429       CreoleParameter paramAnnot = method.getAnnotation(CreoleParameter.class);
430       HiddenCreoleParameter hiddenParamAnnot = method
431               .getAnnotation(HiddenCreoleParameter.class);
432       if(paramAnnot != null || hiddenParamAnnot != null) {
433         if(!method.getName().startsWith("set") || method.getName().length() < 4
434                 || method.getParameterTypes().length != 1) {
435           throw new GateException("Creole parameter annotation found on "
436                   + method
437                   + ", but only setter methods may have this annotation.");
438         }
439         // extract the parameter name from the method name
440         String paramName = Character.toLowerCase(method.getName().charAt(3))
441                 + method.getName().substring(4);
442         if(hiddenParamAnnot != null && !parameterMap.containsKey(paramName)) {
443           parameterMap.put(paramName, null);
444         }
445         if(paramAnnot != null) {
446           Element paramElt = null;
447           if(parameterMap.containsKey(paramName)) {
448             paramElt = parameterMap.get(paramName);
449           }
450           else {
451             paramElt = new Element("PARAMETER").setAttribute("NAME", paramName);
452             if(!"".equals(paramAnnot.disjunction())) {
453               Element disjunctionElt = disjunctionMap.get(paramAnnot
454                       .disjunction());
455               if(disjunctionElt == null) {
456                 disjunctionElt = new Element("OR");
457                 resourceElement.addContent(disjunctionElt);
458                 disjunctionMap.put(paramAnnot.disjunction(), disjunctionElt);
459               }
460               disjunctionElt.addContent(paramElt);
461             }
462             else {
463               resourceElement.addContent(paramElt);
464             }
465             parameterMap.put(paramName, paramElt);
466           }
467 
468           if(paramElt != null) {
469             // here we have a valid element for the current parameter,
470             // which has not been masked by a @HiddenCreoleParameter
471             if(paramElt.getTextTrim().length() == 0) {
472               // need to determine the type
473               paramElt.setText(method.getParameterTypes()[0].getName());
474               // for collections (but not GATE Resources) we also try to determine
475               // the item type.
476               if((!Resource.class.isAssignableFrom(
477                       method.getParameterTypes()[0])) &&
478                   Collection.class.isAssignableFrom(
479                       method.getParameterTypes()[0])) {
480                 determineCollectionElementType(method, paramElt);
481               }
482             }
483 
484             // other attributes
485             addAttribute(paramElt, paramAnnot.comment(), "", "COMMENT");
486             addAttribute(paramElt, paramAnnot.suffixes(), "", "SUFFIXES");
487             addAttribute(paramElt, paramAnnot.defaultValue(),
488                     CreoleParameter.NO_DEFAULT_VALUE, "DEFAULT");
489             addAttribute(paramElt, String.valueOf(paramAnnot.priority()),
490                     String.valueOf(CreoleParameter.DEFAULT_PRIORITY), "PRIORITY");
491 
492             // runtime and optional are based on marker annotations
493             String runtimeParam = "";
494             if(method.isAnnotationPresent(RunTime.class)) {
495               runtimeParam = String.valueOf(method.getAnnotation(RunTime.class)
496                       .value());
497             }
498             addAttribute(paramElt, runtimeParam, "", "RUNTIME");
499 
500             String optionalParam = "";
501             if(method.isAnnotationPresent(Optional.class)) {
502               optionalParam = String.valueOf(method.getAnnotation(
503                       Optional.class).value());
504             }
505             addAttribute(paramElt, optionalParam, "", "OPTIONAL");
506           }
507         }
508       }
509     }
510 
511     // go up the tree
512     Class<?> superclass = resourceClass.getSuperclass();
513     if(superclass != null) {
514       processParameters(superclass, resourceElement, parameterMap,
515               disjunctionMap);
516     }
517 
518     for(Class<?> intf : resourceClass.getInterfaces()) {
519       processParameters(intf, resourceElement, parameterMap, disjunctionMap);
520     }
521   }

523   /**
524    * Given a single-argument method whose parameter is a
525    * {@link Collection}, use the method's generic type information to
526    * determine the collection element type and store it as the
527    * ITEM_CLASS_NAME attribute of the given Element.
528    *
529    * @param method the setter method
530    * @param paramElt the PARAMETER element
531    */
532   private void determineCollectionElementType(Method method, Element paramElt) {
533     if(paramElt.getAttributeValue("ITEM_CLASS_NAME") == null) {
534       Class<?> elementType;
535       CreoleParameter paramAnnot = method.getAnnotation(CreoleParameter.class);
536       if(paramAnnot != null
537               && paramAnnot.collectionElementType() != CreoleParameter.NoElementType.class) {
538         elementType = paramAnnot.collectionElementType();
539       }
540       else {
541         Type paramType = method.getGenericParameterTypes()[0];
542         elementType = findCollectionElementType(paramType);
543       }
544       if(elementType != null) {
545         paramElt.setAttribute("ITEM_CLASS_NAME", elementType.getName());
546       }
547     }
548   }
549 
550   /**
551    * Find the collection element type for the given type.
552    *
553    * @param type the type to use. To be able to find the element type,
554    *          this must be a Class that is assignable from Collection or
555    *          a ParameterizedType whose raw type is assignable from
556    *          Collection.
557    * @return the Class representing the collection element type, or
558    *         <code>null</code> if this cannot be determined
559    */
560   private Class<?> findCollectionElementType(Type type) {
561     return findCollectionElementType(type,
562             new HashMap<TypeVariable<?>, Class<?>>());
563   }
564 
565   /**
566    * Recursive method to find the collection element type for the given
567    * type.
568    *
569    * @param type the type to use
570    * @param tvMap map from type variables to the classes they are
571    *          ultimately bound to. The reflection APIs can tell us that
572    *          List&lt;String&gt; is an instantiation of List&lt;X&gt;
573    *          and List&lt;X&gt; implements Collection&lt;X&gt;, but we
574    *          have to keep track of the fact that X maps to String
575    *          ourselves.
576    * @return the collection element type, or <code>null</code> if it
577    *         cannot be determined.
578    */
579   private Class<?> findCollectionElementType(Type type,
580           Map<TypeVariable<?>, Class<?>> tvMap) {
581     Class<?> rawClass = null;
582     if(type instanceof Class) {
583       // we have a non-parameterized type, but it might be a raw class
584       // that extends a parameterized one (e.g. CustomCollection extends
585       // Set<Integer>) so we still need to look up
586       // the class tree
587       rawClass = (Class<?>)type;
588     }
589     else if(type instanceof ParameterizedType) {
590       Type rawType = ((ParameterizedType)type).getRawType();
591       // if we've reached Collection<T>, look at the tvMap to find what
592       // T maps to and return that
593       if(rawType == Collection.class) {
594         Type collectionTypeArgument = ((ParameterizedType)type)
595                 .getActualTypeArguments()[0];
596         if(collectionTypeArgument instanceof Class<?>) {
597           // e.g. Collection<String>
598           return (Class<?>)collectionTypeArgument;
599         }
600         else if(collectionTypeArgument instanceof TypeVariable<?>) {
601           // e.g. Collection<X>
602           return tvMap.get(collectionTypeArgument);
603         }
604         else {
605           // e.g. Collection<? extends Widget> or Collection<T[]>- we
606           // can't handle this in creole.xml so give up
607           return null;
608         }
609       }
610 
611       // we haven't reached Collection here, so add the type variable
612       // mappings to the tvMap before we look up the tree
613       if(rawType instanceof Class) {
614         rawClass = (Class<?>)rawType;
615         Type[] actualTypeParams = ((ParameterizedType)type)
616                 .getActualTypeArguments();
617         TypeVariable<?>[] formalTypeParams = ((Class<?>)rawType)
618                 .getTypeParameters();
619         for(int i = 0; i < actualTypeParams.length; i++) {
620           if(actualTypeParams[i] instanceof Class) {
621             tvMap.put(formalTypeParams[i], (Class<?>)actualTypeParams[i]);
622           }
623           else if(actualTypeParams[i] instanceof TypeVariable) {
624             tvMap.put(formalTypeParams[i], tvMap.get(actualTypeParams[i]));
625           }
626         }
627       }
628     }
629 
630     // process the superclass, if there is one, and any implemented
631     // interfaces
632     if(rawClass != null) {
633       Type superclass = rawClass.getGenericSuperclass();
634       if(type != null) {
635         Class<?> componentType = findCollectionElementType(superclass, tvMap);
636         if(componentType != null) {
637           return componentType;
638         }
639       }
640 
641       for(Type intf : rawClass.getGenericInterfaces()) {
642         Class<?> componentType = findCollectionElementType(intf, tvMap);
643         if(componentType != null) {
644           return componentType;
645         }
646       }
647     }
648 
649     return null;
650   }

652   /**
653    * Add an attribute with the given value to the given element, but
654    * only if the element does not already have the attribute, and the
655    * value is not equal to the given default value.
656    *
657    * @param paramElt the element
658    * @param value the attribute value (which will be converted to a
659    *          string)
660    * @param defaultValue if value.equals(defaultValue) we do not add the
661    *          attribute.
662    * @param attrName the name of the attribute to add.
663    */
664   private void addAttribute(Element paramElt, Object value,
665           Object defaultValue, String attrName) {
666     if(!defaultValue.equals(value) && paramElt.getAttribute(attrName) == null) {
667       paramElt.setAttribute(attrName, value.toString());
668     }
669   }
