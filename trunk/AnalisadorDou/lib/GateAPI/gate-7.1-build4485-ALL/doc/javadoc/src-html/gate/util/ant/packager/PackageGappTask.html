052 /**
053  * Ant task to copy a gapp file, rewriting any relative paths it
054  * contains to point within the same directory as the target file
055  * location and copy the referenced files into the right locations. The
056  * resulting structure is self-contained and can be packaged up (e.g. in
057  * a zip file) to send to a third party.
058  * 
059  * @author Ian Roberts
060  */

063   /**
064    * Comparator to compare URLs by lexicographic ordering of their
065    * getPath() values.  <code>null</code> compares less-than anything
066    * not <code>null</code>.
067    */
068   public static final Comparator<URL> PATH_COMPARATOR = new Comparator<URL>() {
069     public int compare(URL a, URL b) {
070       if(a == null) {
071         return (b == null) ? 0 : -1;
072       }
073       if(b == null) {
074         return 1;
075       }
076       return a.getPath().compareTo(b.getPath());
077     }
078   };

298   @Override
299   public void execute() throws BuildException {
300     // process the hints
301     for(MappingHint h : hintTasks) {
302       h.perform();
303     }
304 
305     // map to store the necessary file copy operations
306     Map<URL, URL> fileCopyMap = new HashMap<URL, URL>();
307     Map<URL, URL> dirCopyMap = new HashMap<URL, URL>();
308     TreeMap<URL, URL> pluginCopyMap = new TreeMap<URL, URL>(PATH_COMPARATOR);
309 
310     log("Packaging gapp file " + src);
311     // do the work
312     GappModel gappModel = null;
313     URL newFileURL = null;
314     try {
315       URL gateHomeURL = null;
316       // convert gateHome to a URL, if it was provided
317       if(gateHome != null) {
318         gateHomeURL = gateHome.toURI().toURL();
319       }
320       gappModel = new GappModel(src.toURI().toURL(), gateHomeURL);
321       newFileURL = destFile.toURI().toURL();
322       gappModel.setGappFileURL(newFileURL);
323     }
324     catch(MalformedURLException e) {
325       throw new BuildException("Couldn't convert src or dest file to URL", e,
326               getLocation());
327     }
328 
329     // we use TreeSet for these sets so we will process the paths
330     // higher up the directory tree before paths pointing to their
331     // subdirectories.
332     Set<URL> plugins = new TreeSet<URL>(PATH_COMPARATOR);
333     plugins.addAll(gappModel.getPluginURLs());
334     Set<URL> resources = new TreeSet<URL>(PATH_COMPARATOR);
335     resources.addAll(gappModel.getResourceURLs());
336 
337     // process the extraresourcespath elements (if any)
338     processExtraResourcesPaths(resources);
339 
340     // first look at the explicit mapping hints
341     if(mappingHints != null && !mappingHints.isEmpty()) {
342       Iterator<URL> resourcesIt = resources.iterator();
343       while(resourcesIt.hasNext()) {
344         URL resource = resourcesIt.next();
345         for(URL hint : mappingHints.keySet()) {
346           String hintString = hint.toExternalForm();
347           if(resource.equals(hint)
348                   || (hintString.endsWith("/") && resource.toExternalForm()
349                           .startsWith(hintString))) {
350             // found this resource under this hint
351             log("Found resource " + resource + " under mapping hint URL "
352                     + hint, Project.MSG_VERBOSE);
353             String hintTarget = mappingHints.get(hint);
354             URL newResourceURL = null;
355             if(hintTarget == null) {
356               // hint asks to map to an absolute URL
357               log("  Converting to absolute URL", Project.MSG_VERBOSE);
358               newResourceURL = resource;
359             }
360             else {
361               // relativize the URL against the hint source and
362               // construct the new URL relative to the hint target
363               try {
364                 URL mappedHint = new URL(newFileURL, hintTarget);
365                 String resourceRelpath =
366                         PersistenceManager.getRelativePath(hint, resource);
367                 newResourceURL = new URL(mappedHint, resourceRelpath);
368                 fileCopyMap.put(resource, newResourceURL);
369                 if(copyResourceDirs) {
370                   dirCopyMap.put(new URL(resource, "."), new URL(
371                           newResourceURL, "."));
372                 }
373               }
374               catch(MalformedURLException e) {
375                 throw new BuildException("Couldn't construct URL relative to "
376                         + hintTarget + " for " + resource, e, getLocation());
377               }
378               log("  Relocating to " + newResourceURL, Project.MSG_VERBOSE);
379             }
380             // do the relocation
381             gappModel.updatePathForURL(resource, newResourceURL,
382                     hintTarget != null);
383             // we've now dealt with this resource, so don't need to
384             // handle it later
385             resourcesIt.remove();
386             break;
387           }
388         }
389       }
390     }
391 
392     // Any resources that aren't covered by the hints, try and
393     // resolve them relative to the plugins referenced by the
394     // application.
395     Iterator<URL> pluginsIt = plugins.iterator();
396     while(pluginsIt.hasNext()) {
397       URL pluginURL = pluginsIt.next();
398       pluginsIt.remove();
399       URL newPluginURL = null;
400       
401       String pluginName = pluginURL.getFile();
402       log("Processing plugin " + pluginName, Project.MSG_VERBOSE);
403 
404       // first check whether this plugin is a subdirectory of another plugin
405       // we have already processed
406       SortedMap<URL, URL> possibleAncestors = pluginCopyMap.headMap(pluginURL);
407       URL ancestorPlugin = null;
408       if(!possibleAncestors.isEmpty()) ancestorPlugin = possibleAncestors.lastKey();
409       if(ancestorPlugin != null && pluginURL.toExternalForm().startsWith(
410               ancestorPlugin.toExternalForm())) {
411         // this plugin is under one we have already dealt with
412         log("  Plugin is located under another plugin " + ancestorPlugin, Project.MSG_VERBOSE);
413         String relPath = PersistenceManager.getRelativePath(ancestorPlugin, pluginURL);
414         try {
415           newPluginURL = new URL(pluginCopyMap.get(ancestorPlugin), relPath);
416         }
417         catch(MalformedURLException e) {
418           throw new BuildException("Couldn't construct URL relative to plugins/"
419                   + " for " + pluginURL, e, getLocation());
420         }
421       }
422       else {
423         // normal case, this plugin is not a subdir of another plugin
424         boolean addSlash = false;
425         // we will map the plugin whose directory name is X to plugins/X
426         if(pluginName.endsWith("/")) {
427           addSlash = true;
428           pluginName = pluginName.substring(
429                   pluginName.lastIndexOf('/', pluginName.length() - 2) + 1,
430                   pluginName.length() - 1);
431         }
432         else {
433           pluginName = pluginName.substring(pluginName.lastIndexOf('/') + 1);
434         }
435         log("  Plugin name is " + pluginName, Project.MSG_VERBOSE);
436         try {
437           newPluginURL = new URL(newFileURL, "plugins/" + pluginName + (addSlash ? "/" : ""));
438           // a gapp may refer to two or more plugins with the same name.
439           // If plugins/{pluginName} is already taken, try
440           // plugins/{pluginName}-2, plugins/{pluginName}-3, etc.,
441           // until we find one that is available.
442           if(pluginCopyMap.containsValue(newPluginURL)) {
443             int index = 2;
444             do {
445               newPluginURL =
446                       new URL(newFileURL, "plugins/" + pluginName + "-"
447                               + (index++) + (addSlash ? "/" : ""));
448             } while(pluginCopyMap.containsValue(newPluginURL));
449           }
450         }
451         catch(MalformedURLException e) {
452           throw new BuildException("Couldn't construct URL relative to plugins/"
453                   + " for " + pluginURL, e, getLocation());
454         }
455       }
456       log("  Relocating to " + newPluginURL, Project.MSG_VERBOSE);
457 
458       // deal with the plugin URL itself (in the urlList)
459       gappModel.updatePathForURL(pluginURL, newPluginURL, true);
460       pluginCopyMap.put(pluginURL, newPluginURL);
461 
462       // now look for resources located under that plugin
463       String pluginUri = pluginURL.toExternalForm();
464       if(!pluginUri.endsWith("/")) {
465         pluginUri += "/";
466       }
467       Iterator<URL> resourcesIt = resources.iterator();
468       while(resourcesIt.hasNext()) {
469         URL resourceURL = resourcesIt.next();
470         try {
471           if(resourceURL.toExternalForm().startsWith(
472                   pluginUri)) {
473             // found a resource under this plugin, so relocate it to be
474             // under the re-located plugin dir
475             resourcesIt.remove();
476             String resourceRelpath =
477                     PersistenceManager.getRelativePath(pluginURL, resourceURL);
478             log("    Found resource " + resourceURL, Project.MSG_VERBOSE);
479             URL newResourceURL = null;
480             newResourceURL = new URL(newPluginURL, resourceRelpath);
481             log("    Relocating to " + newResourceURL, Project.MSG_VERBOSE);
482             gappModel.updatePathForURL(resourceURL, newResourceURL, true);
483             fileCopyMap.put(resourceURL, newResourceURL);
484             if(copyResourceDirs) {
485               dirCopyMap.put(new URL(resourceURL, "."), new URL(newResourceURL,
486                       "."));
487             }
488           }
489         }
490         catch(MalformedURLException e) {
491           throw new BuildException("Couldn't construct URL relative to "
492                   + newPluginURL + " for " + resourceURL, e, getLocation());
493         }
494       }
495     }
496 
497     // anything left over, handle according to onUnresolved
498     if(!resources.isEmpty()) {
499       switch(onUnresolved) {
500         case fail:
501           // easy case - fail the build
502           log("There were unresolved resources:", Project.MSG_ERR);
503           for(URL res : resources) {
504             log(res.toExternalForm(), Project.MSG_ERR);
505           }
506           log("Either set onUnresolved=\"absolute|recover\" or add the "
507                   + "relevant mapping hints", Project.MSG_ERR);
508           throw new BuildException("There were unresolved resources",
509                   getLocation());
510 
511         case absolute:
512           // convert all unresolved resources to absolute URLs
513           log("There were unresolved resources, which have been made absolute",
514                   Project.MSG_WARN);
515           for(URL res : resources) {
516             gappModel.updatePathForURL(res, res, false);
517             log(res.toExternalForm(), Project.MSG_VERBOSE);
518           }
519           break;
520 
521         case recover:
522           // the clever case - recover by putting all the unresolved
523           // resources into subdirectories of an "application-resources"
524           // directory under the output dir
525           URL unresolvedResourcesDir = null;
526           try {
527             unresolvedResourcesDir =
528                     new URL(newFileURL, "application-resources/");
529           }
530           catch(MalformedURLException e) {
531             throw new BuildException("Can't construct URL relative to "
532                     + newFileURL + " for application-resources", e,
533                     getLocation());
534           }
535           // map to track where under application-resources we should map
536           // each directory that contains unresolved resources
537           TreeMap<URL, URL> unresolvedResourcesSubDirs = new TreeMap<URL, URL>(PATH_COMPARATOR);
538           log("There were unresolved resources, which have been gathered into "
539                   + unresolvedResourcesDir, Project.MSG_INFO);
540           for(URL res : resources) {
541             URL resourceDir = null;
542             try {
543               resourceDir = new URL(res, ".");
544             }
545             catch(MalformedURLException e) {
546               throw new BuildException(
547                       "Can't construct URL to parent directory of " + res, e,
548                       getLocation());
549             }
550             URL targetDir =
551                     getUnresolvedResourcesTarget(unresolvedResourcesSubDirs,
552                             unresolvedResourcesDir, resourceDir);
553             String resName = res.getFile();
554             resName = resName.substring(resName.lastIndexOf('/') + 1);
555             URL newResourceURL = null;
556             try {
557               newResourceURL = new URL(targetDir, resName);
558             }
559             catch(MalformedURLException e) {
560               throw new BuildException("Can't construct URL relative to "
561                       + unresolvedResourcesDir + " for " + resName, e,
562                       getLocation());
563             }
564             gappModel.updatePathForURL(res, newResourceURL, true);
565             fileCopyMap.put(res, newResourceURL);
566             if(copyResourceDirs) {
567               dirCopyMap.put(resourceDir, targetDir);
568             }
569           }
570           break;
571 
572         default:
573           throw new BuildException("Unrecognised UnresolvedAction",
574                   getLocation());
575       }
576     }
577 
578     // write out the fixed GAPP file
579     try {
580       log("Writing modified gapp to " + destFile);
581       gappModel.write();
582     }
583     catch(IOException e) {
584       throw new BuildException("Error writing out modified GAPP file", e,
585               getLocation());
586     }
587 
588     // now copy the files that it references
589     if(fileCopyMap.size() > 0) {
590       log("Copying " + fileCopyMap.size() + " resources");
591     }
592     for(Map.Entry<URL, URL> resEntry : fileCopyMap.entrySet()) {
593       File source = Files.fileFromURL(resEntry.getKey());
594       File dest = Files.fileFromURL(resEntry.getValue());
595       if(source.isDirectory()) {
596         // source URL points to a directory, so create a corresponding
597         // directory dest
598         dest.mkdirs();
599       }
600       else {
601         // source URL doesn't point to a directory, so
602         // ensure parent directory exists
603         dest.getParentFile().mkdirs();
604         if(source.isFile()) {
605           // source URL points to an existing file, copy it
606           try {
607             log("Copying " + source + " to " + dest, Project.MSG_VERBOSE);
608             FileUtils.getFileUtils().copyFile(source, dest);
609           }
610           catch(IOException e) {
611             throw new BuildException(
612                     "Error copying file " + source + " to " + dest, e,
613                     getLocation());
614           }
615         }
616       }
617     }
618 
619     // handle the plugins
620     if(pluginCopyMap.size() > 0) {
621       log("Copying " + pluginCopyMap.size() + " plugins");
622       if(copyPlugins) {
623         log("Also copying complete plugin contents", Project.MSG_VERBOSE);

637             ivyExpander.setDir(dir);
638             ivyExpander.init();
639             ivyExpander.perform();
640           }
641         }
642       }

647       log("Copying " + dirCopyMap.size() + " resource directories");

735   /**
736    * Extract the text values from any &lt;JAR&gt; elements contained in the
737    * referenced creole.xml file.
738    * 
739    * @return a set with one element for each unique &lt;JAR&gt; entry in the
740    *         given creole.xml.
741    */
742   private Set<String> getJars(URL creoleXml) {
743     try {
744       Set<String> jars = new HashSet<String>();
745       // the XPath is a bit ugly, but needed to match the element name
746       // case-insensitively.
747       XPath jarXPath =
748           XPath
749               .newInstance("//*[translate(local-name(), 'jar', 'JAR') = 'JAR']");
750       SAXBuilder builder = new SAXBuilder();
751       Document creoleDoc = builder.build(creoleXml);
752       // technically unsafe, but we know that the above XPath expression
753       // can only match elements.

755       List<Element> jarElts = jarXPath.selectNodes(creoleDoc);
756       for(Element e : jarElts) {
757         jars.add(e.getTextTrim());
758       }
759 
760       return jars;
761     } catch(JDOMException e) {
762       throw new BuildException("Error extracting JAR elements from "
763           + creoleXml, e, getLocation());
764     } catch(IOException e) {
765       throw new BuildException("Error loading " + creoleXml
766           + " to extract JARs", e, getLocation());
767     }
768   }

770   /**
771    * Get a URL for a directory to which the given (unresolved) resource
772    * directory should be mapped.
773    * 
774    * @param unresolvedResourcesSubDirs a map from URLs of directories
775    *          containing unresolved resources to the URLs under the
776    *          target unresolved-resources directory that they will be
777    *          mapped to. This map is updated by this method.
778    * @param unresolvedResourcesDir the top-level application-resources
779    *          directory in the target location.
780    * @param resourceDir a directory containing an unresolved resource.
781    * @return the URL under application-resources to which this directory
782    *         should be mapped. For a resourceDir of the form .../foo,
783    *         the returned URL would typically be
784    *         &lt;applicationResourcesDir&gt;/foo, but if a different
785    *         directory with the same name has already been mapped then
786    *         we will return the first ..../foo-2, foo-3, etc. that is
787    *         not already in use.  If one of the directory's ancestors
788    *         has already been mapped then we return a URL pointing
789    *         to the same relative path inside that ancestor's mapping,
790    *         e.g. if .../foo has already been mapped to a-r/foo-2 then
791    *         .../foo/bar/baz will map to a-r/foo-2/bar/baz.
792    */
793   private URL getUnresolvedResourcesTarget(
794           TreeMap<URL, URL> unresolvedResourcesSubDirs, URL unresolvedResourcesDir,
795           URL resourceDir) throws BuildException {
796     URL targetDir = unresolvedResourcesSubDirs.get(resourceDir);
797     try {
798       if(targetDir == null) {
799         // no exact match, try an ancestor match
800         SortedMap<URL, URL> possibleAncestors = unresolvedResourcesSubDirs.headMap(resourceDir);
801         URL nearestAncestor = null;
802         if(!possibleAncestors.isEmpty()) nearestAncestor = possibleAncestors.lastKey();
803         if(nearestAncestor != null && resourceDir.toExternalForm().startsWith(
804                 nearestAncestor.toExternalForm())) {
805           // found an ancestor mapping, so take the relative path
806           // from the ancestor to this dir and map it to the same
807           // path under the ancestor's mapping.
808           String relPath = PersistenceManager.getRelativePath(nearestAncestor, resourceDir);
809           targetDir = new URL(unresolvedResourcesSubDirs.get(nearestAncestor), relPath);
810         }
811         else {
812           // no ancestors currently mapped, so start a new sub-dir of
813           // unresolvedResourcesDir whose name is the last path
814           // component of the source URL
815           String resourcePath = resourceDir.getFile();
816           if(resourcePath.endsWith("/")) {
817             resourcePath = resourcePath.substring(0, resourcePath.length() - 1);
818           }
819           String targetDirName =
820                   resourcePath.substring(resourcePath.lastIndexOf('/') + 1);
821           if(targetDirName.length() == 0) {
822             // edge case, if the source URL points to the root directory "/"
823             targetDirName = "resources";
824           }
825           // try application-resources/{targetDirName} as the target
826           targetDir = new URL(unresolvedResourcesDir, targetDirName + "/");
827           // if this is already taken, try
828           // application-resources/{targetDirName}-2,
829           // application-resources/{targetDirName}-3, etc., until we find
830           // one that is available.
831           if(unresolvedResourcesSubDirs.containsValue(targetDir)) {
832             int index = 2;
833             do {
834               targetDir =
835                       new URL(unresolvedResourcesDir, targetDirName + "-"
836                               + (index++) + "/");
837             } while(unresolvedResourcesSubDirs.containsValue(targetDir));
838           }
839         }
840         
841         // store the mapping for future use
842         unresolvedResourcesSubDirs.put(resourceDir, targetDir);
843       }
844     }
845     catch(MalformedURLException e) {
846       throw new BuildException("Can't construct target URL for directory "
847               + resourceDir, e, getLocation());
848     }
849 
850     return targetDir;
851   }
