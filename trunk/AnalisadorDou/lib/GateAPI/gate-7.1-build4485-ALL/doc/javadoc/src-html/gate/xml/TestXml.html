211   /*
212    * This method runs ANNIE with defaults on a document, then saves
213    * it as a GATE XML document and loads it back. All the annotations on the
214    * loaded document should be the same as the original ones.
215    *
216    * It also verifies if the matches feature still holds after an export/import to XML
217    */
218   public void testAnnotationConsistencyForSaveAsXml()throws Exception{
219     // Load a document from the test repository
220     //Document origDoc = gate.Factory.newDocument(Gate.getUrl("tests/xml/gateTestSaveAsXML.xml"));
221     String testDoc = gate.util.Files.getGateResourceAsString("gate.ac.uk/tests/xml/gateTestSaveAsXML.xml");
222     Document origDoc = gate.Factory.newDocument(testDoc);
223 
224     // Verifies if the maximum annotation ID on the origDoc is less than the
225     // Annotation ID generator of the document.
226     verifyAnnotationIDGenerator(origDoc);
227 
228     // Load ANNIE with defaults and run it on the document
229     SerialAnalyserController annie = (SerialAnalyserController)
230       PersistenceManager.loadObjectFromFile(new File(new File(
231         Gate.getPluginsHome(), ANNIEConstants.PLUGIN_DIR),
232           ANNIEConstants.DEFAULT_FILE));
233     assertTrue("ANNIE not loaded!", annie != null);
234     Corpus c = Factory.newCorpus("test");
235     c.add(origDoc);
236     annie.setCorpus(c);
237     annie.execute();
238 
239     // SaveAS XML and reload the document into another GATE doc
240     // Export the Gate document called origDoc as XML, into a temp file,
241     // using the working encoding
242     File xmlFile = Files.writeTempFile(origDoc.toXml(),workingEncoding);
243     System.out.println("Saved to temp file :" + xmlFile.toURI().toURL());
244 
245     Document reloadedDoc = gate.Factory.newDocument(xmlFile.toURI().toURL(), workingEncoding);
246     // Verifies if the maximum annotation ID on the origDoc is less than the
247     // Annotation ID generator of the document.
248     verifyAnnotationIDGenerator(reloadedDoc);
249 
250     // Verify if the annotations are identical in the two docs.
251     Map origAnnotMap = buildID2AnnotMap(origDoc);
252     Map reloadedAnnMap = buildID2AnnotMap(reloadedDoc);
253 
254     //Verifies if the reloaded annotations are the same as the original ones
255     verifyIDConsistency(origAnnotMap, reloadedAnnMap);
256 
257     // Build the original Matches map
258     // ID  -> List of IDs
259     Map origMatchesMap = buildMatchesMap(origDoc);
260     // Verify the consistency of matches
261     // Compare every orig annotation pointed by the MatchesMap with the reloadedAnnot
262     // extracted from the reloadedMAp
263     for(Iterator it = origMatchesMap.keySet().iterator(); it.hasNext();){
264       Integer id = (Integer)it.next();
265       Annotation origAnnot = (Annotation) origAnnotMap.get(id);
266       assertTrue("Couldn't find an original annot with ID=" + id, origAnnot != null);
267       Annotation reloadedAnnot = (Annotation) reloadedAnnMap.get(id);
268       assertTrue("Couldn't find a reloaded annot with ID=" + id, reloadedAnnot != null);
269       compareAnnot(origAnnot,reloadedAnnot);
270       // Iterate through the matches list and repeat the comparison
271       List matchesList = (List) origMatchesMap.get(id);
272       for (Iterator itList = matchesList.iterator(); itList.hasNext();){

275         assertTrue("Couldn't find an original annot with ID=" + matchId, origA != null);
276         Annotation reloadedA = (Annotation) reloadedAnnMap.get(matchId);
277         assertTrue("Couldn't find a reloaded annot with ID=" + matchId, reloadedA != null);
278         compareAnnot(origA, reloadedA);
279       }// End for
280     }// End for

282     xmlFile.delete();

285   /**
286    * Builds a Map based on the matches feature of some annotations. The goal is to
287    * use this map to validate the annotations from the reloaded document.
288    * In case no Annot has the matches feat, will return an Empty MAP
289    * @param doc The document of which annotations will be used to construct the map
290    * @return A Map from Annot ID -> Lists of Annot IDs
291    */
292   private Map buildMatchesMap(Document doc){
293     Map matchesMap = new HashMap();
294     // Scan the default annotation set
295     AnnotationSet annotSet = doc.getAnnotations();
296 
297     helperBuildMatchesMap(annotSet, matchesMap);
298     // Scan all named annotation sets
299     if (doc.getNamedAnnotationSets() != null){
300       for ( Iterator namedAnnotSetsIter = doc.getNamedAnnotationSets().values().iterator();
301                                                                 namedAnnotSetsIter.hasNext(); ){
302         helperBuildMatchesMap((gate.AnnotationSet) namedAnnotSetsIter.next(), matchesMap);
303       }// End while
304     }// End if
305     return matchesMap;
306   }// End of buildMatchesMap()
307 
308   /**
309    * This is a helper metod. It scans an annotation set and adds the ID of the annotations
310    * which have the matches feature to the map.
311    * @param sourceAnnotSet  The annotation set investigated
312    * @param aMap
313    */
314   private void helperBuildMatchesMap(AnnotationSet sourceAnnotSet, Map aMap ){
315 
316     for (Iterator it = sourceAnnotSet.iterator(); it.hasNext();){
317       Annotation a = (Annotation) it.next();
318       FeatureMap aFeatMap = a.getFeatures();
319       // Skip those annotations who don't have features
320       if (aFeatMap == null) continue;
321       // Extract the matches feat
322       List matchesVal = (List) aFeatMap.get("matches");
323       if (matchesVal == null) continue;
324       Integer id = a.getId();
325       aMap.put(id,matchesVal);
326     }//End for
327 
328   }// End of helperBuildMatchesMap()

330   /**
331    * This method tests if the generator for new Annotation IDs is greather than the
332    * maximum Annotation ID present in the GATE document. In oter words, it ensures that
333    * new Annotations will receive an UNIQUE ID.
334    *
335    * @param aDoc The GATE document being tested
336    */
337   protected void verifyAnnotationIDGenerator(gate.Document aDoc){
338     // Creates a MAP containing all the annotations of the document.
339     // In doing so, it also tests if there are annotations with the same ID.
340     Map id2AnnotationMap = buildID2AnnotMap(aDoc);
341 
342     if (id2AnnotationMap == null || id2AnnotationMap.isEmpty()){
343       //System.out.println("No annotations found on the document! Nothing to test.");
344       return;
345     }
346 
347     // Get the key set of the Map and sort them
348     Set keysSet = id2AnnotationMap.keySet();
349     TreeSet sortedSet = new TreeSet(keysSet);
350     // Get the highest Annotation ID
351     Integer maxAnnotId =  (Integer) sortedSet.last();
352     // Compare its value to the one hold by the document's ID generator
353     Integer generatorId = ((DocumentImpl)aDoc).getNextAnnotationId();
354 
355 //    System.out.println("maxAnnotid = " + maxAnnotId + " generatorID = " + generatorId);
356 
357     assertTrue("Annotation ID generator["+generatorId+"] on document [" + aDoc.getSourceUrl() +
358             "] was equal or less than the MAX Annotation ID["+maxAnnotId+"] on the document."+
359             " This may lead to Annotation ID conflicts.", generatorId.intValue() > maxAnnotId.intValue());
360 
361 
362   }// End of verifyAnnotationIDGenerator()

442       }// End while
443     }// End if
444     return id2AnnMap;
445   }// End of buildID2AnnotMap()
