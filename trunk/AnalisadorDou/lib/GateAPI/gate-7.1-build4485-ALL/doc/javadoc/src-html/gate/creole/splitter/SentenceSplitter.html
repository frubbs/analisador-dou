070   public Resource init()throws ResourceInstantiationException{
071     //create all the componets
072     FeatureMap params;
073     FeatureMap features;
074 
075     params = Factory.newFeatureMap();
076     if(gazetteerListsURL != null)
077       params.put(DefaultGazetteer.DEF_GAZ_LISTS_URL_PARAMETER_NAME,
078               gazetteerListsURL);
079     params.put(DefaultGazetteer.DEF_GAZ_ENCODING_PARAMETER_NAME, encoding);
080 
081     if (gazetteer == null) {
082       //gazetteer
083       fireStatusChanged("Creating the gazetteer");
084       features = Factory.newFeatureMap();
085       Gate.setHiddenAttribute(features, true);
086 
087       gazetteer = (DefaultGazetteer)Factory.createResource(
088               "gate.creole.gazetteer.DefaultGazetteer",
089               params, features);
090       gazetteer.setName("Gazetteer " + System.currentTimeMillis());
091     }
092     else {
093       gazetteer.setParameterValues(params);
094       gazetteer.reInit();
095     }
096     
097     fireProgressChanged(10);
098 
099     params = Factory.newFeatureMap();
100     if(transducerURL != null)
101       params.put(Transducer.TRANSD_GRAMMAR_URL_PARAMETER_NAME, transducerURL);
102     params.put(Transducer.TRANSD_ENCODING_PARAMETER_NAME, encoding);
103 
104     if (transducer == null) {
105       //transducer
106       fireStatusChanged("Creating the JAPE transducer");
107       features = Factory.newFeatureMap();
108       Gate.setHiddenAttribute(features, true);
109 
110       transducer = (Transducer)Factory.createResource(
111               "gate.creole.Transducer",
112               params, features);
113       transducer.setName("Transducer " + System.currentTimeMillis());
114     }
115     else {
116       transducer.setParameterValues(params);
117       transducer.reInit();
118     }
119     
120     fireProgressChanged(100);
121     fireProcessFinished();
122 
123     return this;
124   }

131   public void execute() throws ExecutionException{
132     interrupted = false;
133     //set the runtime parameters
134     FeatureMap params;
135     if(inputASName != null && inputASName.equals("")) inputASName = null;
136     if(outputASName != null && outputASName.equals("")) outputASName = null;
137     try{
138       fireProgressChanged(0);
139       params = Factory.newFeatureMap();
140       params.put(DefaultGazetteer.DEF_GAZ_DOCUMENT_PARAMETER_NAME, document);
141       params.put(DefaultGazetteer.DEF_GAZ_ANNOT_SET_PARAMETER_NAME, inputASName);
142       gazetteer.setParameterValues(params);
143 
144       params = Factory.newFeatureMap();
145       params.put(Transducer.TRANSD_DOCUMENT_PARAMETER_NAME, document);
146       params.put(Transducer.TRANSD_INPUT_AS_PARAMETER_NAME, inputASName);
147       params.put(Transducer.TRANSD_OUTPUT_AS_PARAMETER_NAME, inputASName);
148       transducer.setParameterValues(params);
149     }catch(Exception e){
150       throw new ExecutionException(e);
151     }
152     ProgressListener pListener = null;
153     StatusListener sListener = null;
154     fireProgressChanged(5);
155 
156     //run the gazetteer
157     if(isInterrupted()) throw new ExecutionInterruptedException(
158         "The execution of the \"" + getName() +
159         "\" sentence splitter has been abruptly interrupted!");
160     pListener = new IntervalProgressListener(5, 10);
161     sListener = new StatusListener(){
162       public void statusChanged(String text){
163         fireStatusChanged(text);
164       }
165     };
166     gazetteer.addProgressListener(pListener);
167     gazetteer.addStatusListener(sListener);
168     gazetteer.execute();
169     gazetteer.removeProgressListener(pListener);
170     gazetteer.removeStatusListener(sListener);
171 
172     //run the transducer
173     if(isInterrupted()) throw new ExecutionInterruptedException(
174         "The execution of the \"" + getName() +
175         "\" sentence splitter has been abruptly interrupted!");
176     pListener = new IntervalProgressListener(11, 90);
177     transducer.addProgressListener(pListener);
178     transducer.addStatusListener(sListener);
179     Benchmark.executeWithBenchmarking(transducer,
180             Benchmark.createBenchmarkId("SentenceSplitterTransducer",
181                     getBenchmarkId()), this, null);
182     transducer.removeProgressListener(pListener);
183     transducer.removeStatusListener(sListener);
184 
185     //get pointers to the annotation sets
186     AnnotationSet inputAS = (inputASName == null) ?
187                             document.getAnnotations() :
188                             document.getAnnotations(inputASName);
189 
190     AnnotationSet outputAS = (outputASName == null) ?
191                              document.getAnnotations() :
192                              document.getAnnotations(outputASName);
193 
194     //copy the results to the output set if they are different
195     if(inputAS != outputAS){
196       outputAS.addAll(inputAS.get(SENTENCE_ANNOTATION_TYPE));
197     }
198 
199     //create one big sentence if none were found
200     AnnotationSet sentences = outputAS.get(SENTENCE_ANNOTATION_TYPE);
201     if(sentences == null || sentences.isEmpty()){
202       //create an annotation covering the entire content
203       try{
204         outputAS.add(new Long(0), document.getContent().size(), 
205                 SENTENCE_ANNOTATION_TYPE, Factory.newFeatureMap());
206       }catch(InvalidOffsetException ioe){
207         throw new GateRuntimeException(ioe);
208       }
209     }else{
210       //add a sentence covering all the tokens after the last sentence
211       Long endSentences = sentences.lastNode().getOffset();
212       AnnotationSet remainingTokens = inputAS.get(TOKEN_ANNOTATION_TYPE, endSentences,
213                                                   inputAS.lastNode().getOffset());
214       if(remainingTokens != null && !remainingTokens.isEmpty()){
215         try{
216           outputAS.add(remainingTokens.firstNode().getOffset(),
217                        remainingTokens.lastNode().getOffset(),
218                        SENTENCE_ANNOTATION_TYPE,
219                        Factory.newFeatureMap());
220         }catch(InvalidOffsetException ioe){
221           throw new ExecutionException(ioe);
222         }
223       }
224     }
225     fireProcessFinished();
226   }//execute()
