024 /**
025  * This class compiles a set of java sources using the user's preferred Java
026  * compiler. The default compiler used is the Eclipse JDT compiler, but this can
027  * be overridden by the user via an option in gate.xml.
028  */
029 public abstract class Javac implements GateConstants {

037   /**
038    * Compiles a set of java sources and loads the compiled classes in the gate
039    * class loader.
040    * 
041    * @param sources
042    *          a map from fully qualified classname to java source
043    * @param classLoader
044    *          the classloader into which the sources should be compiled and
045    *          loaded. Note that this classloader must also have access to all
046    *          the classes required to compile the sources.
047    * @throws GateException
048    *           in case of a compilation error or warning. In the case of
049    *           warnings the compiled classes are loaded before the error is
050    *           raised.
051    */
052   public static void loadClasses(Map<String, String> sources,
053       GateClassLoader classLoader) throws GateException {
054     
055     if(compiler == null) {
056       setCompilerTypeFromUserConfig();
057     }
058     
059     if (classLoader == null) {
060       Err.println("A null classloader was provided, using the Top-Level GATE classloader instead!");
061       classLoader = Gate.getClassLoader();
062     }
063 
064     compiler.compile(sources, classLoader);
065   }
066 
067   /**
068    * Sets the type of compiler to be used, based on the user's configuration.
069    * The default is to use the Eclipse compiler unless the user requests
070    * otherwise.
071    */
072   private static void setCompilerTypeFromUserConfig() throws GateException {
073     // see if the user has expressed a preference
074     String compilerType = Gate.getUserConfig().getString(COMPILER_TYPE_KEY);
075     // if not, use the default
076     if(compilerType == null) {
077       compilerType = DEFAULT_COMPILER;
078     }
079 
080     // We try and load the compiler class first by treating the given
081     // name as a fully qualified class name. If this fails, we prepend
082     // "gate.util.compilers." (so the user can say just "Sun" rather
083     // than "gate.util.compilers.Sun"). If that fails, we try the
084     // default value DEFAULT_COMPILER. If that fails, we give up.
085     try {
086       // first treat the compiler type as a fully qualified class name
087       compiler = createCompilerInstance(compilerType);
088     } catch(GateException ge) {
089       // if it's the default compiler we've just failed to load, give up
090       // now
091       if(DEFAULT_COMPILER.equals(compilerType)) { throw ge; }
092 
093       // we failed to find the class as a FQN, so try relative to
094       // gate.util.compilers
095       compilerType = "gate.util.compilers." + compilerType;
096       try {
097         compiler = createCompilerInstance(compilerType);
098       } catch(GateException ge2) {
099         // if it's the default compiler we've just failed to load, give
100         // up now
101         if(DEFAULT_COMPILER.equals(compilerType)) { throw ge2; }
102 
103         Err.prln("Unable to load compiler class " + compilerType
104             + ", falling back to default of " + DEFAULT_COMPILER);
105         compilerType = DEFAULT_COMPILER;
106         // last try - fall back on the default value. If this fails we
107         // just allow the failure exception to propagate up the stack
108         // from here.
109         compiler = createCompilerInstance(compilerType);
110       }
111     }
112   }
113 
114   private static Javac createCompilerInstance(String compilerType)
115       throws GateException {
116     Class compilerClass = null;
117     try {
118       // first treat the compiler type as a fully qualified class name
119       compilerClass = Gate.getClassLoader().loadClass(compilerType, true);
120     } catch(ClassNotFoundException cnfe) {
121       // ignore exception but leave compilerClass == null
122     }
123 
124     if(compilerClass == null || !Javac.class.isAssignableFrom(compilerClass)) { throw new GateException(
125         "Unable to load Java compiler class " + compilerType); }
126 
127     // At this point we have successfully loaded a compiler class.
128     // Now try and create an instance using a no-argument constructor.
129     try {
130       Constructor<?> noArgConstructor = compilerClass.getConstructor();
131       return (Javac)noArgConstructor.newInstance();
132     } catch(IllegalAccessException iae) {
133       throw new GateException("Cannot access Java compiler class "
134           + compilerType, iae);
135     } catch(InstantiationException ie) {
136       throw new GateException("Cannot instantiate Java compiler class "
137           + compilerType, ie);
138     } catch(NoSuchMethodException nsme) {
139       throw new GateException("Java compiler class " + compilerType
140           + " does not have a no-argument constructor", nsme);
141     } catch(InvocationTargetException ite) {
142       throw new GateException("Exception when constructing Java compiler "
143           + "of type " + compilerType, ite.getCause());
144     } catch(ExceptionInInitializerError eiie) {
145       throw new GateException("Exception when initializing Java compiler "
146           + "class " + compilerType, eiie.getCause());
147     }
148   }

150   /**
151    * Compile a set of Java sources, and load the resulting classes into the GATE
152    * class loader.
153    * 
154    * @param sources
155    *          a map from fully qualified classname to java source
156    * @param classLoader
157    *          the classloader into which the sources should be compiled. Note
158    *          that this classloader must also have access to all the classes
159    *          required to compile the sources.
160    * @throws GateException
161    *           in case of a compilation error or warning. In the case of
162    *           warnings, the compiled classes are loaded before the exception is
163    *           thrown.
164    */
165   public abstract void compile(Map<String, String> sources,
166       GateClassLoader classLoader) throws GateException;
167

169   public void compile(Map<String, String> sources) throws GateException {

171   }
