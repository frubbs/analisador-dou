017 package gate.corpora;
018 
019 import gate.DocumentContent;
020 import gate.util.BomStrippingInputStreamReader;
021 import gate.util.InvalidOffsetException;
022 
023 import java.io.BufferedReader;
024 import java.io.IOException;
025 import java.io.InputStream;
026 import java.net.URL;
027 
028 import org.apache.commons.io.IOUtils;
029 
030 /** Represents the commonalities between all sorts of document contents.
031   */
032 public class DocumentContentImpl implements DocumentContent
033 {
034   /** Debug flag */
035   private static final boolean DEBUG = false;
036 
037   /** Buffer size for reading
038    *  16k is 4 times the block size on most filesystems
039    *  so it should be efficient for most cases
040    *  */
041   private static final int INTERNAL_BUFFER_SIZE  = 16*1024;
042 
043   /** Default construction */
044   public DocumentContentImpl() {
045     content = new String();
046   } // default construction
047 
048   /** Contruction from URL and offsets. */
049   public DocumentContentImpl(URL u, String encoding, Long start, Long end)
050   throws IOException {
051 
052     int readLength = 0;
053     char[] readBuffer = new char[INTERNAL_BUFFER_SIZE];
054 
055     BufferedReader uReader = null;
056     InputStream uStream = null;
057     StringBuffer buf = new StringBuffer();
058 
059     long s = 0, e = Long.MAX_VALUE;
060     if(start != null && end != null) {
061       s = start.longValue();
062       e = end.longValue();
063     }
064 
065     try {
066       uStream = u.openStream();
067       
068       if(encoding != null && !encoding.equalsIgnoreCase("")) {
069         uReader = new BomStrippingInputStreamReader(uStream, encoding, INTERNAL_BUFFER_SIZE);
070       } else {
071         uReader = new BomStrippingInputStreamReader(uStream, INTERNAL_BUFFER_SIZE);
072       };
073   
074       // 1. skip S characters
075       uReader.skip(s);
076   
077       // 2. how many character shall I read?
078       long toRead = e - s;
079   
080       // 3. read gtom source into buffer
081       while (
082         toRead > 0 &&
083         (readLength = uReader.read(readBuffer, 0, INTERNAL_BUFFER_SIZE)) != -1
084       ) {
085         if (toRead <  readLength) {
086           //well, if toRead(long) is less than readLenght(int)
087           //then there can be no overflow, so the cast is safe
088           readLength = (int)toRead;
089         }
090   
091         buf.append(readBuffer, 0, readLength);
092         toRead -= readLength;
093       }
094     }
095     finally {
096       // 4.close reader
097       IOUtils.closeQuietly(uReader);
098       IOUtils.closeQuietly(uStream);
099     }
100 
101     content = new String(buf);
102     originalContent = content;
103   } // Contruction from URL and offsets
104 
105   /** Propagate changes to the document content. */
106   void edit(Long start, Long end, DocumentContent replacement)
107   {
108     int s = start.intValue(), e = end.intValue();
109     String repl = replacement == null ? "" :
110       ((DocumentContentImpl) replacement).content;
111     StringBuffer newContent = new StringBuffer(content);
112     newContent.replace(s, e, repl);
113     content = newContent.toString();
114   } // edit(start,end,replacement)
115 
116   public DocumentContent getContent(Long start, Long end)
117     throws InvalidOffsetException
118   {
119     if(! isValidOffsetRange(start, end))
120       throw new InvalidOffsetException("Invalid offset range "+start+" to "+end+
121               " for document content of length "+this.size());
122 
123     return new DocumentContentImpl(
124       content.substring(start.intValue(), end.intValue())
125     );
126   } // getContent(start, end)
127 
128   /** Returns the String representing the content in case of a textual document.
129     * NOTE: this is a temporary solution until we have a more generic one.
130     */
131   public String toString(){
132     return content;
133   }
134 
135   /** The size of this content (e.g. character length for textual
136     * content).
137     */
138   public Long size() {
139     return new Long(content.length());
140   } // size()
141 
142   /** Check that an offset is valid */
143   boolean isValidOffset(Long offset) {
144     if(offset == null)
145       return false;
146 
147     long o = offset.longValue();
148     long len = content.length();
149     if(o > len || o < 0)
150       return false;
151 
152     return true;
153   } // isValidOffset
154 
155   /** Check that both start and end are valid offsets and that
156     * they constitute a valid offset range
157     */
158   boolean isValidOffsetRange(Long start, Long end) {
159     return
160       isValidOffset(start) && isValidOffset(end) &&
161       start.longValue() <= end.longValue();
162   } // isValidOffsetRange(start,end)
163 
164   /** Two documents are the same if their contents is the same
165    */
166   public boolean equals(Object other) {
167     if (!(other instanceof DocumentContentImpl)) return false;
168 
169     DocumentContentImpl docImpl = (DocumentContentImpl) other;
170     return content.equals(docImpl.toString());
171   } // equals
172 
173   /** Calculate the hash value for the object. */
174   public int hashCode(){ return toString().hashCode(); }
175 
176   /** Just for now - later we have to cater for different types of
177     * content.
178     */
179   String content;
180 
181   /**
182    * For preserving the original content of the document.
183    * The edit command didn't affect on the original content.
184    * If you construct the content by URL the originalContent will keep
185    * whole information retrieved by URL even you set some start and end.
186    */
187   String originalContent;
188 
189   /**
190    * Return the original content of the document received during the loading
191    * phase or on construction from string.
192    */
193   public String getOriginalContent() { return originalContent; }
194 
195   /** For ranges */
196   public DocumentContentImpl(String s)
197     { content = s; originalContent = content; }

200   static final long serialVersionUID = -1426940535575467461L;
201 } // class DocumentContentImpl
