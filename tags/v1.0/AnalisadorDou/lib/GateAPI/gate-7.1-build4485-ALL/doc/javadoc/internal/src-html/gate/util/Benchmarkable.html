16 /**
17  * Resources that want to log their progress or results into a shared
18  * log centrally maintained by GATE, should implement this interface and
19  * use the java.util.Benchmark class to log their entries.
20  * 
21  * @author niraj
22  */
23 public interface Benchmarkable {
24 
25   /**
26    * Returns the benchmark ID of this resource.
27    */
28   public String getBenchmarkId();
29 
30   /**
31    * This method sets the benchmarkID for this resource. The resource
32    * must use this as the prefix for any sub-events it logs.
33    * 
34    * @param benchmarkId the benchmark ID, which must not contain spaces
35    *  as it is already used as a separator in the log, you can use
36    * {@link Benchmark#createBenchmarkId(String, String)} for it. 
37    */
38   public void setBenchmarkId(String benchmarkId);
39 
40 }
