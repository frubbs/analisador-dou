022 /**
023  * A composed tokeniser containing a {@link SimpleTokeniser} and a
024  * {@link gate.creole.Transducer}.
025  * The simple tokeniser tokenises the document and the transducer processes its
026  * output.
027  */

029 public class DefaultTokeniser extends AbstractLanguageAnalyser implements Benchmarkable {

050   /** Initialise this resource, and return it. */
051   public Resource init() throws ResourceInstantiationException{
052     try{
053       //init super object
054       super.init();
055       //create all the componets
056       FeatureMap params;
057       FeatureMap features;
058 
059       params = Factory.newFeatureMap();
060       if(tokeniserRulesURL != null)
061         params.put(SimpleTokeniser.SIMP_TOK_RULES_URL_PARAMETER_NAME,
062                    tokeniserRulesURL);
063       params.put(SimpleTokeniser.SIMP_TOK_ENCODING_PARAMETER_NAME, encoding);
064 
065       if (tokeniser == null) {
066         //tokeniser
067         fireStatusChanged("Creating a tokeniser");
068         if(DEBUG) Out.prln("Parameters for the tokeniser: \n" + params);
069         features = Factory.newFeatureMap();
070         Gate.setHiddenAttribute(features, true);
071         tokeniser = (SimpleTokeniser)Factory.createResource(
072                 "gate.creole.tokeniser.SimpleTokeniser",
073                 params, features);
074         tokeniser.setName("Tokeniser " + System.currentTimeMillis());
075       }
076       else {
077         tokeniser.setParameterValues(params);
078         tokeniser.reInit();
079       }
080       
081       fireProgressChanged(50);
082 
083       params = Factory.newFeatureMap();
084       if(transducerGrammarURL != null)
085         params.put(Transducer.TRANSD_GRAMMAR_URL_PARAMETER_NAME,
086                 transducerGrammarURL);
087       params.put(Transducer.TRANSD_ENCODING_PARAMETER_NAME, encoding);
088 
089       if (transducer == null) {
090         //transducer
091         fireStatusChanged("Creating a Jape transducer");
092         if(DEBUG) Out.prln("Parameters for the transducer: \n" + params);
093         features = Factory.newFeatureMap();
094         Gate.setHiddenAttribute(features, true);
095         transducer = (Transducer)Factory.createResource("gate.creole.Transducer",
096                 params, features);
097         transducer.setName("Transducer " + System.currentTimeMillis());
098       }
099       else {
100         transducer.setParameterValues(params);
101         transducer.reInit();
102       }
103       fireProgressChanged(100);
104       fireProcessFinished();
105       
106     }catch(ResourceInstantiationException rie){
107       throw rie;
108     }catch(Exception e){
109       throw new ResourceInstantiationException(e);
110     }
111     return this;
112   }
113   
114   public void cleanup() {
115     Factory.deleteResource(transducer);
116     Factory.deleteResource(tokeniser);
117   }
118 
119   public void execute() throws ExecutionException{
120     interrupted = false;
121     //set the parameters
122     try{
123       FeatureMap params = Factory.newFeatureMap();
124       fireProgressChanged(0);
125       //tokeniser
126       params.put(SimpleTokeniser.SIMP_TOK_DOCUMENT_PARAMETER_NAME, document);
127       params.put(
128         SimpleTokeniser.SIMP_TOK_ANNOT_SET_PARAMETER_NAME, annotationSetName);
129       tokeniser.setParameterValues(params);
130 
131       //transducer
132       params.clear();
133       params.put(Transducer.TRANSD_DOCUMENT_PARAMETER_NAME, document);
134       params.put(Transducer.TRANSD_INPUT_AS_PARAMETER_NAME, annotationSetName);
135       params.put(Transducer.TRANSD_OUTPUT_AS_PARAMETER_NAME, annotationSetName);
136       transducer.setParameterValues(params);
137     }catch(ResourceInstantiationException rie){
138       throw new ExecutionException(rie);
139     }
140 
141     ProgressListener pListener = null;
142     StatusListener sListener = null;
143     fireProgressChanged(5);
144     pListener = new IntervalProgressListener(5, 50);
145     sListener = new StatusListener(){
146       public void statusChanged(String text){
147         fireStatusChanged(text);
148       }
149     };
150 
151     //tokeniser
152     if(isInterrupted()) throw new ExecutionInterruptedException(
153         "The execution of the \"" + getName() +
154         "\" tokeniser has been abruptly interrupted!");
155     tokeniser.addProgressListener(pListener);
156     tokeniser.addStatusListener(sListener);
157     try{
158       Benchmark.executeWithBenchmarking(tokeniser,
159               Benchmark.createBenchmarkId("simpleTokeniser",
160                       getBenchmarkId()), this, null);
161     }catch(ExecutionInterruptedException eie){
162       throw new ExecutionInterruptedException(
163         "The execution of the \"" + getName() +
164         "\" tokeniser has been abruptly interrupted!");
165     }
166     tokeniser.removeProgressListener(pListener);
167     tokeniser.removeStatusListener(sListener);
168 
169   //transducer
170     if(isInterrupted()) throw new ExecutionInterruptedException(
171         "The execution of the \"" + getName() +
172         "\" tokeniser has been abruptly interrupted!");
173     pListener = new IntervalProgressListener(50, 100);
174     transducer.addProgressListener(pListener);
175     transducer.addStatusListener(sListener);
176 
177     Benchmark.executeWithBenchmarking(transducer,
178             Benchmark.createBenchmarkId("transducer",
179                     getBenchmarkId()), this, null);
180     transducer.removeProgressListener(pListener);
181     transducer.removeStatusListener(sListener);
182   }//execute
