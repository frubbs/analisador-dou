169    * @return set of the Lookups associated with the parameter
170    */
171   public String runMorpher(String word, String category) {
172     affix = null;
173     if(!validCategory(category)) {
174       return word;
175     }
176     
177     foundRule = false;
178     Set states = new HashSet();
179     states.add(initialState);
180     for (int i = 0; i < word.length(); i++) {
181       char ch = word.charAt(i);
182       states = getStates(ch, states);
183       if (states.isEmpty()) {
184         return word;
185       }
186 
187     }
188 
189     // we have all states here
190     // we obtain all RHSes
191     SortedSet<RHS> rhses = new TreeSet<RHS>(new Comparator<RHS>() {
192       public int compare(RHS r1, RHS r2) {
193         return r1.getPatternIndex() - r2.getPatternIndex();
194       }
195     });
196     
197     Iterator iter = states.iterator();
198     while (iter.hasNext()) {
199       FSMState st = (FSMState) iter.next();
200       rhses.addAll(st.getRHSes());
201     }
202 
203     if (rhses.isEmpty()) {
204       return word;
205     }
206 
207     return executeRHSes(rhses, word, category);
208   }

234   protected String executeRHS(String word, String category, RHS rhs) {
235     if (category.equals("*")) {
236       return executeRule(word, rhs);
237     } else if (rhs.isVerb() && vPat.matcher(category).matches()) {
238       return executeRule(word, rhs);
239     } else if (rhs.isNoun() && nPat.matcher(category).matches()) {
240       return executeRule(word, rhs);
241     }
242     return word;
243   }

245   private String executeRule(String word, RHS rhs) {
246     Pattern p = (Pattern) patterns.get(rhs.getPatternIndex());
247 
248     short methodIndex = rhs.getMethodIndex();
249     if (!p.matcher(word).matches()) {
250       foundRule = false;
251       return word;
252     }
253 
254     // call the appropriate function
255     String[] parameters = rhs.getParameters();
256 
257     // set the given word in that morph program
258     morphInst.setInput(word);
259     String answer = null;
260     switch (methodIndex) {
261     case ParsingFunctions.IRREG_STEM:
262       answer = morphInst.irreg_stem(parameters[0], parameters[1]);
263       break;
264     case ParsingFunctions.NULL_STEM:
265       answer = morphInst.null_stem();
266       break;
267     case ParsingFunctions.SEMIREG_STEM:
268       answer = morphInst.semi_reg_stem(Integer.parseInt(parameters[0]),
269           parameters[1]);
270       break;
271     case ParsingFunctions.STEM:
272       answer = morphInst.stem(Integer.parseInt(parameters[0]),
273           parameters[1], parameters[2]);
274       break;
275     default:
276       answer = null;
277       break;
278     }
279     
280     if(answer != null) {
281       this.affix = morphInst.getAffix();
282       foundRule = true;
283       return answer;
284     } else {
285       foundRule = false;
286       return word;
287     }
288   }

315   /**
316    * This method reads each line of the program and interpret them
317    */
318   private void interpretProgram() throws ResourceInstantiationException {
319     // read each line and parse it
320     while (file.hasNext()) {
321       String currentLine = file.getNext();
322 
323       if (currentLine == null || currentLine.trim().length() == 0) {
324         continue;
325       }
326 
327       // remove all the leading spaces
328       currentLine = currentLine.trim();
329 
330       /*
331        * if commandType is 0 ==> defineVars command if commandType is 1
332        * ==> defineRules command if commandType is 2 ==> variable
333        * declaration if commandType is 3 ==> rule declaration otherwise //
334        * unknown generate error
335        */
336       int commandType = findCommandType(currentLine);
337       switch (commandType) {
338       case -1:
339         // comment command
340         continue;
341       case 0:
342         // defineVars command
343         defineVarsCommand();
344         break;
345       case 1:
346         // defineRules command
347         defineRulesCommand();
348         break;
349       case 2:
350         // variable declaration
351         variableDeclarationCommand(currentLine);
352         break;
353       case 3:
354         // rule declaration
355         ruleDeclarationCommand(currentLine);
356         break;
357       default:
358         generateError("Syntax Error at line " + file.getPointer()
359             + " : " + currentLine);
360         break;
361       }
362     } // end while
363   }

365   /**
366    * This method interprets the line and finds out the type of command and
367    * returns the integer indicating the type of the command
368    * 
369    * @param line
370    *            The program command to be interpreted
371    * @return and <tt>int</tt> value
372    */
373   private int findCommandType(String line) {
374 
375     // check for the comment command
376     if (line.substring(0, 2).equals("//") || line.charAt(0) == '#') {
377       return -1;
378     } else if (line.equals("defineVars")) {
379       return 0;
380     } else if (line.equals("defineRules")) {
381       return 1;
382     } else if (isDefineVarSession && line.split("==>").length == 2) {
383       return 2;
384     } else if (isDefineRulesSession &&
385     /*
386      * (line.charAt(0) == '{' || line.charAt(0) == '[' || line.charAt(0) ==
387      * '(' || line.charAt(0) == '\"')
388      */(line.charAt(0) == '<') && line.split("==>").length == 2) {
389       return 3;
390     } else {
391       return Codes.ERROR_CODE;
392     }
393   }

425   /**
426    * This method processes the command to declare the variable
427    * 
428    * @param line
429    */
430   private void variableDeclarationCommand(String line)
431       throws ResourceInstantiationException {
432     // ok so first find the variable name and the value for it
433     String varName = (line.split("==>"))[0].trim();
434     String varValue = (line.split("==>"))[1].trim();
435 
436     // find the type of variable it is
437     int valueType = ParsingFunctions.findVariableType(varValue
438         .trim());
439     if (valueType == Codes.ERROR_CODE) {
440       generateError(varName + " - Variable Syntax Error - see " + "line"
441           + file.getPointer() + " : " + line);
442     }
443 
444     // based on the variable type create the instance
445     Variable varInst = null;
446     switch (valueType) {
447     case Codes.CHARACTER_RANGE_CODE:
448       varInst = new CharacterRange();
449       break;
450     case Codes.CHARACTER_SET_CODE:
451       varInst = new CharacterSet();
452       break;
453     case Codes.STRING_SET_CODE:
454       varInst = new StringSet();
455       break;
456     }
457 
458     // set the values in the variable
459     if (!varInst.set(varName, varValue)) {
460       generateError(varName
461           + " - Syntax Error while assigning value to the "
462           + "variable - see line" + file.getPointer() + " : " + line);
463     }
464 
465     // and finally add the variable in
466     if (!variables.add(varName, varInst.getPattern())) {
467       generateError(varName.trim() + " - Variable already defined - see "
468           + "line " + file.getPointer() + " : " + line);
469     }
470 
471     varInst.resetPointer();
472   }

474   /**
475    * This method processes the command to declare the rule
476    * 
477    * @param line
478    */
479   private void ruleDeclarationCommand(String line)
480       throws ResourceInstantiationException {
481     // lets divide the rule into two parts
482     // LHS and RHS.
483     // LHS is a part which requires to be parsed and
484     // RHS should be checked for the legal function name and valid arguments
485     // we process RHS first and then the LHS
486     String[] ruleParts = line.split("==>");
487     if (ruleParts.length != 2) {
488       generateError("Error in declaring rule at line : "
489           + file.getPointer() + " : " + line);
490     }
491 
492     // now check if the method which has been called in this rule actually
493     // available in the MorphFunction Class
494     String methodCalled = ruleParts[1].trim();
495     if (!isMethodAvailable(methodCalled)) {
496 
497       // no method is not available so print the syntax error
498       generateError("Syntax error - method does not exists - see "
499           + "line " + file.getPointer() + " : " + line);
500     }
501 
502     // so RHS part is Ok
503     // now we need to check if LHS is written properly
504     // and convert it to the pattern that is recognized by the java
505     String category = "";
506     // we need to find out the category
507     int i = 1;
508     for (; i < ruleParts[0].length(); i++) {
509       if (ruleParts[0].charAt(i) == '>')
510         break;
511       category = category + ruleParts[0].charAt(i);
512     }
513 
514     if (i >= ruleParts[0].length()) {
515       generateError("Syntax error - pattern not written properly - see "
516           + "line " + file.getPointer() + " : " + line);
517     }
518 
519     RHS rhs = new RHS(ruleParts[1], category, (short)patterns.size());
520     ruleParts[0] = ruleParts[0].substring(i + 1, ruleParts[0].length())
521         .trim();
522     String regExp = ParsingFunctions.convertToRegExp(
523         ruleParts[0], variables);
524     patterns.add(Pattern.compile(regExp));
525     String[] rules = ParsingFunctions.normlizePattern(regExp);
526     for (int m = 0; m < rules.length; m++) {
527       Set lss = new HashSet();
528       lss.clear();
529       Set newSet = new HashSet();
530       newSet.add(initialState);
531       lss.add(newSet);
532       PatternPart parts[] = ParsingFunctions
533           .getPatternParts(rules[m].trim());
534       for (int j = 0; j < parts.length; j++) {
535         lss = ParsingFunctions.createFSMs(parts[j].getPartString(), parts[j].getType(), lss, this);
536       }
537       Iterator iter = lss.iterator();
538       while (iter.hasNext()) {
539         Set set = (HashSet) iter.next();
540         Iterator subIter = set.iterator();
541         while (subIter.hasNext()) {
542           FSMState st = (FSMState) subIter.next();
543           st.addRHS(rhs);
544         }
545       }
546     }
547     //drawFSM();
548   }

569   private void drawFSM(FSMState st, String space) {
570     CharMap map = st.getTransitionFunction();
571     char[] keys = map.getItemsKeys();
572     if (keys != null) {
573       System.out.println(space + "Child:");

589   /**
590    * This method takes a method signature and searches if the method
591    * 
592    * @param method
593    * @return a <tt>boolean</tt> value.
594    */
595   private boolean isMethodAvailable(String method) {
596     // now first find the name of the method
597     // their parameters and their types
598     int index = method.indexOf("(");
599     if (index == -1 || index == 0
600         || method.charAt(method.length() - 1) != ')') {
601       return false;
602     }
603 
604     String methodName = method.substring(0, index);
605     // now get the parameters
606 
607     String[] parameters;
608     int[] userMethodParams;
609 
610     String arguments = method.substring(index + 1, method.length() - 1);
611     if (arguments == null || arguments.trim().length() == 0) {
612       parameters = null;
613       userMethodParams = null;
614     } else {
615       parameters = method.substring(index + 1, method.length() - 1)
616           .split(",");
617       userMethodParams = new int[parameters.length];
618     }
619 
620     // find the parameter types
621     // here we define only three types of arguments
622     // String, boolean and int
623     if (parameters != null) {
624       for (int i = 0; i < parameters.length; i++) {
625         if (parameters[i].startsWith("\"")
626             && parameters[i].endsWith("\"")) {
627           userMethodParams[i] = 7;
628           parameters[i] = "java.lang.String";
629           continue;
630         } else if (ParsingFunctions.isBoolean(parameters[i])) {
631           userMethodParams[i] = 6;
632           parameters[i] = "boolean";
633         } else if (ParsingFunctions.isInteger(parameters[i])) {
634           userMethodParams[i] = 2;
635           parameters[i] = "int";
636         } else {
637           // type cannot be recognized so generate error
638           return false;
639         }
640       }
641     }
642 
643     // now parameters have been found, so check them with the available
644     // methods
645     // in the morph function
646     Outer: for (int i = 0; i < methods.length; i++) {
647       if (methods[i].getName().equals(methodName)) {
648         // yes method has found now check for the parameters
649         // compatibility
650         Class[] methodParams = methods[i].getParameterTypes();
651         // first check for the number of parameters
652         if (methods[i].getName().equals("null_stem")) {
653           return true;
654         }
655         if (methodParams.length == parameters.length) {
656           // yes arity has matched
657           // now set the precedence
658           int[] paramPrecedence = new int[methodParams.length];
659 
660           // assign precedence
661           for (int j = 0; j < methodParams.length; j++) {
662             if (methodParams[j].getName()
663                 .equals("java.lang.String"))
664               paramPrecedence[j] = 7;
665             else if (methodParams[j].getName().equals("boolean"))
666               paramPrecedence[j] = 6;
667             else if (methodParams[j].getName().equals("int"))
668               paramPrecedence[j] = 2;
669             else
670               return false;
671           }
672 
673           // if we are here that means all the type matched
674           // so valid method declaration
675           return true;
676         }
677       }
678     }
679     // if we are here that means method doesnot found
680     return false;
681   }
