1064     private Set<Annotation> generateAntecedentCandidates(int sentenceNumber,
1065                                                         int quoteNumber ,
1066                                                         int mode) {
1067 
1068       //0. preconditions
1069       Assert.assertTrue(sentenceNumber >=0);
1070       Assert.assertTrue(quoteNumber >=0);
1071       Assert.assertTrue(mode == Quote.ANTEC_AFTER ||
1072                         mode == Quote.ANTEC_BEFORE ||
1073                         mode == Quote.ANTEC_BACK);
1074 
1075       //1. get sentence
1076      Sentence sentence = textSentences[sentenceNumber];
1077 
1078       //2. get the persons
1079       Set<Annotation> antecedents = new HashSet<Annotation>(sentence.getPersons());
1080 
1081       //4. now get the he/she pronouns in the relevant context
1082       AnnotationSet annotations = null;
1083 
1084       switch(mode) {
1085 
1086         case ANTEC_BEFORE:
1087           annotations = defaultAnnotations.getContained(sentence.getStartOffset(),
1088                                                       this.getStartOffset());
1089           break;
1090 
1091         case ANTEC_AFTER:
1092           annotations = defaultAnnotations.getContained(this.getEndOffset(),
1093                                                      sentence.getEndOffset());
1094           break;
1095 
1096         case ANTEC_BACK:
1097           annotations = defaultAnnotations.getContained(sentence.getStartOffset(),
1098                                                      sentence.getEndOffset());
1099           break;
1100       }
1101 
1102       //4. get the pronouns
1103       //restrict to he/she pronouns
1104       if (null != annotations) {
1105         AnnotationSet pronouns = annotations.get(TOKEN_ANNOTATION_TYPE,PRP_RESTRICTION);
1106 
1107         if (null != pronouns) {
1108 
1109           Iterator it = pronouns.iterator();
1110           while (it.hasNext()) {
1111             Annotation currPronoun = (Annotation)it.next();
1112             //add to succPersons only if HE/SHE
1113             String pronounString = (String)currPronoun.getFeatures().get(TOKEN_STRING_FEATURE_NAME);
1114 
1115             if (null != pronounString &&
1116                 (pronounString.equalsIgnoreCase("he") || pronounString.equalsIgnoreCase("she"))
1117                 )
1118               antecedents.add(currPronoun);
1119           }//while
1120         }//if
1121       }//if
1122 
1123 
1124       //3. depending on the mode, may have to restrict persons to these that precede/succeed
1125       //the quoted fragment
1126       //
1127       //for ANTEC_BEFORE, get the ones #preceding# the quote, contained in the sentence where
1128       //the quote *starts*
1129       //
1130       //for ANTEC_AFTER, get the ones #succeeding# the quote, contained in the sentence where
1131       //the quote *ends*
1132       //
1133       //for ANTEC_BACK, we are operating in the context of the sentence previous to the
1134       //sentence where the quote starts. I.e. we're resolbinf a case like
1135       // [sss "q1q1q1q1" s1s1s1s1]["q2q2q2q2"]
1136       //...and we want to get the entities from the s1s1 part - they *succeed* the #previous# quote
1137       //Note that the cirrent sentence is the first one, not the second
1138       //
1139       Iterator itPersons = antecedents.iterator();
1140 
1141       while (itPersons.hasNext()) {
1142         Annotation currPerson = (Annotation)itPersons.next();
1143 
1144         //cut
1145         if (Quote.ANTEC_BEFORE == mode &&
1146             currPerson.getStartNode().getOffset().intValue() > getStartOffset().intValue()) {
1147           //restrict only to persosn preceding
1148           itPersons.remove();
1149         }
1150         else if (Quote.ANTEC_AFTER == mode &&
1151                 currPerson.getStartNode().getOffset().intValue() < getEndOffset().intValue()) {
1152           //restrict only to persons succeeding the quote
1153           itPersons.remove();
1154         }

1156           //this one is tricky

1159           //wew try to find if any quote (prevQuote) exist in this sentence and get the
1160           //persons succeeding it)
1161 
1162           //get prev quote

1164           if (quoteNumber >0) {
