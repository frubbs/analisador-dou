122   /**
123    * Return the next numberOfHits -1 indicates all
124    */
125   public Hit[] next(int numberOfHits) throws SearchException {
126 
127     annicPatterns = new ArrayList<Pattern>();
128 
129     if(!success) {
130       this.annicPatterns = new ArrayList<Pattern>();
131       return getHits();
132     }
133 
134     if(fwdIterationEnded) {
135       this.annicPatterns = new ArrayList<Pattern>();
136       return getHits();
137     }
138 
139     try {
140       if(wasDeleteQuery) {
141         List<String> docIDs = new ArrayList<String>();
142         List<String> setNames = new ArrayList<String>();
143         for(int i = 0; i < luceneHits.length(); i++) {
144           Document luceneDoc = luceneHits.doc(i);
145           String documentID = luceneDoc.get(Constants.DOCUMENT_ID);
146           String annotationSetID = luceneDoc.get(Constants.ANNOTATION_SET_ID);
147           int index = docIDs.indexOf(documentID);
148           if(index == -1) {
149             docIDs.add(documentID);
150             setNames.add(annotationSetID);
151           } else {
152             if(!setNames.get(index).equals(annotationSetID)) {
153               docIDs.add(documentID);
154               setNames.add(annotationSetID);
155             }
156           }
157         }
158 
159         Hit[] toReturn = new Hit[docIDs.size()];
160         for(int i = 0; i < toReturn.length; i++) {
161           toReturn[i] = new Hit(docIDs.get(i), setNames.get(i), 0, 0, "");
162         }
163         return toReturn;
164       }
165 
166       for(; luceneSearchThreadIndex < luceneSearchThreads.size(); luceneSearchThreadIndex++) {
167         LuceneSearchThread lst =
168             luceneSearchThreads.get(luceneSearchThreadIndex);
169         List<Pattern> results = lst.next(numberOfHits);
170         if(results != null) {
171           if(numberOfHits != -1) {
172             numberOfHits -= results.size();
173           }
174 
175           this.annicPatterns.addAll(results);
176           if(numberOfHits == 0) { return getHits(); }
177         }
178       }
179 
180       // if we are here, there wer no sufficient patterns available
181       // so what we do is make success to false so that this method
182       // return null on next call
183       fwdIterationEnded = true;
184       return getHits();
185     } catch(Exception e) {
186       throw new SearchException(e);
187     }
188   }

190   /**
191    * Method retunrs true/false indicating whether results were found or not.
192    */
193   public boolean search(String query, Map<String, Object> parameters)
194       throws SearchException {
195     luceneHits = null;
196     annicPatterns = new ArrayList<Pattern>();
197     annotationTypesMap = new HashMap<String, List<String>>();
198     luceneSearchThreads = new ArrayList<LuceneSearchThread>();
199     luceneSearchThreadIndex = 0;
200     success = false;
201     fwdIterationEnded = false;
202     wasDeleteQuery = false;
203 
204     if(parameters == null)
205       throw new SearchException("Parameters cannot be null");
206 
207     this.parameters = parameters;
208 
209     /*
210      * lets first check if the query is to search the document names This is
211      * used when we only wants to search for documents stored under the specific
212      * corpus
213      */
214     if(parameters.size() == 2
215         && parameters.get(Constants.INDEX_LOCATION_URL) != null) {
216       String corpusID = (String)parameters.get(Constants.CORPUS_ID);
217       String indexLocation = null;
218       try {
219         indexLocation =
220             new File(
221                 ((URL)parameters.get(Constants.INDEX_LOCATION_URL)).toURI())
222                 .getAbsolutePath();
223       } catch(URISyntaxException use) {
224         indexLocation =
225             new File(
226                 ((URL)parameters.get(Constants.INDEX_LOCATION_URL)).getFile())
227                 .getAbsolutePath();
228       }
229 
230       if(corpusID != null && indexLocation != null) {
231         wasDeleteQuery = true;
232         Term term = new Term(Constants.CORPUS_ID, corpusID);
233         TermQuery tq = new TermQuery(term);
234         try {
235           gate.creole.annic.apache.lucene.search.Searcher searcher =
236               new IndexSearcher(indexLocation);
237           // and now execute the query
238           // result of which will be stored in hits
239           luceneHits = searcher.search(tq);
240           success = luceneHits.length() > 0 ? true : false;
241           return success;
242         } catch(IOException ioe) {
243           ioe.printStackTrace();
244           throw new SearchException(ioe);
245         }
246       }
247     }
248 
249     // check for index locations
250     if(parameters.get(Constants.INDEX_LOCATIONS) == null) {
251       String indexLocation;
252       try {
253         indexLocation =
254             new File(((URL)datastore.getIndexer().getParameters()
255                 .get(Constants.INDEX_LOCATION_URL)).toURI()).getAbsolutePath();
256 
257       } catch(URISyntaxException use) {
258         indexLocation =
259             new File(((URL)datastore.getIndexer().getParameters()
260                 .get(Constants.INDEX_LOCATION_URL)).getFile())
261                 .getAbsolutePath();
262       }
263       ArrayList<String> indexLocations = new ArrayList<String>();
264       indexLocations.add(indexLocation);
265       parameters.put(Constants.INDEX_LOCATIONS, indexLocations);
266     }
267 
268     indexLocations =
269         new ArrayList<String>(
270             (List<? extends String>)parameters.get(Constants.INDEX_LOCATIONS));
271 
272     if(indexLocations.size() == 0)
273       throw new SearchException("Corpus is not initialized");
274 
275     // check for valid context window
276     if(parameters.get(Constants.CONTEXT_WINDOW) == null)
277       throw new SearchException("Parameter " + Constants.CONTEXT_WINDOW
278           + " is not provided!");
279 
280     contextWindow =
281         ((Integer)parameters.get(Constants.CONTEXT_WINDOW)).intValue();
282 
283     if(getContextWindow().intValue() <= 0)
284       throw new SearchException("Context Window must be atleast 1 or > 1");
285 
286     if(query == null) throw new SearchException("Query is not initialized");
287 
288     this.query = query;
289     this.corpusToSearchIn = (String)parameters.get(Constants.CORPUS_ID);
290     this.annotationSetToSearchIn =
291         (String)parameters.get(Constants.ANNOTATION_SET_ID);
292 
293     annicPatterns = new ArrayList<Pattern>();
294     annotationTypesMap = new HashMap<String, List<String>>();
295 
296     luceneSearchThreads = new ArrayList<LuceneSearchThread>();
297 
298     // for different indexes, we create a different instance of
299     // indexSearcher
300     // TODO: is this really useful or used to have several indexLocations ?
301     for(int indexCounter = 0; indexCounter < indexLocations.size(); indexCounter++) {
302       String location = indexLocations.get(indexCounter);
303       // we create a separate Thread for each index
304       LuceneSearchThread lst = new LuceneSearchThread();
305       if(lst.search(query, contextWindow, location, corpusToSearchIn,
306           annotationSetToSearchIn, this)) {
307         luceneSearchThreads.add(lst);
308       }
309     }
310 
311     success = luceneSearchThreads.size() > 0 ? true : false;
312     return success;
313   }

353   /**
354    * This method returns a set of annotation set names that are indexed. Each
355    * entry has the following format:
356    * <p>
357    * corpusName;annotationSetName
358    * </p>
359    * where, the corpusName is the name of the corpus the annotationSetName
360    * belongs to.
361    */
362   public String[] getIndexedAnnotationSetNames() throws SearchException {
363     String indexLocation;
364     try {
365       indexLocation =
366           new File(((URL)datastore.getIndexer().getParameters()
367               .get(Constants.INDEX_LOCATION_URL)).toURI()).getAbsolutePath();
368 
369     } catch(URISyntaxException use) {
370       indexLocation =
371           new File(((URL)datastore.getIndexer().getParameters()
372               .get(Constants.INDEX_LOCATION_URL)).getFile()).getAbsolutePath();
373     }
374 
375     annotationTypesMap = new HashMap<String, List<String>>();
376     Set<String> toReturn = new HashSet<String>();
377     try {
378       IndexReader reader = IndexReader.open(indexLocation);
379       try {
380 
381         // lets first obtain stored corpora
382         TermEnum terms =
383             reader.terms(new Term(Constants.ANNOTATION_SET_ID, ""));
384         if(terms == null) { return new String[0]; }
385 
386         // iterating over terms and finding out names of annotation sets indexed
387         Set<String> annotSets = new HashSet<String>();
388         boolean foundAnnotSet = false;
389         do {
390           Term t = terms.term();
391           if(t == null) continue;
392 
393           if(t.field().equals(Constants.ANNOTATION_SET_ID)) {
394             annotSets.add(t.text());
395             foundAnnotSet = true;
396           } else {
397             if(foundAnnotSet) break;
398           }
399         } while(terms.next());
400 
401         
402         // we query for each annotation set
403         // and go through docs with that annotation set in them
404         // to see which corpus they belong to.
405         // we could have done the other way round as well (i.e. 
406         // first asking for corpus ids and then obtainin annotation set ids
407         // but not all documents belong to corpora
408         for(String annotSet : annotSets) {
409           Term term = new Term(Constants.ANNOTATION_SET_ID, annotSet);
410           TermQuery tq = new TermQuery(term);
411           try {
412             gate.creole.annic.apache.lucene.search.Searcher searcher =
413                 new IndexSearcher(indexLocation);
414             try {
415               Hits annotSetHits = searcher.search(tq);
416               for(int i = 0; i < annotSetHits.length(); i++) {
417                 Document luceneDoc = annotSetHits.doc(i);
418                 String corpusID = luceneDoc.get(Constants.CORPUS_ID);
419                 if(corpusID == null) corpusID = "";
420                 toReturn.add(corpusID + ";" + annotSet);
421 
422                 // lets create a boolean query
423                 Term annotSetTerm =
424                     new Term(Constants.ANNOTATION_SET_ID, annotSet);
425                 TermQuery atq = new TermQuery(annotSetTerm);
426 
427                 BooleanQuery bq = new BooleanQuery();
428                 bq.add(tq, true, false);
429                 bq.add(atq, true, false);
430                 gate.creole.annic.apache.lucene.search.Searcher indexFeatureSearcher =
431                     new IndexSearcher(indexLocation);
432                 try {
433                   Hits indexFeaturesHits = searcher.search(bq);
434                   for(int j = 0; j < indexFeaturesHits.length(); j++) {
435                     Document aDoc = indexFeaturesHits.doc(j);
436                     String indexedFeatures =
437                         aDoc.get(Constants.INDEXED_FEATURES);
438                     if(indexedFeatures != null) {
439                       String[] features = indexedFeatures.split(";");
440                       for(String aFeature : features) {
441                         // AnnotationType.FeatureName
442                         int index = aFeature.indexOf(".");
443                         if(index == -1) {
444                           continue;
445                         }
446                         String type = aFeature.substring(0, index);
447                         String featureName = aFeature.substring(index + 1);
448                         String key = corpusID + ";" + annotSet + ";" + type;
449                         List<String> listOfFeatures =
450                             annotationTypesMap.get(key);
451                         if(listOfFeatures == null) {
452                           listOfFeatures = new ArrayList<String>();
453                           annotationTypesMap.put(key, listOfFeatures);
454                         }
455                         if(!listOfFeatures.contains(featureName)) {
456                           listOfFeatures.add(featureName);
457                         }
458                       }
459                     }
460                   }
461                 } finally {
462                   indexFeatureSearcher.close();
463                 }
464               }
465             } finally {
466               searcher.close();
467             }
468           } catch(IOException ioe) {
469             ioe.printStackTrace();
470             throw new SearchException(ioe);
471           }
472         }
473       } finally {
474         reader.close();
475       }
476     } catch(IOException ioe) {
477       throw new SearchException(ioe);
478     }
479     return toReturn.toArray(new String[0]);
480   }
