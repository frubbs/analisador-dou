120   /**
121    * This method runs the coreferencer. It assumes that all the needed parameters
122    * are set. If they are not, an exception will be fired.
123    *
124    * The process goes like this:
125    * - Create a sorted list of Person and JobTitle annotations.
126    * - Loop through the annotations
127    *    If it is a Person, we add it to the top of a stack.
128    *    If it is a job title, we subject it to a series of tests. If it 
129    *      passes, we associate it with the Person annotation at the top
130    *      of the stack
131    */
132   public void execute() throws ExecutionException{
133 
134     Annotation[] nominalArray;
135 
136     //0. preconditions
137     if (null == this.document) {
138       throw new ExecutionException("[coreference] Document is not set!");
139     }
140 
141     //1. preprocess
142     preprocess();
143 
144     // Out.println("Total annotations: " + defaultAnnotations.size());
145 
146     // Get a sorted array of Tokens.
147     // The tests for job titles often require getting previous and subsequent
148     // tokens, so to save work, we create a single, sorted list of 
149     // tokens.
150     Annotation[] tokens = defaultAnnotations.get(TOKEN_ANNOTATION_TYPE).
151         toArray(new Annotation[0]);
152     java.util.Arrays.sort(tokens, new OffsetComparator());
153 
154     // The current token is the token at the start of the current annotation.
155     int currentToken = 0;
156 
157     // get Person entities
158     //FeatureMap personConstraint = new SimpleFeatureMapImpl();
159     //personConstraint.put(LOOKUP_MAJOR_TYPE_FEATURE_NAME,
160     //                          PERSON_CATEGORY);
161     HashSet personConstraint = new HashSet();
162     personConstraint.add(PERSON_CATEGORY);
163     AnnotationSet people =
164       this.defaultAnnotations.get(personConstraint);
165 
166     // get all JobTitle entities
167     //FeatureMap constraintJobTitle = new SimpleFeatureMapImpl();
168     //constraintJobTitle.put(LOOKUP_MAJOR_TYPE_FEATURE_NAME, JOBTITLE_CATEGORY);
169     HashSet jobTitleConstraint = new HashSet();
170     jobTitleConstraint.add(JOBTITLE_CATEGORY);
171     
172     AnnotationSet jobTitles = 
173       this.defaultAnnotations.get(jobTitleConstraint);
174 
175     FeatureMap orgNounConstraint = new SimpleFeatureMapImpl();
176     orgNounConstraint.put(LOOKUP_MAJOR_TYPE_FEATURE_NAME,
177                           ORGANIZATION_NOUN_CATEGORY);
178     AnnotationSet orgNouns =
179       this.defaultAnnotations.get(LOOKUP_CATEGORY, orgNounConstraint);
180 
181     HashSet orgConstraint = new HashSet();
182     orgConstraint.add(ORGANIZATION_CATEGORY);
183 
184     AnnotationSet organizations =
185       this.defaultAnnotations.get(orgConstraint);
186 
187     // combine them into a list of nominals
188     Set<Annotation> nominals = new HashSet();
189     if (people != null) {
190       nominals.addAll(people);
191     }
192     if (jobTitles != null) {
193       nominals.addAll(jobTitles);
194     }
195     if (orgNouns != null) {
196       nominals.addAll(orgNouns);
197     }
198     if (organizations != null) {
199       nominals.addAll(organizations);
200     }
201 
202     //  Out.println("total nominals: " + nominals.size());
203 
204     // sort them according to offset
205     nominalArray = nominals.toArray(new Annotation[0]);
206     java.util.Arrays.sort(nominalArray, new OffsetComparator());
207     
208     ArrayList<Annotation> previousPeople = new ArrayList<Annotation>();
209     ArrayList<Annotation> previousOrgs = new ArrayList<Annotation>();
210     
211         
212     // process all nominals
213     for (int i=0; i<nominalArray.length; i++) {
214       Annotation nominal = (Annotation)nominalArray[i];
215       
216       // Find the current place in the tokens array
217       currentToken = advanceTokenPosition(nominal, currentToken, tokens);
218       
219       //Out.print("processing nominal [" + stringValue(nominal) + "] ");
220       
221       if (nominal.getType().equals(PERSON_CATEGORY)) {
222   // Add each Person entity to the beginning of the people list
223   // but don't add pronouns
224   Object[] personTokens = getSortedTokens(nominal);
225     
226   if (personTokens.length == 1) {
227     Annotation personToken = (Annotation) personTokens[0];
228     
229     String personCategory = (String) 
230       personToken.getFeatures().get(TOKEN_CATEGORY_FEATURE_NAME);
231     if (personCategory.equals("PP") ||
232         personCategory.equals("PRP") ||
233         personCategory.equals("PRP$") ||
234         personCategory.equals("PRPR$")) {
235         //Out.println("ignoring personal pronoun");
236         continue;
237     }
238   }
239   
240   previousPeople.add(0, nominal);
241   //Out.println("added person");
242       }
243       else if (nominal.getType().equals(JOBTITLE_CATEGORY)) {
244     
245   // Look into the tokens to get some info about POS.
246   Object[] jobTitleTokens = getSortedTokens(nominal);
247   
248   Annotation lastToken = (Annotation)
249     jobTitleTokens[jobTitleTokens.length - 1];
250   
251   // Don't associate if the job title is not a singular noun
252   String tokenCategory = (String) 
253     lastToken.getFeatures().get(TOKEN_CATEGORY_FEATURE_NAME);
254   // UNCOMMENT FOR SINGULAR PROPER NOUNS (The President, the Pope)
255   //if (! tokenCategory.equals("NN") &&
256   //! tokenCategory.equals("NNP")) {
257   if (! tokenCategory.equals("NN")) {
258       // Out.println("Not a singular noun");
259     continue;
260   }
261   
262   // Don't associate it if it's part of a Person (eg President Bush)
263   if (overlapsAnnotations(nominal, people)) {
264       //Out.println("overlapping annotation");
265     continue;
266   }
267 
268   Annotation previousToken;
269         String previousValue;
270 
271   // Don't associate it if it's proceeded by a generic marker
272         if (currentToken != 0) {
273           previousToken = (Annotation) tokens[currentToken - 1];
274           previousValue = (String) 
275       previousToken.getFeatures().get(TOKEN_STRING_FEATURE_NAME);
276           if (previousValue.equalsIgnoreCase("a") ||
277               previousValue.equalsIgnoreCase("an") ||
278               previousValue.equalsIgnoreCase("other") ||
279               previousValue.equalsIgnoreCase("another")) {
280               //Out.println("indefinite");
281       continue;
282           }
283         }
284 
285   // nominals immediately followed by Person annotations:
286   // BAD:
287   //   Chairman Bill Gates               (title)
288   // GOOD:
289   //   secretary of state, Colin Powell  (inverted appositive)
290   //   the home secretary David Blunkett (same but no comma, 
291   //                                      possible in transcriptions)
292   // "the" is a good indicator for apposition
293   
294   // Luckily we have an array of all Person annotations in order...
295   if (i < nominalArray.length - 1) {
296     Annotation nextAnnotation = (Annotation) nominalArray[i+1];
297     if (nextAnnotation.getType().equals(PERSON_CATEGORY)) {
298       // is it preceded by a definite article?
299       previousToken = (Annotation) tokens[currentToken - 1];
300       previousValue = (String) 
301         previousToken.getFeatures().get(TOKEN_STRING_FEATURE_NAME);
302       
303       // Get all tokens between this and the next person
304       int interveningTokens =
305         countInterveningTokens(nominal, nextAnnotation,
306              currentToken, tokens);
307       if (interveningTokens == 0 && 
308         ! previousValue.equalsIgnoreCase("the")) {
309       
310         // There is nothing between the job title and the person,
311         // like "Chairman Gates" -- do nothing.
312         //Out.println("immediately followed by Person");
313         continue;
314       }
315       else if (interveningTokens == 1) {
316         String tokenString =
317           (String) getFollowingToken(nominal,
318              currentToken, tokens)
319       .getFeatures().get(TOKEN_STRING_FEATURE_NAME);
320         //Out.print("STRING VALUE [" + tokenString + "] ");
321         if (! tokenString.equals(",") &&
322     ! tokenString.equals("-")) {
323     //Out.println("nominal and person separated by NOT [,-]");
324     continue;
325         }
326       }
327       
328       // Did we get through all that? Then we must have an 
329       // apposition.
330       
331       anaphor2antecedent.put(nominal, nextAnnotation);
332       //Out.println("associating with " +
333       //  stringValue(nextAnnotation));
334       continue;
335       
336     }
337   }
338   
339   // If we have no possible antecedents, create a new Person
340   // annotation.
341   if (previousPeople.size() == 0) {
342     FeatureMap personFeatures = new SimpleFeatureMapImpl();
343     personFeatures.put("ENTITY_MENTION_TYPE", "NOMINAL");
344     this.defaultAnnotations.add(nominal.getStartNode(),
345               nominal.getEndNode(),
346               PERSON_CATEGORY,
347               personFeatures);
348     //Out.println("creating as new Person");
349     continue;
350   }
351 
352   // Associate this entity with the most recent Person
353   int personIndex = 0;
354   
355   Annotation previousPerson =
356     (Annotation) previousPeople.get(personIndex);
357   
358   // Don't associate if the two nominals are not the same gender
359   String personGender = (String) 
360     previousPerson.getFeatures().get(PERSON_GENDER_FEATURE_NAME);
361   String jobTitleGender = (String) 
362           nominal.getFeatures().get(PERSON_GENDER_FEATURE_NAME);
363   if (personGender != null && jobTitleGender != null) {
364           if (! personGender.equals(jobTitleGender)) {
365             //Out.println("wrong gender: " + personGender + " " +
366             //            jobTitleGender);
367       continue;
368     }
369   }
370   
371   //Out.println("associating with " +
372   //  previousPerson.getFeatures()
373   //  .get(TOKEN_STRING_FEATURE_NAME));
374   
375   anaphor2antecedent.put(nominal, previousPerson);
376       }
377       else if (nominal.getType().equals(ORGANIZATION_CATEGORY)) {
378         // Add each organization entity to the beginning of
379   // the organization list
380   previousOrgs.add(0, nominal);
381   //Out.println("added organization");
382       }
383       else if (nominal.getType().equals(LOOKUP_CATEGORY)) {
384   // Don't associate it if we have no organizations
385   if (previousOrgs.size() == 0) {
386     //Out.println("no orgs");
387     continue;
388   }
389     
390   // Look into the tokens to get some info about POS.
391   Annotation[] orgNounTokens =
392     this.defaultAnnotations.get(TOKEN_ANNOTATION_TYPE,
393               nominal.getStartNode().getOffset(),
394               nominal.getEndNode().getOffset()).toArray(new Annotation[0]);
395   java.util.Arrays.sort(orgNounTokens, new OffsetComparator());
396   Annotation lastToken = (Annotation)
397     orgNounTokens[orgNounTokens.length - 1];
398   
399   // Don't associate if the org noun is not a singular noun
400   if (! lastToken.getFeatures().get(TOKEN_CATEGORY_FEATURE_NAME)
401       .equals("NN")) {
402       //Out.println("Not a singular noun");
403       continue;
404   }
405   
406   //Out.println("organization noun");
407   // Associate this entity with the most recent Person
408   anaphor2antecedent.put(nominal, previousOrgs.get(0));
409       }
410     }
411 
412     // This method does the dirty work of actually adding new annotations and
413     // coreferring.
414     generateCorefChains(anaphor2antecedent);
415   }

417   /**
418    * This method specifies whether a given annotation overlaps any of a 
419    * set of annotations. For instance, JobTitles occasionally are
420    * part of Person annotations.
421    * 
422    */
423   private boolean overlapsAnnotations(Annotation a,
424                                       AnnotationSet annotations) {
425     Iterator<Annotation> iter = annotations.iterator();
426     while (iter.hasNext()) {
427       Annotation current = iter.next();
428       if (a.overlaps(current)) {
429         return true;
430       }
431     }
432       
433     return false;
434   }

436   /** Use this method to keep the current token pointer at the right point
437    * in the token list */
438   private int advanceTokenPosition(Annotation target, int currentPosition,
439            Object[] tokens) {
440     long targetOffset = target.getStartNode().getOffset().longValue();
441     long currentOffset = ((Annotation) tokens[currentPosition])
442       .getStartNode().getOffset().longValue();
443     
444     if (targetOffset > currentOffset) {
445       while (targetOffset > currentOffset) {
446   currentPosition++;
447   currentOffset = ((Annotation) tokens[currentPosition])
448           .getStartNode().getOffset().longValue();
449       }
450     }
451     else if (targetOffset < currentOffset) {
452       while (targetOffset < currentOffset) {
453   currentPosition--;
454   currentOffset = ((Annotation) tokens[currentPosition])
455           .getStartNode().getOffset().longValue();
456       }
457     }
458     
459     return currentPosition;
460   }
461 
462   /** Return the number of tokens between the end of annotation 1 and the
463    * beginning of annotation 2. Will return 0 if they are not in order */
464   private int countInterveningTokens(Annotation first, Annotation second,
465              int currentPosition, Object[] tokens) {
466     int interveningTokens = 0;
467 
468     long startOffset = first.getEndNode().getOffset().longValue();
469     long endOffset = second.getStartNode().getOffset().longValue();
470     
471     long currentOffset = ((Annotation) tokens[currentPosition])
472       .getStartNode().getOffset().longValue();
473     
474     while (currentOffset < endOffset) {
475       if (currentOffset >= startOffset) {
476         interveningTokens++;
477       }
478       currentPosition++;
479       currentOffset = ((Annotation) tokens[currentPosition])
480   .getStartNode().getOffset().longValue();
481     }
482     return interveningTokens;
483   }

485   /** Get the next token after an annotation */
486   private Annotation getFollowingToken(Annotation current, int currentPosition,
487                Object[] tokens) {
488     long endOffset = current.getEndNode().getOffset().longValue();
489     long currentOffset = ((Annotation) tokens[currentPosition])
490       .getStartNode().getOffset().longValue();
491     while (currentOffset < endOffset) {
492       currentPosition++;
493       currentOffset = ((Annotation) tokens[currentPosition])
494   .getStartNode().getOffset().longValue();
495     }
496     return (Annotation) tokens[currentPosition];
497   }

499   /** Get the text of an annotation */
500   private String stringValue(Annotation ann) {
501     Object[] tokens = getSortedTokens(ann);
502   
503     StringBuffer output = new StringBuffer();
504     for (int i=0;i<tokens.length;i++) {
505       Annotation token = (Annotation) tokens[i];
506       output.append(token.getFeatures().get(TOKEN_STRING_FEATURE_NAME));
507       if (i < tokens.length - 1) {
508         output.append(" ");
509       }
510     }
511     return output.toString();
512   }

514   /** Get a sorted array of the tokens that make up a given annotation. */
515   private Annotation[] getSortedTokens(Annotation a) {
516     Annotation[] annotationTokens =
517       this.defaultAnnotations.get(TOKEN_ANNOTATION_TYPE,
518           a.getStartNode().getOffset(),
519           a.getEndNode().getOffset()).toArray(new Annotation[0]);
520     java.util.Arrays.sort(annotationTokens, new OffsetComparator());
521     return annotationTokens;
522   }
