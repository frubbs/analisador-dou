034 /**
035  * Utility class for managing a set of GATE relations (usually each annotation

037  */

053    * associated with an {@link AnnotationSet} used the annotation set name as a

074   /**
075    * Indexes for relations by member. Each element in the list refers to a given

080    * (representing a relation member) to a {@link BitSet} indicating which of
081    * the relation indexes (in {@link #relations}) correspond to relations that 
082    * contain the given annotation (i.e. member) on the position 
083    * <code>pos</code>. 
084    */
085   protected List<Map<Integer, BitSet>> indexesByMember;

087   /**
088    * Factory method that gets a named {@link RelationSet} from a document. 
089    * {@link RelationSet}s are stored inside a special document feature, which is
090    * where this method retrieves them from. If no relation set is associated
091    * with the given name, then a new one is created and returned.
092    * 
093    * To get the relation set associated with a given annotation set use 
094    * {@link #getRelations(AnnotationSet)} instead.
095    * 
096    * @param document the document for which the relations are being requested.
097    * @param name the name for the relation set. This can be any arbitrary 
098    * {@link String} or even <code>null</code>. However, by convention, names of 
099    * annotation sets (and <code>null</code> for the default annotation set) are
100    * used to name the <i>principal</i> relation set for each annotation set.   
101    * @return the {@link RelationSet} requested. 
102    */
103   public static RelationSet getRelations(Document document, String name) {
104     Object relMapObj = document.getFeatures().get(
105       RELATIONS_MAP_DOCUMENT_FEATURE_NAME);
106     Map<String, RelationSet> relationsMap = null;
107     if(relMapObj != null) {
108       if(relMapObj instanceof Map) {
109         relationsMap = (Map)relMapObj;
110       } else {
111         document.getFeatures().remove(RELATIONS_MAP_DOCUMENT_FEATURE_NAME);
112         log.warn("Invalid value for feature \"" + 
113             RELATIONS_MAP_DOCUMENT_FEATURE_NAME + "\" on document \"" + 
114             document.getName() + "\" has been removed.");
115       }      
116     }
117     if(relationsMap == null) {
118       relationsMap = new HashMap<String, RelationSet>();
119       document.getFeatures().put(RELATIONS_MAP_DOCUMENT_FEATURE_NAME, 
120         relationsMap);
121     }
122     RelationSet relSet = relationsMap.get(name);
123     if(relSet == null) {
124       relSet = new RelationSet();
125       relationsMap.put(name, relSet);
126     }
127     return relSet;
128   }

131    * Factory method that gets the {@link RelationSet} associated with an 
132    * {@link AnnotationSet}.
133    * @param annSet the annotation set for which the relations are being

136    * and returned.
137    */
138   public static RelationSet getRelations(AnnotationSet annSet) {
139     return getRelations(annSet.getDocument(), annSet.getName());  
140   }

142   protected RelationSet() {
143     relations = new ArrayList<Relation>();
144     indexByType = new HashMap<String, BitSet>();
145     indexesByMember = new ArrayList<Map<Integer,BitSet>>();
146     deleted = new BitSet();

149   /**
150    * Creates a new {@link Relation} and adds it to this set. Uses the default 
151    * relation implementation at {@link SimpleRelation}.
152    * @param type the type for the new relation.
153    * @param members the annotation IDs for the annotations that are members in 
154    * this relation.
155    * @return the newly created {@link Relation} instance.
156    */
157   public Relation addRelation(String type, int... members) {
158     Relation rel = new SimpleRelation(type, members);
159     addRelation(rel);
160     return rel;
161   }

164   /**
165    * Adds an externally-created {@link Relation} instance.
166    * @param rel the {@link Relation} to be added.
167    */
168   public void addRelation(Relation rel) {
169     int relIdx = relations.size();
170     relations.add(rel);
171     BitSet sameType = indexByType.get(rel.getType());
172     if(sameType == null) {
173       sameType = new BitSet(relations.size());
174       indexByType.put(rel.getType(), sameType);
175     }
176     sameType.set(relIdx);
177 
178     // widen the index by member list, if needed
179     for(int i = indexesByMember.size(); i < rel.getMembers().length; i++) {
180       indexesByMember.add(new HashMap<Integer, BitSet>());
181     }
182     for(int memeberPos = 0; memeberPos < rel.getMembers().length; memeberPos++) {
183       int member = rel.getMembers()[memeberPos];
184       Map<Integer, BitSet> indexByMember = indexesByMember.get(memeberPos);
185       BitSet sameMember = indexByMember.get(member);
186       if(sameMember == null) {
187         sameMember = new BitSet(relations.size());
188         indexByMember.put(member, sameMember);
189       }
190       sameMember.set(relIdx);
191     }
192   }

202   /**
203    * Finds relations based on their type.
204    * @param type the type of relation being sought.
205    * @return the list of all relations in this {@link RelationSet} that have the
206    * required type.
207    */
208   public List<Relation> getRelations(String type) {
209     List<Relation> res = new ArrayList<Relation>();
210     BitSet rels = indexByType.get(type);
211     if(rels != null) {
212       rels.andNot(deleted);
213       for(int relPos = 0; relPos < relations.size(); relPos++){
214         if(rels.get(relPos)) res.add(relations.get(relPos));  
215       }
216     }
217     return res;
218   }

220   /**
221    * Finds relations based on their members.
222    * @param members an array containing annotation IDs. If a constraint is not
223    * required for a given member position, then the {@link #ANY}. value should 
224    * be used.
225    * @return all the relations that have the given annotation IDs (members)
226    * on the specified positions.
227    */
228   public List<Relation> getRelations(int... members) {
229     // get the lists of relations for each member
230     BitSet[] postingLists = new BitSet[indexesByMember.size()];
231     for(int  i = 0; i < postingLists.length; i++) {
232       if(i < members.length && members[i] >= 0) {
233         postingLists[i] = indexesByMember.get(i).get(members[i]);
234       } else {
235         postingLists[i] = null;
236       }
237     }
238     return intersection(postingLists);
239   }

241   public List<Relation> getRelations(String type, int... members) {
242     // get the lists of relations for each member
243     BitSet[] postingLists = new BitSet[indexesByMember.size() + 1];
244     for(int  i = 0; i < postingLists.length; i++) {
245       if(i < members.length && members[i] >= 0) {
246         postingLists[i] = indexesByMember.get(i).get(members[i]);
247       } else {
248         postingLists[i] = null;
249       }
250     }
251     postingLists[postingLists.length - 1] = indexByType.get(type);
252     return intersection(postingLists);
253   }

255   /**
256    * Deletes the specified relation.
257    * @param relation the relation to be deleted.
258    * @return <code>true</code> if the given relation was deleted, or 
259    * <code>false</code> if it was not found.
260    */
261   public boolean deleteRelation(Relation relation) {
262     int relIdx = relations.indexOf(relation);
263     if(relIdx >= 0){
264       deleted.set(relIdx);
265       relations.set(relIdx, null);
266       return true;
267     } else {
268       return false;
269     }
270   }
271   
272   /**
273    * Calculates the intersection of a set of lists containing relation indexes.
274    * @param indexLists the list to be intersected.
275    * @return the list of relations contained in all the supplied index lists. 
276    */
277   protected List<Relation> intersection(BitSet... indexLists) {
278     BitSet relIds = new BitSet(relations.size());
279     relIds.set(0, relations.size() - 1);
280     relIds.andNot(deleted);
281     for(BitSet aList : indexLists) {
282       if(aList != null) {
283         relIds.and(aList);
284         if(relIds.isEmpty()) break;
285       }
286     }
287     
288     List<Relation> res = new ArrayList<Relation>();
289     for(int relIdx = 0; relIdx < relations.size(); relIdx++){
290       if(relIds.get(relIdx)) res.add(relations.get(relIdx));
291     }
292     return res;
293   }

295   /* (non-Javadoc)
296    * @see java.lang.Object#toString()
297    */
298   @Override
299   public String toString() {
300     StringBuilder str = new StringBuilder();
301     str.append("[");
302     boolean first = true;
303     for(int  i = 0; i < relations.size(); i++) {
304       if(!deleted.get(i)) {
305         if(first) {
306           first = false;
307         } else{ 
308           str.append("; ");
309         }
310         String relStr = relations.get(i).toString();
311         relStr = relStr.replaceAll(";", Matcher.quoteReplacement("\\;"));
312         if(!relations.get(i).getClass().equals(SimpleRelation.class)) {
313           relStr = "(" + relations.get(i).getClass().getName() + ")" + relStr;  
314         }
315         str.append(relStr);
316       }
317     }
318     str.append("]");
319     return str.toString();
320   }
