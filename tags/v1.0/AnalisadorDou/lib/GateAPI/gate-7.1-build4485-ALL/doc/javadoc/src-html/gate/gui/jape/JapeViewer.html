183   @Override
184   public void setTarget(Object target) {
185     if(target == null) {
186      throw new NullPointerException("JAPE viewer received a null target");
187     }
188     // check that the target is one we can work with - it needs to be a
189     // LanguageAnalyser that has grammarURL and encoding parameters.
190     boolean targetOK = true;
191     if(target instanceof LanguageAnalyser) {
192       try {
193         ((LanguageAnalyser)target).getParameterValue("grammarURL");
194         ((LanguageAnalyser)target).getParameterValue("encoding");
195       } catch(ResourceInstantiationException rie) {
196         targetOK = false;
197       }
198     } else {
199       targetOK = false;
200     }
201      
202     if(!targetOK) {
203       throw new IllegalArgumentException(
204               "The GATE jape viewer can only be used with a GATE jape transducer!\n"
205                       + target.getClass().toString()
206                       + " is not a GATE Jape Transducer!");
207     }
208 
209     if(transducer != null && transducer instanceof AbstractProcessingResource) {
210       ((AbstractProcessingResource)transducer).removeProgressListener(this);
211     }
212 
213     transducer = (LanguageAnalyser)target;
214     URL japeFileURL = null;
215     
216     try {
217       japeFileURL = (URL)transducer.getParameterValue("grammarURL");
218     }
219     catch (ResourceInstantiationException rie) {
220       //ignore this for now and let the null catch take over
221       rie.printStackTrace();
222     }
223 
224     if(japeFileURL == null) {
225       textArea.setText("The source for this JAPE grammar is not available!");
226       remove(treeScroll);
227       return;
228     }
229 
230     String japePhaseName = japeFileURL.getFile();
231     japePhaseName = japePhaseName.substring(japePhaseName.lastIndexOf("/") + 1,
232             japePhaseName.length() - 5);
233     treePhases.setModel(new DefaultTreeModel(new DefaultMutableTreeNode(
234             japePhaseName)));
235     treePhases.setSelectionRow(0);
236 
237     readJAPEFileContents(japeFileURL);
238     if(transducer instanceof AbstractProcessingResource) {
239       ((AbstractProcessingResource)transducer).addProgressListener(this);
240     }
241   }

243   private void readJAPEFileContents(URL url) {
244     if(treePhases.getLastSelectedPathComponent() == null) return;
245     updating = true;
246 
247     try {
248       Reader japeReader = null;
249       String encoding = (String)transducer.getParameterValue("encoding");
250       if(encoding == null) {
251         japeReader = new BomStrippingInputStreamReader(url.openStream());
252       }
253       else {
254         japeReader = new BomStrippingInputStreamReader(url.openStream(), encoding);
255       }
256       BufferedReader br = new BufferedReader(japeReader);
257       String content = br.readLine();
258       StringBuilder japeFileContents = new StringBuilder();
259       List<Integer> lineOffsets = new ArrayList<Integer>();
260 
261       while(content != null) {
262         lineOffsets.add(japeFileContents.length());
263 
264         // replace tabs with spaces otherwise the highlighting fails
265         // TODO work out why this is needed and fix it properly
266         japeFileContents.append(content.replaceAll("\t", "   ")).append("\n");
267         content = br.readLine();
268       }
269 
270       textArea.setText(japeFileContents.toString());
271       textArea.updateUI();
272       br.close();
273 
274       ParseCpslTokenManager tokenManager = new ParseCpslTokenManager(
275               new SimpleCharStream(
276                       new StringReader(japeFileContents.toString())));
277 
278       StyledDocument doc = textArea.getStyledDocument();
279 
280       doc.setCharacterAttributes(0, japeFileContents.length(), defaultStyle,
281               true);
282 
283       ((DefaultMutableTreeNode)treePhases.getSelectionPath()
284               .getLastPathComponent()).removeAllChildren();
285 
286       Token t;
287       while((t = tokenManager.getNextToken()).kind != 0) {
288 
289         Token special = t.specialToken;
290         while(special != null) {
291           Style style = colorMap.get(special.kind);
292           if(style != null) {
293             int start = lineOffsets.get(special.beginLine - 1)
294                     + special.beginColumn - 1;
295             int end = lineOffsets.get(special.endLine - 1) + special.endColumn
296                     - 1;
297             doc.setCharacterAttributes(start, end - start + 1, style, true);
298           }
299 
300           special = special.specialToken;
301         }
302 
303         Style style = colorMap.get(t.kind);
304 
305         if(style != null) {
306           int start = lineOffsets.get(t.beginLine - 1) + t.beginColumn - 1;
307           int end = lineOffsets.get(t.endLine - 1) + t.endColumn - 1;
308           doc.setCharacterAttributes(start, end - start + 1, style, true);
309         }
310 
311         if(t.kind == ParseCpslConstants.path) {
312           ((DefaultMutableTreeNode)treePhases.getSelectionPath()
313                   .getLastPathComponent()).add(new DefaultMutableTreeNode(t
314                   .toString()));
315         }
316       }
317     }
318     catch(IOException ioe) {
319       throw new GateRuntimeException(ioe);
320     } catch(ResourceInstantiationException rie) {
321       throw new GateRuntimeException(rie);
322     }
323 
324     if(treePhases.getSelectionRows() != null
325             && treePhases.getSelectionRows().length > 0)
326       treePhases.expandRow(treePhases.getSelectionRows()[0]);
327 
328     updating = false;
329   }
