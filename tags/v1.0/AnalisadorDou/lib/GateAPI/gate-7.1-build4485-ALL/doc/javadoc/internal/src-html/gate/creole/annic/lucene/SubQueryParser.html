045   /**
046    * Method retrieves wild card characters after the closing bracket.
047    * 
048    * @param brClPos
049    * @param query
050    * @return
051    */
052   private static String findWildCardString(int brClPos, String query) {
053     String wcs = "";
054     if(brClPos + 1 < query.length()) {
055       if(query.charAt(brClPos + 1) == '*' || query.charAt(brClPos + 1) == '+' || query.charAt(brClPos + 1) == '?') {
056         wcs = query.charAt(brClPos + 1) + "";
057         // ok so lets fetch the number
058         for(int i = brClPos + 2; i < query.length(); i++) {
059           if(Character.isDigit(query.charAt(i))) {
060             wcs += query.charAt(i);
061           }
062           else {
063             break;
064           }
065         }
066       }
067     }
068     return wcs;
069   }

071   /**
072    * This method, interprets the wild cards and convert query
073    * accordingly. For example: (A)+3 is converted into ((A) | ((A)(A)) |
074    * ((A)(A)(A)))
075    * 
076    * @param query
077    * @return
078    * @throws SearchException
079    */
080   private static String extractWildcards(String query) throws SearchException {
081     outer: while(true) {
082       char ch = ' ', pre = ' ';
083       for(int i = 0; i < query.length(); i++) {
084         pre = ch;
085         ch = query.charAt(i);
086 
087         // check if it is an open bracket
088         // it is if it doesn't follow the '\' escape sequence
089         if(isOpenBracket(ch, pre)) {
090 
091           // so find out where it gets closed
092           int brClPos = findBracketClosingPosition(i + 1, query);
093           if(brClPos == -1) {
094             throw new SearchException("unbalanced brackets",
095               "a closing bracket ()) is missing for this opening bracket", query, i);
096           }
097 
098           String wildCardString = findWildCardString(brClPos, query);
099           int wcsLen = 0;
100           boolean atLeastOne = false;
101 
102           // at least once
103           int repeatClause = 1;
104 
105           if(wildCardString.length() != 0) {
106             if(wildCardString.length() == 1) {
107               // if there is only wildcard char sign
108               // we consider it as 1
109               wcsLen = 1;
110             }
111             else {
112               atLeastOne = (wildCardString.charAt(0) == '*' || wildCardString.charAt(0) == '?') ? false : true;
113               // now find out the number of Times we need to
114               // duplicate the bracketClause
115               repeatClause = Integer.parseInt(wildCardString.substring(1,
116                       wildCardString.length()));
117               wcsLen = wildCardString.length();
118             }
119 
120             String previous = query.substring(0, i);
121             String after = query
122                     .substring(brClPos + wcsLen + 1, query.length());
123             String sToRepeat = query.substring(i, brClPos + 1);
124 
125             String newString = "(";
126             for(int k = 1; k <= repeatClause; k++) {
127               newString += "(";
128               for(int subK = 0; subK < k; subK++) {
129                 newString += sToRepeat;
130               }
131               newString += ")";
132               if(k + 1 <= repeatClause) {
133                 newString += " | ";
134               }
135             }
136 
137             if(!atLeastOne) {
138               newString += "| {__o__}";
139             }
140 
141             newString += ")";
142             query = previous + newString + after;
143             continue outer;
144           }
145         }
146       }
147 
148       // if we are here
149       // that means no whildcard left
150       return query;
151     }

154   /**
155    * this method parses the query and returns the different queries
156    * converted into the OR normalized form 
157    * for e.g. ({A}|{B}){C} 
158    * this will be converted into ({A}{C}) | ({B}{C}) 
159    * and the arrayList consists of 
160    * 1. {A}{C} 
161    * 2. {B}{C}
162    * @throws SearchException
163    */
164   public static ArrayList<String> parseQuery(String q1) throws SearchException {
165 
166     // arraylist to return - will contain all the OR normalized queries
167     ArrayList<String> queries = new ArrayList<String>();
168 
169     // remove all extra spaces from the query
170     q1 = q1.trim();
171 
172     // we add opening and closing brackets explicitly
173     q1 = "( " + q1 + " )";
174 
175     q1 = extractWildcards(q1);
176     // add the main Query in the arraylist
177     queries.add(q1);
178 
179     for(int index = 0; index < queries.size(); index++) {
180       // get the query to be parsed
181       String query = (String)queries.get(index);
182 
183       // current character and the previous character
184       char ch = ' ', pre = ' ';
185 
186       // if query is ORed
187       // we need duplication
188       // for example: {A}({B}|{C})
189       // the normalized form will be
190       // {A}{B}
191       // {A}{C}
192       // here we need {A} to be duplicated two times
193       boolean duplicated = false;
194       int dupliSize = 0;
195       String data = "";
196 
197       // we need to look into one query at a time and parse it
198       for(int i = 0; i < query.length(); i++) {
199         pre = ch;
200         ch = query.charAt(i);
201 
202         // check if it is an open bracket
203         // it is if it doesn't follow the '\' escape sequence
204         if(isOpenBracket(ch, pre)) {
205 
206           // so find out where it gets closed
207           int brClPos = findBracketClosingPosition(i + 1, query);
208           if(brClPos == -1) {
209             throw new SearchException("unbalanced brackets",
210               "a closing bracket ()) is missing for this opening bracket", query, i);
211           }
212 
213           // see if there are any OR operators in it
214           ArrayList<String> orTokens = findOrTokens(query.substring(i + 1, brClPos));
215 
216           // orTokens will have
217           // for eg. {A} | ({B}{C})
218           // then {A}
219           // and ({B}{C})
220           // so basically findOrTokens find out all the tokens around
221           // | operator
222           if(orTokens.size() > 1) {
223             String text = "";
224 
225             // data contains all the buffered character before the
226             // current positions
227             // for example "ABC" ({B} | {C})
228             // here "ABC" will be in data
229             // and {B} and {C} in orTokens
230             if(!duplicated && data.length() > 0) {
231               text = data;
232               data = "";
233             }
234             else {
235               if(index == queries.size() - 1) {
236                 // this is the case where we would select the
237                 // text as ""
238                 text = "";
239               }
240               else {
241                 text = (String)queries.get(queries.size() - 1);
242               }
243             }
244 
245             // so we need to duplicate the text orTokens.size()
246             // times
247             // for example "ABC" ({B} | {C})
248             // text = "ABC"
249             // orTokens {B} {C}
250             // so two queries will be added
251             // 1. "ABC"
252             // 2. "ABC"
253 
254             queries = duplicate(queries, text, dupliSize, orTokens.size());
255             // and tokens will be added
256             // 1. "ABC" {B}
257             // 2. "ABC" {C}
258             queries = writeTokens(orTokens, queries, dupliSize);
259 
260             // text is duplicated so make it true
261             duplicated = true;
262 
263             // and how many times it was duplicated
264             if(dupliSize == 0) dupliSize = 1;
265             dupliSize *= orTokens.size();
266           }
267           else {
268             // what if the there is only one element between ( and )
269             // it is not an 'OR' query
270 
271             // check how many times we have duplicated the text
272             if(dupliSize == 0) {
273               // if zero and the text buffered is ""
274               // we simply add "" as a separate Query
275               // otherwise add the buffered data as a separate
276               // Query
277               if(data.length() == 0)
278                 queries.add("");
279               else queries.add(data);
280 
281               // because we simply needs to add it only once
282               // but still we have copied it as a separate query
283               // so say duplicated = true
284               duplicated = true;
285               data = "";
286               // and ofcourse the size of the duplication will be
287               // only 1
288               dupliSize = 1;
289             }
290             // and we need to add all the contents between two
291             // brackets in the last duplicated
292             // queries
293             queries = writeStringInAll(query.substring(i + 1, brClPos),
294                     dupliSize, queries);
295           }
296           i = brClPos;
297         }
298         else if(isClosingBracket(ch, pre)) {
299           throw new SearchException("unbalanced brackets",
300             "a opening bracket (() is missing for this closing bracket", query, i);
301         }
302         else {
303           if(duplicated) {
304             queries = writeCharInAll(ch, dupliSize, queries);
305           }
306           else {
307             data += "" + ch;
308           }
309         }
310       }
311 
312       boolean scan = scanQueryForOrOrBracket(query);
313       if(scan) {
314         queries.remove(index);
315         index--;
316       }
317     }
318 
319     ArrayList<String> queriesToReturn = new ArrayList<String>();
320     for(int i = 0; i < queries.size(); i++) {
321       String q = queries.get(i);
322       if(q.trim().length() == 0) {
323         continue;
324       }
325       else if(queriesToReturn.contains(q.trim())) {
326         continue;
327       }
328       else {
329         queriesToReturn.add(q.trim());
330       }
331     }
332     return queriesToReturn;
333   }

335   /**
336    * This method checks if query has either | or ( in it.
337    */
338   public static boolean scanQueryForOrOrBracket(String query) {
339     int index = 0;
340     int index1 = 0;
341     do {
342       index = query.indexOf('|', index);
343       if(index == 0) {
344         return true;
345       }
346       else if(index > 0) {
347         // we have found it but we need to check if it is an escape
348         // sequence
349         if(query.charAt(index - 1) == '\\') {
350           // yes it is an escape sequence
351           // lets search for the next one
352         }
353         else {
354           return true;
355         }
356       }
357 
358       // if we are here that means it was not found
359       index1 = query.indexOf('(', index1);
360       if(index1 == 0) {
361         return true;
362       }
363       else if(index1 > 0) {
364         // we have found it
365         if(query.charAt(index1 - 1) == '\\') {
366           // yes it is an escape sequence
367           continue;
368         }
369         else {
370           return true;
371         }
372       }
373 
374     } while(index >= 0 && index1 >= 0);
375 
376     return false;
377   }

380    * This is a helper method that helps in duplicating the provided tokens.
381    * @param tokens
382    * @param queries
383    * @param dupliSize
384    * @return
385    */
386   private static ArrayList<String> writeTokens(ArrayList<String> tokens, ArrayList<String> queries,

388     if(dupliSize == 0) dupliSize = 1;
389

392       for(int i = 1; i <= tokens.size(); i++) {
393         String token = tokens.get(i - 1);

398                 .get(queries.size() - (j * tokens.size() + i));

402       }
403     }
404 
405     // and now remove
406     for(int i = 0; i < qToRemove.size(); i++) {
407       queries.remove(qToRemove.get(i));
408     }
409 
410     return queries;
411   }

442    * then {A}

444    * so basically findOrTokens find out all the tokens around
445    * | operator
446    */
447   public static ArrayList<String> findOrTokens(String query) {
448     int balance = 0;
449     char pre = ' ';
450     char ch = ' ';
451     ArrayList<String> ors = new ArrayList<String>();
452

455       pre = ch;
456       ch = query.charAt(i);
457       if(isOpenBracket(ch, pre)) {
458         balance++;
459         s += "" + ch;
460         continue;
461       }
462 
463       if(isClosingBracket(ch, pre) && balance > 0) {
464         balance--;
465         s += "" + ch;
466         continue;
467       }
468 
469       if(isOrSym(ch, pre)) {
470         if(balance > 0) {
471           s += "" + ch;
472           continue;
473         }
474         else {
475           ors.add(s);
476           s = "";
477           continue;
478         }
479       }
480 
481       s += "" + ch;
482     }
483 
484     if(s.length() > 0) ors.add(s);
485 
486     return ors;
487   }

551   /**
552    * Returns if the character is bracket used to mark boundary of a token or an escape character.
553    * @param ch
554    * @param pre
555    * @return
556    */
557   private static boolean isOpenBracket(char ch, char pre) {
558     if(ch == '(' && pre != '\\')
