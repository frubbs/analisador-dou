067 /** Top-level entry point for GATE test suite;
068   * "main" will run the JUnit test runner interface.
069   * <P>
070   * Many tests require access to files; generally these files are located
071   * on Web servers. In cases where there is no net connection, or the
072   * Web servers are down, the test files are searched for in the file system
073   * or Jar code base that the system has been loaded from. The search
074   * order for test files is like this:
075   * <UL>
076   * <LI>
077   * <A HREF=http://derwent.dcs.shef.ac.uk:80/gate.ac.uk/>
078   * http://derwent.dcs.shef.ac.uk:80/gate.ac.uk/</A>
079   * <LI>
080   * <A HREF=http://gate.ac.uk:80/>http://gate.ac.uk:80/</A>
081   * <LI>
082   * <A HREF=http://localhost:80/gate.ac.uk/>http://localhost:80/gate.ac.uk/</A>
083   * <LI>
084   * the file system location that the classes came from, e.g.
085   * <TT>z:\gate\classes</TT>, or <TT>jar:....gate.jar</TT>.
086   * </UL>
087   * This search order can be modified by parameters to the main
088   * function (see below).
089   */
090 
091 public class TestGate {
092 
093   /** Debug flag */
094   private static final boolean DEBUG = false;
095 
096   /** Status flag for normal exit. */
097   private static final int STATUS_NORMAL = 0;
098 
099   /** Status flag for error exit. */
100   private static final int STATUS_ERROR = 1;
101 
102   /** GATE test suite. Every test case class has to be
103     * registered here.
104     */
105   public static Test suite() throws Exception {
106     // inialise the library. we re-throw any exceptions thrown by
107     // init, after printing them out, because the junit gui doesn't
108     // say anything more informative than "can't invoke suite" if there's
109     // an exception here...
110 
111     try {
112       //get the config if set through a property
113       String configFile = System.getProperty("gate.config");
114       if(configFile != null && configFile.length() > 0){
115         File f = new File(configFile);
116         try {
117           URL u = f.toURI().toURL();
118         } catch(MalformedURLException e) {
119           Err.prln("Bad initialisation file: " + configFile);
120           Err.prln(e);
121           System.exit(STATUS_ERROR);
122         }
123         Gate.setSiteConfigFile(f);
124       }
125       Gate.init();
126     } catch(GateException e) {
127       Out.prln("can't initialise GATE library! exception = " + e);
128       throw(e);
129     }
130 
131     TestSuite suite = new TestSuite();
132 
133     try {
134       ////////////////////////////////////////////////
135       // Test bench
136       ////////////////////////////////////////////////
137       // set this true to run all tests; false to run the just one below
138       String testName = System.getProperty("gate.testcase");
139       if(testName != null) {
140         // single test class specified in a system property, so run just
141         // that test
142         Class testClass = Class.forName(testName);
143         Method suiteMethod = testClass.getMethod("suite");
144         Test theSuite = (Test)suiteMethod.invoke(null);
145         suite.addTest(theSuite);
146       } else {
147         suite.addTest(TestAnnic.suite());
148         // no test name specified, so run them all
149 
150         //WordNet has been moved into a plugin along with the test
151         //suite.addTest(TestWordNet.suite());
152         suite.addTest(TestIndex.suite());
153         suite.addTest(TestPersist.suite());
154         suite.addTest(TestControllers.suite());
155 
156         // the db isn't usually available so this will always fail
157         //suite.addTest(TestSecurity.suite());
158         suite.addTest(TestConfig.suite());
159         suite.addTest(TestAnnotation.suite());
160         suite.addTest(TestEmail.suite());
161 
162         suite.addTest(TestXml.suite());
163         suite.addTest(TestHtml.suite());
164         suite.addTest(TestSgml.suite());
165         suite.addTest(TestXSchema.suite());
166 
167         suite.addTest(TestCreole.suite());
168         suite.addTest(TestFiles.suite());
169         suite.addTest(TestJavac.suite());
170         suite.addTest(TestJape.suite());
171         suite.addTest(TestTemplate.suite());
172         /* The TCL tests rely on the application being started from the
173          * gate directory. This is not possible from the nightly build script.
174          */
175 //        suite.addTest(TestJacl.suite());
176         suite.addTest(TestDocument.suite());
177         suite.addTest(TestDocumentStaxUtils.suite());
178         suite.addTest(TestTikaFormats.suite());
179         suite.addTest(TestRBTreeMap.suite());
180         suite.addTest(TestCorpus.suite());
181         suite.addTest(TestSerialCorpus.suite());
182         suite.addTest(TestDiffer.suite());
183         //suite.addTest(TestIaa.suite()); Moved this to the IAA plugin
184         suite.addTest(TestAnnotationMerging.suite());
185         suite.addTest(TestClassificationMeasures.suite());
186 //no longer needed as replaced by testPR
187 //        suite.addTest(TestTokeniser.suite());
188         suite.addTest(TestGazetteer.suite());
189 //        suite.addTest(TestSplitterTagger.suite());
190         suite.addTest(TestFeatureMap.suite());
191         suite.addTest(TestTools.suite());
192         suite.addTest(TestCreoleAnnotationHandler.suite());
193         suite.addTest(TestPR.suite());
194         suite.addTest(TestMorph.suite());
195         // suite.addTest(TestMaxentWrapper.suite());
196 
197         //test ontotext gazetteer
198         suite.addTest(TestHashGazetteer.suite());
199         suite.addTest(TestRepositioningInfo.suite());
200         suite.addTest(TestFlexibleGazetteer.suite());
201 
202         // Dynamic creole registration
203         suite.addTest(DynamicRegistrationTest.suite());
204 
205       } // if(allTests)
206 
207     } catch(Exception e) {
208       Out.prln("can't add tests! exception = " + e);
209       throw(e);
210     }
211 
212     return suite;
213   } // suite
214 
215 } // class TestGate
