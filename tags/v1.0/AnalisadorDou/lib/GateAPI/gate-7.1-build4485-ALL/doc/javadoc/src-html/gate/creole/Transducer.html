082   /*
083    * private void writeObject(ObjectOutputStream oos) throws IOException {
084    * Out.prln("writing transducer"); oos.defaultWriteObject();
085    * Out.prln("finished writing transducer"); } // writeObject
086    */
087   /**
088    * This method is the one responsible for initialising the transducer. It
089    * assumes that all the needed parameters have been already set using the
090    * appropiate setXXX() methods.
091    *
092    * @return a reference to <b>this</b>
093    */
094   public Resource init() throws ResourceInstantiationException {
095     try {
096       fireProgressChanged(0);
097 
098       initCustomConstraints();
099 
100       if(binaryGrammarURL != null) {
101         ObjectInputStream s = new ObjectInputStream(binaryGrammarURL
102                 .openStream());
103         batch = (gate.jape.Batch)s.readObject();
104       } else if(grammarURL != null) {
105         if(encoding != null) {
106           batch = new Batch(grammarURL, encoding, new InternalStatusListener());
107           if(enableDebugging != null) {
108             batch.setEnableDebugging(enableDebugging.booleanValue());
109           } else {
110             batch.setEnableDebugging(false);
111           }
112           batch.setOntology(ontology);
113         } else {
114           throw new ResourceInstantiationException("encoding is not set!");
115         }
116       } else {
117         throw new ResourceInstantiationException(
118                 "Neither grammarURL or binaryGrammarURL parameters are set!");
119       }
120     } catch(Exception e) {
121       String message = "Error while parsing the grammar ";
122       if(grammarURL != null) message += "(" + grammarURL.toExternalForm() + ")";
123       message += ":";
124       throw new ResourceInstantiationException(message, e);
125     } finally {
126       fireProcessFinished();
127     }
128     actionContext = initActionContext();
129     batch.setActionContext(actionContext);
130     batch.addProgressListener(new IntervalProgressListener(0, 100));
131     return this;
132   }

134   /**
135    * Method that initialises the ActionContext. This method can be overridden
136    * if somebody wants to extend the Transducer PR class and provide their own
137    * subclass of DefaultActionContext to add some functionality.
138    * 
139    * @return a DefaultActionContext object
140    */
141   protected DefaultActionContext initActionContext() {
142     return new DefaultActionContext();
143   }

147   /**
148    * Implementation of the run() method from {@link java.lang.Runnable}. This
149    * method is responsible for doing all the processing of the input document.
150    */
151   public void execute() throws ExecutionException {
152     interrupted = false;
153     if(document == null) throw new ExecutionException("No document provided!");
154     if(inputASName != null && inputASName.equals("")) inputASName = null;
155     if(outputASName != null && outputASName.equals("")) outputASName = null;
156     // the action context always reflects, for each document executed,
157     // the current PR features and the corpus, if present
158     actionContext.setCorpus(corpus);
159     actionContext.setPRFeatures(features);
160     actionContext.setPR(this);
161     try {
162       batch.transduce(document, inputASName == null
163               ? document.getAnnotations()
164               : document.getAnnotations(inputASName), outputASName == null
165               ? document.getAnnotations()
166               : document.getAnnotations(outputASName));
167     } catch(JapeException je) {
168       throw new ExecutionException(je);
169     }
170   }

183   /**
184    * Loads any custom operators and annotation accessors into the ConstraintFactory.
185    * @throws ResourceInstantiationException
186    */
187   protected void initCustomConstraints() throws ResourceInstantiationException {
188     //Load operators
189     if (operators != null) {
190       for(String opName : operators) {
191         Class<? extends ConstraintPredicate> clazz = null;
192         try {
193           clazz = Class.forName(opName, true, Gate.getClassLoader())
194                         .asSubclass(ConstraintPredicate.class);
195         }
196         catch(ClassNotFoundException e) {
197           //if couldn't find it that way, try with current thread class loader
198           try {
199             clazz = Class.forName(opName, true,
200                 Thread.currentThread().getContextClassLoader())
201                   .asSubclass(ConstraintPredicate.class);
202           }
203           catch(ClassNotFoundException e1) {
204             throw new ResourceInstantiationException("Cannot load class for operator: " + opName, e1);
205           }
206         }
207         catch(ClassCastException cce) {
208           throw new ResourceInstantiationException("Operator class '" + opName + "' must implement ConstraintPredicate");
209         }
210 
211         //instantiate an instance of the class so can get the operator string
212         try {
213           ConstraintPredicate predicate = clazz.newInstance();
214           String opSymbol = predicate.getOperator();
215           //now store it in ConstraintFactory
216           Factory.getConstraintFactory().addOperator(opSymbol, clazz);
217         }
218         catch(Exception e) {
219           throw new ResourceInstantiationException("Cannot instantiate class for operator: " + opName, e);
220         }
221       }
222     }
223 
224     //Load annotationAccessors
225     if (annotationAccessors != null) {
226       for(String accessorName : annotationAccessors) {
227         Class<? extends AnnotationAccessor> clazz = null;
228         try {
229           clazz = Class.forName(accessorName, true, Gate.getClassLoader())
230                      .asSubclass(AnnotationAccessor.class);
231         }
232         catch(ClassNotFoundException e) {
233           //if couldn't find it that way, try with current thread class loader
234           try {
235             clazz = Class.forName(accessorName, true,
236                 Thread.currentThread().getContextClassLoader())
237                    .asSubclass(AnnotationAccessor.class);
238           }
239           catch(ClassNotFoundException e1) {
240             throw new ResourceInstantiationException("Cannot load class for accessor: " + accessorName, e1);
241           }
242         }
243         catch(ClassCastException cce) {
244           throw new ResourceInstantiationException("Operator class '" + accessorName + "' must implement AnnotationAccessor");
245         }
246 
247         //instantiate an instance of the class so can get the meta-property name string
248         try {
249           AnnotationAccessor aa = clazz.newInstance();
250           String accSymbol = (String)aa.getKey();
251           //now store it in ConstraintFactory
252           Factory.getConstraintFactory().addMetaProperty(accSymbol, clazz);
253         }
254         catch(Exception e) {
255           throw new ResourceInstantiationException("Cannot instantiate class for accessor: " + accessorName, e);
256         }
257 
258       }
259     }
260   }

262   /**
263    * Sends a serialized (binary) copy of this transducer to the specified output stream.
264    * Note that this is the same function used by the "Serialize Transducer" menu item
265    * allowing the same functionality to be accessed via code as well as the GUI.

269     out.flush();

272   /**
273    * Saves the Jape Transuder to the binary file.
274    *
275    * @author niraj
276    */
277   protected class SerializeTransducerAction extends javax.swing.AbstractAction {
278     public SerializeTransducerAction() {
279       super("Serialize Transducer");
280       putValue(SHORT_DESCRIPTION, "Serializes the Transducer as binary file");
281     }
282 
283     public void actionPerformed(java.awt.event.ActionEvent evt) {
284       Runnable runnable = new Runnable() {
285         public void run() {
286           JFileChooser fileChooser = MainFrame.getFileChooser();
287           fileChooser.setFileFilter(fileChooser.getAcceptAllFileFilter());
288           fileChooser.setFileSelectionMode(JFileChooser.FILES_ONLY);
289           fileChooser.setMultiSelectionEnabled(false);
290           if(fileChooser.showSaveDialog(null) == JFileChooser.APPROVE_OPTION) {
291             File file = fileChooser.getSelectedFile();
292             try {
293               MainFrame.lockGUI("Serializing JAPE Transducer...");
294               FileOutputStream out = new FileOutputStream(file);
295               ObjectOutputStream s = new ObjectOutputStream(out);
296            serialize(s);
297               s.close();
298               out.close();
299             } catch(IOException ioe) {
300               JOptionPane.showMessageDialog(MainFrame.getInstance(), "Error!\n" + ioe.toString(),
301                       "GATE", JOptionPane.ERROR_MESSAGE);
302               ioe.printStackTrace(Err.getPrintWriter());
303             } finally {
304               MainFrame.unlockGUI();
305             }
306           }
307         }
308       };
309       Thread thread = new Thread(runnable, "Transduer Serialization");
310       thread.setPriority(Thread.MIN_PRIORITY);
311       thread.start();
312     }
313   }
