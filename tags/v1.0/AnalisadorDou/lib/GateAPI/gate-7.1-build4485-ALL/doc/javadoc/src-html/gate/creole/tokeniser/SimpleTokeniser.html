032 /** Implementation of a Unicode rule based tokeniser.
033  * The tokeniser gets its rules from a file an {@link java.io.InputStream
034  * InputStream} or a {@link java.io.Reader Reader} which should be sent to one
035  * of the constructors.
036  * The implementations is based on a finite state machine that is built based
037  * on the set of rules.
038  * A rule has two sides, the left hand side (LHS)and the right hand side (RHS)
039  * that are separated by the &quot;&gt;&quot; character. The LHS represents a
040  * regular expression that will be matched against the input while the RHS
041  * describes a Gate2 annotation in terms of annotation type and attribute-value
042  * pairs.
043  * The matching is done using Unicode enumarated types as defined by the {@link
044  * java.lang.Character Character} class. At the time of writing this class the
045  * suported Unicode categories were:
046  * <ul>
047  * <li>UNASSIGNED
048  * <li>UPPERCASE_LETTER
049  * <li>LOWERCASE_LETTER
050  * <li>TITLECASE_LETTER
051  * <li>MODIFIER_LETTER
052  * <li>OTHER_LETTER
053  * <li>NON_SPACING_MARK
054  * <li>ENCLOSING_MARK
055  * <li>COMBINING_SPACING_MARK
056  * <li>DECIMAL_DIGIT_NUMBER
057  * <li>LETTER_NUMBER
058  * <li>OTHER_NUMBER
059  * <li>SPACE_SEPARATOR
060  * <li>LINE_SEPARATOR
061  * <li>PARAGRAPH_SEPARATOR
062  * <li>CONTROL
063  * <li>FORMAT
064  * <li>PRIVATE_USE
065  * <li>SURROGATE
066  * <li>DASH_PUNCTUATION
067  * <li>START_PUNCTUATION
068  * <li>END_PUNCTUATION
069  * <li>CONNECTOR_PUNCTUATION
070  * <li>OTHER_PUNCTUATION
071  * <li>MATH_SYMBOL
072  * <li>CURRENCY_SYMBOL
073  * <li>MODIFIER_SYMBOL
074  * <li>OTHER_SYMBOL
075  * </ul>
076  * The accepted operators for the LHS are "+", "*" and "|" having the usual
077  * interpretations of "1 to n occurences", "0 to n occurences" and
078  * "boolean OR".
079  * For instance this is a valid LHS:
080  * <br>"UPPERCASE_LETTER" "LOWERCASE_LETTER"+
081  * <br>meaning an uppercase letter followed by one or more lowercase letters.
082  *
083  * The RHS describes an annotation that is to be created and inserted in the
084  * annotation set provided in case of a match. The new annotation will span the
085  * text that has been recognised. The RHS consists in the annotation type
086  * followed by pairs of attributes and associated values.
087  * E.g. for the LHS above a possible RHS can be:<br>
088  * Token;kind=upperInitial;<br>
089  * representing an annotation of type &quot;Token&quot; having one attribute
090  * named &quot;kind&quot; with the value &quot;upperInitial&quot;<br>
091  * The entire rule willbe:<br>
092  * <pre>"UPPERCASE_LETTER" "LOWERCASE_LETTER"+ > Token;kind=upperInitial;</pre>
093  * <br>
094  * The tokeniser ignores all the empty lines or the ones that start with # or
095  * //.
096  *
097  */
098 @CreoleResource(name="GATE Unicode Tokeniser", comment="A customisable Unicode tokeniser.", helpURL="http://gate.ac.uk/userguide/sec:annie:tokeniser", icon="tokeniser")

175   /** Parses one input line containing a tokeniser rule.
176    * This will create the necessary FSMState objects and the links
177    * between them.
178    *
179    * @param line the string containing the rule
180    */
181   void parseRule(String line)throws TokeniserException{
182     //ignore comments
183     if(line.startsWith("#")) return;
184 
185     if(line.startsWith("//")) return;
186 
187     StringTokenizer st = new StringTokenizer(line, "()+*|\" \t\f>", true);
188     FSMState newState = new FSMState(this);
189 
190     initialState.put(null, newState);
191     FSMState finalState = parseLHS(newState, st, LHStoRHS);
192     String rhs = "";
193 
194     if(st.hasMoreTokens()) rhs = st.nextToken("\f");
195 
196     if(rhs.length() > 0)finalState.setRhs(rhs);
197   } // parseRule

199   /** Parses a part or the entire LHS.
200    *
201    * @param startState a FSMState object representing the initial state for
202    *     the small FSM that will recognise the (part of) the rule parsed by this
203    *     method.
204    * @param st a {@link java.util.StringTokenizer StringTokenizer} that
205    *     provides the input
206    * @param until the string that marks the end of the section to be
207    *     recognised. This method will first be called by {@link
208    *     #parseRule(String)} with &quot; &gt;&quot; in order to parse the entire
209    *     LHS. when necessary it will make itself another call to {@link #parseLHS
210    *     parseLHS} to parse a region of the LHS (e.g. a
211    *     &quot;(&quot;,&quot;)&quot; enclosed part.
212    */
213   FSMState parseLHS(FSMState startState, StringTokenizer st, String until)
214        throws TokeniserException{
215 
216     FSMState currentState = startState;
217     boolean orFound = false;
218     List<FSMState> orList = new LinkedList<FSMState>();
219     String token;
220     token = skipIgnoreTokens(st);
221 
222     if(null == token) return currentState;
223 
224     FSMState newState;
225     Integer typeId;
226     UnicodeType uType;
227 
228     bigwhile: while(!token.equals(until)){
229       if(token.equals("(")){//(..)
230         newState = parseLHS(currentState, st,")");
231       } else if(token.equals("\"")){//"unicode_type"
232         String sType = parseQuotedString(st, "\"");
233         newState = new FSMState(this);
234         typeId = stringTypeIds.get(sType);
235 
236         if(null == typeId)
237           throw new InvalidRuleException("Invalid type: \"" + sType + "\"");
238         else uType = new UnicodeType(typeId.intValue());
239 
240         currentState.put(uType ,newState);
241       } else {// a type with no quotes
242         String sType = token;
243         newState = new FSMState(this);
244         typeId = stringTypeIds.get(sType);
245 
246         if(null == typeId)
247           throw new InvalidRuleException("Invalid type: \"" + sType + "\"");
248         else uType = new UnicodeType(typeId.intValue());
249 
250         currentState.put(uType ,newState);
251       }
252       //treat the operators
253       token = skipIgnoreTokens(st);
254       if(null == token) throw
255         new InvalidRuleException("Tokeniser rule ended too soon!");
256 
257       if(token.equals("|")) {
258 
259         orFound = true;
260         orList.add(newState);
261         token = skipIgnoreTokens(st);
262         if(null == token) throw
263           new InvalidRuleException("Tokeniser rule ended too soon!");
264 
265         continue bigwhile;
266       } else if(orFound) {//done parsing the "|"
267         orFound = false;
268         orList.add(newState);
269         newState = new FSMState(this);
270         Iterator<FSMState> orListIter = orList.iterator();
271 
272         while(orListIter.hasNext())
273           orListIter.next().put(null, newState);
274         orList.clear();
275       }
276 
277       if(token.equals("+")) {
278 
279         newState.put(null,currentState);
280         currentState = newState;
281         newState = new FSMState(this);
282         currentState.put(null,newState);
283         token = skipIgnoreTokens(st);
284 
285         if(null == token) throw
286           new InvalidRuleException("Tokeniser rule ended too soon!");
287       } else if(token.equals("*")) {
288 
289         currentState.put(null,newState);
290         newState.put(null,currentState);
291         currentState = newState;
292         newState = new FSMState(this);
293         currentState.put(null,newState);
294         token = skipIgnoreTokens(st);
295 
296         if(null == token) throw
297           new InvalidRuleException("Tokeniser rule ended too soon!");
298       }
299       currentState = newState;
300     }
301     return currentState;
302   } // parseLHS

332   /** Skips the ignorable tokens from the input returning the first significant
333    * token.
334    * The ignorable tokens are defined by {@link #ignoreTokens a set}
335    */
336   protected static String skipIgnoreTokens(StringTokenizer st){
337     Iterator<String> ignorables;
338     boolean ignorableFound = false;
339     String currentToken;
340 
341     while(true){
342       if(st.hasMoreTokens()){
343         currentToken = st.nextToken();
344         ignorables = ignoreTokens.iterator();
345         ignorableFound = false;
346 
347         while(!ignorableFound && ignorables.hasNext()){
348           if(currentToken.equals(ignorables.next()))
349             ignorableFound = true;
350         }
351 
352         if(!ignorableFound) return currentToken;
353       } else return null;
354     }

357   /* Computes the lambda-closure (aka epsilon closure) of the given set of
358    * states, that is the set of states that are accessible from any of the
359    * states in the given set using only unrestricted transitions.
360    * @return a set containing all the states accessible from this state via
361    * transitions that bear no restrictions.
362    */
363   /**
364    * Converts the finite state machine to a deterministic one.
365    *
366    * @param s
367    */
368   private AbstractSet<FSMState> lambdaClosure(Set<FSMState> s){
369 
370     //the stack/queue used by the algorithm
371     LinkedList<FSMState> list = new LinkedList<FSMState>(s);
372 
373     //the set to be returned
374     AbstractSet<FSMState> lambdaClosure = new HashSet<FSMState>(s);
375 
376     FSMState top;
377     FSMState currentState;
378     Set nextStates;
379     Iterator statesIter;
380 
381     while(!list.isEmpty()) {
382       top = list.removeFirst();
383       nextStates = top.nextSet(null);
384 
385       if(null != nextStates){
386         statesIter = nextStates.iterator();
387 
388         while(statesIter.hasNext()) {
389           currentState = (FSMState)statesIter.next();
390           if(!lambdaClosure.contains(currentState)){
391             lambdaClosure.add(currentState);
392             list.addFirst(currentState);
393           }//if(!lambdaClosure.contains(currentState))
394         }//while(statesIter.hasNext())
395 
396       }//if(null != nextStates)
397     }
398     return lambdaClosure;
399   } // lambdaClosure

401   /** Converts the FSM from a non-deterministic to a deterministic one by
402    * eliminating all the unrestricted transitions.
403    */
404   void eliminateVoidTransitions() throws TokeniserException {
405 
406     //kalina:clear() faster than init() which is called with init()
407     newStates.clear();
408     Set<Set> sdStates = new HashSet<Set>();
409     LinkedList<Set> unmarkedDStates = new LinkedList<Set>();
410     DFSMState dCurrentState = new DFSMState(this);
411     Set<FSMState> sdCurrentState = new HashSet<FSMState>();
412 
413     sdCurrentState.add(initialState);
414     sdCurrentState = lambdaClosure(sdCurrentState);
415     newStates.put(sdCurrentState, dCurrentState);
416     sdStates.add(sdCurrentState);
417 
418     //find out if the new state is a final one
419     Iterator<FSMState> innerStatesIter = sdCurrentState.iterator();
420     String rhs;
421     FSMState currentInnerState;
422     Set<String> rhsClashSet = new HashSet<String>();
423     boolean newRhs = false;
424 
425     while(innerStatesIter.hasNext()){
426       currentInnerState = innerStatesIter.next();
427       if(currentInnerState.isFinal()){
428         rhs = currentInnerState.getRhs();
429         rhsClashSet.add(rhs);
430         dCurrentState.rhs = rhs;
431         newRhs = true;
432       }
433     }
434 
435     if(rhsClashSet.size() > 1){
436       Err.println("Warning, rule clash: " +  rhsClashSet +
437                          "\nSelected last definition: " + dCurrentState.rhs);
438     }
439 
440     if(newRhs)dCurrentState.buildTokenDesc();
441     rhsClashSet.clear();
442     unmarkedDStates.addFirst(sdCurrentState);
443     dInitialState = dCurrentState;
444     Set<FSMState> nextSet;
445 
446     while(!unmarkedDStates.isEmpty()){
447       //Out.println("\n\n=====================" + unmarkedDStates.size());
448       sdCurrentState = unmarkedDStates.removeFirst();
449       for(int type = 0; type < maxTypeId; type++){
450       //Out.print(type);
451         nextSet = new HashSet<FSMState>();
452         innerStatesIter = sdCurrentState.iterator();
453 
454         while(innerStatesIter.hasNext()){
455           currentInnerState = innerStatesIter.next();
456           Set<FSMState> tempSet = currentInnerState.nextSet(type);
457           if(null != tempSet) nextSet.addAll(tempSet);
458         }//while(innerStatesIter.hasNext())
459 
460         if(!nextSet.isEmpty()){
461           nextSet = lambdaClosure(nextSet);
462           dCurrentState = newStates.get(nextSet);
463 
464           if(dCurrentState == null){
465 
466             //we have a new DFSMState
467             dCurrentState = new DFSMState(this);
468             sdStates.add(nextSet);
469             unmarkedDStates.add(nextSet);
470 
471             //check to see whether the new state is a final one
472             innerStatesIter = nextSet.iterator();
473             newRhs =false;
474 
475             while(innerStatesIter.hasNext()){
476               currentInnerState = innerStatesIter.next();
477               if(currentInnerState.isFinal()){
478                 rhs = currentInnerState.getRhs();
479                 rhsClashSet.add(rhs);
480                 dCurrentState.rhs = rhs;
481                 newRhs = true;
482               }
483             }
484 
485             if(rhsClashSet.size() > 1){
486               Err.println("Warning, rule clash: " +  rhsClashSet +
487                             "\nSelected last definition: " + dCurrentState.rhs);
488             }
489 
490             if(newRhs)dCurrentState.buildTokenDesc();
491             rhsClashSet.clear();
492             newStates.put(nextSet, dCurrentState);
493           }
494           newStates.get(sdCurrentState).put(type,dCurrentState);
495         } // if(!nextSet.isEmpty())
496 
497       } // for(byte type = 0; type < 256; type++)
498 
499     } // while(!unmarkedDStates.isEmpty())
500 
501   } // eliminateVoidTransitions

503   /** Returns a string representation of the non-deterministic FSM graph using
504    * GML (Graph modelling language).

571   /**
572    * The method that does the actual tokenisation.
573    */
574   public void execute() throws ExecutionException {
575     interrupted = false;
576     AnnotationSet annotationSet;
577     //check the input
578     if(document == null) {
579       throw new ExecutionException(
580         "No document to tokenise!"
581       );
582     }
583 
584     if(annotationSetName == null ||
585        annotationSetName.equals("")) annotationSet = document.getAnnotations();
586     else annotationSet = document.getAnnotations(annotationSetName);
587 
588     fireStatusChanged(
589         "Tokenising " + document.getName() + "...");
590 
591     String content = document.getContent().toString();
592     int length = content.length();
593     char currentChar;
594 
595     DFSMState graphPosition = dInitialState;
596 
597     //the index of the first character of the token trying to be recognised
598     int tokenStart = 0;
599 
600     //the index of the last character of the last token recognised
601     int lastMatch = -1;
602 
603     DFSMState lastMatchingState = null;
604     DFSMState nextState;
605     String tokenString;
606     int charIdx = 0;
607     int oldCharIdx = 0;
608     FeatureMap newTokenFm;
609 
610     while(charIdx < length){
611       currentChar = content.charAt(charIdx);
612 //      Out.println(
613 //      currentChar + typesMnemonics[Character.getType(currentChar)+128]);
614       nextState = graphPosition.next(typeIds.get(
615                   new Integer(Character.getType(currentChar))).intValue());
616 
617       if( null != nextState ) {
618         graphPosition = nextState;
619         if(graphPosition.isFinal()) {
620           lastMatch = charIdx;
621           lastMatchingState = graphPosition;
622         }
623         charIdx ++;
624       } else {//we have a match!
625         newTokenFm = Factory.newFeatureMap();
626 
627         if (null == lastMatchingState) {
628           tokenString = content.substring(tokenStart, tokenStart +1);
629           newTokenFm.put("type","UNKNOWN");
630           newTokenFm.put(TOKEN_STRING_FEATURE_NAME, tokenString);
631           newTokenFm.put(TOKEN_LENGTH_FEATURE_NAME,
632                          Integer.toString(tokenString.length()));
633 
634           try {
635             annotationSet.add(new Long(tokenStart),
636                               new Long(tokenStart + 1),
637                               "DEFAULT_TOKEN", newTokenFm);
638           } catch (InvalidOffsetException ioe) {
639             //This REALLY shouldn't happen!
640             ioe.printStackTrace(Err.getPrintWriter());
641           }
642           // Out.println("Default token: " + tokenStart +
643           //             "->" + tokenStart + " :" + tokenString + ";");
644           charIdx  = tokenStart + 1;
645         } else {
646           tokenString = content.substring(tokenStart, lastMatch + 1);
647           newTokenFm.put(TOKEN_STRING_FEATURE_NAME, tokenString);
648           newTokenFm.put(TOKEN_LENGTH_FEATURE_NAME,
649                          Integer.toString(tokenString.length()));
650 
651           for(int i = 1; i < lastMatchingState.getTokenDesc().length; i++){
652             newTokenFm.put(lastMatchingState.getTokenDesc()[i][0],
653                            lastMatchingState.getTokenDesc()[i][1]);
654           //Out.println(lastMatchingState.getTokenDesc()[i][0] + "=" +
655           //                       lastMatchingState.getTokenDesc()[i][1]);
656           }
657 
658 
659           try {
660             annotationSet.add(new Long(tokenStart),
661                             new Long(lastMatch + 1),
662                             lastMatchingState.getTokenDesc()[0][0], newTokenFm);
663           } catch(InvalidOffsetException ioe) {
664             //This REALLY shouldn't happen!
665             throw new GateRuntimeException(ioe.toString());
666           }
667 
668           // Out.println(lastMatchingState.getTokenDesc()[0][0] +
669           //              ": " + tokenStart + "->" + lastMatch +
670           //              " :" + tokenString + ";");
671           charIdx = lastMatch + 1;
672         }
673 
674         lastMatchingState = null;
675         graphPosition = dInitialState;
676         tokenStart = charIdx;
677       }
678 
679       if((charIdx - oldCharIdx > 256)){
680         fireProgressChanged((100 * charIdx )/ length );
681         oldCharIdx = charIdx;
682         if(isInterrupted()) throw new ExecutionInterruptedException();
683       }
684 
685     } // while(charIdx < length)
686 
687     if (null != lastMatchingState) {
688       tokenString = content.substring(tokenStart, lastMatch + 1);
689       newTokenFm = Factory.newFeatureMap();
690       newTokenFm.put(TOKEN_STRING_FEATURE_NAME, tokenString);
691       newTokenFm.put(TOKEN_LENGTH_FEATURE_NAME,
692                      Integer.toString(tokenString.length()));
693 
694       for(int i = 1; i < lastMatchingState.getTokenDesc().length; i++){
695         newTokenFm.put(lastMatchingState.getTokenDesc()[i][0],
696                        lastMatchingState.getTokenDesc()[i][1]);
697       }
698 
699 
700       try {
701         annotationSet.add(new Long(tokenStart),
702                           new Long(lastMatch + 1),
703                           lastMatchingState.getTokenDesc()[0][0], newTokenFm);
704       } catch(InvalidOffsetException ioe) {
705         //This REALLY shouldn't happen!
706         throw new GateRuntimeException(ioe.toString());
707       }
708 
709     }
710 
711     reset();
712     fireProcessFinished();
713     fireStatusChanged("Tokenisation complete!");
714   } // run

823   /** The static initialiser will inspect the class {@link java.lang.Character}
824     * using reflection to find all the public static members and will map them
825     * to ids starting from 0.
826     * After that it will build all the static data: {@link #typeIds}, {@link
827     * #maxTypeId}, {@link #typeMnemonics}, {@link #stringTypeIds}
828     */
829   static{
830     Field[] characterClassFields;
831 
832     try{
833       characterClassFields = Class.forName("java.lang.Character").getFields();
834     }catch(ClassNotFoundException cnfe){
835       throw new LuckyException("Could not find the java.lang.Character class!");
836     }
837 
838     Collection<Field> staticFields = new LinkedList<Field>();
839     // JDK 1.4 introduced directionality constants that have the same values as
840     //character types; we need to skip those as well
841     for(int i = 0; i< characterClassFields.length; i++)
842       if(Modifier.isStatic(characterClassFields[i].getModifiers()) &&
843          characterClassFields[i].getName().indexOf("DIRECTIONALITY") == -1)
844         staticFields.add(characterClassFields[i]);
845 
846     Map<Integer, Integer> tempTypeIds = new HashMap<Integer, Integer>();
847     maxTypeId = staticFields.size() -1;
848     typeMnemonics = new String[maxTypeId + 1];
849     Map<String, Integer> tempStringTypeIds = new HashMap<String, Integer>();
850 
851     
852     
853     Iterator<Field> staticFieldsIter = staticFields.iterator();
854     Field currentField;
855     int currentId = 0;
856     String fieldName;
857 
858     try {
859       while(staticFieldsIter.hasNext()){
860         currentField = staticFieldsIter.next();
861         if(currentField.getType().toString().equals("byte")){
862           fieldName = currentField.getName();
863           tempTypeIds.put(new Integer(currentField.getInt(null)),
864                                     new Integer(currentId));
865           typeMnemonics[currentId] = fieldName;
866           tempStringTypeIds.put(fieldName, new Integer(currentId));
867           currentId++;
868         }
869       }
870     } catch(Exception e) {
871       throw new LuckyException(e.toString());
872     }
873     
874     typeIds = Collections.unmodifiableMap(tempTypeIds);
875     stringTypeIds = Collections.unmodifiableMap(tempStringTypeIds);
876 
877     ignoreTokens = new HashSet<String>();
878     ignoreTokens.add(" ");
879     ignoreTokens.add("\t");
880     ignoreTokens.add("\f");
881   }
