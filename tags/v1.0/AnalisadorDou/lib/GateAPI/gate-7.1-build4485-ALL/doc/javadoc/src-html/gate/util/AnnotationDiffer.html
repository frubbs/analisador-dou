022 /**
023  * This class provides the logic used by the Annotation Diff tool. It starts 
024  * with two collections of annotation objects, one of key annotations
025  * (representing the gold standard) and one of response annotations 
026  * (representing the system's responses). It will then pair the keys and 
027  * responses in a way that maximises the score. Each key - response pair gets a 
028  * score of {@link #CORRECT_VALUE} (2), {@link #PARTIALLY_CORRECT_VALUE} (1) or 
029  * {@link #WRONG_VALUE} (0)depending on whether the two annotations match are 
030  * overlapping or completely unmatched. Each pairing also has a type of 
031  * {@link #CORRECT_TYPE}, {@link #PARTIALLY_CORRECT_TYPE}, 
032  * {@link #SPURIOUS_TYPE} or {@link #MISSING_TYPE} further detailing the type of
033  * error for the wrong matches (<i>missing</i> being the keys that weren't 
034  * matched to a response while  <i>spurious</i> are the responses that were 
035  * over-generated and are not matching any key.
036  *   
037  * Precision, recall and f-measure are also calculated.
038  */
039 public class AnnotationDiffer {
040 
041   /**
042    * Constructor to be used when you have a collection of AnnotationDiffer
043    * and want to consider it as only one AnnotationDiffer.
044    * Then you can only use the methods getPrecision/Recall/FMeasure...().
045    * @param differs collection to be regrouped in one AnnotationDiffer
046    */
047   public AnnotationDiffer(Collection<AnnotationDiffer> differs) {
048     correctMatches = 0;
049     partiallyCorrectMatches = 0;
050     missing = 0;
051     spurious = 0;
052     int keyCount = 0;
053     int responseCount = 0;
054     for (AnnotationDiffer differ : differs) {
055       // set the correct, partial, spurious and missing values to be
056       // the sum of those in the collection
057       correctMatches += differ.getCorrectMatches();
058       partiallyCorrectMatches += differ.getPartiallyCorrectMatches();
059       missing += differ.getMissing();
060       spurious += differ.getSpurious();
061       keyCount += differ.getKeysCount();
062       responseCount += differ.getResponsesCount();
063     }
064     keyList = new ArrayList<Annotation>(Collections.nCopies(keyCount, (Annotation) null));
065     responseList = new ArrayList<Annotation>(Collections.nCopies(responseCount, (Annotation) null));
066   }
067 
068   public AnnotationDiffer() {
069     // empty constructor
070   }

072   /**
073    * Interface representing a pairing between a key annotation and a response 
074    * one.
075    */
076   public static interface Pairing{
077     /**
078      * Gets the key annotation of the pairing. Can be <tt>null</tt> (for 
079      * spurious matches).
080      * @return an {@link Annotation} object.
081      */
082     public Annotation getKey();
083 
084     public int getResponseIndex();
085     public int getValue();
086     public int getKeyIndex();
087     public int getScore();
088     public void setType(int i);
089     public void remove();
090     
091     /**
092      * Gets the response annotation of the pairing. Can be <tt>null</tt> (for 
093      * missing matches).
094      * @return an {@link Annotation} object.
095      */
096     public Annotation getResponse();
097     
098     /**
099      * Gets the type of the pairing, one of {@link #CORRECT_TYPE},
100      * {@link #PARTIALLY_CORRECT_TYPE}, {@link #SPURIOUS_TYPE} or
101      * {@link #MISSING_TYPE},
102      * @return an <tt>int</tt> value.
103      */
104     public int getType();
105   }
106 
107   /**
108    * Computes a diff between two collections of annotations.
109    * @param key the collection of key annotations.
110    * @param response the collection of response annotations.
111    * @return a list of {@link Pairing} objects representing the pairing set
112    * that results in the best score.
113    */
114   public List<Pairing> calculateDiff(Collection<Annotation> key, Collection<Annotation> response){
115     
116     //initialise data structures
117     if(key == null || key.size() == 0)
118       keyList = new ArrayList<Annotation>();
119     else
120       keyList = new ArrayList<Annotation>(key);
121 
122     if(response == null || response.size() == 0)
123       responseList = new ArrayList<Annotation>();
124     else
125       responseList = new ArrayList<Annotation>(response);
126 
127     if(correctAnnotations != null) {
128       correctAnnotations.clear();
129     }
130     else {
131       correctAnnotations = new HashSet<Annotation>();
132     }
133     if(partiallyCorrectAnnotations != null) {
134       partiallyCorrectAnnotations.clear();
135     }
136     else {
137       partiallyCorrectAnnotations = new HashSet<Annotation>();
138     }
139     if(missingAnnotations != null) {
140       missingAnnotations.clear();
141     }
142     else {
143       missingAnnotations = new HashSet<Annotation>();
144     }
145     if(spuriousAnnotations != null) {
146       spuriousAnnotations.clear();
147     }
148     else {
149       spuriousAnnotations = new HashSet<Annotation>();
150     }
151     
152     keyChoices = new ArrayList<List<Pairing>>(keyList.size());
153     keyChoices.addAll(Collections.nCopies(keyList.size(), (List<Pairing>) null));
154     responseChoices = new ArrayList<List<Pairing>>(responseList.size());
155     responseChoices.addAll(Collections.nCopies(responseList.size(), (List<Pairing>) null));
156 
157     possibleChoices = new ArrayList<Pairing>();
158 
159     //1) try all possible pairings
160     for(int i = 0; i < keyList.size(); i++){
161       for(int j =0; j < responseList.size(); j++){
162         Annotation keyAnn = (Annotation)keyList.get(i);
163         Annotation resAnn = (Annotation)responseList.get(j);
164         PairingImpl choice = null;
165         if(keyAnn.coextensive(resAnn)){
166           //we have full overlap -> CORRECT or WRONG
167           if(keyAnn.isCompatible(resAnn, significantFeaturesSet)){
168             //we have a full match
169             choice = new PairingImpl(i, j, CORRECT_VALUE);
170           }else{
171             //the two annotations are coextensive but don't match
172             //we have a missmatch
173             choice = new PairingImpl(i, j, MISMATCH_VALUE);
174           }
175         }else if(keyAnn.overlaps(resAnn)){
176           //we have partial overlap -> PARTIALLY_CORRECT or WRONG
177           if(keyAnn.isPartiallyCompatible(resAnn, significantFeaturesSet)){
178             choice = new PairingImpl(i, j, PARTIALLY_CORRECT_VALUE);
179           }else{
180             choice = new PairingImpl(i, j, WRONG_VALUE);
181           }
182         }
183 
184         //add the new choice if any
185         if (choice != null) {
186           addPairing(choice, i, keyChoices);
187           addPairing(choice, j, responseChoices);
188           possibleChoices.add(choice);
189         }
190       }//for j
191     }//for i
192 
193     //2) from all possible pairings, find the maximal set that also
194     //maximises the total score
195     Collections.sort(possibleChoices, new PairingScoreComparator());
196     Collections.reverse(possibleChoices);
197     finalChoices = new ArrayList<Pairing>();
198     correctMatches = 0;
199     partiallyCorrectMatches = 0;
200     missing = 0;
201     spurious = 0;
202 
203     while(!possibleChoices.isEmpty()){
204       PairingImpl bestChoice = (PairingImpl)possibleChoices.remove(0);
205       bestChoice.consume();
206       finalChoices.add(bestChoice);
207       switch(bestChoice.value){
208         case CORRECT_VALUE:{
209           correctAnnotations.add(bestChoice.getResponse());
210           correctMatches++;
211           bestChoice.setType(CORRECT_TYPE);
212           break;
213         }
214         case PARTIALLY_CORRECT_VALUE:{
215           partiallyCorrectAnnotations.add(bestChoice.getResponse());
216           partiallyCorrectMatches++;
217           bestChoice.setType(PARTIALLY_CORRECT_TYPE);
218           break;
219         }
220         case MISMATCH_VALUE:{
221           //this is a missing and a spurious annotations together
222           missingAnnotations.add(bestChoice.getKey());
223           missing ++;
224           spuriousAnnotations.add(bestChoice.getResponse());
225           spurious ++;
226           bestChoice.setType(MISMATCH_TYPE);
227           break;
228         }
229         case WRONG_VALUE:{
230           if(bestChoice.getKey() != null){
231             //we have a missed key
232             if(missingAnnotations == null) missingAnnotations = new HashSet<Annotation>();
233             missingAnnotations.add(bestChoice.getKey());
234             missing ++;
235             bestChoice.setType(MISSING_TYPE);
236           }
237           if(bestChoice.getResponse() != null){
238             //we have a spurious response
239             if(spuriousAnnotations == null) spuriousAnnotations = new HashSet<Annotation>();
240             spuriousAnnotations.add(bestChoice.getResponse());
241             spurious ++;
242             bestChoice.setType(SPURIOUS_TYPE);
243           }
244           break;
245         }
246         default:{
247           throw new GateRuntimeException("Invalid pairing type: " +
248                   bestChoice.value);
249         }
250       }
251     }
252     //add choices for the incorrect matches (MISSED, SPURIOUS)
253     //get the unmatched keys
254     for(int i = 0; i < keyChoices.size(); i++){
255       List<Pairing> aList = keyChoices.get(i);
256       if(aList == null || aList.isEmpty()){
257         if(missingAnnotations == null) missingAnnotations = new HashSet<Annotation>();
258         missingAnnotations.add((Annotation)(keyList.get(i)));
259         Pairing choice = new PairingImpl(i, -1, WRONG_VALUE);
260         choice.setType(MISSING_TYPE);
261         finalChoices.add(choice);
262         missing ++;
263       }
264     }
265 
266     //get the unmatched responses
267     for(int i = 0; i < responseChoices.size(); i++){
268       List<Pairing> aList = responseChoices.get(i);
269       if(aList == null || aList.isEmpty()){
270         if(spuriousAnnotations == null) spuriousAnnotations = new HashSet<Annotation>();
271         spuriousAnnotations.add((Annotation)(responseList.get(i)));
272         PairingImpl choice = new PairingImpl(-1, i, WRONG_VALUE);
273         choice.setType(SPURIOUS_TYPE);
274         finalChoices.add(choice);
275         spurious ++;
276       }
277     }
278 
279     return finalChoices;
280   }

282   /**
283    * Gets the strict precision (the ratio of correct responses out of all the 
284    * provided responses).
285    * @return a <tt>double</tt> value.
286    */
287   public double getPrecisionStrict(){
288     if(responseList.size() == 0) {
289       return 1.0;
290     }
291     return correctMatches/(double)responseList.size();
292   }
293 
294   /**
295    * Gets the strict recall (the ratio of key matched to a response out of all 
296    * the keys).
297    * @return a <tt>double</tt> value.
298    */  
299   public double getRecallStrict(){
300     if(keyList.size() == 0) {
301       return 1.0;
302     }
303     return correctMatches/(double)keyList.size();
304   }
305 
306   /**
307    * Gets the lenient precision (where the partial matches are considered as
308    * correct).
309    * @return a <tt>double</tt> value.
310    */
311   public double getPrecisionLenient(){
312     if(responseList.size() == 0) {
313       return 1.0;
314     }
315     return ((double)correctMatches + partiallyCorrectMatches) / (double)responseList.size();
316   }
317 
318   /**
319    * Gets the average of the strict and lenient precision values.
320    * @return a <tt>double</tt> value.
321    */
322   public double getPrecisionAverage() {
323     return ((double)getPrecisionLenient() + getPrecisionStrict()) / (double)(2.0);
324   }
325 
326   /**
327    * Gets the lenient recall (where the partial matches are considered as
328    * correct).
329    * @return a <tt>double</tt> value.
330    */
331   public double getRecallLenient(){
332     if(keyList.size() == 0) {
333       return 1.0;
334     }
335     return ((double)correctMatches + partiallyCorrectMatches) / (double)keyList.size();
336   }
337 
338   /**
339    * Gets the average of the strict and lenient recall values.
340    * @return a <tt>double</tt> value.
341    */
342   public double getRecallAverage() {
343     return ((double) getRecallLenient() + getRecallStrict()) / (double)(2.0);
344   }
345 
346   /**
347    * Gets the strict F-Measure (the harmonic weighted mean of the strict
348    * precision and the strict recall) using the provided parameter as relative 
349    * weight. 
350    * @param beta The relative weight of precision and recall. A value of 1 
351    * gives equal weights to precision and recall. A value of 0 takes the recall 
352    * value completely out of the equation.
353    * @return a <tt>double</tt>value.
354    */
355   public double getFMeasureStrict(double beta){
356     double precision = getPrecisionStrict();
357     double recall = getRecallStrict();
358     double betaSq = beta * beta;
359     double answer = (double)(((double)(betaSq + 1) * precision * recall ) / (double)(betaSq * precision + recall));
360     if(Double.isNaN(answer)) answer = 0.0;
361     return answer;
362   }
363 
364   /**
365    * Gets the lenient F-Measure (F-Measure where the lenient precision and 
366    * recall values are used) using the provided parameter as relative weight. 
367    * @param beta The relative weight of precision and recall. A value of 1 
368    * gives equal weights to precision and recall. A value of 0 takes the recall 
369    * value completely out of the equation.
370    * @return a <tt>double</tt>value.
371    */
372   public double getFMeasureLenient(double beta){
373     double precision = getPrecisionLenient();
374     double recall = getRecallLenient();
375     double betaSq = beta * beta;
376     double answer = (double)(((double)(betaSq + 1) * precision * recall) / ((double)betaSq * precision + recall));
377     if(Double.isNaN(answer)) answer = 0.0;
378     return answer;
379   }
380 
381   /**
382    * Gets the average of strict and lenient F-Measure values.
383    * @param beta The relative weight of precision and recall. A value of 1 
384    * gives equal weights to precision and recall. A value of 0 takes the recall 
385    * value completely out of the equation.
386    * @return a <tt>double</tt>value.
387    */  
388   public double getFMeasureAverage(double beta) {
389     double answer = ((double)getFMeasureLenient(beta) + (double)getFMeasureStrict(beta)) / (double)(2.0);
390     return answer;
391   }

458   /**
459    * Prints to System.out the pairings that are not correct.
460    */
461   public void printMissmatches(){
462     //get the partial correct matches
463     for (Pairing aChoice : finalChoices) {
464       switch(aChoice.getValue()){
465         case PARTIALLY_CORRECT_VALUE:{
466           System.out.println("Missmatch (partially correct):");
467           System.out.println("Key: " + keyList.get(aChoice.getKeyIndex()).toString());
468           System.out.println("Response: " + responseList.get(aChoice.getResponseIndex()).toString());
469           break;
470         }
471       }
472     }
473 
474     //get the unmatched keys
475     for(int i = 0; i < keyChoices.size(); i++){
476       List<Pairing> aList = keyChoices.get(i);
477       if(aList == null || aList.isEmpty()){
478         System.out.println("Missed Key: " + keyList.get(i).toString());
479       }
480     }
481 
482     //get the unmatched responses
483     for(int i = 0; i < responseChoices.size(); i++){
484       List<Pairing> aList = responseChoices.get(i);
485       if(aList == null || aList.isEmpty()){
486         System.out.println("Spurious Response: " + responseList.get(i).toString());
487       }
488     }
489   }

493   /**
494    * Performs some basic checks over the internal data structures from the last
495    * run.
496    * @throws Exception
497    */
498   void sanityCheck()throws Exception{
499     //all keys and responses should have at most one choice left
500     for (List<Pairing> choices : keyChoices)  {
501       if(choices != null){
502         if(choices.size() > 1){
503           throw new Exception("Multiple choices found!");
504         }
505         else if(!choices.isEmpty()){
506           //size must be 1
507           Pairing aChoice = choices.get(0);
508           //the SAME choice should be found for the associated response
509           List<?> otherChoices = (List<?>) responseChoices.get(aChoice.getResponseIndex());
510           if(otherChoices == null ||
511              otherChoices.size() != 1 ||
512              otherChoices.get(0) != aChoice){
513             throw new Exception("Reciprocity error!");
514           }
515         }
516       }
517     }

527           //the SAME choice should be found for the associated response
528           List<?> otherChoices = (List<?>) keyChoices.get(aChoice.getKeyIndex());
529           if(otherChoices == null){
530             throw new Exception("Reciprocity error : null!");
531           }else if(otherChoices.size() != 1){
532             throw new Exception("Reciprocity error: not 1!");
533           }else if(otherChoices.get(0) != aChoice){
534             throw new Exception("Reciprocity error: different!");
535           }
536         }
537       }
538     }

664      * type - sum of all the types of all OTHER mutually exclusive choices

666     void calculateScore(){
667       //this needs to be a set so we don't count conflicts twice
668       Set<Pairing> conflictSet = new HashSet<Pairing>();
669       //add all the choices from the same response annotation
670       conflictSet.addAll(responseChoices.get(responseIndex));
671       //add all the choices from the same key annotation
672       conflictSet.addAll(keyChoices.get(keyIndex));
673       //remove this choice from the conflict set
674       conflictSet.remove(this);
675       score = value;
676       for (Pairing item : conflictSet) {
677         score -= item.getValue();
678       }
679       scoreCalculated = true;
680     }
