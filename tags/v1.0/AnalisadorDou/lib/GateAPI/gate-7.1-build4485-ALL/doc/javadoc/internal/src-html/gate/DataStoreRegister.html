033   private static final long serialVersionUID = 1L;

035   /**
036    * All the DataStore classes available. This is a map of class name to
037    * descriptive text.
038    */
039   public static Map<String,String> getDataStoreClassNames() {
040     Map<String,String> names = new HashMap<String,String>();
041 
042     // TODO: no plugability here at present.... at some future point there should

047     names.put("gate.persist.SerialDataStore", "SerialDataStore: file-based storage using Java serialisation");
048 
049     names.put("gate.persist.LuceneDataStoreImpl", "Lucene Based Searchable DataStore");
050     
051     // docservice
052     try {
053       if (Class.forName("gleam.docservice.gate.DocServiceDataStore", true, Gate.getClassLoader()) != null) {
054         names.put("gleam.docservice.gate.DocServiceDataStore",
055             "SAFE DocService DataStore");
056       }
057     } catch (ClassNotFoundException e) {
058     }
059     
060     return names;
061   } // getDataStoreClassNames()

063   /**
064    * Adds the specified element to this set if it is not already present.
065    * Overriden here for event registration code.
066    */
067   public boolean add(DataStore o) {
068     return super.add(o);
069   } // add
070 
071   /**
072    * Removes the given element from this set if it is present. Overriden here
073    * for event registration code.
074    */
075   public boolean remove(Object o) {
076     boolean res = super.remove(o);
077     if (res) {
078       fireDatastoreClosed(new CreoleEvent((DataStore) o, CreoleEvent.DATASTORE_CLOSED));
079       removeSecurityData((DataStore) o);
080     }
081     return res;
082   } // remove
083 
084   /**
085    * Removes all of the elements from this set. Overriden here for event
086    * registration code.
087    */
088   public void clear() {
089     Set datastores = new HashSet(this);
090     super.clear();
091 
092     Iterator iter = datastores.iterator();
093     while (iter.hasNext()) {
094       fireDatastoreClosed(new CreoleEvent((DataStore) iter.next(), CreoleEvent.DATASTORE_CLOSED));
095     } // while
096   } // clear()

139   /**
140    * Removes a previously registered {@link gate.event.CreoleListener} from the
141    * list of listeners for this DataStoreRegister. Normally the only listener
142    * that is registered with the DataStoreRegister is the {@link CreoleRegister}
143    * which can be obtained through {@link Gate#getCreoleRegister()}
144    */
145   public synchronized void removeCreoleListener(CreoleListener l) {
146     if (creoleListeners != null && creoleListeners.contains(l)) {
147       Vector v = (Vector) creoleListeners.clone();
148       v.removeElement(l);
149       creoleListeners = v;
150     }
151   } // removeCreoleListener(CreoleListener l)
152 
153   /**
154    * Registers a new {@link gate.event.CreoleListener} with this
155    * DataStoreRegister. Normally the only listener that is registered with the
156    * DataStoreRegister is the {@link CreoleRegister} which can be obtained
157    * through {@link Gate#getCreoleRegister()}
158    */
159   public synchronized void addCreoleListener(CreoleListener l) {
160     Vector v = creoleListeners == null ? new Vector(2) : (Vector) creoleListeners.clone();
161     if (!v.contains(l)) {
162       v.addElement(l);
163       creoleListeners = v;
164     }// if
165   }// addCreoleListener(CreoleListener l)
166 
167   /**
168    * Notifies all registered {@link gate.event.CreoleListener}s that a
169    * {@link DataStore} has been opened. Normally the only listener that is
170    * registered with the DataStoreRegister is the {@link CreoleRegister} which
171    * can be obtained through {@link Gate#getCreoleRegister()}
172    */
173   protected void fireDatastoreOpened(CreoleEvent e) {
174     if (creoleListeners != null) {
175       Vector listeners = creoleListeners;
176       int count = listeners.size();
177       for (int i = 0; i < count; i++) {
178         ((CreoleListener) listeners.elementAt(i)).datastoreOpened(e);
179       } // for
180     } // if
181   } // fireDatastoreOpened(CreoleEvent e)
182 
183   /**
184    * Notifies all registered {@link gate.event.CreoleListener}s that a new
185    * {@link DataStore} has been created. Normally the only listener that is
186    * registered with the DataStoreRegister is the {@link CreoleRegister} which
187    * can be obtained through {@link Gate#getCreoleRegister()}
188    */
189   protected void fireDatastoreCreated(CreoleEvent e) {
190     if (creoleListeners != null) {
191       Vector listeners = creoleListeners;
192       int count = listeners.size();
193       for (int i = 0; i < count; i++) {
194         ((CreoleListener) listeners.elementAt(i)).datastoreCreated(e);
195       } // for
196     } // if
197   } // fireDatastoreCreated(CreoleEvent e)
198 
199   /**
200    * Notifies all registered {@link gate.event.CreoleListener}s that a
201    * {@link DataStore} has been closed. Normally the only listener that is
202    * registered with the DataStoreRegister is the {@link CreoleRegister} which
203    * can be obtained through {@link Gate#getCreoleRegister()}
204    */
205   protected void fireDatastoreClosed(CreoleEvent e) {
206     if (creoleListeners != null) {
207       Vector listeners = creoleListeners;
208       int count = listeners.size();
209       for (int i = 0; i < count; i++) {
210         ((CreoleListener) listeners.elementAt(i)).datastoreClosed(e);
211       } // for
212     } // if
213   } // fireDatastoreClosed(CreoleEvent e)

215   /** */
216   private transient Vector creoleListeners;

218 } // class DataStoreRegister
