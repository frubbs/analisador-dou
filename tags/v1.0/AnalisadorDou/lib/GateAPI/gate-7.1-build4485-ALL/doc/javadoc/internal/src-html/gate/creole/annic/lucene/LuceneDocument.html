466   private boolean createTokens(gate.Document gateDocument, AnnotationSet set) {
467     String gateContent = gateDocument.getContent().toString();
468     int start = -1;
469     for(int i = 0; i < gateContent.length(); i++) {
470       char c = gateContent.charAt(i);
471       if(Character.isWhitespace(c)) {
472         if(start != -1) {
473           FeatureMap features = gate.Factory.newFeatureMap();
474           String string = gateContent.substring(start, i);
475           if(string.trim().length() > 0) {
476             features.put("string", string);
477             try {
478               set.add(new Long(start), new Long(i), Constants.ANNIC_TOKEN,
479                 features);
480             }
481             catch(InvalidOffsetException ioe) {
482               ioe.printStackTrace();
483               return false;
484             }
485           }
486           start = i + 1;
487         }
488       }
489       else {
490         if(start == -1) start = i;
491       }
492     }
493     if(start == -1) return false;
494     if(start < gateContent.length()) {
495       FeatureMap features = gate.Factory.newFeatureMap();
496       String string = gateContent.substring(start, gateContent.length());
497       if(string.trim().length() > 0) {
498         features.put("string", string);
499         try {
500           set.add(new Long(start), new Long(gateContent.length()),
501             Constants.ANNIC_TOKEN, features);
502         }
503         catch(InvalidOffsetException ioe) {
504           ioe.printStackTrace();
505           return false;
506         }
507       }
508     }
509     return true;
510   }

514    * method replaces all those incompatible characters with '_'.
515    * 
516    * @param name
517    * @return
518    */
519   private String getCompatibleName(String name) {
520     return name.replaceAll("[\\/:\\*\\?\"<>|]", "_");
521   }

523   /**
524    * This method, given a tokenstream and file name, writes the tokenstream on
525    * the provided location.
526    * 
527    * @param tokenStream
528    * @param fileName
529    * @param location
530    * @throws Exception
531    */
532   private void writeOnDisk(List<Token> tokenStream, String folderName,
533     String fileName, String location) throws Exception {
534 
535     // before we write it on a disk, we need to change its name to
536     // underlying file system name
537     fileName = getCompatibleName(fileName);
538     folderName = getCompatibleName(folderName);
539 
540     if(location.startsWith("file:/"))
541       location = location.substring(6, location.length());
542 
543     if(location.charAt(1) != ':') location = "/" + location;
544 
545     File locationFile = new File(location);
546     File folder = new File(locationFile, Constants.SERIALIZED_FOLDER_NAME);
547     if(!folder.exists()) {
548       folder.mkdirs();
549     }
550     if(!folder.exists()) { throw new IOException(
551       "Directory could not be created :" + folder.getAbsolutePath()); }
552 
553     folder = new File(folder, folderName);
554     if(!folder.exists()) {
555       folder.mkdirs();
556     }
557 
558     if(!folder.exists()) { throw new IOException(
559       "Directory could not be created :" + folder.getAbsolutePath()); }
560 
561     File outputFile = new File(folder, fileName + ".annic");
562     ObjectOutput output = null;
563     OutputStream file = new FileOutputStream(outputFile);
564     OutputStream buffer = new BufferedOutputStream(file);
565     output = new ObjectOutputStream(buffer);
566     output.writeObject(tokenStream);
567     if(output != null) {
568       output.close();
569     }
570   }

572   /**
573    * Internal class used for storing the offsets of annotations.
574    * 
575    * @author niraj
576    * 
577    */
578   private class OffsetGroup {
579     Long startOffset;
580 
581     Long endOffset;
582   }
583 
584   /**
585    * This method given a GATE document and other required parameters, for each
586    * annotation of type indexUnitAnnotationType creates a separate list of
587    * baseTokens underlying in it.
588    * 
589    * @param document
590    * @param inputAs
591    * @param featuresToExclude
592    * @param baseTokenAnnotationType
593    * @param indexUnitAnnotationType
594    * @return
595    */
596   private List<Token>[] getTokens(gate.Document document,
597     AnnotationSet inputAs, List<String> featuresToInclude,
598     List<String> featuresToExclude, String baseTokenAnnotationType,
599     AnnotationSet baseTokenSet, String indexUnitAnnotationType,
600     AnnotationSet indexUnitSet, Set<String> indexedFeatures) {
601 
602     boolean excludeFeatures = false;
603     boolean includeFeatures = false;
604 
605     // if include features are provided, we donot look at the exclude
606     // features
607     if(!featuresToInclude.isEmpty()) {
608       includeFeatures = true;
609     }
610     else if(!featuresToExclude.isEmpty()) {
611       excludeFeatures = true;
612     }
613 
614     HashSet<OffsetGroup> unitOffsetsSet = new HashSet<OffsetGroup>();
615     if(indexUnitAnnotationType == null
616       || indexUnitAnnotationType.trim().length() == 0 || indexUnitSet == null
617       || indexUnitSet.size() == 0) {
618       // the index Unit Annotation Type is not specified
619       // therefore we consider the entire document as a single unit
620       OffsetGroup group = new OffsetGroup();
621       group.startOffset = new Long(0);
622       group.endOffset = document.getContent().size();
623       unitOffsetsSet.add(group);
624     }
625     else {
626       Iterator<Annotation> iter = indexUnitSet.iterator();
627       while(iter.hasNext()) {
628         Annotation annotation = iter.next();
629         OffsetGroup group = new OffsetGroup();
630         group.startOffset = annotation.getStartNode().getOffset();
631         group.endOffset = annotation.getEndNode().getOffset();
632         unitOffsetsSet.add(group);
633       }
634     }
635 
636     Set<String> allTypes = new HashSet<String>();
637 
638     for(String aType : inputAs.getAllTypes()) {
639       if(aType.indexOf(".") > -1 || aType.indexOf("=") > -1
640         || aType.indexOf(";") > -1 || aType.indexOf(",") > -1) {
641         System.err
642           .println("Annotations of type "
643             + aType
644             + " cannot be indexed as the type name contains one of the ., =, or ; character");
645         continue;
646       }
647       allTypes.add(aType);
648     }
649 
650     if(baseTokenSet != null && baseTokenSet.size() > 0) {
651       allTypes.remove(baseTokenAnnotationType);
652     }
653 
654     if(indexUnitSet != null && indexUnitSet.size() > 0)
655       allTypes.remove(indexUnitAnnotationType);
656 
657     AnnotationSet toUseSet = new AnnotationSetImpl(document);
658     for(String type : allTypes) {
659       for(Annotation a : inputAs.get(type)) {
660         try {
661           toUseSet.add(a.getStartNode().getOffset(),
662             a.getEndNode().getOffset(), a.getType(), a.getFeatures());
663         }
664         catch(InvalidOffsetException ioe) {
665           throw new GateRuntimeException(ioe);
666         }
667       }
668     }
669 
670     ArrayList<Token> toReturn[] = new ArrayList[unitOffsetsSet.size()];
671     Iterator<OffsetGroup> iter = unitOffsetsSet.iterator();
672     int counter = 0;
673     while(iter.hasNext()) {
674       OffsetGroup group = iter.next();
675       ArrayList<Token> newTokens = new ArrayList<Token>();
676       ArrayList<Annotation> tokens =
677         new ArrayList<Annotation>(toUseSet.getContained(group.startOffset,
678           group.endOffset));
679 
680       // add tokens from the baseTokenSet
681       if(baseTokenSet != null && baseTokenSet.size() != 0) {
682         tokens.addAll(baseTokenSet.getContained(group.startOffset,
683           group.endOffset));
684       }
685 
686       if(tokens == null || tokens.size() == 0) return null;
687 
688       Collections.sort(tokens, new OffsetComparator());
689 
690       int position = -1;
691       for(int i = 0; i < tokens.size(); i++) {
692         byte inc = 1;
693         Annotation annot = tokens.get(i);
694         String type = annot.getType();
695         // if the feature is specified in featuresToExclude -exclude it
696         if(excludeFeatures && featuresToExclude.contains(type)) continue;
697 
698         // if the feature is not sepcified in the include features -
699         // exclude it
700         if(includeFeatures && !featuresToInclude.contains(type)) continue;
701 
702         int startOffset = annot.getStartNode().getOffset().intValue();
703         int endOffset = annot.getEndNode().getOffset().intValue();
704         String text =
705           document.getContent().toString().substring(startOffset, endOffset);
706         if(text == null) {
707           continue;
708         }
709 
710 
711         Token token1 = new Token(type, startOffset, endOffset, "*");
712 
713         // each token has four values
714         // String, int, int, String
715         // we add extra info of position
716         if(i > 0) {
717           if(annot.getStartNode().getOffset().longValue() == tokens.get(i - 1)
718             .getStartNode().getOffset().longValue()) {
719             token1.setPositionIncrement(0);
720             inc = 0;
721           }
722         }
723 
724         position += inc;
725         token1.setPosition(position);
726         newTokens.add(token1);
727 
728         if(!type.equals(baseTokenAnnotationType)
729           || (annot.getFeatures().get("string") == null)) {
730           // we need to create one string feature for this
731           Token tk1 = new Token(text, startOffset, endOffset, type + ".string");
732           indexedFeatures.add(type + ".string");
733           tk1.setPositionIncrement(0);
734           tk1.setPosition(position);
735           newTokens.add(tk1);
736         }
737 
738         // now find out the features and add them
739         FeatureMap features = annot.getFeatures();
740         Iterator fIter = features.keySet().iterator();
741         while(fIter.hasNext()) {
742           String type1 = (String)fIter.next();
743           // if the feature is specified in featuresToExclude -exclude
744           // it
745           if(excludeFeatures && featuresToExclude.contains(type + "." + type1)) {
746             continue;
747           }
748 
749           // if the feature is not sepcified in the include features -
750           // exclude it
751           if(includeFeatures && !featuresToInclude.contains(type + "." + type1))
752             continue;
753 
754           Object tempText = features.get(type1);
755           if(tempText == null) continue;
756 
757           String text1 = (String)tempText.toString();
758           // we need to qualify the type names
759           // for each annotation type feature we add AT.Feature=="**" to be able
760           // to search for it
761           // to calculate stats
762 
763           Token tempToken =
764             new Token(text1, startOffset, endOffset, type + "." + type1);
765           indexedFeatures.add(type + "." + type1);
766           tempToken.setPositionIncrement(0);
767           tempToken.setPosition(position);
768           newTokens.add(tempToken);
769 
770           Token onlyATFeature =
771             new Token(type + "." + type1, startOffset, endOffset, "**");
772           onlyATFeature.setPosition(position);
773           onlyATFeature.setPositionIncrement(0);
774           newTokens.add(onlyATFeature);
775 
776         }
777       }
778       toReturn[counter] = newTokens;
779       counter++;
780     }
781     return toReturn;
782   }
