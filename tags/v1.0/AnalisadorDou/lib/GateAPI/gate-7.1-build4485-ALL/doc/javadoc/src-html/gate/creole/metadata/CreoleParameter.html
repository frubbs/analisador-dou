016 package gate.creole.metadata;
017 
018 import java.lang.annotation.Documented;
019 import java.lang.annotation.ElementType;
020 import java.lang.annotation.Retention;
021 import java.lang.annotation.RetentionPolicy;
022 import java.lang.annotation.Target;
023 import java.util.Collection;
024 
025 /**
026  * <p>
027  * Annotation used to define a parameter to a CREOLE resource. This annotation
028  * should be used to annotate the <code>set</code> method corresponding to the
029  * parameter. The parameter's name is inferred from the method name, and its
030  * type is inferred from the type of the method's argument. When annotating a
031  * method whose argument type is a subtype of {@link Collection}, GATE also
032  * attempts to infer the collection element type from any generic type
033  * arguments given. If the collection type is a raw type then you will need to
034  * supply the collectionElementType explicitly in the annotation.
035  * </p>
036  *
037  * <p>
038  * Parameters can be marked as optional or runtime parameters using the
039  * appropriate additional annotations.
040  * </p>
041  *
042  * @see Optional
043  * @see RunTime
044  */
045 @Documented
046 @Target( {ElementType.METHOD})
047 @Retention(RetentionPolicy.RUNTIME)
048 public @interface CreoleParameter {
049   /**
050    * Dummy type used to signify that no value has been supplied for {@link #collectionElementType()}.
051    */
052   public static interface NoElementType {}
053   
054   /**
055    * Special value used to signify the absence of a default value for a
056    * parameter.
057    */
058   public static final String NO_DEFAULT_VALUE = "____NO_DEFAULT____";
059   
060   /**
061    * The default priority value assumed if no explicit priority
062    * is set.
063    */
064   public static final int DEFAULT_PRIORITY = Integer.MAX_VALUE;
065   
066   /**
067    * The item class name for parameters whose type is a Collection, Set
068    * or List. When annotating a field or get method with a parameterised type
069    * (e.g. <code>List&lt;String&gt;</code>), the correct value can often be
070    * inferred automatically, but if a value is specified here it takes
071    * precedence.
072    */
073   Class<?> collectionElementType() default NoElementType.class;
074 
075   /**
076    * The default value for the parameter, expressed as a string. For
077    * non-string parameters, follow the rules given in the <a
078    * href="http://gate.ac.uk/userguide/sec:creole-model:config:xml">user
079    * guide</a>. If not specified, the default is <code>null</code>.
080    */
081   String defaultValue() default NO_DEFAULT_VALUE;
082 
083   /**
084    * A descriptive comment about this parameter.
085    */
086   String comment() default "";
087 
088   /**
089    * Semicolon-separated list of file suffixes accepted by default for
090    * this parameter. For parameters of type {@link java.net.URL}, the
091    * GUI provides a button to locate the desired file in a Java file
092    * chooser. The suffixes list defines the default file name filter
093    * used by the file chooser.
094    */
095   String suffixes() default "";
096 
097   /**
098    * If this parameter is part of a disjunction, set an ID here. All
099    * parameters sharing the same disjunction ID will be grouped together
100    * in a single disjunction. If not specified, the parameter will not
101    * be considered as part of a disjunction.
102    */
103   String disjunction() default "";
104   
105   /**
106    * If this parameter is part of a disjunction, the order in which
107    * the disjunctive parameters are listed is determined by their
108    * priority values.  Parameters with smaller priority values are
109    * considered "more important" than those with larger values (i.e.
110    * 1 is higher priority than 10).  When creating a new resource
111    * instance in GATE Developer, the parameters dialog offers the
112    * parameters in each disjunction in priority order, so you should
113    * assign priorities to your parameters such that the most important
114    * parameter in each disjunction is the most frequent use case.  For
115    * example, in a JAPE transducer the grammarURL parameter (for
116    * loading a .jape file) is considered more important than the
117    * binaryGrammarURL (for loading a serialized transducer).
118    */
119   int priority() default Integer.MAX_VALUE;
120 }
