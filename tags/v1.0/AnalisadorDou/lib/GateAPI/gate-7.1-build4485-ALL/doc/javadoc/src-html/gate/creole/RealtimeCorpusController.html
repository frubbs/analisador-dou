041 /**
042  * A custom GATE controller that interrupts the execution over a document when a
043  * specified amount of time has elapsed. It also ignores all errors/exceptions 
044  * that may occur during execution and simply carries on with the next document
045  * when that happens.
046  */
047 @CreoleResource(name = "Real-Time Corpus Pipeline",
048     comment = "A serial controller for PR pipelines over corpora which "
049         + "limits the run time of each PR.",
050     icon = "application-realtime",
051     helpURL = "http://gate.ac.uk/userguide/sec:creole-model:applications")
052 public class RealtimeCorpusController extends SerialAnalyserController {

078   public RealtimeCorpusController(){

081       prof = new Profiler();

083       prof.printToSystemOut(true);
084       timeMap = new HashMap<String,Long>();

088   protected class DocRunner implements Callable<Object>{
089     
090     public DocRunner(Document document) {
091       this.document = document;
092     }
093     
094     public Object call() {
095       try {
096         // save a reference to the executor thread
097         currentWorkingThread = Thread.currentThread();
098         // run the system over the current document
099         // set the doc and corpus
100         for(int j = 0; j < prList.size(); j++) {
101           ((LanguageAnalyser)prList.get(j)).setDocument(document);
102           ((LanguageAnalyser)prList.get(j)).setCorpus(corpus);
103         }
104         interrupted = false;
105         // execute the PRs
106         // check all the PRs have the right parameters
107         checkParameters();
108         if(DEBUG) {
109           prof.initRun("Execute controller [" + getName() + "]");
110         }
111 
112         // execute all PRs in sequence
113         interrupted = false;
114         for(int j = 0; j < prList.size(); j++) {
115           if(isInterrupted())
116             throw new ExecutionInterruptedException("The execution of the "
117                     + getName() + " application has been abruptly interrupted!");
118 
119           if(Thread.currentThread().isInterrupted()) {
120             Err.println("Execution on document " + document.getName()
121                     + " has been stopped");
122             break;
123           }
124 
125           try {
126             runComponent(j);
127           } catch(ThreadDeath td) {
128             // we got stopped
129             throw td;
130           } catch(Throwable e) {
131             if(threadDying){
132               // we're in the process of stopping 
133               Err.println("Execution on document " + document.getName()
134                   + " has been stopped");
135               // stop running the rest of the PRs
136               break;
137             } else {
138               // the thread was not in the process of being stopped: 
139               // actual exception during processing: throw upwards
140               throw e;
141             }
142           }
143           if(DEBUG) {
144             prof.checkPoint("~Execute PR ["
145                     + ((ProcessingResource)prList.get(j)).getName() + "]");
146             Long timeOfPR = timeMap.get(((ProcessingResource)prList.get(j))
147                     .getName());
148             if(timeOfPR == null)
149               timeMap.put(((ProcessingResource)prList.get(j)).getName(),
150                       new Long(prof.getLastDuration()));
151             else timeMap.put(((ProcessingResource)prList.get(j)).getName(),
152                     new Long(timeOfPR.longValue() + prof.getLastDuration()));
153             Out.println("Time taken so far by "
154                     + ((ProcessingResource)prList.get(j)).getName()
155                     + ": "
156                     + timeMap.get(((ProcessingResource)prList.get(j)).getName()));
157           }
158         }
159       }
160       catch(ThreadDeath td) {
161         // special case as we need to re-throw this one
162         Err.prln("Execution on document " + document.getName()
163                 + " has been stopped");
164         throw (td);
165       }
166       catch(Throwable cause) {
167         logger.info("Execution on document " + document.getName()
168                 + " has caused an error (ignored):\n=========================", cause);
169         logger.info("=========================\nError ignored...\n");
170       }
171       finally {
172         // remove the reference to the thread, as we're now done
173         currentWorkingThread = null;
174         // unset the doc and corpus
175         for(int j = 0; j < prList.size(); j++) {
176           ((LanguageAnalyser)prList.get(j)).setDocument(null);
177           ((LanguageAnalyser)prList.get(j)).setCorpus(null);
178         }
179 
180         if(DEBUG) {
181           prof.checkPoint("Execute controller [" + getName() + "] finished");
182         }
183       }
184       
185       return null;
186     }
187     private Document document;
188   }

196   @Override
197   public Resource init() throws ResourceInstantiationException {
198     // we normally require 2 threads: one to execute the PRs and another one to
199     // to execute the job stoppers. More threads are created as required.  We
200     // use a custom ThreadFactory that returns daemon threads so we don't block
201     // GATE from exiting if this controller has not been properly disposed of.
202     threadSource = Executors.newSingleThreadExecutor(new ThreadFactory() {
203       private ThreadFactory dtf = Executors.defaultThreadFactory();
204       public Thread newThread(Runnable r) {
205         Thread t = dtf.newThread(r);
206         t.setDaemon(true);
207         return t;
208       }
209     });
210     return super.init();
211   }

213   /** Run the Processing Resources in sequence. */
214   @SuppressWarnings("deprecation")
215   public void executeImpl() throws ExecutionException{
216     interrupted = false;
217     if(corpus == null) throw new ExecutionException(
218       "(SerialAnalyserController) \"" + getName() + "\":\n" +
219       "The corpus supplied for execution was null!");
220     //iterate through the documents in the corpus
221     for(int i = 0; i < corpus.size(); i++){
222       if(isInterrupted()) throw new ExecutionInterruptedException(
223         "The execution of the " + getName() +
224         " application has been abruptly interrupted!");
225 
226       boolean docWasLoaded = corpus.isDocumentLoaded(i);
227       Document doc = (Document)corpus.get(i);
228       // start the execution, in the separate thread
229       threadDying = false;
230       Future<?> docRunnerFuture = threadSource.submit(new DocRunner(doc));
231       // how long have we already waited 
232       long waitSoFar = 0;
233       // check if we should use graceful stop first 
234       if (graceful != -1 && (timeout == -1 || graceful < timeout )) {
235         try {
236           docRunnerFuture.get(graceful, TimeUnit.MILLISECONDS);
237         } catch(TimeoutException e) {
238           // we waited the graceful period, and the task did not finish
239           // -> interrupt the job (nicely)
240           threadDying = true;
241           waitSoFar += graceful;
242           logger.info("Execution timeout, attempting to gracefully stop worker thread...");
243           // interrupt the working thread - we can't cancel the future as
244           // that would cause future get() calls to fail immediately with
245           // a CancellationException
246           Thread t = currentWorkingThread;
247           if(t != null) {
248             t.interrupt();
249           }
250           for(int j = 0; j < prList.size(); j++){
251             ((Executable)prList.get(j)).interrupt();
252           }
253           // next check scheduled for 
254           // - half-time between graceful and timeout, or
255           // - graceful-and-a-half (if no timeout)
256           long waitTime = (timeout != -1) ? 
257                           (timeout - graceful) / 2 : 
258                           (graceful / 2);
259           try {
260             docRunnerFuture.get(waitTime, TimeUnit.MILLISECONDS);
261           } catch(TimeoutException e1) {
262             // the mid point has been reached: try nullify
263             threadDying = true;
264             waitSoFar += waitTime;
265             logger.info("Execution timeout, attempting to induce exception in order to stop worker thread...");
266             for(int j = 0; j < prList.size(); j++){
267               ((LanguageAnalyser)prList.get(j)).setDocument(null);
268               ((LanguageAnalyser)prList.get(j)).setCorpus(null);
269             }            
270           } catch(InterruptedException e1) {
271             // the current thread (not the execution thread!) was interrupted
272             // throw it forward
273             Thread.currentThread().interrupt();
274           } catch(java.util.concurrent.ExecutionException e2) {
275             throw new ExecutionException(e2);
276           }
277         } catch(java.util.concurrent.ExecutionException e) {
278           throw new ExecutionException(e);
279         } catch(InterruptedException e) {
280           // the current thread (not the execution thread!) was interrupted
281           // throw it forward
282           Thread.currentThread().interrupt();
283         }
284       }
285       // wait before we call stop()
286       if(timeout != -1) {
287         long waitTime = timeout - waitSoFar;
288         if(waitTime > 0) {
289           try {
290             docRunnerFuture.get(waitTime, TimeUnit.MILLISECONDS);
291           } catch(TimeoutException e) {
292             // we're out of time: stop the thread
293             threadDying = true;
294             logger.info("Execution timeout, worker thread will be forcibly terminated!");
295             // using a volatile variable instead of synchronisation
296             Thread theThread = currentWorkingThread;
297             if(theThread != null) {
298               theThread.stop();
299               try {
300                 // and wait for it to actually die
301                 docRunnerFuture.get();
302               } catch(InterruptedException e2) {
303                 // current thread has been interrupted: 
304                 Thread.currentThread().interrupt();
305               } catch(java.util.concurrent.ExecutionException ee) {
306                 if(ee.getCause() instanceof ThreadDeath) {
307                   // we have just caused this  
308                 } else {
309                   logger.error("Real Time Controller Malfunction", ee);
310                 }
311               }
312             }
313           } catch(InterruptedException e) {
314             // the current thread (not the execution thread!) was interrupted
315             // throw it forward
316             Thread.currentThread().interrupt();
317           } catch(java.util.concurrent.ExecutionException e) {
318             throw new ExecutionException(e);
319           }
320         } else {
321           // stop now!
322           threadDying = true;
323           logger.info("Execution timeout, worker thread will be forcibly terminated!");
324           // using a volatile variable instead of synchronisation
325           Thread theThread = currentWorkingThread;
326           if(theThread != null){
327             theThread.stop();
328             try {
329               // and wait for it to actually die
330               docRunnerFuture.get();
331             } catch(InterruptedException e) {
332               // current thread has been interrupted: 
333               Thread.currentThread().interrupt();
334             } catch(java.util.concurrent.ExecutionException ee) {
335               if(ee.getCause() instanceof ThreadDeath) {
336                 // we have just caused this  
337               } else {
338                 logger.error("Real Time Controller Malfunction", ee);
339               }
340             }
341           }
342         }
343       }
344       
345       String docName = doc.getName();
346       // at this point we finished execution (one way or another)
347       if(!docWasLoaded){
348         //trigger saving
349         getCorpus().unloadDocument(doc);
350         //close the previously unloaded Doc
351         Factory.deleteResource(doc);
352       }
353 
354       // global progress bar depends on this status message firing at the end
355       // of processing for each document.
356       fireStatusChanged("Finished running " + getName() + " on document " +
357           docName);
358     }
359   }
