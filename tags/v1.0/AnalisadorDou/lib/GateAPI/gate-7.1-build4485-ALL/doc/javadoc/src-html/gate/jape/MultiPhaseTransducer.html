132   /** Transduce the document by running each phase in turn. */
133   public void transduce(Document doc, AnnotationSet input,
134                         AnnotationSet output) throws JapeException,
135                                                      ExecutionException {
136     interrupted = false;
137     ProgressListener pListener = null;
138     StatusListener sListener = null;
139     pListener = new ProgressListener(){
140       public void processFinished(){
141         donePhases ++;
142         if(donePhases == phasesCnt) fireProcessFinished();
143       }
144 
145       public void progressChanged(int i){
146         int value = (donePhases * 100 + i)/phasesCnt;
147         fireProgressChanged(value);
148       }
149 
150       int phasesCnt = phases.size();
151       int donePhases = 0;
152     };
153 
154     sListener = new StatusListener(){
155       public void statusChanged(String text){
156         fireStatusChanged(text);
157       }
158     };
159 
160     for(Iterator i = phases.iterator(); i.hasNext(); ) {
161       Transducer t = (Transducer) i.next();
162 
163       if(isInterrupted()) throw new ExecutionInterruptedException(
164         "The execution of the \"" + getName() +
165         "\" Jape transducer has been abruptly interrupted!");
166 
167       try {
168         fireStatusChanged("Transducing " + doc.getName() +
169                              " (Phase: " + t.getName() + ")...");
170         String savedBenchmarkID = null;
171         String phaseBenchmarkID = null;
172         if(Benchmark.isBenchmarkingEnabled()) {
173           savedBenchmarkID = t.getBenchmarkId();
174           this.benchmarkFeatures.put(Benchmark.DOCUMENT_NAME_FEATURE, doc.getName());
175           phaseBenchmarkID = Benchmark.createBenchmarkId("phase__" + t.getName(), this.getBenchmarkId());
176           t.setBenchmarkId(phaseBenchmarkID);
177         }
178         long startTime = Benchmark.startPoint();
179         t.addProgressListener(pListener);
180         t.addStatusListener(sListener);
181 
182         t.setActionContext(actionContext);
183         t.transduce(doc, input, output);
184         t.removeProgressListener(pListener);
185         t.removeStatusListener(sListener);
186         if(Benchmark.isBenchmarkingEnabled()) {
187           Benchmark.checkPoint(startTime, phaseBenchmarkID, this, benchmarkFeatures);
188           t.setBenchmarkId(savedBenchmarkID);
189         }
190         fireStatusChanged("");
191       } catch(JapeException e) {
192         String location = "phase " + t.getName() + ", document " + doc.getName();
193         e.setLocation(location);
194         throw e;
195       }
196     }
197 
198     cleanUp();
199   } // transduce

254   @Override
255   public void runControllerExecutionStartedBlock(
256     ActionContext ac, Controller c, Ontology o) throws ExecutionException {
257     for(Iterator i = phases.iterator(); i.hasNext(); ) {
258       Transducer t = (Transducer) i.next();
259       t.runControllerExecutionStartedBlock(ac, c, o);
260     }

262   @Override
263   public void runControllerExecutionFinishedBlock(
264     ActionContext ac, Controller c, Ontology o) throws ExecutionException {
265     for(Iterator i = phases.iterator(); i.hasNext(); ) {
266       Transducer t = (Transducer) i.next();
267       t.runControllerExecutionFinishedBlock(ac, c, o);
268     }
269   }
270   @Override
271   public void runControllerExecutionAbortedBlock(
272     ActionContext ac, Controller c, Throwable throwable, Ontology o) throws ExecutionException {
273     for(Iterator i = phases.iterator(); i.hasNext(); ) {
274       Transducer t = (Transducer) i.next();
275       t.runControllerExecutionAbortedBlock(ac, c,throwable, o);
276     }
