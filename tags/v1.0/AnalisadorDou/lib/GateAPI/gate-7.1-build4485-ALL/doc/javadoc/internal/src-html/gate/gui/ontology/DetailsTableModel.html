039 /**
040  * A DataModel that is created when a node is selected in the ontology
041  * tree. It contains information such as direct/all sub/super classes,
042  * equivalent classes, instances, properties/ property values and so on.
043  * The information from this model is then shown in the right hand side
044  * panel of the ontology editor.
045  * 
046  * @author niraj
047  *

074   public int getRowCount() {
075     int count = detailGroups.length;

077       if(detailGroups[j].isExpanded()) {

112   public boolean isCellEditable(int rowIndex, int columnIndex) {
113     if (columnIndex != VALUE_COLUMN) { return false; }
114     Object object = getItemForRow(rowIndex);
115     if (!(object instanceof PropertyValue)) { return false; }
116     RDFProperty property = ((PropertyValue) object).getProperty();
117     return property instanceof AnnotationProperty
118         || property instanceof DatatypeProperty;
119   }

121   public void setValueAt(Object value, int row, int col) {
122     Object object = getItemForRow(row);
123     switch (col) {
124       case EXPANDED_COLUMN:
125         if (object instanceof DetailsGroup) {
126           DetailsGroup detailsgroup = (DetailsGroup) object;
127           detailsgroup.setExpanded((Boolean) value);
128         }
129         break;
130       case VALUE_COLUMN:
131         if (object instanceof PropertyValue) {
132           PropertyValue propertyValue = (PropertyValue) object;
133           RDFProperty property = propertyValue.getProperty();
134           if (property instanceof AnnotationProperty) {
135             // update the ontology
136             oResource.removeAnnotationPropertyValue((AnnotationProperty)
137               property, (Literal) propertyValue.getValue());
138             oResource.addAnnotationPropertyValue((AnnotationProperty)
139               property, new Literal((String) value));
140             // update the data structure for this table
141             setItemForRow(row, new PropertyValue(
142               property, new Literal((String) value)));
143           } else if (property instanceof DatatypeProperty) {
144             boolean isValidValue = ((DatatypeProperty)property)
145               .getDataType().isValidValue((String) value);
146             if (!isValidValue) {
147               JOptionPane.showMessageDialog(MainFrame.getInstance(),
148                 "Incompatible value: " + value +
149                 "\nUse a value of type " + ((DatatypeProperty)property)
150                 .getDataType().getXmlSchemaURIString()
151                   .replaceFirst("http://www.w3.org/2001/XMLSchema#", ""));
152               return;
153             }
154             try {
155               ((OInstance)oResource).removeDatatypePropertyValue(
156                 (DatatypeProperty)property, (Literal)propertyValue.getValue());
157               ((OInstance)oResource).addDatatypePropertyValue(
158                 (DatatypeProperty)property, new Literal((String) value,
159                   ((DatatypeProperty)property).getDataType()));
160               setItemForRow(row, new PropertyValue(
161                 property, new Literal((String) value)));
162             } catch(InvalidValueException e) {
163               JOptionPane.showMessageDialog(MainFrame.getInstance(),
164                       "Incompatible value");
165               e.printStackTrace();
166               return;
167             }
168           }
169         }
170         break;
171     }
172     // redraw the table
173     fireTableDataChanged();
174   }

176   protected Object getItemForRow(int row) {
177     int groupRow = 0;
178     for (int groupIndex = 0; groupRow <= row; groupIndex++) {
179       if (groupRow == row) {
180         return detailGroups[groupIndex];
181       }
182       int groupSize = 1 + (detailGroups[groupIndex].isExpanded() ?
183         detailGroups[groupIndex].getSize() : 0);
184       if (groupRow + groupSize > row) {
185         return detailGroups[groupIndex].getValueAt(row - groupRow - 1);
186       }
187       groupRow += groupSize;
188     }
189     return null;
190   }
191 
192   protected void setItemForRow(int row, Object value) {
193     int groupRow = 0;
194     for (int groupIndex = 0; groupRow <= row; groupIndex++) {
195       if (groupRow == row) {
196         detailGroups[groupIndex].setExpanded((Boolean) value);
197         return;
198       }
199       int groupSize = 1 + (detailGroups[groupIndex].isExpanded() ?
200         detailGroups[groupIndex].getSize() : 0);
201       if (groupRow + groupSize > row) {
202         detailGroups[groupIndex].setValueAt(row - groupRow - 1, value);
203         return;
204       }
205       groupRow += groupSize;
206     }
207   }

209   public Object getValueAt(int row, int col) {
210     Object object = getItemForRow(row);
211     switch(col) {
212       case EXPANDED_COLUMN:
213         return (object instanceof DetailsGroup) ?
214                  ((DetailsGroup) object).getSize() > 0 ?
215                    ((DetailsGroup) object).isExpanded() ?
216                      "expanded" : "closed" : "empty" : null;
217       case LABEL_COLUMN:
218         return object;
219       case VALUE_COLUMN:
220         if (object instanceof DetailsGroup) {
221           return "";
222         } else if (object instanceof PropertyValue) {
223           PropertyValue property = (PropertyValue) object;
224           if(property.getValue() instanceof Literal) {
225             return ((Literal)property.getValue()).getValue();
226           } else {
227             return property.getValue().toString();
228           }
229         } else if (object instanceof KeyValuePair) {
230             KeyValuePair keyValuePair = (KeyValuePair) object;
231             return keyValuePair.getValue().toString();
232         } else if (object instanceof RDFProperty) {
233           RDFProperty property = (RDFProperty) object;
234           if (property instanceof DatatypeProperty) {
235             return ((DatatypeProperty)property).getDataType()
236               .getXmlSchemaURIString();
237           } else if (!(property instanceof AnnotationProperty)) {
238             Set<OResource> set = property.getRange();
239             if (set == null || set.isEmpty()) {
240               return "[ALL CLASSES]";
241             } else {
242               String s = "[";
243               boolean firstTime = true;
244               for(OResource res : set) {
245                 if (!firstTime) {
246                   s += ",";
247                 } else {
248                   firstTime = false;
249                 }
250                 s += res.getName();
251               }
252               s += "]";
253               return s;
254             }
255           } else {
256             return "[ALL RESOURCES]";
257           }
258         } else {
259           return object.toString();
260         }
261       case DELETE_COLUMN:
262         return object;
263     }
264     return null;
265   }

267   public void setItem(OResource oResource) {
268     this.oResource = oResource;
269     if(oResource instanceof OClass) {
270       detailGroups = new DetailsGroup[] {resourceInfo, directSuperClasses, allSuperClasses,
271           directSubClasses, allSubClasses, equivalentClasses, propertyTypes,
272           propertyValues, instances};
273       OClass tclass = (OClass)oResource;
274       resourceInfo.getValues().clear();
275       if(tclass instanceof Restriction) {
276           resourceInfo.getValues().addAll(Utils.getDetailsToAdd(tclass));
277       } else {
278         resourceInfo.getValues().add(tclass);
279         resourceInfo.getValues().add(new KeyValuePair(tclass, "URI", tclass.getONodeID().toString(), false));
280         resourceInfo.getValues().add(new KeyValuePair(tclass, "TYPE", "Ontology Class", false));
281       }
282       
283       // direct super classes
284       Set<OClass> set = tclass.getSuperClasses(OConstants.Closure.DIRECT_CLOSURE);
285       directSuperClasses.getValues().clear();
286       directSuperClasses.getValues().addAll(set);
287       Collections.sort(directSuperClasses.getValues(), itemComparator);
288 
289       // all super classes
290       Set<OClass> set1 = tclass.getSuperClasses(OConstants.Closure.TRANSITIVE_CLOSURE);
291       allSuperClasses.getValues().clear();
292       allSuperClasses.getValues().addAll(set1);
293       Collections.sort(allSuperClasses.getValues(), itemComparator);
294       
295 
296       // direct subclasses
297       Set<OClass> set2 = tclass.getSubClasses(OConstants.Closure.DIRECT_CLOSURE);
298       directSubClasses.getValues().clear();
299       directSubClasses.getValues().addAll(set2);
300       Collections.sort(directSubClasses.getValues(), itemComparator);
301       
302       // all sub classes
303       Set<OClass> set3 = tclass.getSubClasses(OConstants.Closure.TRANSITIVE_CLOSURE);
304       allSubClasses.getValues().clear();
305       allSubClasses.getValues().addAll(set3);
306       Collections.sort(allSubClasses.getValues(), itemComparator);
307 
308       // equivalent classes
309       Set<OClass> set4 = tclass.getEquivalentClasses();
310       equivalentClasses.getValues().clear();
311       equivalentClasses.getValues().addAll(set4);
312       Collections.sort(equivalentClasses.getValues(), itemComparator);
313       
314       // properties with resource as domain
315       propertyTypes.getValues().clear();
316       Set<RDFProperty> dprops = tclass.getPropertiesWithResourceAsDomain();
317       propertyTypes.getValues().addAll(dprops);
318       Collections.sort(propertyTypes.getValues(), itemComparator);
319       
320 
321       // annotation property values
322       propertyValues.getValues().clear();
323       Set<AnnotationProperty> props = tclass.getSetAnnotationProperties();
324       if(props != null) {
325         Iterator<AnnotationProperty> apIter = props.iterator();
326         while(apIter.hasNext()) {
327           AnnotationProperty ap = apIter.next();
328           List<Literal> literals = tclass.getAnnotationPropertyValues(ap);
329           for(int i = 0; i < literals.size(); i++) {
330             PropertyValue pv = new PropertyValue(ap, literals.get(i));
331             propertyValues.getValues().add(pv);
332           }
333         }
334       }
335 
336       // instances
337       Set<OInstance> set5 = ontology.getOInstances(tclass,
338               OConstants.Closure.DIRECT_CLOSURE);
339       instances.getValues().clear();
340       if(set5 != null) {
341         instances.getValues().addAll(set5);
342         Collections.sort(instances.getValues(), itemComparator);
343       }
344     }
345     else if(oResource instanceof OInstance) {
346       OInstance oinstance = (OInstance)oResource;
347       detailGroups = (new DetailsGroup[] {resourceInfo, directTypes, allTypes,
348           sameAsInstances, propertyTypes, propertyValues});
349 
350       resourceInfo.getValues().clear();
351       resourceInfo.getValues().add(oinstance);
352       resourceInfo.getValues().add(new KeyValuePair(oinstance, "URI", oinstance.getOURI().toString(), false));
353       resourceInfo.getValues().add(new KeyValuePair(oinstance, "TYPE", "Ontology Instance", false));
354       
355       // direct classes
356       Set<OClass> set1 = oinstance.getOClasses(OConstants.Closure.DIRECT_CLOSURE);
357       directTypes.getValues().clear();
358       if(set1 != null) {
359         for(OClass aClass : set1) {
360           directTypes.getValues().addAll(Utils.getDetailsToAdd(aClass));
361         }
362       }
363 
364       // all classes
365       Set<OClass> set2 = oinstance.getOClasses(OConstants.Closure.TRANSITIVE_CLOSURE);
366       allTypes.getValues().clear();
367       if(set2 != null) {
368         for(OClass aClass : set2) {
369           allTypes.getValues().addAll(Utils.getDetailsToAdd(aClass));
370         }
371       }
372 
373       Set<OInstance> set3 = oinstance.getSameInstance();
374       sameAsInstances.getValues().clear();
375       if(set3 != null) {
376         sameAsInstances.getValues().addAll(set3);
377         Collections.sort(sameAsInstances.getValues(), itemComparator);
378       }
379 
380       propertyTypes.getValues().clear();
381       Set<RDFProperty> dprops = oinstance.getPropertiesWithResourceAsDomain();
382       propertyTypes.getValues().addAll(dprops);
383       
384       propertyValues.getValues().clear();
385       Set<AnnotationProperty> apProps = oinstance.getSetAnnotationProperties();
386       Set<DatatypeProperty> dtProps = oinstance.getSetDatatypeProperties();
387       Set<ObjectProperty> obProps = oinstance.getSetObjectProperties();
388       Set<RDFProperty> rdfProp = oinstance.getSetRDFProperties();
389 
390       for(AnnotationProperty ap : apProps) {
391         List<Literal> literals = oinstance.getAnnotationPropertyValues(ap);
392         for(int i = 0; i < literals.size(); i++) {
393           PropertyValue pv = new PropertyValue(ap, literals.get(i));
394           propertyValues.getValues().add(pv);
395         }
396       }
397 
398       for(DatatypeProperty dt : dtProps) {
399         List<Literal> literals = oinstance.getDatatypePropertyValues(dt);
400         for(int i = 0; i < literals.size(); i++) {
401           PropertyValue pv = new PropertyValue(dt, literals.get(i));
402           propertyValues.getValues().add(pv);
403         }
404       }
405 
406       for(ObjectProperty ob : obProps) {
407         List<OInstance> oinstances = oinstance.getObjectPropertyValues(ob);
408         for(int i = 0; i < oinstances.size(); i++) {
409           PropertyValue pv = new PropertyValue(ob, oinstances.get(i));
410           propertyValues.getValues().add(pv);
411         }
412       }
413 
414       for(RDFProperty rd : rdfProp) {
415         List<OValue> oinstances = oinstance.getRDFPropertyOValues(rd);
416         for(int i = 0; i < oinstances.size(); i++) {
417           PropertyValue pv = new PropertyValue(rd, oinstances.get(i));
418           propertyValues.getValues().add(pv);
419         }
420       }
421     }
422     fireTableDataChanged();
423   }
