016 package gate.jape.constraint;
017 
018 import java.util.Collection;
019 import java.util.Collections;
020 
021 import gate.Annotation;
022 import gate.AnnotationSet;
023 import gate.jape.Constraint;
024 import gate.jape.JapeException;
025 
026 /**
027  * Predicate whose {@link #getValue()} property may be set to a
028  * Constraint itself, allowing for recursive evaluations.
029  *
030  * @version $Revision$
031  * @author esword
032  */
033 public abstract class EmbeddedConstraintPredicate extends AbstractConstraintPredicate {
034 
035   protected Constraint valueConstraint;
036   protected String annotType;
037 
038   public EmbeddedConstraintPredicate() {
039     super();
040   }
041 
042   public EmbeddedConstraintPredicate(AnnotationAccessor accessor, Object value) {
043     super(accessor, value);
044   }
045 
046   /**
047    * Sets up environment for concrete class to do the specific matching check
048    */
049   public boolean doMatch(Object annotValue, AnnotationSet context)
050           throws JapeException {
051 
052     Annotation annot = (Annotation)annotValue;
053     AnnotationSet containedSet = doMatch(annot, context);
054 
055     Collection<Annotation> filteredSet = filterMatches(containedSet);
056 
057     return !filteredSet.isEmpty();
058   }
059 
060   protected abstract AnnotationSet doMatch(Annotation annot, AnnotationSet as);
061 
062   /**
063    * If there are attribute constraints, filter the set.
064    */
065   protected Collection<Annotation> filterMatches(AnnotationSet containedSet) {
066     if (containedSet == null)
067       return Collections.emptySet();
068 
069     if (valueConstraint == null || containedSet.isEmpty()) {
070       return containedSet;
071     }
072     else {
073       return valueConstraint.matches(containedSet, null, containedSet);
074     }
075   }
076 
077   /**
078    * If the given value is a {@link Constraint}, then check if there
079    * are any additional attribute/feature-checks on the constraint. If
080    * so, then store the constraint for use during matching calls. If
081    * not, then only the annotation type for the constraint is stored
082    * since the full constraint is not needed.
083    */
084   @Override
085   public void setValue(Object v) {
086     if(v instanceof Constraint) {
087       Constraint c = (Constraint)v;
088       annotType = c.getAnnotType();
089       if(!c.getAttributeSeq().isEmpty()) {
090         // store full constraint for later use. It's stored in the
091         // main value object for toString purposes.
092         valueConstraint = c;
093         value = c;
094       }
095     }
096 
097     // if the given value is not a constraint, then just store it
098     // directly as the annotationType
099     if(annotType == null && valueConstraint == null) {
100       value = v;
101       annotType = String.valueOf(v);
102     }
103   }
104 
105   public String getAnnotType() {
106     return annotType;
107   }
108 
109   @Override
110   public String toString() {
111     StringBuilder str = new StringBuilder();
112     str.append(getOperator()).append(" ");
113     if(annotType != null)  str.append(annotType);
114     if(valueConstraint != null){
115       str.append(": ").append(valueConstraint);
116     }
117     return str.toString();
118   }
119   
120   @Override
121   public int hashCode() {
122     final int prime = 31;
123     int result = super.hashCode();
124     result = prime * result + ((annotType == null) ? 0 : annotType.hashCode());
125     result =
126         prime * result
127             + ((valueConstraint == null) ? 0 : valueConstraint.hashCode());
128     return result;
129   }
130 
131   @Override
132   public boolean equals(Object obj) {
133     if(this == obj) return true;
134     if(!super.equals(obj)) return false;
135     if(!(obj instanceof EmbeddedConstraintPredicate)) return false;
136     EmbeddedConstraintPredicate other = (EmbeddedConstraintPredicate)obj;
137     if(annotType == null) {
138       if(other.annotType != null) return false;
139     } else if(!annotType.equals(other.annotType)) return false;
140     if(valueConstraint == null) {
141       if(other.valueConstraint != null) return false;
142     } else if(!valueConstraint.equals(other.valueConstraint)) return false;
143     return true;
144   }
