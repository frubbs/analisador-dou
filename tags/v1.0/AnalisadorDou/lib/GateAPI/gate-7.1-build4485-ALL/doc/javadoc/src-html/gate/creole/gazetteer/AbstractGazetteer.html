018 import java.util.*;
019 
020 import gate.FeatureMap;
021 import gate.creole.ResourceInstantiationException;
022 import gate.creole.metadata.CreoleParameter;
023 import gate.creole.metadata.CreoleResource;

027 /**AbstractGazetteer
028  * This class implements the common-for-all methods of the Gazetteer interface*/
029 public abstract class AbstractGazetteer
030   extends gate.creole.AbstractLanguageAnalyser implements Gazetteer {
031 
032   /** the set of gazetteer listeners */
033   protected Set listeners = new HashSet();
034 
035   /** Used to store the annotation set currently being used for the newly
036    * generated annotations*/
037   protected String annotationSetName;
038 
039   /** the encoding of the gazetteer */
040   protected String encoding = "UTF-8";
041 
042   /**
043    * The value of this property is the URL that will be used for reading the
044    * lists that define this Gazetteer
045    */
046   protected java.net.URL listsURL;
047 
048   /**
049    * Should this gazetteer be case sensitive. The default value is true.
050    */
051   protected Boolean caseSensitive = new Boolean(true);
052 
053   /**
054    * Should this gazetteer only match whole words. The default value is
055    * <tt>true</tt>.
056    */
057   protected Boolean wholeWordsOnly = new Boolean(true);
058 
059   /**
060    * Should this gazetteer only match the longest string starting from any 
061    * offset? This parameter is only relevant when the list of lookups contains
062    * proper prefixes of other entries (e.g when both &quot;Dell&quot; and 
063    * &quot;Dell Europe&quot; are in the lists). The default behaviour (when this
064    * parameter is set to <tt>true</tt>) is to only match the longest entry, 
065    * &quot;Dell Europe&quot; in this example. This is the default GATE gazetteer
066    * behaviour since version 2.0. Setting this parameter to <tt>false</tt> will 
067    * cause the gazetteer to match all possible prefixes.
068    */
069   protected Boolean longestMatchOnly = new Boolean(true);
070   
071   /** the linear definition of the gazetteer */
072   protected LinearDefinition definition;
073 
074   /** reference to mapping definition info
075    *  allows filling of Lookup.ontologyClass according to a list*/
076   protected MappingDefinition mappingDefinition;
077 
078 
079   /**
080    * Sets the AnnotationSet that will be used at the next run for the newly
081    * produced annotations.
082    */
083   @RunTime
084   @Optional
085   @CreoleParameter(comment="The annotation set to be used for the generated annotations")
086   public void setAnnotationSetName(String newAnnotationSetName) {
087     annotationSetName = newAnnotationSetName;
088   }
089 
090   /**
091    * Gets the AnnotationSet that will be used at the next run for the newly
092    * produced annotations.
093    */
094   public String getAnnotationSetName() {
095     return annotationSetName;
096   }
097 
098   @CreoleParameter(comment="The encoding used for reading the definitions", defaultValue="UTF-8")
099   public void setEncoding(String newEncoding) {
100     encoding = newEncoding;
101   }
102 
103   public String getEncoding() {
104     return encoding;
105   }
106 
107   public java.net.URL getListsURL() {
108     return listsURL;
109   }
110 
111   @CreoleParameter(comment="The URL to the file with list of lists", suffixes="def", defaultValue="resources/gazetteer/lists.def")
112   public void setListsURL(java.net.URL newListsURL) {
113     listsURL = newListsURL;
114   }
115 
116   @CreoleParameter(comment="Should this gazetteer diferentiate on case?", defaultValue="true")
117   public void setCaseSensitive(Boolean newCaseSensitive) {
118     caseSensitive = newCaseSensitive;
119   }
120 
121   public Boolean getCaseSensitive() {
122     return caseSensitive;
123   }
124 
125   public void setMappingDefinition(MappingDefinition mapping) {
126     mappingDefinition = mapping;
127   }
128 
129   public MappingDefinition getMappingDefinition(){
130     return mappingDefinition;
131   }
132   
133   /**
134    * @return the longestMatchOnly
135    */
136   public Boolean getLongestMatchOnly() {
137     return longestMatchOnly;
138   }
139 
140   /**
141    * @param longestMatchOnly the longestMatchOnly to set
142    */
143   @RunTime
144   @CreoleParameter(comment="Should this gazetteer only match the longest string starting from any offset?", defaultValue="true")
145   public void setLongestMatchOnly(Boolean longestMatchOnly) {
146     this.longestMatchOnly = longestMatchOnly;
147   }
148 
149   /**Gets the linear definition of this gazetteer. there is no parallel
150    * set method because the definition is loaded through the listsUrl
151    * on init().
152    * @return the linear definition of the gazetteer */
153   public LinearDefinition getLinearDefinition() {
154     return definition;
155   }
156 
157   public void reInit() throws ResourceInstantiationException {
158     super.reInit();
159     fireGazetteerEvent(new GazetteerEvent(this,GazetteerEvent.REINIT));
160   }//reInit()
161 
162   /**
163    * fires a Gazetteer Event
164    * @param ge Gazetteer Event to be fired
165    */
166   public void fireGazetteerEvent(GazetteerEvent ge) {
167     Iterator li = listeners.iterator();
168     while ( li.hasNext()) {
169       GazetteerListener gl = (GazetteerListener) li.next();
170       gl.processGazetteerEvent(ge);
171     }
172   }
173 
174   /**
175    * Registers a Gazetteer Listener
176    * @param gl Gazetteer Listener to be registered
177    */
178   public void addGazetteerListener(GazetteerListener gl){
179     if ( null!=gl )
180       listeners.add(gl);
181   }
182 
183   /**
184    * Gets the value for the {@link #wholeWordsOnly} parameter.
185    * @return a Boolean value.
186    */
187   public Boolean getWholeWordsOnly() {
188     return wholeWordsOnly;
189   }
190 
191   /**
192    * Sets the value for the {@link #wholeWordsOnly} parameter.
193    * @param wholeWordsOnly a Boolean value.
194    */
195   @RunTime
196   @CreoleParameter(comment="Should this gazetteer only match whole words?", defaultValue="true")
197   public void setWholeWordsOnly(Boolean wholeWordsOnly) {
198     this.wholeWordsOnly = wholeWordsOnly;
199   }
200 
201 }//class AbstractGazetteer
