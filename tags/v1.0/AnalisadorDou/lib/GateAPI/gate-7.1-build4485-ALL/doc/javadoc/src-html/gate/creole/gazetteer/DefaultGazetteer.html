044 /** This component is responsible for doing lists lookup. The implementation is
045  * based on finite state machines.
046  * The phrases to be recognised should be listed in a set of files, one for
047  * each type of occurrences.
048  * The gazetteer is build with the information from a file that contains the set
049  * of lists (which are files as well) and the associated type for each list.
050  * The file defining the set of lists should have the following syntax:
051  * each list definition should be written on its own line and should contain:
052  * <ol>
053  * <li>the file name (required) </li>
054  * <li>the major type (required) </li>
055  * <li>the minor type (optional)</li>
056  * <li>the language(s) (optional) </li>
057  * </ol>
058  * The elements of each definition are separated by &quot;:&quot;.
059  * The following is an example of a valid definition: <br>
060  * <code>personmale.lst:person:male:english</code>
061  * Each list file named in the lists definition file is just a list containing
062  * one entry per line.
063  * When this gazetteer will be run over some input text (a Gate document) it
064  * will generate annotations of type Lookup having the attributes specified in
065  * the definition file.
066  */

139   /** Reads one lists (one file) of phrases
140    *

142    * @param add if <b>true</b> will add the phrases found in the list to the ones

144    *     list will be removed from the list of phrases recognised by this
145    *     gazetteer.
146    */
147    protected void readList(LinearNode node, boolean add) 
148        throws ResourceInstantiationException{
149     String listName, majorType, minorType, languages,annotationType;
150     if ( null == node ) {
151       throw new ResourceInstantiationException(" LinearNode node is null ");
152     }
153 
154     listName = node.getList();
155     majorType = node.getMajorType();
156     minorType = node.getMinorType();
157     languages = node.getLanguage();
158     annotationType = node.getAnnotationType();
159     GazetteerList gazList = (GazetteerList)listsByNode.get(node);
160     if (null == gazList) {
161       throw new ResourceInstantiationException("gazetteer list not found by node");
162     }
163 
164     Iterator iline = gazList.iterator();
165     
166     // create default lookup for entries with no arbitrary features
167     Lookup defaultLookup = new Lookup(listName,majorType, minorType, languages,annotationType);
168     defaultLookup.list = node.getList();
169     if ( null != mappingDefinition){
170       MappingNode mnode = mappingDefinition.getNodeByList(defaultLookup.list);
171       if (null!=mnode){
172         defaultLookup.oClass = mnode.getClassID();
173         defaultLookup.ontology = mnode.getOntologyID();
174       }
175     }//if mapping def
176     
177     Lookup lookup;
178     String entry; // the actual gazetteer entry text
179     while(iline.hasNext()){
180       GazetteerNode gazNode = (GazetteerNode)iline.next();
181       entry = gazNode.getEntry();
182       
183       Map features = gazNode.getFeatureMap();
184       if (features == null) {
185         lookup = defaultLookup;
186       } else {
187         // create a new Lookup object with features
188         lookup = new Lookup(listName, majorType, minorType, languages,annotationType);
189         lookup.list = node.getList();
190         if(null != mappingDefinition) {
191           MappingNode mnode = mappingDefinition.getNodeByList(lookup.list);
192           if(null != mnode) {
193             lookup.oClass = mnode.getClassID();
194             lookup.ontology = mnode.getOntologyID();
195           }
196         }// if mapping def
197         lookup.features = features;
198       } 
199       
200       if(add)addLookup(entry, lookup);
201       else removeLookup(entry, lookup);
202     }

208    * @param lookup the description of the annotation to be added when this
209    *     phrase is recognised
210    */
211   public void addLookup(String text, Lookup lookup) {
212     char currentChar;
213     FSMState currentState = initialState;
214     FSMState nextState;
215     Lookup oldLookup;
216     boolean isSpace;
217 
218     for(int i = 0; i< text.length(); i++) {
219         currentChar = text.charAt(i);
220         isSpace = Character.isSpaceChar(currentChar) || Character.isWhitespace(currentChar);
221         if(isSpace) currentChar = ' ';
222         else currentChar = (caseSensitive.booleanValue()) ?
223                           currentChar :
224                           Character.toUpperCase(currentChar) ;
225       nextState = currentState.next(currentChar);
226       if(nextState == null){
227         nextState = new FSMState(this);
228         currentState.put(currentChar, nextState);
229         if(isSpace) nextState.put(' ',nextState);
230       }
231       currentState = nextState;
232     } //for(int i = 0; i< text.length(); i++)
233 
234     currentState.addLookup(lookup);
235     //Out.println(text + "|" + lookup.majorType + "|" + lookup.minorType);
236 
237   } // addLookup

300   /**
301    * This method runs the gazetteer. It assumes that all the needed parameters
302    * are set. If they are not, an exception will be fired.
303    */
304   public void execute() throws ExecutionException{
305     interrupted = false;
306     AnnotationSet annotationSet;
307     //check the input
308     if(document == null) {
309       throw new ExecutionException(
310         "No document to process!"
311       );
312     }
313 
314     if(annotationSetName == null ||
315        annotationSetName.equals("")) annotationSet = document.getAnnotations();
316     else annotationSet = document.getAnnotations(annotationSetName);
317 
318     fireStatusChanged("Performing look-up in " + document.getName() + "...");
319     String content = document.getContent().toString();
320     int length = content.length();
321     char currentChar;
322     FSMState currentState = initialState;
323     FSMState nextState;
324     FSMState lastMatchingState = null;
325     int matchedRegionEnd = 0;
326     int matchedRegionStart = 0;
327     int charIdx = 0;
328     int oldCharIdx = 0;
329     FeatureMap fm;
330     Lookup currentLookup;
331 
332     while(charIdx < length) {
333       currentChar = content.charAt(charIdx);
334       if( Character.isSpaceChar(currentChar) || Character.isWhitespace(currentChar) ) currentChar = ' ';
335       else currentChar = caseSensitive.booleanValue() ?
336                           currentChar :
337                           Character.toUpperCase(currentChar);
338       nextState = currentState.next(currentChar);
339       if(nextState == null) {
340         //the matching stopped
341         //if we had a successful match then act on it;
342         if(lastMatchingState != null){
343           createLookups(lastMatchingState, matchedRegionStart, matchedRegionEnd, 
344                   annotationSet);
345           lastMatchingState = null;
346         }
347         //reset the FSM
348         charIdx = matchedRegionStart + 1;
349         matchedRegionStart = charIdx;
350         currentState = initialState;
351       } else{//go on with the matching
352         currentState = nextState;
353         //if we have a successful state then store it
354         if(currentState.isFinal() &&
355            (
356             (!wholeWordsOnly.booleanValue())
357              ||
358             ((matchedRegionStart == 0 ||
359              !isWordInternal(content.charAt(matchedRegionStart - 1)))
360              &&
361              (charIdx + 1 >= content.length()   ||
362              !isWordInternal(content.charAt(charIdx + 1)))
363             )
364            )
365           ){
366           //we have a new match
367           //if we had an existing match and we need to annotate prefixes, then 
368           //apply it
369           if(!longestMatchOnly && lastMatchingState != null){
370             createLookups(lastMatchingState, matchedRegionStart, 
371                     matchedRegionEnd, annotationSet);
372           }
373           matchedRegionEnd = charIdx;
374           lastMatchingState = currentState;
375         }
376         charIdx ++;
377         if(charIdx == content.length()){
378           //we can't go on, use the last matching state and restart matching
379           //from the next char
380           if(lastMatchingState != null){
381             //let's add the new annotation(s)
382             createLookups(lastMatchingState, matchedRegionStart, 
383                     matchedRegionEnd, annotationSet);
384             lastMatchingState = null;
385           }
386           //reset the FSM
387           charIdx = matchedRegionStart + 1;
388           matchedRegionStart = charIdx;
389           currentState = initialState;
390         }
391       }
392       //fire the progress event
393       if(charIdx - oldCharIdx > 256) {
394         fireProgressChanged((100 * charIdx )/ length );
395         oldCharIdx = charIdx;
396         if(isInterrupted()) throw new ExecutionInterruptedException(
397             "The execution of the " + getName() +
398             " gazetteer has been abruptly interrupted!");
399       }
400     } // while(charIdx < length)
401     //we've finished. If we had a stored match, then apply it.
402     if(lastMatchingState != null) {
403       createLookups(lastMatchingState, matchedRegionStart, 
404               matchedRegionEnd, annotationSet);
405     }
406     fireProcessFinished();
407     fireStatusChanged("Look-up complete!");
408   } // execute

412    * Creates the Lookup annotations according to a gazetteer match.
413    * @param matchingState the final FSMState that was reached while matching. 
414    * @param matchedRegionStart the start of the matched text region.
415    * @param matchedRegionEnd the end of the matched text region.

419   protected void createLookups(FSMState matchingState, long matchedRegionStart,

426       if (null!= currentLookup.oClass && null!=currentLookup.ontology){
427         fm.put(LOOKUP_CLASS_FEATURE_NAME,currentLookup.oClass);
428         fm.put(LOOKUP_ONTOLOGY_FEATURE_NAME,currentLookup.ontology);
429       }
430 
431       if(null != currentLookup.minorType)
432         fm.put(LOOKUP_MINOR_TYPE_FEATURE_NAME, currentLookup.minorType);
433       if(null != currentLookup.languages)
434         fm.put(LOOKUP_LANGUAGE_FEATURE_NAME, currentLookup.languages);      
435       if(null != currentLookup.features) {
436         fm.putAll(currentLookup.features);
437       }
438       try{
439 //        if(currentLookup.annotationType==null || "".equals(currentLookup.annotationType)){
440 //          annotationSet.add(new Long(matchedRegionStart),
441 //                          new Long(matchedRegionEnd + 1),
442 //                          LOOKUP_ANNOTATION_TYPE,
443 //                          fm);
444 //        }else{
445           annotationSet.add(new Long(matchedRegionStart),
446                           new Long(matchedRegionEnd + 1),
447                           currentLookup.annotationType, //this pojo attribute will have Lookup as a default tag.
448                           fm);
449        // }
450       } catch(InvalidOffsetException ioe) {
451         throw new GateRuntimeException(ioe.toString());
452       }
453     }//while(lookupIter.hasNext())
454   }

487     char currentChar;
488     FSMState currentState = initialState;
489     FSMState nextState;
490     Lookup oldLookup;
491 
492     for(int i = 0; i< singleItem.length(); i++) {

531   /**
532    * class implementing the map using binary search by char as key
533    * to retrieve the corresponding object.
534    */
535   public static class CharMap implements Serializable
536   {
537       char[] itemsKeys = null;
538       Object[] itemsObjs = null;
539 
540       /**
541        * resize the containers by one, leaving empty element at position 'index'
542        */
543       void resize(int index)
544       {
545           int newsz = itemsKeys.length + 1;
546           char[] tempKeys = new char[newsz];
547           Object[] tempObjs = new Object[newsz];
548           System.arraycopy(itemsKeys, 0, tempKeys, 0, index);
549           System.arraycopy(itemsObjs, 0, tempObjs, 0, index);
550           System.arraycopy(itemsKeys, index, tempKeys, index + 1, newsz - index - 1);
551           System.arraycopy(itemsObjs, index, tempObjs, index + 1, newsz - index - 1);
552 
553           itemsKeys = tempKeys;
554           itemsObjs = tempObjs;
555       } // resize
556 
557   /**
558    * get the object from the map using the char key
559    */
560       Object get(char key)
561       {
562           if (itemsKeys == null) return null;
563           int index = Arrays.binarySearch(itemsKeys, key);
564           if (index<0)
565               return null;
566           return itemsObjs[index];
567       }
568   /**
569    * put the object into the char map using the char as the key
570    */
571       Object put(char key, Object value)
572       {
573           if (itemsKeys == null)
574           {
575               itemsKeys = new char[1];
576               itemsKeys[0] = key;
577               itemsObjs = new Object[1];
578               itemsObjs[0] = value;
579               return value;
580           }// if first time
581           int index = Arrays.binarySearch(itemsKeys, key);
582           if (index<0)
583           {
584               index = ~index;
585               resize(index);
586               itemsKeys[index] = key;
587               itemsObjs[index] = value;
588           }
589           return itemsObjs[index];
590       } // put
