32   public ListSelectionModel getSelectionModel();
33   
34   /**
35    * Provides the annotation 
36    */
37   public AnnotationData getAnnotationAtRow(int row);
38 
39   /**

41    * @param aData the annotation for which the row is required
42    * @return a positive int value if the annotation is found or -1 otherwise 
43    */
44   public int getRowForAnnotation(AnnotationData aData);
45 }
