168   /** This method is intended to be used in conjunction with
169     * getSelectedParameters(). The method will not instantiate the resource
170     * like {@link #show(ResourceData)} but it is intended to collect the params
171     * required to instantiate a resource. Returns true if the user pressed OK
172     * and false otherwise.
173     */
174   public synchronized boolean show(ResourceData rData, String aTitle) {
175     this.resourceData = rData;
176     if (aTitle != null) setTitle(aTitle);
177     nameField.setText("");
178     parametersEditor.init(null,
179                           rData.getParameterList().getInitimeParameters());
180     pack();
181     nameField.requestFocusInWindow();
182     userCanceled = true;
183     setModal(true);
184     setLocationRelativeTo(getOwner());
185     super.setVisible(true);
186     dispose();
187     if(userCanceled) return false;
188     else return true;
189   }//show();

201   public synchronized void show(ResourceData rData) {
202     if(rData != null) {
203       this.resourceData = rData;
204       nameField.setText("");
205       parametersEditor.init(null,
206                             rData.getParameterList().getInitimeParameters());
207       pack();
208       setLocationRelativeTo(getOwner());
209     } else {
210       // dialog already populated
211     }
212     //default case when the dialog just gets closed 
213     userCanceled = true;
214     //show the dialog
215     setVisible(true);
216     if(userCanceled){
217       //release resources
218       dispose();
219       return;
220     } else {
221       Runnable runnable = new Runnable(){
222         public void run(){
223           //create the new resource
224           FeatureMap params = parametersEditor.getParameterValues();
225 
226           Resource res;
227           gate.event.StatusListener sListener =
228             (gate.event.StatusListener)Gate.getListeners().
229                                        get("gate.event.StatusListener");
230           if(sListener != null) sListener.statusChanged("Loading " +
231                                                         nameField.getText() +
232                                                         "...");
233 
234           gate.event.ProgressListener pListener =
235             (gate.event.ProgressListener)Gate.getListeners().
236                                          get("gate.event.ProgressListener");
237           if(pListener != null){
238             pListener.progressChanged(0);
239           }
240           boolean success = true;
241           try {
242             long startTime = System.currentTimeMillis();
243             FeatureMap features = Factory.newFeatureMap();
244             String name = nameField.getText();
245             if(name == null || name.length() == 0) name = null;
246             res = Factory.createResource(resourceData.getClassName(), params,
247                                          features, name);
248             long endTime = System.currentTimeMillis();
249             if(sListener != null) sListener.statusChanged(
250                 nameField.getText() + " loaded in " +
251                 NumberFormat.getInstance().format(
252                 (double)(endTime - startTime) / 1000) + " seconds");
253             if(pListener != null) pListener.processFinished();
254           } catch(ResourceInstantiationException rie){
255             success = false;
256             JOptionPane.showMessageDialog(getOwner(),
257                                           "Resource could not be created!\n" +
258                                           rie.toString(),
259                                           "GATE", JOptionPane.ERROR_MESSAGE);
260             rie.printStackTrace(Err.getPrintWriter());
261             res = null;
262             if(sListener != null) sListener.statusChanged("Error loading " +
263                                                           nameField.getText() +
264                                                           "!");
265             if(pListener != null) pListener.processFinished();
266           }catch(Throwable thr){
267             success = false;
268             JOptionPane.showMessageDialog(getOwner(),
269                     "Unhandled error!\n" +
270                     thr.toString(),
271                     "GATE", JOptionPane.ERROR_MESSAGE);
272             thr.printStackTrace(Err.getPrintWriter());
273             res = null;
274             if(sListener != null) sListener.statusChanged("Error loading " +
275                                                 nameField.getText() +
276                                                 "!");
277             if(pListener != null) pListener.processFinished();
278           } finally {
279             if(!success) {
280               // re-show the dialog, to allow the suer to correct the entry
281               SwingUtilities.invokeLater(new Runnable() {
282                 public void run() {
283                   show(null);    
284                 }
285               });
286             }
287             else {
288               dispose();
289             }
290           }
291         }//public void run()
292       };
293       Thread thread = new Thread(runnable, "");
294       thread.setPriority(Thread.MIN_PRIORITY);
295       thread.start();
296     }
297   }// public synchronized Resource show(ResourceData rData)

299   @Override
300   public void dispose() {
301     // when we have finished with the dialog release the data about the resource
302     // we were trying to create so that, at a later date, we can garbage collect
303     // the classloader if the plugin is unloaded
304     resourceData = null;
305   }
