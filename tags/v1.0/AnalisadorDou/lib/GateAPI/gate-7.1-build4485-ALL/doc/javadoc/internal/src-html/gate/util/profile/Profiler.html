019 /**
020  *  This is a simple "code-internal" profiler which can be used
021  *  by introducing calls to its methods (mostly, checkPoint()) in  the code.
022  *  It allows detection and reporting of time and memory usage
023  *  between execution points as well as for the whole run. In addition allows
024  *  for collection/reporting of the resources usage by categories (types of
025  *  activity performed between the check point and the previous one).
026  *  The results are currently just dumped to the output.
027  *  <p>
028  *  Important feature is the possibility of the memory usage measurment code
029  *  to call the Garbage Collector following a specific strategy that allows
030  *  the real memory usage to be deteceted. This feature can be switched OFF
031  *  if appropriate, for example, if the *real* run-time memory usage have to
032  *  be measured rather than the precise but ideallistic one provided otherwise.
033  *  <p>
034  *  The profiler can be switched OFF using the enable(boolean) method. In such
035  *  case it ignores any calls to its checkPoint(...) method, so, basically does
036  *  nothing. This feature is usefull in cases when the profiler have to be
037  *  "stripped" from the code in order to allow non-disturbed execution.
038  *  <p>
039  *  The profiler takes care to calculate the time exlusive its footprint. It
040  *  takes account for both "net" time and the absolute duration. Also in
041  *  relation to the execution time measurment - it depends on the speed of the
042  *  machine as well as on the other applications running in parallel.
043  */

181   /**
182    * To be called at all execution points of interest. Detects the time

184    * execution point

190   /**
191    * In addition to the variant of the method with two parameters allows:
192    * a set of categories (identified by strings) to which the preceeding
193    * fragment of code belongs; flag determining whether the description of
194    * the execution point to be displayed; flag determining whether the
195    * statistics to be shown
196    */
197   public String checkPoint(String execPointDescr, String categories[],
198       boolean showDescr, boolean showStats, boolean memoryCheck)
199   {
200     if (!m_enabled)
201       return "";
202 
203     long currTime = System.currentTimeMillis() - m_startTime;
204     m_lastDuration = currTime - m_lastCheckTime;
205 
206     if (memoryCheck) {
207       long oldMemory = m_currMemory;
208 
209       if (m_garbageCollection) {
210         do {
211           m_currMemory = m_rt.totalMemory() - m_rt.freeMemory();
212           m_rt.gc();
213           try {
214             wait(300);
215           } catch(Exception e) {
216             //ignore this as it should never really happen
217           }
218           m_rt.gc();
219         } while (m_currMemory > m_rt.totalMemory() - m_rt.freeMemory());
220       }
221       else {
222         m_currMemory = m_rt.totalMemory() - m_rt.freeMemory();
223       }
224 
225       m_currMemory /= 1000;
226       m_maxMemory = Math.max(m_maxMemory, m_currMemory);
227       m_diffMemory = m_currMemory - oldMemory;
228     } // if (memoryCheck)
229 
230     m_lastCheckTime = System.currentTimeMillis() - m_startTime;
231     m_profilerTime += (m_lastCheckTime - currTime);
232 
233     checkCategories(categories);
234     return showResults(execPointDescr, showDescr, showStats);
235   } // checkPoint

237   private void checkCategories(String categs[]) {
238     int size = categs.length;
239     String categ;
240     long last, sum;
241     Long l;
242     for (int i=0; i<size; i++) {
243       categ = categs[i].toUpperCase();
244       l = (Long)m_categorySums.get(categ);
245       sum = (l==null) ? 0 : l.longValue();
246       sum += m_lastDuration;
247       m_categorySums.put(categ, new Long(sum));
248       m_categoryLasts.put(categ, new Long(m_lastDuration));
249     } // for
250   } // checkCategories

252   private String showResults(String execPointDescr, boolean showDescr,
253       boolean showStats)
254   {
255     StringBuffer buff = new StringBuffer(500);
256     if (showDescr) {
257       buff.append("---------LOG: ");
258       buff.append(execPointDescr);
259       buff.append("---------");
260     }
261 
262     if (showStats) {
263       buff.append("\nMemory: ");
264       buff.append(m_currMemory);
265       buff.append("k; change: ");
266       buff.append(m_diffMemory);
267       buff.append("k; max: ");
268       buff.append(m_maxMemory);
269       buff.append("k; Net time:   ");
270       buff.append(printTime(getNetRunTime()));
271       buff.append("; since prev. millisecs: ");
272       buff.append(m_lastDuration);
273 //            buff.append("; profiler time: ");
274 //            buff.append(printTime(m_profilerTime));
275 //            buff.append("; duration: ");
276 //            buff.append(printTime(m_lastCheckTime));
277     }
278 
279     if (buff.length() > 0 && m_doPrintToStdOut) {
280         log.debug(buff.toString());
281     }
282     return buff.toString();
283   } // showResults

323   /**
324    * An auxiliary routine printing time in "nnn.nns" format
325    */
326   public String printTime(long timeMillis) {
327     long round = timeMillis/1000;
328     long remaind = (timeMillis % 1000)/10;
329     StringBuffer buff = new StringBuffer(10);
330     buff.append(round);
331     buff.append(".");
332     buff.append(remaind);
333     buff.append("s");
334     return buff.toString();
335   } // printTime

337   /**
338    * An auxiliary routine printing in a string speed
339    */
340   public String printSpeed(long timeMillis,
341       double whatever, String whateverMeasure)
342   {
343     double speed1000 = (double) whatever/ timeMillis;
344     long round = (long)((double)speed1000*1000);
345     long remaind =  (long)(((double)speed1000*100000) - 100 * round);
346     StringBuffer buff = new StringBuffer(10);
347     buff.append(round);
348     buff.append(".");
349     buff.append(remaind);
350     buff.append(whateverMeasure);
351     buff.append("/s");
352     return buff.toString();
353   } // printTime
