0171   /**
0172    * This editor implementation is designed to enforce schema compliance. This
0173    * method will return <tt>false</tt> if the current annotation type does not
0174    * have a schema or if the features of the current annotation do not comply
0175    * with the schema.
0176    * 
0177    * @see gate.gui.annedit.OwnedAnnotationEditor#editingFinished()
0178    */
0179   public boolean editingFinished() {
0180     if(annotation == null) return true;
0181     // if the dialog is hidden, we've missed the train and we can't force
0182     // compliance for the old annotation any more. Just give up and
0183     // allow further editing
0184     if(!dialog.isVisible()) return true;
0185     if(!schemasByType.containsKey(annotation.getType())) return false;
0186     // we need to check that:
0187     // 1) all required features have values
0188     // 2) all features known by schema that have values, comply with the schema
0189     if(annotation == null) return true;
0190     AnnotationSchema aSchema = schemasByType.get(annotation.getType());
0191     if(aSchema.getFeatureSchemaSet() == null
0192         || aSchema.getFeatureSchemaSet().isEmpty()) {
0193       // known type but no schema restrictions -> OK
0194       return true;
0195     }
0196     FeatureMap annotationFeatures = annotation.getFeatures();
0197     Map<String, FeatureSchema> featureSchemaByName =
0198         new HashMap<String, FeatureSchema>();
0199     // store all the feature schemas, and check the required ones
0200     for(FeatureSchema aFeatureSchema : aSchema.getFeatureSchemaSet()) {
0201       featureSchemaByName.put(aFeatureSchema.getFeatureName(), aFeatureSchema);
0202       Object featureValue =
0203           annotationFeatures == null ? null : annotationFeatures
0204               .get(aFeatureSchema.getFeatureName());
0205       if(aFeatureSchema.isRequired() && featureValue == null) return false;
0206     }
0207     // check all the actual values for compliance
0208     for(Object featureName : annotationFeatures.keySet()) {
0209       Object featureValue = annotationFeatures.get(featureName);
0210       FeatureSchema fSchema = featureSchemaByName.get(featureName);
0211       if(fSchema != null) {
0212         // this is a schema feature
0213         if(fSchema.getFeatureValueClass().equals(Boolean.class)
0214             || fSchema.getFeatureValueClass().equals(Integer.class)
0215             || fSchema.getFeatureValueClass().equals(Short.class)
0216             || fSchema.getFeatureValueClass().equals(Byte.class)
0217             || fSchema.getFeatureValueClass().equals(Float.class)
0218             || fSchema.getFeatureValueClass().equals(Double.class)) {
0219           if(featureValue instanceof String) {
0220             // try to convert numbers
0221             try {
0222               if (fSchema.getFeatureValueClass().equals(Integer.class)) {
0223                 featureValue = Integer.valueOf((String) featureValue);
0224               } else if (fSchema.getFeatureValueClass().equals(Short.class)) {
0225                 featureValue = Short.valueOf((String) featureValue);
0226               } else if (fSchema.getFeatureValueClass().equals(Byte.class)) {
0227                 featureValue = Byte.valueOf((String) featureValue);
0228               } else if (fSchema.getFeatureValueClass().equals(Double.class)) {
0229                 featureValue = Double.valueOf((String) featureValue);
0230               } else if (fSchema.getFeatureValueClass().equals(Float.class)) {
0231                 featureValue = Float.valueOf((String) featureValue);
0232               }
0233               annotationFeatures.put(featureName, featureValue);
0234             } catch (NumberFormatException e) {
0235               // could not convert
0236               return false;
0237             }            
0238           } else if(!fSchema.getFeatureValueClass().isAssignableFrom(
0239               featureValue.getClass())) {
0240             // not a String, nor the exact correct class: invalid value type
0241             return false;
0242           }
0243         } else if(fSchema.getFeatureValueClass().equals(String.class)) {
0244           if(fSchema.getPermittedValues() != null
0245               && !fSchema.getPermittedValues().contains(featureValue)) {
0246             // invalid value
0247             return false;
0248           }
0249         }
0250       }
0251     }
0252     return true;
0253   }

0325           if(x < 0) x = 0;

0345         // correct position
0346         if(y > maxY) y = maxY;

0348         dialogLocation.move(x, y);

0350       } catch(BadLocationException ble) {

1016   protected void moveAnnotation(AnnotationSet set, Annotation oldAnnotation,
1017       Long newStartOffset, Long newEndOffset) throws InvalidOffsetException {
1018     // Moving is done by deleting the old annotation and creating a new one.
1019     // If this was the last one of one type it would mess up the gui which

1025       // create a clone of the annotation that will be deleted, to act as a

1028           set.add(oldAnnotation.getStartNode(), oldAnnotation.getStartNode(),
1029               oldAnnotation.getType(), oldAnnotation.getFeatures());
1030       tempAnn = set.get(tempAnnId);
1031     }
1032     Integer oldID = oldAnnotation.getId();
1033     set.remove(oldAnnotation);
1034     set.add(oldID, newStartOffset, newEndOffset, oldAnnotation.getType(),
1035         oldAnnotation.getFeatures());
1036     Annotation newAnn = set.get(oldID);
1037     // update the selection to the new annotation
1038     getOwner().selectAnnotation(new AnnotationDataImpl(set, newAnn));
1039     editAnnotation(newAnn, set);
1040     // remove the temporary annotation
1041     if(tempAnn != null) set.remove(tempAnn);
1042     owner.annotationChanged(newAnn, set, null);
1043   }
