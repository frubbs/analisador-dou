032 /**
033  * Stack of annotations in a JPanel.
034  * <br><br>
035  * To use, respect this order:<br><code>
036  * AnnotationStack stackPanel = new AnnotationStack(...);<br>
037  * stackPanel.set...(...);<br>
038  * stackPanel.clearAllRows();<br>
039  * stackPanel.addRow(...);<br>
040  * stackPanel.addAnnotation(...);<br>
041  * stackPanel.drawStack();</code>
042  */
043 public class AnnotationStack extends JPanel {
044 
045   public AnnotationStack() {
046     super();
047     init();
048   }

050   /**
051    * @param maxTextLength maximum number of characters for the text,
052    * if too long an ellipsis is added in the middle
053    * @param maxFeatureValueLength maximum number of characters
054    *  for a feature value
055    */
056   public AnnotationStack(int maxTextLength, int maxFeatureValueLength) {
057     super();
058     this.maxTextLength = maxTextLength;
059     this.maxFeatureValueLength = maxFeatureValueLength;
060     init();
061   }

073   /**
074    * Add a row to the annotation stack.
075    *
076    * @param set set name for the annotation, may be null
077    * @param type annotation type
078    * @param feature feature name, may be null
079    * @param lastColumnButton button at the end of the column, may be null
080    * @param shortcut replace the header of the row, may be null

121   /**
122    * Draw the annotation stack in a JPanel with a GridBagLayout.
123    */
124   public void drawStack() {
125 
126     // clear the panel
127     removeAll();
128 
129     boolean textTooLong = text.length() > maxTextLength;
130     int upperBound = text.length() - (maxTextLength/2);
131 
132     GridBagConstraints gbc = new GridBagConstraints();
133     gbc.gridx = 0;
134     gbc.gridy = 0;
135     gbc.fill = GridBagConstraints.BOTH;
136 
137     /**********************
138      * First row of text *
139      *********************/
140 
141     gbc.gridwidth = 1;
142     gbc.insets = new java.awt.Insets(10, 10, 10, 10);
143     JLabel labelTitle = new JLabel("Context");
144     labelTitle.setOpaque(true);
145     labelTitle.setBackground(Color.WHITE);
146     labelTitle.setBorder(new CompoundBorder(
147       new EtchedBorder(EtchedBorder.LOWERED,
148         new Color(250, 250, 250), new Color(250, 250, 250).darker()),
149       new EmptyBorder(new Insets(0, 2, 0, 2))));
150     labelTitle.setToolTipText("Expression and its context.");
151     add(labelTitle, gbc);
152     gbc.insets = new java.awt.Insets(10, 0, 10, 0);
153 
154     int expressionStart = contextBeforeSize;
155     int expressionEnd = text.length() - contextAfterSize;
156 
157     // for each character
158     for (int charNum = 0; charNum < text.length(); charNum++) {
159 
160       gbc.gridx = charNum + 1;
161       if (textTooLong) {
162         if (charNum == maxTextLength/2) {
163           // add ellipsis dots in case of a too long text displayed
164           add(new JLabel("..."), gbc);
165           // skip the middle part of the text if too long
166           charNum = upperBound + 1;
167           continue;
168         } else if (charNum > upperBound) {
169           gbc.gridx -= upperBound - (maxTextLength/2) + 1;
170         }
171       }
172 
173       // set the text and color of the feature value
174       JLabel label = new JLabel(text.substring(charNum, charNum+1));
175       if (charNum >= expressionStart && charNum < expressionEnd) {
176         // this part is matched by the pattern, color it
177         label.setBackground(new Color(240, 201, 184));
178       } else {
179         // this part is the context, no color
180         label.setBackground(Color.WHITE);
181       }
182       label.setOpaque(true);
183 
184       // get the word from which belongs the current character charNum
185       int start = text.lastIndexOf(" ", charNum);
186       int end = text.indexOf(" ", charNum);
187       String word = text.substring(
188         (start == -1) ? 0 : start,
189         (end == -1) ? text.length() : end);
190       // add a mouse listener that modify the query
191       label.addMouseListener(textMouseListener.createListener(word));
192       add(label, gbc);
193     }
194 
195       /************************************
196        * Subsequent rows with annotations *
197        ************************************/
198 
199     // for each row to display
200     for (StackRow stackRow : stackRows) {
201       String type = stackRow.getType();
202       String feature = stackRow.getFeature();
203       if (feature == null) { feature = ""; }
204       String shortcut = stackRow.getShortcut();
205       if (shortcut == null) { shortcut = ""; }
206 
207       gbc.gridy++;
208       gbc.gridx = 0;
209       gbc.gridwidth = 1;
210       gbc.insets = new Insets(0, 0, 3, 0);
211 
212       // add the header of the row
213       JLabel annotationTypeAndFeature = new JLabel();
214       String typeAndFeature = type + (feature.equals("") ? "" : ".") + feature;
215       annotationTypeAndFeature.setText(!shortcut.equals("") ?
216         shortcut : stackRow.getSet() != null ?
217           stackRow.getSet() + "#" + typeAndFeature : typeAndFeature);
218       annotationTypeAndFeature.setOpaque(true);
219       annotationTypeAndFeature.setBackground(Color.WHITE);
220       annotationTypeAndFeature.setBorder(new CompoundBorder(
221         new EtchedBorder(EtchedBorder.LOWERED,
222           new Color(250, 250, 250), new Color(250, 250, 250).darker()),
223         new EmptyBorder(new Insets(0, 2, 0, 2))));
224       if (feature.equals("")) {
225         annotationTypeAndFeature.addMouseListener(
226           headerMouseListener.createListener(type));
227       } else {
228         annotationTypeAndFeature.addMouseListener(
229           headerMouseListener.createListener(type, feature));
230       }
231       gbc.insets = new java.awt.Insets(0, 10, 3, 10);
232       add(annotationTypeAndFeature, gbc);
233       gbc.insets = new java.awt.Insets(0, 0, 3, 0);
234 
235       // add all annotations for this row
236       HashMap<Integer,TreeSet<Integer>> gridSet =
237         new HashMap<Integer,TreeSet<Integer>>();
238       int gridyMax = gbc.gridy;
239       for(StackAnnotation ann : stackRow.getAnnotations()) {
240         gbc.gridx = ann.getStartNode().getOffset().intValue()
241                   - expressionStartOffset + contextBeforeSize + 1;
242         gbc.gridwidth = ann.getEndNode().getOffset().intValue()
243                       - ann.getStartNode().getOffset().intValue();
244         if (gbc.gridx == 0) {
245           // column 0 is already the row header
246           gbc.gridwidth -= 1;
247           gbc.gridx = 1;
248         } else if (gbc.gridx < 0) {
249           // annotation starts before displayed text
250           gbc.gridwidth += gbc.gridx - 1;
251           gbc.gridx = 1;
252         }
253         if (gbc.gridx + gbc.gridwidth > text.length()) {
254           // annotation ends after displayed text
255           gbc.gridwidth = text.length() - gbc.gridx + 1;
256         }
257         if(textTooLong) {
258           if(gbc.gridx > (upperBound + 1)) {
259             // x starts after the hidden middle part
260             gbc.gridx -= upperBound - (maxTextLength / 2) + 1;
261           }
262           else if(gbc.gridx > (maxTextLength / 2)) {
263             // x starts in the hidden middle part
264             if(gbc.gridx + gbc.gridwidth <= (upperBound + 3)) {
265               // x ends in the hidden middle part
266               continue; // skip the middle part of the text
267             }
268             else {
269               // x ends after the hidden middle part
270               gbc.gridwidth -= upperBound - gbc.gridx + 2;
271               gbc.gridx = (maxTextLength / 2) + 2;
272             }
273           }
274           else {
275             // x starts before the hidden middle part
276             if(gbc.gridx + gbc.gridwidth < (maxTextLength / 2)) {
277               // x ends before the hidden middle part
278               // do nothing
279             }
280             else if(gbc.gridx + gbc.gridwidth < upperBound) {
281               // x ends in the hidden middle part
282               gbc.gridwidth = (maxTextLength / 2) - gbc.gridx + 1;
283             }
284             else {
285               // x ends after the hidden middle part
286               gbc.gridwidth -= upperBound - (maxTextLength / 2) + 1;
287             }
288           }
289         }
290         if(gbc.gridwidth == 0) {
291           gbc.gridwidth = 1;
292         }
293 
294         JLabel label = new JLabel();
295         Object object = ann.getFeatures().get(feature);
296         String value = (object == null) ? " " : Strings.toString(object);
297         if(value.length() > maxFeatureValueLength) {
298           // show the full text in the tooltip
299           label.setToolTipText((value.length() > 500) ?
300             "<html><textarea rows=\"30\" cols=\"40\" readonly=\"readonly\">"
301             + value.replaceAll("(.{50,60})\\b", "$1\n") + "</textarea></html>" :
302             ((value.length() > 100) ?
303               "<html><table width=\"500\" border=\"0\" cellspacing=\"0\">"
304                 + "<tr><td>" + value.replaceAll("\n", "<br>")
305                 + "</td></tr></table></html>" :
306               value));
307           if(stackRow.getCrop() == CROP_START) {
308             value = "..." + value.substring(
309               value.length() - maxFeatureValueLength - 1);
310           }
311           else if(stackRow.getCrop() == CROP_END) {
312             value = value.substring(0, maxFeatureValueLength - 2) + "...";
313           }
314           else {// cut in the middle
315             value = value.substring(0, maxFeatureValueLength / 2) + "..."
316               + value.substring(value.length() - (maxFeatureValueLength / 2));
317           }
318         }
319         label.setText(value);
320         label.setBackground(AnnotationSetsView.getColor(stackRow.getSet(),ann.getType()));
321         label.setBorder(BorderFactory.createLineBorder(Color.BLACK, 1));
322         label.setOpaque(true);
323 
324         label.addMouseListener(annotationMouseListener.createListener(
325           stackRow.getSet(), type, String.valueOf(ann.getId())));
326 
327         // show the feature values in the tooltip
328         if (!ann.getFeatures().isEmpty()) {
329           String width = (Strings.toString(ann.getFeatures()).length() > 100) ?
330             "500" : "100%";
331           String toolTip = "<html><table width=\"" + width
332             + "\" border=\"0\" cellspacing=\"0\" cellpadding=\"4\">";
333           Color color = (Color) UIManager.get("ToolTip.background");
334           float[] hsb = Color.RGBtoHSB(
335             color.getRed(), color.getGreen(), color.getBlue(), null);
336           color = Color.getHSBColor(hsb[0], hsb[1],
337             Math.max(0f, hsb[2] - hsb[2]*0.075f)); // darken the color
338           String hexColor = Integer.toHexString(color.getRed()) +
339             Integer.toHexString(color.getGreen()) +
340             Integer.toHexString(color.getBlue());
341           boolean odd = false; // alternate background color every other row
342           for(Map.Entry<Object, Object> map : ann.getFeatures().entrySet()) {
343             toolTip +="<tr align=\"left\""
344               + (odd?" bgcolor=\"#"+hexColor+"\"":"")+"><td><strong>"
345               + map.getKey() + "</strong></td><td>"
346               + ((Strings.toString(map.getValue()).length() > 500) ?
347               "<textarea rows=\"20\" cols=\"40\" cellspacing=\"0\">"
348                 + StringEscapeUtils.escapeHtml(Strings.toString(map.getValue()).replaceAll("(.{50,60})\\b", "$1\n"))
349                 + "</textarea>" :
350               StringEscapeUtils.escapeHtml(Strings.toString(map.getValue())).replaceAll("\n", "<br>"))
351               + "</td></tr>";
352             odd = !odd;
353           }
354           label.setToolTipText(toolTip + "</table></html>");
355         } else {
356           label.setToolTipText("No features.");
357         }
358 
359         if(!feature.equals("")) {
360           label.addMouseListener(annotationMouseListener.createListener(
361             stackRow.getSet(), type, feature, Strings.toString(
362               ann.getFeatures().get(feature)), String.valueOf(ann.getId())));
363         }
364         // find the first empty row span for this annotation
365         int oldGridy = gbc.gridy;
366         for(int y = oldGridy; y <= (gridyMax + 1); y++) {
367           // for each cell of this row where spans the annotation
368           boolean xSpanIsEmpty = true;
369           for(int x = gbc.gridx;
370               (x < (gbc.gridx + gbc.gridwidth)) && xSpanIsEmpty; x++) {
371             xSpanIsEmpty = !(gridSet.containsKey(x)
372               && gridSet.get(x).contains(y));
373           }
374           if(xSpanIsEmpty) {
375             gbc.gridy = y;
376             break;
377           }
378         }
379         // save the column x and row y of the current value
380         TreeSet<Integer> ts;
381         for(int x = gbc.gridx; x < (gbc.gridx + gbc.gridwidth); x++) {
382           ts = gridSet.get(x);
383           if(ts == null) {
384             ts = new TreeSet<Integer>();
385           }
386           ts.add(gbc.gridy);
387           gridSet.put(x, ts);
388         }
389         add(label, gbc);
390         gridyMax = Math.max(gridyMax, gbc.gridy);
391         gbc.gridy = oldGridy;
392       }
393 
394       // add a button at the end of the row
395       gbc.gridwidth = 1;
396       if (stackRow.getLastColumnButton() != null) {
397         // last cell of the row
398         gbc.gridx = Math.min(text.length(), maxTextLength) + 1;
399         gbc.insets = new Insets(0, 10, 3, 0);
400         gbc.fill = GridBagConstraints.NONE;
401         gbc.anchor = GridBagConstraints.WEST;
402         add(stackRow.getLastColumnButton(), gbc);
403         gbc.insets = new Insets(0, 0, 3, 0);
404         gbc.fill = GridBagConstraints.BOTH;
405         gbc.anchor = GridBagConstraints.CENTER;
406       }
407 
408       // set the new gridy to the maximum row we put a value
409       gbc.gridy = gridyMax;
410     }
411 
412     if (lastRowButton != null) {
413       // add a configuration button on the last row
414       gbc.insets = new java.awt.Insets(0, 10, 0, 10);
415       gbc.gridx = 0;
416       gbc.gridy++;
417       add(lastRowButton, gbc);
418     }
419 
420     // add an empty cell that takes all remaining space to
421     // align the visible cells at the top-left corner
422     gbc.gridy++;
423     gbc.gridx = Math.min(text.length(), maxTextLength) + 1;
424     gbc.gridwidth = GridBagConstraints.REMAINDER;
425     gbc.gridheight = GridBagConstraints.REMAINDER;
426     gbc.weightx = 1;
427     gbc.weighty = 1;
428     add(new JLabel(""), gbc);
429 
430     validate();
431     updateUI();
432   }

434   /**
435    * Extension of a MouseInputAdapter that adds a method
436    * to create new Listeners from it.<br>
437    * You must overriden the createListener method.
438    */
439   public static class StackMouseListener extends MouseInputAdapter {
440     /**
441      * There is 3 cases for the parameters of createListener:
442      * <ol>
443      * <li>first line of text -> createListener(word)
444      * <li>first column, header -> createListener(type),
445      *   createListener(type, feature)
446      * <li>annotation -> createListener(set, type, annotationId),
447      *   createListener(set, type, feature, value, annotationId)
448      * </ol>
449      * @param parameters see above
450      * @return a MouseInputAdapter for the text, header or annotation
451      */
452     public MouseInputAdapter createListener(String... parameters) {
453       return null;
454     }
455   }

457   /**
458    * Annotation that doesn't belong to an annotation set
459    * and with id always equal to -1.<br>
460    * Allows to create an annotation without document, nodes, annotation set,
461    * and keep compatibility with gate.Annotation.
462    * <br>
463    * This class is only for AnnotationStack internal use
464    * as it won't work with most of the methods that use gate.Annotation.
465    */
466   private static class StackAnnotation extends gate.annotation.AnnotationImpl {
467     StackAnnotation(Integer id, Node start, Node end, String type,
468                          FeatureMap features) {
