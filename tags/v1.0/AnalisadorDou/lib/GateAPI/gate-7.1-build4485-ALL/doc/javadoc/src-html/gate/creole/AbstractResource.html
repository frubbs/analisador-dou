075   /**
076    * Gets the value of a parameter for a resource.
077    * @param resource the resource from which the parameter value will be
078    * obtained
079    * @param paramaterName the name of the parameter
080    * @return the current value of the parameter
081    */
082   public static Object getParameterValue(Resource resource,
083                                          String paramaterName)
084                 throws ResourceInstantiationException{
085     // get the beaninfo for the resource bean, excluding data about Object
086     BeanInfo resBeanInf = null;
087     try {
088       resBeanInf = getBeanInfo(resource.getClass());
089     } catch(Exception e) {
090       throw new ResourceInstantiationException(
091         "Couldn't get bean info for resource " + resource.getClass().getName()
092         + Strings.getNl() + "Introspector exception was: " + e
093       );
094     }
095     PropertyDescriptor[] properties = resBeanInf.getPropertyDescriptors();
096 
097     //find the property we're interested on
098     if(properties == null){
099       throw new ResourceInstantiationException(
100         "Couldn't get properties info for resource " +
101         resource.getClass().getName());
102     }
103     boolean done = false;
104     int i = 0;
105     Object value = null;
106     while(!done && i < properties.length){
107       PropertyDescriptor prop = properties[i];
108       if(prop.getName().equals(paramaterName)){
109         Method getMethod = prop.getReadMethod();
110         if(getMethod == null){
111           throw new ResourceInstantiationException(
112             "Couldn't get read accessor method for parameter " + paramaterName +
113             " in " + resource.getClass().getName());
114         }
115         // call the get method with the parameter value
116         Object[] args = new Object[0];
117         try {
118           value = getMethod.invoke(resource, args);
119         } catch(Exception e) {
120           throw new ResourceInstantiationException(
121             "couldn't invoke get method: " + e
122           );
123         }
124         done = true;
125       }//if(prop.getName().equals(paramaterName))
126       i++;
127     }//while(!done && i < properties.length)
128     if(done) return value;
129     else throw new ResourceInstantiationException(
130             "Couldn't find parameter named " + paramaterName +
131             " in " + resource.getClass().getName());
132   }

134   /**
135    * Sets the value for a specified parameter for a resource.
136    *
137    * @param resource the resource for which the parameter value will be set
138    * @param paramaterName the name for the parameter
139    * @param parameterValue the value the parameter will receive
140    */
141   public static void setParameterValue(Resource resource, BeanInfo resBeanInf,
142                                        String paramaterName,
143                                        Object parameterValue)
144               throws ResourceInstantiationException{
145     PropertyDescriptor[] properties = resBeanInf.getPropertyDescriptors();
146     //find the property we're interested on
147     if(properties == null){
148       throw new ResourceInstantiationException(
149         "Couldn't get properties info for resource " +
150         resource.getClass().getName());
151     }
152     boolean done = false;
153     int i = 0;
154     while(!done && i < properties.length){
155       PropertyDescriptor prop = properties[i];
156       if(prop.getName().equals(paramaterName)){
157         Method setMethod = prop.getWriteMethod();
158         if(setMethod == null){
159           throw new ResourceInstantiationException(
160             "Couldn't get write accessor method for parameter " +
161             paramaterName + " in " + resource.getClass().getName());
162         }
163 
164         // convert the parameter to the right type eg String -> URL
165         if(parameterValue != null){
166           Class propertyType = prop.getPropertyType();
167           Class typeToCreate = propertyType;
168           if(Parameter.substituteClasses.containsKey(propertyType)) {
169             typeToCreate = Parameter.substituteClasses.get(propertyType);
170           }
171           Class paramType = parameterValue.getClass();
172           if(!propertyType.isAssignableFrom(paramType)) {
173             try {
174               Constructor mostSpecificConstructor =
175                 Tools.getMostSpecificConstructor(typeToCreate, paramType);
176               parameterValue = mostSpecificConstructor
177                  .newInstance( new Object[]{parameterValue} );
178             } catch(Exception e) {
179               //this didn't work; if the parameter value is String
180               //try to use the Parameter implementation for finding the
181               //value
182               if(String.class.isAssignableFrom(paramType)){
183                 ResourceData rData = (ResourceData)Gate.getCreoleRegister().
184                   get(resource.getClass().getName());
185                 ParameterList pList = rData.getParameterList();
186                 Parameter param = null;
187                 Iterator disjIter = pList.getInitimeParameters().iterator();
188                 while(param == null && disjIter.hasNext()){
189                   Iterator paramIter = ((List)disjIter.next()).iterator();
190                   while(param == null && paramIter.hasNext()){
191                     Parameter aParam = (Parameter)paramIter.next();
192                     if(aParam.getName().equals(paramaterName)) param = aParam;
193                   }
194                 }
195                 disjIter = pList.getRuntimeParameters().iterator();
196                 while(param == null && disjIter.hasNext()){
197                   Iterator paramIter = ((List)disjIter.next()).iterator();
198                   while(param == null && paramIter.hasNext()){
199                     Parameter aParam = (Parameter)paramIter.next();
200                     if(aParam.getName().equals(paramaterName)) param = aParam;
201                   }
202                 }
203                 if(param != null){
204                   try{
205                     parameterValue = param.calculateValueFromString(
206                             (String)parameterValue);
207                   }catch(ParameterException pe){
208                     throw new ResourceInstantiationException(pe);
209                   }
210                 }else{
211                   //this should never happen
212                   throw new LuckyException("Unknown parameter " + paramaterName +
213                           " for resource " + resource.getClass().getName() +
214                           "!");
215                 }
216               }else{
217                 throw new ResourceInstantiationException(
218                   "Error converting " + parameterValue.getClass() +
219                   " to " + propertyType + ": " + e.toString()
220                 );
221               }
222             }
223           }
224         }//if(parameterValue != null)
225 
226         // call the set method with the parameter value
227         Object[] args = new Object[1];
228         args[0] = parameterValue;
229         try {
230           setMethod.invoke(resource, args);
231         } catch(Exception e) {
232           e.printStackTrace(Err.getPrintWriter());
233           throw new ResourceInstantiationException(
234             "couldn't invoke set method for " + paramaterName +
235             " on " + resource.getClass().getName() + ": " + e);
236         }
237         done = true;
238       }//if(prop.getName().equals(paramaterName))
239       i++;
240     }//while(!done && i < properties.length)
241     if(!done) throw new ResourceInstantiationException(
242                           "Couldn't find parameter named " + paramaterName +
243                           " in " + resource.getClass().getName());
244   }//public void setParameterValue(String paramaterName, Object parameterValue)

275   /**
276    * Adds listeners to a resource.
277    * @param listeners The listeners to be registered with the resource. A
278    * {@link java.util.Map} that maps from fully qualified class name (as a
279    * string) to listener (of the type declared by the key).
280    * @param resource the resource that listeners will be registered to.
281    */
282   public static void setResourceListeners(Resource resource, Map listeners)
283   throws
284     IntrospectionException, InvocationTargetException,
285     IllegalAccessException, GateException
286   {
287     // get the beaninfo for the resource bean, excluding data about Object
288     BeanInfo resBeanInfo = getBeanInfo(resource.getClass());
289 
290     // get all the events the bean can fire
291     EventSetDescriptor[] events = resBeanInfo.getEventSetDescriptors();
292 
293     // add the listeners
294     if (events != null) {
295       EventSetDescriptor event;
296       for(int i = 0; i < events.length; i++) {
297         event = events[i];
298 
299         // did we get such a listener?
300         Object listener =
301           listeners.get(event.getListenerType().getName());
302         if(listener != null) {
303           Method addListener = event.getAddListenerMethod();
304 
305           // call the set method with the parameter value
306           Object[] args = new Object[1];
307           args[0] = listener;
308           addListener.invoke(resource, args);
309         }
310       } // for each event
311     }   // if events != null
312   } // setResourceListeners()
313

352   /**
353    * Checks whether the provided {@link Resource} has values for all the
354    * required parameters from the provided list of parameters.
355    *
356    * @param resource the resource being checked
357    * @param parameters is a {@link List} of {@link List} of {@link Parameter}
358    * representing a list of parameter disjunctions (e.g. the one returned by
359    * {@link ParameterList#getRuntimeParameters()}).
360    * @return <tt>true</tt> if all the required parameters have non null values,
361    * <tt>false</tt> otherwise.
362    * @throws {@link ResourceInstantiationException} if problems occur while
363    * inspecting the parameters for the resource. These will normally be
364    * introspection problems and are usually caused by the lack of a parameter
365    * or of the read accessor for a parameter.
366    */
367   public static boolean checkParameterValues(Resource resource,
368                                              List parameters)
369                 throws ResourceInstantiationException{
370     Iterator disIter = parameters.iterator();
371     while(disIter.hasNext()){
372       List disjunction = (List)disIter.next();
373       boolean required = !((Parameter)disjunction.get(0)).isOptional();
374       if(required){
375         //at least one parameter in the disjunction must have a value
376         boolean valueSet = false;
377         Iterator parIter = disjunction.iterator();
378         while(!valueSet && parIter.hasNext()){
379           Parameter par = (Parameter)parIter.next();
380           valueSet = (resource.getParameterValue(par.getName()) != null);
381         }
382         if(!valueSet) return false;
383       }
384     }
385     return true;
386   }

432   /**
433    * Get the current values of the given parameters from the given
434    * resource.
435    *
436    * @param res the resource
437    * @param params a list of parameter disjunctions such as would
438    *         be returned by {@link ParameterList#getInitimeParameters()}
439    *         or similar.
440    */
441   public static FeatureMap getParameterValues(Resource res,
442           List<List<Parameter>> params) throws ResourceInstantiationException {
443     FeatureMap fm = Factory.newFeatureMap();
444     for(List<Parameter> parDisjunction : params) {
445       for(Parameter p : parDisjunction) {
446         fm.put(p.getName(), res.getParameterValue(p.getName()));
447       }
448     }
449     return fm;
450   }
