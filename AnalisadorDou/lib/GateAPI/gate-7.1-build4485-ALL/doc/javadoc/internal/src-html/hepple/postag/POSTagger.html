161   /**
162    * Runs the tagger over a set of sentences.
163    * @param sentences a {@link java.util.List} of {@link java.util.List}s
164    * of words to be tagged. Each list is a sentence represented as a list of
165    * words.
166    * @return a {@link java.util.List} of {@link java.util.List}s of
167    * {@link java.lang.String}[]. A list of tagged sentences, each sentence
168    * being itself a list having pairs of strings as elements with
169    * the word on the first position and the tag on the second.
170    */
171   public List runTagger(List sentences){
172     List output = new ArrayList();
173     List taggedSentence = new ArrayList();
174     Iterator sentencesIter = sentences.iterator();
175     while(sentencesIter.hasNext()){
176       List sentence = (List)sentencesIter.next();
177       Iterator wordsIter = sentence.iterator();
178       while(wordsIter.hasNext()){
179         String newWord = (String)wordsIter.next();
180         oneStep(newWord, taggedSentence);
181       }//while(wordsIter.hasNext())
182       //finished adding all the words from a sentence, add six more
183       //staarts to flush all words out of the tagging buffer
184       for(int i = 0; i < 6; i++){
185         oneStep(staart, taggedSentence);
186       }
187       //we have a new finished sentence
188       output.add(taggedSentence);
189       taggedSentence = new ArrayList();
190     }//while(sentencesIter.hasNext())
191     return output;
192   }

209   /**
210    * Adds a new word to the window of 7 words (on the last position) and tags
211    * the word currently in the middle (i.e. on position 3). This function
212    * also reads the word on the first position and adds its tag to the
213    * taggedSentence structure as this word would be lost at the next advance.
214    * If this word completes a sentence then it returns true otherwise it
215    * returns false.
216    * @param word the new word
217    * @param taggedSentence a List of pairs of strings representing the results
218    * of tagging the current sentence so far.
219    * @return returns true if a full sentence is now tagged, otherwise false.
220    */
221   protected boolean oneStep(String word, List taggedSentence){
222     //add the new word at the end of the text window
223     for (int i=1 ; i<7 ; i++) {
224       wordBuff[i-1] = wordBuff[i];
225       tagBuff[i-1] = tagBuff[i];
226       lexBuff[i-1] = lexBuff[i];
227     }
228     wordBuff[6] = word;
229     lexBuff[6] = classifyWord(word);
230     tagBuff[6] = lexBuff[6][0];
231 
232     //apply the rules to the word in the middle of the text window
233     //Try to fire a rule for the current lexical entry. It may be the case that
234     //no rule applies.
235     List rulesToApply = (List)rules.get(lexBuff[3][0]);
236     if(rulesToApply != null && rulesToApply.size() > 0){
237       Iterator rulesIter = rulesToApply.iterator();
238       //find the first rule that applies, fire it and stop.
239       while(rulesIter.hasNext() && !((Rule)rulesIter.next()).apply(this)){}
240     }
241 
242     //save the tagged word from the first position
243     String taggedWord = wordBuff[0];
244     if(taggedWord != staart){
245       taggedSentence.add(new String[]{taggedWord, tagBuff[0]});
246       if(wordBuff[1] == staart){
247         //wordTag[0] was the end of a sentence
248         return true;
249       }//if(wordBuff[1] == staart)
250     }//if(taggedWord != staart)
251     return false;
252 
253   }//protected List oneStep(String word, List taggedSentence)

255   /**
256    * Reads the rules from the rules input file
257    */
258   public void readRules(URL rulesURL) throws IOException, InvalidRuleException{
259     BufferedReader rulesReader = null;
260     InputStream rulesStream = null;
261     try {
262       rulesStream = rulesURL.openStream();
263       if(encoding == null) {
264         rulesReader = new BomStrippingInputStreamReader(rulesStream);
265       } else {
266         rulesReader = new BomStrippingInputStreamReader(rulesStream, this.encoding);
267       }
268   
269       String line;
270       Rule newRule;
271   
272       line = rulesReader.readLine();
273       while(line != null){
274         List<String> ruleParts = new ArrayList<String>();
275         StringTokenizer tokens = new StringTokenizer(line);
276         while (tokens.hasMoreTokens()) ruleParts.add(tokens.nextToken());
277         if (ruleParts.size() < 3) throw new InvalidRuleException(line);
278   
279         newRule = createNewRule((String)ruleParts.get(2));
280         newRule.initialise(ruleParts);
281         List<Rule> existingRules = (List)rules.get(newRule.from);
282         if(existingRules == null){
283           existingRules = new ArrayList<Rule>();
284           rules.put(newRule.from, existingRules);
285         }
286         existingRules.add(newRule);
287   
288         line = rulesReader.readLine();
289       }//while(line != null)
290     }
291     finally {
292       IOUtils.closeQuietly(rulesReader);
293       IOUtils.closeQuietly(rulesStream);
294     }
295   }//public void readRules()

301   /**
302    * Attempts to classify an unknown word.
303    * @param wd the word to be classified
304    */
305   protected String[] classifyWord(String wd){
306     String[] result;
307 
308     if (staart.equals(wd)) return staartLex;
309 
310     List categories = (List)lexicon.get(wd);
311     if(categories != null){
312       result = new String[categories.size()];
313       for(int i = 0; i < result.length; i++){
314         result[i] = (String)categories.get(i);
315       }
316       return result;
317     }
318 
319     //no lexical entry for the word. Try to guess
320     if ('A' <= wd.charAt(0) && wd.charAt(0) <= 'Z') return deflex_NNP;
321 
322     for (int i=1 ; i < wd.length()-1 ; i++)
323       if (wd.charAt(i) == '-') return deflex_JJ;
324 
325     for (int i=0 ; i < wd.length() ; i++)
326       if ('0' <= wd.charAt(i) && wd.charAt(i) <= '9') return deflex_CD;
327 
328     if (wd.endsWith("ed") ||
329         wd.endsWith("us") ||
330         wd.endsWith("ic") ||
331         wd.endsWith("ble") ||
332         wd.endsWith("ive") ||
333         wd.endsWith("ary") ||
334         wd.endsWith("ful") ||
335         wd.endsWith("ical") ||
336         wd.endsWith("less")) return deflex_JJ;
337 
338     if (wd.endsWith("s")) return deflex_NNS;
339 
340     if (wd.endsWith("ly")) return deflex_RB;
341 
342     if (wd.endsWith("ing")) return deflex_VBG;
343 
344     return deflex_NN;
345   }//private String[] classifyWord(String wd)
