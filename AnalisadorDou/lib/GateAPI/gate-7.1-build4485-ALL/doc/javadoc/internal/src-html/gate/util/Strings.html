070     String line = null;

072 
073     try {
074       for(int lineNum = startLine; ( line = reader.readLine() ) != null; lineNum++) {
075         String pad;
076         if(lineNum < 10) pad = " ";
077         else pad = "";

081 
082     return result.toString();

085   /**
086    * A method to unescape Java strings, returning a string containing escape
087    * sequences into the respective character. i.e. "\" followed by "t" is turned
088    * into the tab character.
089    *
090    * @param str the string to unescape
091    * @return a new unescaped string of the one passed in
092    */
093   public static String unescape(String str) {
094     if (str == null) return str;
095 
096     StringBuilder sb = new StringBuilder(); // string to build
097 
098     StringBuilder unicodeStr = new StringBuilder(4); // store unicode sequences
099 
100     boolean inUnicode = false;
101     boolean hadSlash = false;
102 
103     for (char ch: str.toCharArray()) {
104       if (inUnicode) {
105         unicodeStr.append(ch);
106         if (unicodeStr.length() == 4) {
107           try {
108             int unicodeValue = Integer.parseInt(unicodeStr.toString(), 16);
109             sb.append((char) unicodeValue);
110             unicodeStr.setLength(0);
111             inUnicode = false;
112             hadSlash = false;
113           } catch (NumberFormatException e) {
114             throw new RuntimeException("Couldn't parse unicode value: " + unicodeStr, e);
115           }
116         }
117         continue;
118       }
119       if (hadSlash) {
120         hadSlash = false;
121         switch (ch) {
122           case '\\':
123             sb.append('\\');
124             break;
125           case '\'':
126             sb.append('\'');
127             break;
128           case '\"':
129             sb.append('"');
130             break;
131           case 'r':
132             sb.append('\r');
133             break;
134           case 'f':
135             sb.append('\f');
136               break;
137           case 't':
138             sb.append('\t');
139             break;
140           case 'n':
141             sb.append('\n');
142             break;
143           case 'b':
144             sb.append('\b');
145             break;
146           case 'u':
147             inUnicode = true;
148             break;
149           default :
150             sb.append(ch);
151             break;
152         }
153         continue;
154       } else if (ch == '\\') {
155         hadSlash = true;
156         continue;
157       }
158       sb.append(ch);
159     }
160     if (hadSlash) {
161       sb.append('\\');
162     }
163     return sb.toString();
164   }

166   /**
167    * Converts a size given as a number of bytes (e.g. a file size) to a 
168    * human-readable string.
169    * @param bytes the size to be converted.
170    * @param base10 if <code>true</code> then the multiplier used is 10^3 (i.e. 
171    * 1000 bytes are reported as 1kB, etc.); otherwise 1024 is used as a 
172    * multiplier (1024 bytes is 1KiB, etc.).
173    * @return a human readable version of a file size
174    */
175   public static String humanReadableByteCount(long bytes, boolean base10) {
176     int unit = base10 ? 1000 : 1024;
177     if (bytes < unit) return bytes + " B";
178     int exp = (int) (Math.log(bytes) / Math.log(unit));
179     String pre = (base10 ? "kMGTPE" : "KMGTPE").charAt(exp-1) + (base10 ? "" : "i");
180     return String.format("%.1f %sB", bytes / Math.pow(unit, exp), pre);
181   }

183   /**
184    * Convert about any object to a human readable string.<br>
185    * Use {@link Arrays#deepToString(Object[])} to convert an array or
186    * a collection.
187    * @param object object to be converted to a string
188    * @return a string representation of the object, the empty string if null.
189    */
190   public static String toString(Object object) {
191     if (object == null) {
192       return "";
193     } else if (object instanceof Object[]) {
194       return Arrays.deepToString((Object[])object);
195     } else if (object instanceof Collection) {
196       return Arrays.deepToString(((Collection)object).toArray());
197     } else {
198       return object.toString();
199     }
200   }

202   /**
203    * Create a String representation of a Set of String with the format
204    * [value, value].
205    * Escape with a backslash the separator character.
206    * @param set set to convert to one String
207    * @return a String that represent the set
208    * @see #toSet(String, String)
209    */
210   public static String toString(LinkedHashSet<String> set) {
211     StringBuilder builder = new StringBuilder();
212     builder.append('[');
213     for (String element : set) {
214       // escape element separator
215       String escapedElement = element.replaceAll(",", "\\\\,");
216       builder.append(escapedElement).append(", ");
217     }
218     if (builder.length() > 1) { // remove last element separator
219       builder.delete(builder.length()-2, builder.length());
220     }
221     builder.append("]");
222     return builder.toString();
223   }

225   /**
226    * Create a String representation of a Map of String*String with the format
227    * {key=value, key=value}.
228    * Escape with a backslash the separator characters.
229    * @param map map to convert to one String
230    * @return a String that represent the map
231    * @see #toMap(String)
232    */
233   public static String toString(Map<String, String> map) {
234     StringBuilder builder = new StringBuilder();
235     builder.append('{');
236     for (Map.Entry<String, String> entry : map.entrySet()) {
237       // escape element separator
238       String escapedKey = entry.getKey().replaceAll("=", "\\\\=")
239         .replaceAll(",", "\\\\,");
240       String escapedValue = entry.getValue().replaceAll("=", "\\\\=")
241         .replaceAll(",", "\\\\,");
242       builder.append(escapedKey).append("=").append(escapedValue).append(", ");
243     }
244     if (builder.length() > 1) { // remove last element separator
245       builder.delete(builder.length()-2, builder.length());
246     }
247     builder.append("}");
248     return builder.toString();
249   }

251   /**
252    * Get back a Set of String from its String representation.
253    * Unescape backslashed separator characters.
254    * @param string String to convert to a List
255    * @param separator String that delimits the element of the set
256    * @return a linked hashset
257    * @see #toString(java.util.LinkedHashSet)
258    */
259   public static LinkedHashSet<String> toSet(String string, String separator) {
260     LinkedHashSet<String> set = new LinkedHashSet<String>();
261     if (string == null
262      || string.length() < 3) {
263       return set;
264     }
265     // remove last character
266     String value = string.substring(0, string.length()-1);
267     int index = 1;
268     int startIndex = 1;
269     int separatorIndex;
270     // split on element separator
271     while ((separatorIndex = value.indexOf(separator, index)) != -1) {
272       // check that the separator is not an escaped character
273       if (value.charAt(separatorIndex-1) != '\\') {
274         set.add(value.substring(startIndex, separatorIndex)
275           // unescape separator
276           .replaceAll("\\\\"+separator.charAt(0), ""+separator.charAt(0)));
277         startIndex = separatorIndex + separator.length();
278       }
279       index = separatorIndex + separator.length();
280     }
281     // last element of the set
282     set.add(value.substring(startIndex));
283     return set;
284   }

286   /**
287    * Get back a Map of String*String from its String representation.
288    * Unescape backslashed separator characters.
289    * @param string String to convert to a Map
290    * @return a Map
291    * @see #toString(java.util.Map)
292    */
293   public static Map<String, String> toMap(String string) {
294     Map<String, String> map = new HashMap<String, String>();
295     if (string == null
296      || string.length() < 3) {
297       return map;
298     }
299     Set<String> firstList = toSet(string, ", ");
300     for (String element : firstList) {
301       LinkedHashSet<String> secondList = toSet("[" + element + "]", "=");
302       if (secondList.size() == 2) {
303         Iterator<String> iterator = secondList.iterator();
304         map.put(iterator.next(), iterator.next());
305       } else {
306         Err.prln("Ignoring element: [" + element + "]");
307         Err.prln("Expecting: [key=value]");
308       }
309     }
310     return map;
311   }

313   /**
314    * Crop the text in the middle if too long.
315    * @param text text to crop
316    * @param maxLength maximum length of the text
317    * @return cropped text if needed otherwise the same text
318    */
319   public static String crop(String text, int maxLength) {
320     if (text.length() > maxLength) {
321       text = text.substring(0, (maxLength/2)-2) + "..."
322         + text.substring(text.length() - (maxLength/2));
323     }
324     return text;
325   }
