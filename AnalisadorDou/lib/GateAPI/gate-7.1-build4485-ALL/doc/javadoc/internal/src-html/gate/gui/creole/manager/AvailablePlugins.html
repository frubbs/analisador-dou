319   private void filterRows(String rowFilter) {
320     final String filter = rowFilter.trim().toLowerCase();
321     final String previousURL =
322             mainTable.getSelectedRow() == -1 ? "" : (String)mainTable
323                     .getValueAt(mainTable.getSelectedRow(),
324                             mainTable.convertColumnIndexToView(NAME_COLUMN));
325     if(filter.length() < 2) {
326       // one character or less, don't filter rows
327       visibleRows.clear();
328       visibleRows.addAll(Gate.getKnownPlugins());
329     } else {
330       // filter rows case insensitively on each plugin URL and its resources
331       visibleRows.clear();
332       for(int i = 0; i < Gate.getKnownPlugins().size(); i++) {
333         Gate.DirectoryInfo dInfo =
334                 Gate.getDirectoryInfo(Gate.getKnownPlugins().get(i));
335         String url = dInfo.getUrl().toString();
336         String resources = "";
337         for(int j = 0; j < dInfo.getResourceInfoList().size(); j++) {
338           resources +=
339                   dInfo.getResourceInfoList().get(j).getResourceName() + " ";
340         }
341         if(url.toLowerCase().contains(filter)
342                 || resources.toLowerCase().contains(filter)) {
343           visibleRows.add(Gate.getKnownPlugins().get(i));
344         }
345       }
346     }
347 
348     mainTableModel.fireTableDataChanged();
349 
350     if(mainTable.getRowCount() > 0) {
351       SwingUtilities.invokeLater(new Runnable() {
352         @Override
353         public void run() {
354           mainTable.setRowSelectionInterval(0, 0);
355           if(filter.length() < 2 && previousURL != null
356                   && !previousURL.equals("")) {
357             // reselect the last selected row based on its name and url values
358             for(int row = 0; row < mainTable.getRowCount(); row++) {
359               String url =
360                       (String)mainTable.getValueAt(row,
361                               mainTable.convertColumnIndexToView(NAME_COLUMN));
362               if(url.contains(previousURL)) {
363                 mainTable.setRowSelectionInterval(row, row);
364                 mainTable.scrollRectToVisible(mainTable.getCellRect(row, 0,
365                         true));
366                 break;
367               }
368             }
369           }
370         }
371       });
372     }
373   }

375   private Boolean getLoadNow(URL url) {
376     Boolean res = loadNowByURL.get(url);
377     if(res == null) {
378       res = Gate.getCreoleRegister().getDirectories().contains(url);
379       loadNowByURL.put(url, res);
380     }
381     return res;
382   }

387       res = Gate.getAutoloadPlugins().contains(url);
388       loadAlwaysByURL.put(url, res);
389     }
390     return res;
391   }

446     @Override

448       Gate.DirectoryInfo dInfo = Gate.getDirectoryInfo(visibleRows.get(row));
449       if(dInfo == null) { return null; }
450       switch(column){
451         case NAME_COLUMN:
452           return dInfo.toHTMLString();
453         case ICON_COLUMN:
454           if(!dInfo.isValid()) return invalidIcon;
455           if(dInfo.isRemotePlugin()) return remoteIcon;
456           if(dInfo.isCorePlugin()) return coreIcon;
457           if(dInfo.isUserPlugin()) return userIcon;
458           return otherIcon;
459         case LOAD_NOW_COLUMN:
460           return getLoadNow(dInfo.getUrl());
461         case LOAD_ALWAYS_COLUMN:
462           return getLoadAlways(dInfo.getUrl());
463         default:
464           return null;
465       }
466     }

491           mainTable.requestFocusInWindow();
492           break;
493       }
494     }

543   protected boolean unsavedChanges() {
544 
545     @SuppressWarnings("unchecked")
546     Set<URL> creoleDirectories = Gate.getCreoleRegister().getDirectories();
547 
548     Iterator<URL> pluginIter = loadNowByURL.keySet().iterator();
549     while(pluginIter.hasNext()) {

551       boolean load = loadNowByURL.get(aPluginURL);
552       boolean loaded = creoleDirectories.contains(aPluginURL);
553       if(load && !loaded) { return true; }
554       if(!load && loaded) { return true; }
555     }
556 
557     pluginIter = loadAlwaysByURL.keySet().iterator();
558     while(pluginIter.hasNext()) {
559       URL aPluginURL = pluginIter.next();
560       boolean load = loadAlwaysByURL.get(aPluginURL);
561       boolean loaded = Gate.getAutoloadPlugins().contains(aPluginURL);
562       if(load && !loaded) { return true; }
563       if(!load && loaded) { return true; }
564     }

566     return false;

569   protected Set<URL> updateAvailablePlugins() {
570 
571     @SuppressWarnings("unchecked")
572     Set<URL> creoleDirectories = Gate.getCreoleRegister().getDirectories();
573 
574     // update the data structures to reflect the user's choices
575     Iterator<URL> pluginIter = loadNowByURL.keySet().iterator();
576     
577     Set<URL> toLoad = new HashSet<URL>();
578     while(pluginIter.hasNext()) {
579       URL aPluginURL = pluginIter.next();
580       boolean load = loadNowByURL.get(aPluginURL);
581       boolean loaded = creoleDirectories.contains(aPluginURL);
582       if(load && !loaded) {
583         // remember that we need to load this plugin
584         toLoad.add(aPluginURL);
585       }
586       if(!load && loaded) {
587         // remove the directory
588         Gate.getCreoleRegister().removeDirectory(aPluginURL);
589       }
590     }
591 
592     pluginIter = loadAlwaysByURL.keySet().iterator();
593     while(pluginIter.hasNext()) {
594       URL aPluginURL = pluginIter.next();
595       boolean load = loadAlwaysByURL.get(aPluginURL);
596       boolean loaded = Gate.getAutoloadPlugins().contains(aPluginURL);
597       if(load && !loaded) {
598         // set autoload to true
599         Gate.addAutoloadPlugin(aPluginURL);
600       }
601       if(!load && loaded) {
602         // set autoload to false
603         Gate.removeAutoloadPlugin(aPluginURL);
604       }
605     }
606     
607     while(!toLoad.isEmpty()) {
608       //lets finally try loading all the plugings
609       int numToLoad = toLoad.size();
610       List<Throwable> errors = new ArrayList<Throwable>();
611       
612       pluginIter = toLoad.iterator();
613       while(pluginIter.hasNext()) {
614         URL aPluginURL = pluginIter.next();
615         
616         // load the directory
617         try {
618           Gate.getCreoleRegister().registerDirectories(aPluginURL);
619           pluginIter.remove();
620         } catch(Throwable ge) {
621           //TODO suppress the errors unless we are going to break out of the loop
622           //ge.printStackTrace();
623           errors.add(ge);
624         }
625       }
626       
627       if (numToLoad == toLoad.size()) {
628         //we tried loading all the plugins and yet
629         //we didn't actually achieve anything
630         for (Throwable t : errors) {
631           t.printStackTrace();
632         }        
633         
634         break;
635       }
636     }
637     
638     loadNowByURL.clear();
639     loadAlwaysByURL.clear();
640     
641     return toLoad;
642   }
