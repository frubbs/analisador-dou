17 package gate.util;
18 
19 /** An inherited class from  SAX exception in the GATE packages. Can be used
20   * to catch any internal exception thrown by the GATE SAX libraries.
21   */
22 public class GateSaxException extends org.xml.sax.SAXException {
23   /** Debug flag */
24   private static final boolean DEBUG = false;
25 
26   public GateSaxException(String aMessage, Exception anException) {
27     super(aMessage,anException);
28   }
29 
30   public GateSaxException(String aMessage) {
31     super(aMessage);
32   }
33 
34   public GateSaxException(Exception anException) {
35     super(anException.toString());
36   }
37 } // GateSaxException
