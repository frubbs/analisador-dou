029 /**
030  * A Class that defines how each node in the ontology tree should look
031  * like (e.g. Icon, Text and Tooltips etc).
032  * 
033  * @author niraj
034  * 
035  */
036 public class OntoTreeCellRenderer extends DefaultTreeCellRenderer {
037   private static final long serialVersionUID = 3256445798102610225L;
038   public Component getTreeCellRendererComponent(JTree jtree, Object obj,
039           boolean flag, boolean flag1, boolean flag2, int i, boolean flag3) {
040     if(obj != null && (obj instanceof DefaultMutableTreeNode)) {
041       javax.swing.Icon icon = null;
042       String s = null;
043       Object obj1 = ((DefaultMutableTreeNode)obj).getUserObject();
044       if(obj1 != null && obj1 instanceof OResourceNode) {
045         obj1 = ((OResourceNode)obj1).getResource();
046       }
047 
048       if(obj1 instanceof OResource) {
049         Color color = Color.BLACK;
050         setForeground(color);
051       }
052       
053       if(obj1 instanceof Restriction) {
054         icon = MainFrame.getIcon("ontology-restriction");
055         s = ((OClass)obj1).getName();
056         if(((OClass)obj1).getONodeID().isAnonymousResource())
057           setToolTipText(((OClass)obj1).getONodeID().toString()
058                   + " is a restriction");
059         else setToolTipText(((OClass)obj1).getONodeID().toString());
060       }
061       else if(obj1 instanceof OClass) {
062         icon = MainFrame.getIcon("ontology-class");
063         s = ((OClass)obj1).getName();
064         if(((OClass)obj1).getONodeID().isAnonymousResource())
065           setToolTipText(((OClass)obj1).getONodeID().toString()
066                   + " is an annonymous class");
067         else setToolTipText(((OClass)obj1).getONodeID().toString());
068       }
069       else if(obj1 instanceof OInstance) {
070         icon = MainFrame.getIcon("ontology-instance");
071         s = ((OInstance)obj1).getName();
072         setToolTipText(((OInstance)obj1).getONodeID().toString());
073       }
074       else if(obj1 instanceof AnnotationProperty) {
075         icon = MainFrame.getIcon("ontology-annotation-property");
076         s = ((AnnotationProperty)obj1).getName();
077         setToolTipText(((AnnotationProperty)obj1).getONodeID().toString());
078       }
079       else if(obj1 instanceof DatatypeProperty) {
080         icon = MainFrame.getIcon("ontology-datatype-property");
081         s = ((DatatypeProperty)obj1).getName();
082         setToolTipText(((DatatypeProperty)obj1).getONodeID().toString());
083       }
084       else if(obj1 instanceof SymmetricProperty) {
085         icon = MainFrame.getIcon("ontology-symmetric-property");
086         s = ((SymmetricProperty)obj1).getName();
087         setToolTipText(((SymmetricProperty)obj1).getONodeID().toString());
088       }
089       else if(obj1 instanceof TransitiveProperty) {
090         icon = MainFrame.getIcon("ontology-transitive-property");
091         s = ((TransitiveProperty)obj1).getName();
092         setToolTipText(((TransitiveProperty)obj1).getONodeID().toString());
093       }
094       else if(obj1 instanceof ObjectProperty) {
095         icon = MainFrame.getIcon("ontology-object-property");
096         s = ((ObjectProperty)obj1).getName();
097         setToolTipText(((ObjectProperty)obj1).getONodeID().toString());
098       }
099       else if(obj1 instanceof RDFProperty) {
100         icon = MainFrame.getIcon("ontology-rdf-property");
101         s = ((RDFProperty)obj1).getName();
102         setToolTipText(((RDFProperty)obj1).getONodeID().toString());
103       }
104       if(icon != null) {
105         if(flag1)
106           setOpenIcon(icon);
107         else setClosedIcon(icon);
108         if(flag2) setLeafIcon(icon);
109       }
110       super
111               .getTreeCellRendererComponent(jtree, s, flag, flag1, flag2, i,
112                       flag3);
113     }
114     else {
115       super.getTreeCellRendererComponent(jtree, obj, flag, flag1, flag2, i,
116               flag3);
117     }
118     return this;
119   }
120 }
