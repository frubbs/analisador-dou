028 /**
029   * A CPSL rule. Has an LHS, RHS and a name, and a priority.
030   */
031 public class Rule extends Transducer
032 implements JapeConstants, java.io.Serializable
033 {
034   /** Debug flag */
035   private static final boolean DEBUG = false;
036 
037   /** Construction */
038   public Rule(
039     String name, int position, int priority,
040     LeftHandSide lhs, RightHandSide rhs
041   ) {
042     this.name = name;
043     this.position = position;
044     this.priority = priority;
045     this.lhs = lhs;
046     this.rhs = rhs;
047   } // Construction
048   
049   /** Copy constructor */
050   public Rule(Rule existingRule) {
051     this.name = existingRule.name;
052     this.position = existingRule.position;
053     this.priority = existingRule.priority;
054     this.lhs = existingRule.lhs;
055     this.rhs = new RightHandSide(existingRule.rhs);
056   }
057 
058   /** The LHS or pattern of the rule. */
059   private LeftHandSide lhs;
060 
061   /** The RHS or action of the rule. */
062   private RightHandSide rhs;
063 
064   /** The priority of the rule. */
065   private int priority;
066 
067   /** Get the rule priority. */
068   public int getPriority() { return priority; }
069 
070   /** The rule's position in sequence (e.g. order in file). */
071   private int position;
072 
073   /** Get the rule's position in sequence (e.g. order in file). */
074   public int getPosition() { return position; }
075 
076   /** If we're pending (have matched), get the position we want to fire in,
077     * else -1.
078     */
079   public int pending() {
080     return pendingPosition;
081   } // pending
082 
083   /** If we matched but didn't fire yet, this is our pending position. */
084   private int pendingPosition = -1;
085 
086   /** Flag for end of document during getNextMatch. */
087   private boolean weFinished = false;
088 
089   /** Have we hit the end of the document without matching? */
090   public boolean finished() {
091     return weFinished;
092   } // finished
093 
094   /** Finish: replace dynamic data structures with Java arrays; called
095     * after parsing. WARNING:
096     * bad choice of names: this is not related to the weFinished
097     * member or the finished method!
098     */
099   public void finish(GateClassLoader classloader) {
100     lhs.finish();
101   } // finish
102 
103 
104 
105 
106   /** Apply the RHS of this rule (LHS must have been matched first). */
107   public void transduce(Document doc, AnnotationSet inputAS,
108                         AnnotationSet outputAS) throws JapeException {
109     // the righthand side does the transduction, using bindings from lhs */
110     if(DEBUG) Out.println("applying rule " + name);
111 //    rhs.transduce(doc);
112     /*Debug.pr(
113       this, "Rule.transduce: annotations after transduction: " +
114       doc.selectAnnotations("Name", new FeatureMap()).toString() +
115       Debug.getNl()
116     );*/
117 
118     // clear the caches of matched annotations in the LHS
119 //    reset();
120     //Debug.pr(this, "LHS after reset: " + lhs.toString());
121 
122   } // transduce

127 
128   /** Clean up (delete action class files, for e.g.). */
129   public void cleanUp() {
130     RightHandSide.cleanUp();
131   } // cleanUp

134   /** Create a string representation of the object. */
135   public String toString() { return toString(""); }
136 
137   /** Create a string representation of the object. */
138   public String toString(String pad) {
139     String newline = Strings.getNl();
140     String newPad = Strings.addPadding(pad, INDENT_PADDING);
141 
142     StringBuffer buf = new StringBuffer(
143       pad + "Rule: name(" + name + "); position(" + position + "); priority(" +
144       priority + "); pendingPosition(" + pendingPosition + "); " +
145       "weFinished(" + weFinished + "); lhs(" + newline +
146       lhs.toString(newPad) + newline + pad + "); rhs(" + newline +
147       rhs.toString(newPad) + newline + pad + ");"
148     );
149 
150     buf.append(newline + pad + ") Rule." + newline);
151 
152     return buf.toString();
153   } // toString

155   //needed by FSM
156   public LeftHandSide getLHS(){
157     return lhs;
158   }
159   public RightHandSide getRHS(){
160     return rhs;
161   }

170   //ProcessProgressReporter implementation ends here
