104   /** Mapping Tree Cell Renderer distinguishes nodes, originating from ontology classes from
105    *  nodes, originating from gazetteer lists. */
106   class MappingTreeCR extends DefaultTreeCellRenderer {
107 
108      private static final long serialVersionUID = 3546924666926085169L;
109 
110     /**
111      * Sets the value of the current tree cell to <code>value</code>.
112      * If <code>selected</code> is true, the cell will be drawn as if
113      * selected. If <code>expanded</code> is true the node is currently
114      * expanded and if <code>leaf</code> is true the node represets a
115      * leaf anf if <code>hasFocus</code> is true the node currently has
116      * focus. <code>tree</code> is the JTree the receiver is being
117      * configured for.
118      * Returns the Component that the renderer uses to draw the value.
119      *
120      * @return  Component that the renderer uses to draw the value.
121      */
122     public Component getTreeCellRendererComponent(JTree tree, Object value,
123            boolean selected, boolean expanded,
124            boolean leaf, int row, boolean hasFocus) {
125       super.getTreeCellRendererComponent(
126         tree, value, selected, expanded, leaf, row, hasFocus);
127 
128       if ( value instanceof ClassNode ) {
129         ClassNode cn = (ClassNode) value;
130         Object source = cn.getSource();
131         if ( source instanceof MappingNode ) {
132           setIcon(MainFrame.getIcon(GAZ_ICON));
133         } // if gaz list
134       } // if node
135       return this;
136     } // getTreeCellRendererComponent()
137 
138   } // class MappingTreeCR

140   /**The mouse adapter listens to the entire mouse activity and invokes a popup menu if
141    * right click. */
142   class MyMouseAdapter extends MouseAdapter{
143 
144       public MyMouseAdapter(){
145       }
146 
147       public void mouseClicked(MouseEvent e){
148           TreePath path=MappingTreeView.this.getSelectionPath();
149           
150           ClassNode node =null;
151           if (SwingUtilities.isLeftMouseButton(e)) {
152             if (2 == e.getClickCount()) {
153               if( path != null){
154                 node = (ClassNode) path.getLastPathComponent();
155                 if ( node.getSource() instanceof MappingNode ) {
156                   MappingNode mn = (MappingNode)node.getSource();
157                   gaze.displayList(mn.getList());
158                 }
159               } // if !=null
160             } // double click
161           }  // left
162 
163           if(SwingUtilities.isRightMouseButton(e)){
164             if( path != null){
165               node = (ClassNode) path.getLastPathComponent();
166               if ( node.getSource() instanceof MappingNode ) {
167                 removePuP.setEnabled(true);
168                 insertPuP.setEnabled(false);
169               } else {
170                 removePuP.setEnabled(false);
171                 insertPuP.setEnabled(true);
172               }
173               pupMenu.show(MappingTreeView.this,e.getX(),e.getY());
174             }
175           }
176       }
177   } //class MyMouseAdapter

180   /*Action Listener of the remove pop up menu item */
181   class RemoveAL implements ActionListener{
182     public void actionPerformed(ActionEvent e) {
183       ClassNode node = (ClassNode)MappingTreeView.this.getLastSelectedPathComponent();
184       Object source = node.getSource();
185       if (source instanceof MappingNode) {
186         TreePath pp = MappingTreeView.this.getAnchorSelectionPath().getParentPath();
187         if (null!=pp) {
188           ClassNode pNode = (ClassNode)pp.getLastPathComponent();
189           Vector<ClassNode> kids = pNode.children();
190           kids.remove(node);
191           pNode.setChildren(kids);
192           mapping.remove(source);
193           MappingTreeView.this.updateUI();
194           gaze.updateMappingUI();
195         } // pp ! null
196       }// if map node
197     } // actionPerformed()
198   } // class RemoveAL

201   /*Action Listener of the insert pop up menu item */
202   class InsertAL implements ActionListener {
203     public void actionPerformed(ActionEvent e) {
204       ClassNode node = (ClassNode)MappingTreeView.this.getLastSelectedPathComponent();
205       Object source = node.getSource();
206       if (source instanceof OClass) {
207         List lists = gaze.getLists();
208         Collections.sort(lists);
209 
210         Object result = JOptionPane.showInputDialog(MappingTreeView.this,
211                           "Map selected ontology class to a gazetteer list:",
212                           "Insert Mapping Node",
213                           JOptionPane.PLAIN_MESSAGE,
214                           null,lists.toArray(),null);
215         if (null != result) {
216           OClass oc = (OClass) source;
217           MappingNode mn = new MappingNode((String)result,
218               oc.getOntology().getURL().toString(),
219               node.toString());
220           mapping.add(mn);
221           ClassNode cn = new ClassNode(mn);
222           Vector<ClassNode> kids = node.children();
223           kids.add(cn);
224           MappingTreeView.this.updateUI();
225           gaze.updateMappingUI();
226         } // null!=result
227       }// if map node
228     } // actionPerformed()
229   } // class InsertAL
