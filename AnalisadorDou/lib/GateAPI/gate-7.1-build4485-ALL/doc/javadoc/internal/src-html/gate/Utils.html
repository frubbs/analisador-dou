037 /**
038  * Various utility methods to make often-needed tasks more easy and
039  * using up less code.  In Java code (or JAPE grammars) you may wish to
040  * <code>import static gate.Utils.*</code> to access these methods without
041  * having to qualify them with a class name.  In Groovy code, this class can be
042  * used as a category to inject each utility method into the class of its first
043  * argument, e.g.
044  * <pre>
045  * Document doc = // ...
046  * Annotation ann = // ...
047  * use(gate.Utils) {
048  *   println "Annotation has ${ann.length()} characters"
049  *   println "and covers the string \"${doc.stringFor(ann)}\""
050  * }
051  * </pre>
052  *
053  * @author Johann Petrak, Ian Roberts
054  */

056   /**
057    * Return the length of the document content covered by an Annotation as an
058    * int -- if the content is too long for an int, the method will throw
059    * a GateRuntimeException. Use getLengthLong(SimpleAnnotation ann) if
060    * this situation could occur.
061    * @param ann the annotation for which to determine the length
062    * @return the length of the document content covered by this annotation.
063    */
064   public static int length(SimpleAnnotation ann) {
065     long len = lengthLong(ann);
066     if (len > java.lang.Integer.MAX_VALUE) {
067       throw new GateRuntimeException(
068               "Length of annotation too big to be returned as an int: "+len);
069     } else {
070       return (int)len;
071     }
072   }

305   /**
306    * Return a the subset of annotations from the given annotation set
307    * that start exactly at the given offset.
308    *
309    * @param annotationSet the set of annotations from which to select
310    * @param atOffset the offset where the annoation to be returned should start
311    * @return an annotation set containing all the annotations from the original
312    * set that start at the given offset
313    */
314   public static AnnotationSet getAnnotationsAtOffset(
315           AnnotationSet annotationSet, Long atOffset) {
316     // this returns all annotations that start at this atOffset OR AFTER!
317     AnnotationSet tmp = annotationSet.get(atOffset);
318     // so lets filter ...
319     AnnotationSet ret = new AnnotationSetImpl(annotationSet.getDocument());
320     Iterator<Annotation> it = tmp.iterator();
321     while(it.hasNext()) {
322       Annotation ann = it.next();
323       if(ann.getStartNode().getOffset().equals(atOffset)) {
324         ret.add(ann);
325       }
326     }
327     return ret;
328   }

659   /** 
660    * This method can be used to check if a ProcessingResource has 
661    * a chance to be run in the given controller with the current settings.
662    * <p>
663    * That means that for a non-conditional controller, the method will return
664    * true if the PR is part of the controller. For a conditional controller,
665    * the method will return true if it is part of the controller and at least
666    * once (if the same PR is contained multiple times) it is not disabled.
667    * 
668    * @param controller 
669    * @param pr
670    * @return true or false depending on the conditions explained above.
671    */
672   public static boolean isEnabled(Controller controller, ProcessingResource pr) {
673     Collection<ProcessingResource> prs = controller.getPRs();
674     if(!prs.contains(pr)) {
675       return false;
676     }
677     if(controller instanceof ConditionalSerialController) {
678       Collection<RunningStrategy> rss = 
679         ((ConditionalSerialController)controller).getRunningStrategies();
680       for(RunningStrategy rs : rss) {
681         // if we find at least one occurrence of the PR that is not disabled
682         // return true
683         if(rs.getPR().equals(pr) && 
684            rs.getRunMode() != RunningStrategy.RUN_NEVER) {
685           return true;
686         }
687       }
688       // if we get here, no occurrence of the PR has found or none that
689       // is not disabled, so return false
690       return false;
691     }
692     return true;
693   }
