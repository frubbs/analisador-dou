092     }
093       
094       if(file.exists()) {

099 
100     String baseTokenAnnotationType = (String)parameters
101             .get(Constants.BASE_TOKEN_ANNOTATION_TYPE);
102     if(baseTokenAnnotationType.indexOf(".") > -1 || baseTokenAnnotationType.indexOf("=") > -1

108     String indexUnitAnnotationType = (String)parameters
109             .get(Constants.INDEX_UNIT_ANNOTATION_TYPE);
110     
111     if(DEBUG) {

114     }
115 
116     if(baseTokenAnnotationType == null
117             || baseTokenAnnotationType.trim().length() == 0) {
118       baseTokenAnnotationType = Constants.ANNIC_TOKEN;

251       }

253     isDeleted = dir.delete();
254     if(!isDeleted) {
255       throw new IndexException("Can't delete directory" + dir.getAbsolutePath());

467     }
468 
469     if(!file.exists()) return;
470 
471     java.io.FileReader fileReader = new java.io.FileReader(file);
472 
473     try {
474       // other wise read this and

488   /**

491    * 
492    * @throws IOException
493    */
494   private void writeParametersToDisk() throws IOException {
495     // we create a hashmap to store index definition in the index
496     // directory
497     URL location = (URL)parameters.get(Constants.INDEX_LOCATION_URL);
498     File file = null;
499     try {
500       file = new File(new File(location.toURI()), "LuceneIndexDefinition.xml");
501     } catch(URISyntaxException use) {
502       file = new File(location.getFile(), "LuceneIndexDefinition.xml");
503     }
504 
505     java.io.FileWriter fileWriter = new java.io.FileWriter(file);
506     Map indexInformation = new HashMap();
507     Iterator iter = parameters.keySet().iterator();
508     while(iter.hasNext()) {
509       Object key = iter.next();
510       if(key.equals(Constants.INDEX_LOCATION_URL)) continue;
511       indexInformation.put(key, parameters.get(key));
512     }
513 
514     indexInformation.put(Constants.CORPUS_INDEX_FEATURE,
515             Constants.CORPUS_INDEX_FEATURE_VALUE);
516     if(corpus != null)
517       indexInformation.put(Constants.CORPUS_SIZE, new Integer(corpus
518               .getDocumentNames().size()));
519 
520     // we would use XStream library to store annic patterns
521     com.thoughtworks.xstream.XStream xstream = new com.thoughtworks.xstream.XStream();
522 
523     // Saving is accomplished just using XML serialization of
524     // the map.
525     try {
526       xstream.toXML(indexInformation, fileWriter);
527     }
528     finally {
529       fileWriter.close();
530     }
