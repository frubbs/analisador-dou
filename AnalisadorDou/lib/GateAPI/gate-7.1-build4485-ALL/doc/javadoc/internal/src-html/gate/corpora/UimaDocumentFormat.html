029 /**
030  * UIMA XCAS and XMICAS document formats.
031  */
032 @CreoleResource(name = "UIMA Document Format", isPrivate = true,
033     autoinstances = {@AutoInstance(hidden = true)})
034 public class UimaDocumentFormat extends XmlDocumentFormat {
035   /** Debug flag */
036   private static final boolean DEBUG = false;
037 
038   public void unpackMarkup(Document doc, RepositioningInfo repInfo,
039           RepositioningInfo ampCodingInfo) throws DocumentFormatException {
040     super.unpackMarkup(doc, repInfo, ampCodingInfo);
041     unpackCasMarkup(doc);
042   }
043 
044   /**
045    * Convert UIMA CAS markups to GATE markups.
046    * @param doc XML document already parsed
047    * @throws DocumentFormatException error when parsing the file
048    */
049   private void unpackCasMarkup(Document doc)
050     throws DocumentFormatException {
051 
052     AnnotationSet inputAS = doc.getAnnotations("Original markups");
053     AnnotationSet outputAS = doc.getAnnotations("Original markups");
054 
055     // set format specific names
056     String casPrefix;
057     String idName;
058     if (!inputAS.get("CAS").isEmpty()) {
059       casPrefix = "uima.cas.";
060       idName = "_id";
061     } else if (!inputAS.get("xmi:XMI").isEmpty()) {
062       casPrefix = "cas:";
063       idName = "xmi:id";
064     } else {
065       throw new DocumentFormatException("The document \"" + doc.getName()
066         + "\" is neither of XCAS nor XMICAS format.");
067     }
068 
069     // get array/list contained elements annotations
070     for (Annotation annotation : inputAS) {
071       if (annotation.getType().matches(casPrefix + "[a-zA-Z]+(List|Array)")) {
072         try {
073           String elements = doc.getContent().getContent(
074             annotation.getStartNode().getOffset(),
075             annotation.getEndNode().getOffset()).toString();
076           // add contained values as a feature to the array annotation
077           if (!elements.trim().equals("")) {
078             annotation.getFeatures().put("elements", elements);
079           }
080         } catch (InvalidOffsetException e) {
081           throw new DocumentFormatException(e);
082         }
083       }
084     }
085 
086     // get document content from SOFA annotations
087     Set<Annotation> sofaSet = inputAS.get(casPrefix + "Sofa");
088     if (sofaSet.size() > 1) {
089       Out.prln("More than one UIMA SOFA, annotation offsets won't be correct.");
090     }
091     String documentContent = "";
092     for (Annotation annotation : sofaSet) {
093       documentContent += (String) annotation.getFeatures().get("sofaString");
094     }
095     doc.setContent(new DocumentContentImpl(documentContent));
096 
097     // remove SOFA annotations
098     inputAS.removeAll(sofaSet);
099 
100     // remove non document annotations
101     inputAS.removeAll(inputAS.get("CAS"));
102     inputAS.removeAll(inputAS.get("xmi:XMI"));
103     inputAS.removeAll(inputAS.get("cas:NULL"));
104 
105     // get the views members, views will be added later as annotation sets
106     List<List<String>> viewList = new ArrayList<List<String>>();
107     for (Annotation view : inputAS.get(casPrefix + "View")) {
108       viewList.add(Arrays.asList(((String)
109         view.getFeatures().get("members")).split("\\s+")));
110     }
111     inputAS.removeAll(inputAS.get(casPrefix + "View"));
112 
113     // fill a map with the id as key and the entity name as value
114     // this is specific to the Temis Luxid CAS format
115     Map<String, String> entityMap = new HashMap<String,String>();
116     for (Annotation entity : inputAS.get("com.temis.uima.Entity")) {
117       FeatureMap features = entity.getFeatures();
118       entityMap.put((String) features.get(idName),
119         (String) features.get("value"));
120     }
121 
122     try {
123       // for each UIMA annotation
124       for (Annotation annotation : new HashSet<Annotation>(inputAS)) {
125 
126         FeatureMap features = Factory.newFeatureMap();
127         features.putAll(annotation.getFeatures());
128         String start = (String) features.get("begin");
129         String end = (String) features.get("end");
130         String id = (String) features.get(idName);
131         features.remove("begin"); // UIMA feature
132         features.remove("end"); // UIMA feature
133         features.remove("isEmptyAndSpan"); // GATE feature
134         features.remove("_indexed"); // UIMA XCAS feature
135 
136         if (start == null || end == null) {
137           // no offsets so add it as a GATE document feature
138           features.remove(idName);
139           for (Map.Entry<Object,Object> entry : features.entrySet()) {
140             doc.getFeatures().put(annotation.getType()
141               + '_' + id + '.' + entry.getKey(), entry.getValue());
142           }
143         } else { // offsets so add it as a GATE document annotation
144           String entityReference = (String) features.get("_ref_entity");
145           String type = entityMap.containsKey(entityReference)?
146             entityMap.get(entityReference) : annotation.getType();
147           Integer gateId = outputAS.add(
148             Long.valueOf(start), Long.valueOf(end), type, features);
149           int viewCount = 0;
150           for (List<String> viewMembers : viewList) {
151             if (viewMembers.contains(id)) {
152               // add the annotation to the annotation set
153               doc.getAnnotations("CasView" + viewCount)
154                 .add(outputAS.get(gateId));
155             }
156             viewCount++;
157           }
158         }
159         // delete UIMA annotation
160         inputAS.remove(annotation);
161       }
162     } catch (InvalidOffsetException e) {
163       throw new DocumentFormatException("Couldn't create annotation.", e);
164     }
165   }

167   public Resource init() throws ResourceInstantiationException {
168     // Register XML mime type
169     MimeType mime = new MimeType("text", "xmi+xml");
170     // Register the class handler for this mime type
171     mimeString2ClassHandlerMap.put(mime.getType() + "/" + mime.getSubtype(),
172       this);
173     // Register the mime type with mine string
174     mimeString2mimeTypeMap.put(mime.getType() + "/" + mime.getSubtype(), mime);
175     // Register file suffixes for this mime type
176     suffixes2mimeTypeMap.put("xcas", mime);
177     suffixes2mimeTypeMap.put("xmicas", mime);
178     suffixes2mimeTypeMap.put("xmi", mime);
179     // Register magic numbers for this mime type
180     magic2mimeTypeMap.put("<CAS version=\"2\">", mime);
181     magic2mimeTypeMap.put("xmlns:cas=", mime);
182     // Set the mimeType for this language resource
183     setMimeType(mime);
184     return this;
185   }
