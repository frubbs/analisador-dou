055   /**
056    * Execute this controller. This implementation takes care of informing any
057    * {@link ControllerAwarePR}s of the start and end of execution, and
058    * delegates to the {@link #executeImpl()} method to do the real work.
059    * Subclasses should override {@link #executeImpl()} rather than this method.
060    */
061   public void execute() throws ExecutionException {
062 
063     // inform ControllerAware PRs that execution has started
064     for(ControllerAwarePR pr : getControllerAwarePRs()) {
065       pr.controllerExecutionStarted(this);
066     }
067     Throwable thrown = null;
068     try {
069       if(Benchmark.isBenchmarkingEnabled()) {
070         // write a start marker to the benchmark log for this
071         // controller as a whole
072         Benchmark.startPoint(getBenchmarkId());
073       }
074       // do the real work
075       this.executeImpl();
076     }
077     catch(Throwable t) {
078       thrown = t;
079     }
080     finally {
081       if(thrown == null) {
082         // successfully completed
083         for(ControllerAwarePR pr : getControllerAwarePRs()) {
084           pr.controllerExecutionFinished(this);
085         }
086       }
087       else {
088         // aborted
089         for(ControllerAwarePR pr : getControllerAwarePRs()) {
090           pr.controllerExecutionAborted(this, thrown);
091         }
092 
093         // rethrow the aborting exception or error
094         if(thrown instanceof Error) {
095           throw (Error)thrown;
096         }
097         else if(thrown instanceof RuntimeException) {
098           throw (RuntimeException)thrown;
099         }
100         else if(thrown instanceof ExecutionException) {
101           throw (ExecutionException)thrown;
102         }
103         else {
104           // we have a checked exception that isn't one executeImpl can
105           // throw. This shouldn't be possible, but just in case...
106           throw new UndeclaredThrowableException(thrown);
107         }
108       }
109     }
110 
111   }

113   /**
114    * Get the set of PRs from this controller that implement
115    * {@link ControllerAwarePR}. If there are no such PRs in this controller, an
116    * empty set is returned. This implementation simply filters the collection
117    * returned by {@link Controller#getPRs()}, override this method if your
118    * subclass admits a more efficient implementation.
119    */
120   protected Set<ControllerAwarePR> getControllerAwarePRs() {
121     Set<ControllerAwarePR> returnSet = null;
122     for(Object pr : getPRs()) {
123       if(pr instanceof ControllerAwarePR) {
124         if(returnSet == null) {
125           returnSet = new HashSet<ControllerAwarePR>();
126         }
127         returnSet.add((ControllerAwarePR)pr);
128       }
129     }
130 
131     if(returnSet == null) {
132       // optimization - don't waste time creating a new set in the most
133       // common case where there are no Controller aware PRs
134       return Collections.emptySet();
135     }
136     else {
137       return returnSet;
138     }
139   }

299    * A progress listener used to convert a 0..100 interval into a smaller one
300    */
301   protected class IntervalProgressListener implements ProgressListener {
302     public IntervalProgressListener(int start, int end) {
303       this.start = start;
304       this.end = end;
305     }
306 
307     public void progressChanged(int i) {
308       fireProgressChanged(start + (end - start) * i / 100);
309     }
310 
311     public void processFinished() {
312       fireProgressChanged(end);
313     }
314 
315     int start;
316     int end;
317   }// CustomProgressListener

328   /**
329    * Checks whether all the contained PRs have all the required runtime
330    * parameters set.
331    * 
332    * @return a {@link List} of {@link ProcessingResource}s that have required
333    *         parameters with null values if they exist <tt>null</tt>
334    *         otherwise.
335    * @throws {@link ResourceInstantiationException}
336    *           if problems occur while inspecting the parameters for one of the
337    *           resources. These will normally be introspection problems and are
338    *           usually caused by the lack of a parameter or of the read accessor
339    *           for a parameter.
340    */
341   public List getOffendingPocessingResources()
342     throws ResourceInstantiationException {
343     // take all the contained PRs
344     ArrayList badPRs = new ArrayList(getPRs());
345     // remove the ones that no parameters problems
346     Iterator prIter = getPRs().iterator();
347     while(prIter.hasNext()) {
348       ProcessingResource pr = (ProcessingResource)prIter.next();
349       ResourceData rData =
350         (ResourceData)Gate.getCreoleRegister().get(pr.getClass().getName());
351       if(AbstractResource.checkParameterValues(pr, rData.getParameterList()
352         .getRuntimeParameters())) {
353         badPRs.remove(pr);
354       }
355     }
356     return badPRs.isEmpty() ? null : badPRs;
357   }
