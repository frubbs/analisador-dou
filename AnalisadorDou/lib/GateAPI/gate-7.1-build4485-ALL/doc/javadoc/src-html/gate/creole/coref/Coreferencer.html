139   /** --- */
140   private void generateCorefChains() throws GateRuntimeException {
141 
142     // 1. get the resolved corefs
143     HashMap ana2ant = this.pronominalModule.getResolvedAnaphora();
144 
145     // 2. get the outout annotation set
146     String asName = getAnnotationSetName();
147     AnnotationSet outputSet = null;
148 
149     if(null == asName || asName.equals("")) {
150       outputSet = getDocument().getAnnotations();
151     }
152     else {
153       outputSet = getDocument().getAnnotations(asName);
154     }
155 
156     // 3. generate new annotations
157     Iterator it = ana2ant.entrySet().iterator();
158     while(it.hasNext()) {
159       Map.Entry currLink = (Map.Entry)it.next();
160       Annotation anaphor = (Annotation)currLink.getKey();
161       Annotation antecedent = (Annotation)currLink.getValue();
162 
163       if(DEBUG) {
164         AnnotationSet corefSet = getDocument().getAnnotations("COREF");
165         Long antOffset = new Long(0);
166 
167         if(null != antecedent) {
168           antOffset = antecedent.getStartNode().getOffset();
169         }
170 
171         FeatureMap features = new SimpleFeatureMapImpl();
172         features.put("antecedent", antOffset);
173         corefSet.add(anaphor.getStartNode(), anaphor.getEndNode(), "COREF",
174                 features);
175       }
176 
177       // do we have antecedent?
178       if(null == antecedent) {
179         continue;
180       }
181 
182       // get the ortho-matches of the antecedent
183       List matches = null;
184       Object matchesObj = antecedent.getFeatures().get(
185           ANNOTATION_COREF_FEATURE_NAME);
186       if(matchesObj != null) {
187         if(matchesObj instanceof List) {
188           matches = (List)matchesObj;  
189         } else {
190           log.warn("Illegal value for " + ANNOTATION_COREF_FEATURE_NAME + 
191               " feature was ignored.");
192         }
193       }
194         
195       if(null == matches) {
196         matches = new ArrayList();
197         matches.add(antecedent.getId());
198         antecedent.getFeatures().put(ANNOTATION_COREF_FEATURE_NAME, matches);
199         // check if the document has a list of matches
200         // if yes, simply add the new list to it
201         // if not, create it and add the list of matches to it
202         if(document.getFeatures().containsKey(DOCUMENT_COREF_FEATURE_NAME)) {
203           Map matchesMap = (Map)document.getFeatures().get(
204                   DOCUMENT_COREF_FEATURE_NAME);
205           List matchesList = (List)matchesMap.get(getAnnotationSetName());
206           if(matchesList == null) {
207             matchesList = new ArrayList();
208             matchesMap.put(getAnnotationSetName(), matchesList);
209           }
210           matchesList.add(matches);
211         }
212         else {
213           Map matchesMap = new HashMap();
214           List matchesList = new ArrayList();
215           matchesMap.put(getAnnotationSetName(), matchesList);
216           matchesList.add(matches);
217         }// if else
218       }// if matches == null
219 
220       FeatureMap features = new SimpleFeatureMapImpl();
221       features.put(COREF_TYPE_FEATURE_NAME, "PRONOUN");
222       features.put(ANNOTATION_COREF_FEATURE_NAME, matches);
223       features.put(COREF_ANTECEDENT_FEATURE_NAME, antecedent.getStartNode()
224               .getOffset());
225 
226       //see if the annotation we want to add already exists
227       AnnotationSet existing = outputSet.get(antecedent.getType(), anaphor
228               .getStartNode().getOffset(), anaphor.getEndNode().getOffset());
229 
230       if(existing.size() > 0) {
231         //if it exists simply update the existing annotation
232         Annotation annot = existing.iterator().next();
233         annot.getFeatures().putAll(features);
234         matches.add(annot.getId());
235       }
236       else {
237         //if it doesn't exist create a new annotation
238         matches.add(outputSet.add(anaphor.getStartNode(), anaphor.getEndNode(),
239                 antecedent.getType(), features));
240       }
241     }
242   }
