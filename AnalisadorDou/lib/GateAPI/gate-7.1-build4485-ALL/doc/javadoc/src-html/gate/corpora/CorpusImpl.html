344   /**
345    * Fills the provided corpus with documents created on the fly from
346    * selected files in a directory. Uses a {@link FileFilter} to select
347    * which files will be used and which will be ignored. A simple file
348    * filter based on extensions is provided in the Gate distribution (
349    * {@link gate.util.ExtensionFileFilter}).
350    * 
351    * @param corpus the corpus to be populated
352    * @param directory the directory from which the files will be picked.
353    *          This parameter is an URL for uniformity. It needs to be a
354    *          URL of type file otherwise an InvalidArgumentException
355    *          will be thrown.
356    * @param filter the file filter used to select files from the target
357    *          directory. If the filter is <tt>null</tt> all the files
358    *          will be accepted.
359    * @param encoding the encoding to be used for reading the documents
360    * @param recurseDirectories should the directory be parsed
361    *          recursively?. If <tt>true</tt> all the files from the
362    *          provided directory and all its children directories (on as
363    *          many levels as necessary) will be picked if accepted by
364    *          the filter otherwise the children directories will be
365    *          ignored.
366    * @throws java.io.IOException if a file doesn't exist
367    */
368   public static void populate(Corpus corpus, URL directory, FileFilter filter,
369           String encoding, String mimeType, boolean recurseDirectories)
370           throws IOException {
371 
372     // check input
373     if(!directory.getProtocol().equalsIgnoreCase("file"))
374       throw new IllegalArgumentException(
375               "The URL provided is not of type \"file:\"!");
376 
377     File dir = Files.fileFromURL(directory);
378     if(!dir.exists()) throw new FileNotFoundException(dir.toString());
379 
380     if(!dir.isDirectory())
381       throw new IllegalArgumentException(dir.getAbsolutePath()
382               + " is not a directory!");
383 
384     File[] files;
385     // populate the corpus
386     if(recurseDirectories) {
387       files = Files.listFilesRecursively(dir, filter);
388     }
389     else {
390       files = dir.listFiles(filter);
391     }
392 
393     if(files == null) {
394       return;
395     }
396 
397     // sort the files alphabetically regardless of their paths
398     Arrays.sort(files, new Comparator<File>() {
399       public int compare(File f1, File f2) {
400         return f1.getName().compareTo(f2.getName());
401       }
402     });
403 
404     // create the GATE documents
405     for(File file : files) {
406       if(file.isDirectory()) {
407         continue;
408       }
409       StatusListener sListener = (StatusListener)Gate.getListeners().get(
410               "gate.event.StatusListener");
411       if(sListener != null)
412         sListener.statusChanged("Reading: " + file.getName());
413       String docName = file.getName() + "_" + Gate.genSym();
414       FeatureMap params = Factory.newFeatureMap();
415       params.put(Document.DOCUMENT_URL_PARAMETER_NAME, file.toURI().toURL());
416       if(encoding != null)
417         params.put(Document.DOCUMENT_ENCODING_PARAMETER_NAME, encoding);
418       if(mimeType != null)
419         params.put(Document.DOCUMENT_MIME_TYPE_PARAMETER_NAME, mimeType);
420 
421       try {
422         Document doc = (Document)Factory.createResource(DocumentImpl.class
423                 .getName(), params, null, docName);
424         corpus.add(doc);
425         if(corpus.getLRPersistenceId() != null) {
426           // persistent corpus -> unload the document
427           corpus.unloadDocument(doc);
428           Factory.deleteResource(doc);
429         }
430       }
431       catch(Throwable t) {
432         String nl = Strings.getNl();
433         Err.prln("WARNING: Corpus.populate could not instantiate document" + nl
434                 + "  Document name was: " + docName + nl + "  Exception was: "
435                 + t + nl + nl);
436         t.printStackTrace();
437       }
438       if(sListener != null) sListener.statusChanged(file.getName() + " read");
439     }
440 
441   }// public static void populate

503    *

505    * @param singleConcatenatedFile the trec file.
506    * @param documentRootElement text between this element (start and
507    *          end) is considered for creating a new document.
508    * @param encoding the encoding of the trec file.

511    * @param documentType type of the document it is (i.e. xml, html etc)

516   @Deprecated

518   public static long populate(Corpus corpus, URL singleConcatenatedFile,
519       String documentRootElement, String encoding,
520       int numberOfDocumentsToExtract, String documentNamePrefix,
521       DocType documentType) throws IOException {
522     String mimeType = null;
523 
524     if(DocType.XML.equals(documentType))
525       mimeType = "text/xml";
526     else if(DocType.HTML.equals(documentType)) mimeType = "text/html";
527 
528     return populate(corpus, singleConcatenatedFile, documentRootElement,
529         encoding, numberOfDocumentsToExtract, documentNamePrefix, mimeType, true);

532   public static long populate(Corpus corpus, URL singleConcatenatedFile,
533       String documentRootElement, String encoding,
534       int numberOfDocumentsToExtract, String documentNamePrefix,
535       String mimeType, boolean includeRootElement) throws IOException { 
536     
537     StatusListener sListener = (StatusListener)gate.Gate.getListeners().get("gate.event.StatusListener");
538     
539     // obtain the root element that user has provided
540     // content between the start and end of root element is considered
541     // for creating documents
542     documentRootElement = documentRootElement.toLowerCase();
543 
544     // document name prefix could be an empty string
545     documentNamePrefix = documentNamePrefix == null ? "" : documentNamePrefix
546             .trim()
547             + "_";
548 
549     // we start a new document when we find <documentRootElement> and
550     // close it when we find </documentRootElement>
551     BufferedReader br = null;
552     try {
553       
554       if(encoding != null && encoding.trim().length() != 0) {
555         br = new BomStrippingInputStreamReader(singleConcatenatedFile.openStream(),
556                 encoding, 10485760);
557       }
558       else {
559         br = new BomStrippingInputStreamReader(singleConcatenatedFile.openStream(),
560                 10485760);
561       }
562 
563       // reading line by line
564       String line = br.readLine();
565 
566       // this is where we store document content
567       StringBuilder documentString = new StringBuilder();
568 
569       // toggle switch to indicate search for start element
570       boolean searchingForStartElement = true;
571 
572       // keeping count of number of documents extracted
573       int count = 1;
574 
575       // length in bytes read so far (to return)
576       long lengthInBytes = 0;
577 
578       // continue until reached the end of file
579       while(line != null) {
580 
581         // lowercase the line in order to match documentRootElement in any case
582         String lowerCasedLine = line.toLowerCase();
583 
584         // if searching for startElement?
585         if(searchingForStartElement) {
586 
587           // may be its with attributes
588           int index = lowerCasedLine.indexOf("<" + documentRootElement + " ");
589 
590           // may be no attributes?
591           if(index == -1) {
592             index = lowerCasedLine.indexOf("<" + documentRootElement + ">");
593           }
594 
595           // if index <0, we are out of the content boundaries, so simply
596           // skip the current line and start reading from the next line
597           if(index != -1) {
598             // if found, that's the first line
599             line = line.substring(index);
600             searchingForStartElement = false;
601           }
602           else {
603             line = br.readLine();
604           }
605         }
606         else {
607 
608           // now searching for last element
609           int index = lowerCasedLine.indexOf("</" + documentRootElement + ">");
610 
611           // if not found.. this is the content of a new document
612           if(index == -1) {
613             documentString.append(line + "\n");
614             line = br.readLine();
615           }
616           else {
617 
618             // found.. then end the document
619             documentString.append(line.substring(0, index + documentRootElement.length() + 3));
620 
621             // getting ready for the next document
622             searchingForStartElement = true;
623 
624             // here lets create a new document create the doc
625             if(sListener != null) sListener.statusChanged("Creating Document Number :" + count);
626             
627             String docName = documentNamePrefix + count + "_" + Gate.genSym();
628             
629             String docContent = documentString.toString();
630             
631             if (!includeRootElement)
632               docContent = docContent.substring(docContent.indexOf(">")+1, docContent.lastIndexOf("<"));
633             
634             FeatureMap params = Factory.newFeatureMap();
635             if (mimeType != null) params.put(Document.DOCUMENT_MIME_TYPE_PARAMETER_NAME, mimeType);            
636             params.put(Document.DOCUMENT_STRING_CONTENT_PARAMETER_NAME, docContent);
637             if(encoding != null && encoding.trim().length() > 0)
638               params.put(Document.DOCUMENT_ENCODING_PARAMETER_NAME, encoding); 
639             
640             // calculate the length
641             lengthInBytes += docContent.getBytes().length;            
642 
643             try {
644               Document doc = (Document)Factory.createResource(DocumentImpl.class.getName(), params, null, docName);
645               count++;
646               corpus.add(doc);
647               
648               if(corpus.getLRPersistenceId() != null) {
649                 // persistent corpus -> unload the document
650                 corpus.unloadDocument(doc);
651                 Factory.deleteResource(doc);
652               }
653               
654               // already extracted requested num of documents?
655               if((count - 1) == numberOfDocumentsToExtract) break;
656             }
657             catch(Throwable t) {
658               String nl = Strings.getNl();
659               Err.prln("WARNING: Corpus.populate could not instantiate document" + nl
660                   + "  Document name was: " + docName + nl
661                   + "  Exception was: " + t + nl + nl);
662               t.printStackTrace();
663             }
664             
665             documentString = new StringBuilder();
666             if(sListener != null) sListener.statusChanged(docName + " created!");
667            
668             line = line.substring(index + documentRootElement.length() + 3);
669             if (line.trim().equals("")) line = br.readLine();
670           }
671         }
672       }
673       return lengthInBytes;
674     }
675     finally {
676       if(br != null) br.close();
677     }
678   }// public static void populate
