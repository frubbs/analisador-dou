130     }
131     if(badPRs != null && !badPRs.isEmpty()) { throw new ExecutionException(

133         + "runtime parameters:\n" + badPRs.toString()); }

136   /** Run the Processing Resources in sequence. */
137   protected void executeImpl() throws ExecutionException {
138     // check all the PRs have the right parameters
139     checkParameters();
140 
141     if(log.isDebugEnabled()) {
142       prof.initRun("Execute controller [" + getName() + "]");
143     }
144 
145     // execute all PRs in sequence
146     interrupted = false;
147     for(int i = 0; i < prList.size(); i++) {
148       if(isInterrupted()) {
149 
150       throw new ExecutionInterruptedException("The execution of the "
151         + getName() + " application has been abruptly interrupted!"); }
152 
153       runComponent(i);
154       if(log.isDebugEnabled()) {
155         prof.checkPoint("~Execute PR ["
156           + ((ProcessingResource)prList.get(i)).getName() + "]");
157         Long timeOfPR =
158           (Long)timeMap.get(((ProcessingResource)prList.get(i)).getName());
159         if(timeOfPR == null)
160           timeMap.put(((ProcessingResource)prList.get(i)).getName(), new Long(
161             prof.getLastDuration()));
162         else timeMap.put(((ProcessingResource)prList.get(i)).getName(),
163           new Long(timeOfPR.longValue() + prof.getLastDuration()));
164         log.debug("Time taken so far by "
165           + ((ProcessingResource)prList.get(i)).getName() + ": "
166           + timeMap.get(((ProcessingResource)prList.get(i)).getName()));
167 
168       }
169     }
170 
171     if(log.isDebugEnabled()) {
172       prof.checkPoint("Execute controller [" + getName() + "] finished");
173     }
174     fireStatusChanged("Finished running " + getName());
175 
176   } // executeImpl()

192   /**
193    * Executes a {@link ProcessingResource}.
194    */
195   protected void runComponent(int componentIndex) throws ExecutionException {
196     ProcessingResource currentPR =
197       (ProcessingResource)prList.get(componentIndex);
198 
199     // create the listeners
200     FeatureMap listeners = Factory.newFeatureMap();
201     listeners.put("gate.event.StatusListener", sListener);
202     int componentProgress = 100 / prList.size();
203     listeners.put("gate.event.ProgressListener", new IntervalProgressListener(
204       componentIndex * componentProgress, (componentIndex + 1)
205         * componentProgress));
206 
207     // add the listeners
208     try {
209       AbstractResource.setResourceListeners(currentPR, listeners);
210     }
211     catch(Exception e) {
212       // the listeners setting failed; nothing important
213       log.error("Could not set listeners for " + currentPR.getClass().getName()
214         + "\n" + e.toString() + "\n...nothing to lose any sleep over.");
215     }
216 
217     benchmarkFeatures.put(Benchmark.PR_NAME_FEATURE, currentPR.getName());
218 
219     long startTime = System.currentTimeMillis();
220     // run the thing
221     Benchmark.executeWithBenchmarking(currentPR,
222             Benchmark.createBenchmarkId(Benchmark.PR_PREFIX + currentPR.getName(),
223                     getBenchmarkId()), this, benchmarkFeatures);
224 
225     benchmarkFeatures.remove(Benchmark.PR_NAME_FEATURE);
226 
227     // calculate the time taken by the PR
228     long timeTakenByThePR = System.currentTimeMillis() - startTime;
229     Long time = prTimeMap.get(currentPR.getName());
230     if(time == null) {
231       time = new Long(0);
232     }
233     time = new Long(time.longValue() + timeTakenByThePR);
234     prTimeMap.put(currentPR.getName(), time);
235 
236 
237     // remove the listeners
238     try {
239       AbstractResource.removeResourceListeners(currentPR, listeners);
240     }
241     catch(Exception e) {
242       // the listeners removing failed; nothing important
243       log.error("Could not clear listeners for "
244         + currentPR.getClass().getName() + "\n" + e.toString()
245         + "\n...nothing to lose any sleep over.");
246     }
247   }// protected void runComponent(int componentIndex)

265           prsInOtherControllers.addAll(((Controller)aController).getPRs());
266         }

273 //            Factory.deleteResource((Resource)aPr);
274 //          }

288   /**
289    * Duplicate this controller.  We perform a default duplication of
290    * the controller itself, then recursively duplicate its contained
291    * PRs and add these duplicates to the copy.
292    */
293   public Resource duplicate(Factory.DuplicationContext ctx)
294           throws ResourceInstantiationException {
295     // duplicate this controller in the default way - this handles
296     // subclasses nicely
297     Controller c = (Controller)Factory.defaultDuplicate(this, ctx);
298     
299     // duplicate each of our PRs
300     List<ProcessingResource> newPRs = new ArrayList<ProcessingResource>();
301     for(Object pr : prList) {
302       newPRs.add((ProcessingResource)Factory.duplicate((Resource)pr, ctx));
303     }
304     // and set this duplicated list as the PRs of the copy
305     c.setPRs(newPRs);
306     
307     return c;
308   }

319   public void resourceUnloaded(CreoleEvent e) {
320     // remove all occurences of the resource from this controller
321     if(e.getResource() instanceof ProcessingResource)
322     // while(prList.remove(e.getResource()));
323       // remove all occurrences of this PR from the controller
324       // Use the controller's remove method (rather than prList's so that
325       // subclasses of this controller type can add their own functionality
326       while(remove((ProcessingResource)e.getResource()))
327         ;
328     // remove links in parameters
329     for(int i = 0; i < prList.size(); i++) {
330       ProcessingResource aPr = (ProcessingResource)prList.get(i);
331       ResourceData rData =
332         (ResourceData)Gate.getCreoleRegister().get(aPr.getClass().getName());
333       if(rData != null) {
334         Iterator rtParamDisjIter =
335           rData.getParameterList().getRuntimeParameters().iterator();
336         while(rtParamDisjIter.hasNext()) {
337           List aDisjunction = (List)rtParamDisjIter.next();
338           Iterator rtParamIter = aDisjunction.iterator();
339           while(rtParamIter.hasNext()) {
340             Parameter aParam = (Parameter)rtParamIter.next();
341             String paramName = aParam.getName();
342             try {
343               if(aPr.getParameterValue(paramName) == e.getResource()) {
344                 aPr.setParameterValue(paramName, null);
345               }
346             }
347             catch(ResourceInstantiationException rie) {
348               // nothing to do - this should never happen
349               throw new GateRuntimeException(rie);
350             }
351           }
352         }
353       }
354     }
