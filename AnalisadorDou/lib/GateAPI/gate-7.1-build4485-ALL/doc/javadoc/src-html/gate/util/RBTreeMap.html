1512   /**
1513     * Linear time tree building algorithm from sorted data.  Can accept keys
1514     * and/or values from iterator or stream. This leads to too many
1515     * parameters, but seems better than alternatives.  The four formats
1516     * that this method accepts are:
1517     *
1518     *    1) An iterator of Map.Entries.  (it != null, defaultVal == null).
1519     *    2) An iterator of keys.         (it != null, defaultVal != null).

1521     *            (it == null, defaultVal == null).

1523     *
1524     * It is assumed that the comparator of the RBTreeMap is already set prior
1525     * to calling this method.
1526     *
1527     * @param size the number of keys (or key-value pairs) to be read from
1528     *        the iterator or stream.

1530     *        or keys read from this iterator.

1532     *        possibly values read from this stream in serialized form.
1533     *        Exactly one of it and str should be non-null.
1534     * @param defaultVal if non-null, this default value is used for
1535     *        each value in the map.  If null, each value is read from
1536     *        iterator or stream, as described above.
1537     * @throws IOException propagated from stream reads. This cannot
1538     *         occur if str is null.

1540     *         This cannot occur if str is null.

1542   private void buildFromSorted(int size, java.util.Iterator it,
1543                                 java.io.ObjectInputStream str,
1544                                 Object defaultVal)

1552   /**
1553     * Recursive "helper method" that does the real work of the
1554     * of the previous method.  Identically named parameters have
1555     * identical definitions.  Additional parameters are documented below.
1556     * It is assumed that the comparator and size fields of the RBTreeMap are
1557     * already set prior to calling this method.  (It ignores both fields.)
1558     *
1559     * @param level the current level of tree. Initial call should be 0.
1560     * @param lo the first element index of this subtree. Initial should be 0.
1561     * @param hi the last element index of this subtree.  Initial should be
1562     *        size-1.
1563     * @param redLevel the level at which nodes should be red.
1564     *        Must be equal to computeRedLevel for tree of this size.
1565     */
1566   private static Entry buildFromSorted(int level, int lo, int hi,
1567                                        int redLevel,
1568                                        java.util.Iterator it,
1569                                        java.io.ObjectInputStream str,
1570                                        Object defaultVal)
1571     throws  java.io.IOException, ClassNotFoundException {
1572     /*
1573      * Strategy: The root is the middlemost element. To get to it, we
1574      * have to first recursively construct the entire left subtree,
1575      * so as to grab all of its elements. We can then proceed with right
1576      * subtree.
1577      *
1578      * The lo and hi arguments are the minimum and maximum
1579      * indices to pull out of the iterator or stream for current subtree.
1580      * They are not actually indexed, we just proceed sequentially,
1581      * ensuring that items are extracted in corresponding order.
1582      */
1583 
1584     if (hi < lo) return null;
1585 
1586     int mid = (lo + hi) / 2;
1587 
1588     Entry left  = null;
1589     if (lo < mid)
1590       left = buildFromSorted(level+1, lo, mid - 1, redLevel,
1591                                it, str, defaultVal);
1592 
1593     // extract key and/or value from iterator or stream
1594     Object key;
1595     Object value;
1596 
1597     if (it != null) { // use iterator
1598 
1599       if (defaultVal==null) {
1600         Map.Entry entry = (Map.Entry) it.next();
1601         key = entry.getKey();
1602         value = entry.getValue();
1603       } else {
1604         key = it.next();
1605         value = defaultVal;
1606       }
1607     } else { // use stream
1608       key = str.readObject();
1609       value = (defaultVal != null ? defaultVal : str.readObject());
1610     }
1611 
1612     Entry middle =  new Entry(key, value, null);
1613 
1614     // color nodes in non-full bottommost level red
1615     if (level == redLevel)
1616       middle.color = RED;
1617 
1618     if (left != null) {
1619       middle.left = left;
1620       left.parent = middle;
1621     }
1622 
1623     if (mid < hi) {
1624       Entry right = buildFromSorted(level+1, mid+1, hi, redLevel,
1625                                     it, str, defaultVal);
1626       middle.right = right;
1627       right.parent = middle;
1628     }
1629 
1630     return middle;
1631 
1632   } // Entry buildFromSorted

1634   /**
1635     * Find the level down to which to assign all nodes BLACK.  This is the
1636     * last `full' level of the complete binary tree produced by
1637     * buildTree. The remaining nodes are colored RED. (This makes a `nice'
1638     * set of color assignments wrt future insertions.) This level number is
