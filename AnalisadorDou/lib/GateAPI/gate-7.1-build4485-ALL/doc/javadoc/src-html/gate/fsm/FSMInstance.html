026 /**
027   * The objects of this class represent instances of working Finite State
028   * Machine during parsing a gate document (annotation set).
029   * In order to completely define the state a FSM is in one needs to store
030   * information regarding:
031   * -the position in the FSM transition graph
032   * -the position in the annotation graph
033   * -the set of bindings that occured up to the current state.
034   *  note that a set of bindings is an object of type Map that maps names
035   * (java.lang.String) to bags of annotations (gate.AnnotationSet)
036   */
037 public class FSMInstance implements Comparable, Cloneable, Serializable {

042   /** Creates a new FSMInstance object.
043     * @param supportGraph the transition graph of the FSM
044     * @param FSMPosition the state this instance will be in
045     * @param startNode the node in the AnnotationSet where this FSM instance
046     * started the matching
047     * @param AGPosition the node in the AnnotationSet up to which this FSM Instance
048     * advanced during the matching.
049     * @param bindings a HashMap that maps from labels (objects of type String)
050     * to sets of annotations (objects of type AnnotationSet). This map stores
051     * all the bindings that took place during the matching process.
052     * This FSMInstance started the matching on an AnnotationSet from "startNode"
053     * and advanced to "AGPosition"; during this process it traversed the path in
054     * the transition graph "supportGraph" from the initial state to
055     * "FSMPosition" and made the bindings stored in "bindings".
056     */
057   public FSMInstance(FSM supportGraph, State FSMPosition,
058                      Node startNode, Node AGPosition,
059                      HashMap<String, AnnotationSet> bindings,
060                      Document document) {
061 
062     this.supportGraph = supportGraph;
063     this.FSMPosition = FSMPosition;
064     this.startNode = startNode;
065     this.AGPosition = AGPosition;
066     this.bindings = bindings;
067     this.document = document;
068     length = AGPosition.getOffset().longValue() -
069              startNode.getOffset().longValue();
070     fileIndex = FSMPosition.getFileIndex();
071     priority = FSMPosition.getPriority();
072   }

145   public boolean equals(Object other){
146     if (other == null) return false;
147     if(other instanceof FSMInstance){
148       FSMInstance otherFSM = (FSMInstance)other;
149       boolean result = length == otherFSM.length &&
150              priority == otherFSM.priority &&
151              fileIndex == otherFSM.fileIndex &&
152              bindings.equals(otherFSM.bindings) &&
153              FSMPosition.getAction().equals(otherFSM.FSMPosition.getAction());
154       return result;
155     }else{
156       //equals should never throw an exception it should just return null
157       //throw new ClassCastException(other.getClass().toString());
158       return false;
159     }
160   }

193   /** Implementation of the compareTo method required by the Comparable
194     * interface. The comparison is based on the size of the matched region and
195     * the index in the definition file of the rule associated to this FSM
196     * instance (which needs to be in a final state)
197     * The order imposed by this method is the priority needed in case of a
198     * multiple match.
199     */
200   public int compareTo(Object obj) {
201     if (obj instanceof FSMInstance) {
202       if(obj == this) return 0;
203       FSMInstance other = (FSMInstance)obj;
204       if(length < other.getLength()) return -1;
205       else if(length > other.getLength()) return 1;
206       //equal length
207       else if(priority < other.priority) return -1;
208       else if(priority > other.priority) return 1;
209       //equal priority
210       else return other.fileIndex - fileIndex;
211     } else throw new ClassCastException(
212                     "Attempt to compare a FSMInstance object to an object " +
213                     "of type " + obj.getClass()+"!");
214   }

216   /** Returns a textual representation of this FSM instance.
217     */
218   public String toString() {
219     String res = "";
220     RightHandSide rhs = getFSMPosition().getAction();
221     if(rhs != null){
222       res += rhs.getPhaseName() + "." + rhs.getRuleName() + ": \"";
223       try{
224         res += document.getContent().getContent(
225                         getStartAGPosition().getOffset(),
226                         getAGPosition().getOffset()).toString() + "\"";
227       }catch(InvalidOffsetException ioe){
228         ioe.printStackTrace(Err.getPrintWriter());
229       }
230 
231       Iterator<String> labelIter = bindings.keySet().iterator();
232       res += "\n{";
233       while(labelIter.hasNext()){
234         String label = (String)labelIter.next();
235         Collection<Annotation> annots = bindings.get(label);
236         res += "\n" + label + ": ";
237         Iterator<Annotation> annIter = annots.iterator();
238         while(annIter.hasNext()){
239           Annotation ann  = annIter.next();
240           res += ann.getType() + "(\"";
241           try{
242             res += document.getContent().
243                             getContent(ann.getStartNode().getOffset(),
244                                        ann.getEndNode().getOffset()).toString();
245           }catch(InvalidOffsetException ioe){
246             ioe.printStackTrace(Err.getPrintWriter());
247           }
248           res += "\") ";
249         }
250       }
251       res += "\n}";
252     }else{
253       res +=  "FSM position :" + FSMPosition.getIndex() +
254               "\nFirst matched ANN at:" + startNode.getId() +
255               "\nLast matched ANN at :" + AGPosition.getId() +
256               "\nPriority :" + priority +
257               "\nFile index :" + fileIndex +
258               "\nBindings     :" + bindings;
259     }
260     return res;
261   }

295   /** Static method that provides new FSM instances. This method handles some
296     * basic object pooling in order to reuse the FSMInstance objects.
297     * This is considered to be a good idea because during jape transducing
298     * a large number of FSMIntances are needed for short periods.
299     */
300   public static FSMInstance getNewInstance(FSM supportGraph, State FSMPosition,
301                                            Node startNode, Node AGPosition,
302                                            HashMap<String, AnnotationSet> bindings,
303                                            Document doc) {
304     FSMInstance res;
305     if(myInstances.isEmpty()) res = new FSMInstance(supportGraph, FSMPosition,
306                                                     startNode, AGPosition,
307                                                     bindings, doc);
308     else {
309       res = (FSMInstance)myInstances.removeFirst();
310       res.supportGraph = supportGraph;
311       res.FSMPosition = FSMPosition;
312       res.startNode = startNode;
313       res.AGPosition = AGPosition;
314       res.bindings = bindings;
315     }
316     return res;
317   }
