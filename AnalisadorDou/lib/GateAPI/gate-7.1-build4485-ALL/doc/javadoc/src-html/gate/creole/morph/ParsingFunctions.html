022   /**
023    * This method takes the value of the variable and tells the user what type
024    * of value is from CharacterRange, CharacterSet, StringSet
025    * 
026    * @param varValue
027    *            value for which to find the variable type
028    * @return ERROR_CODE = -4, STRING_SET_CODE = 0, CHARACTER_RANGE_CODE = 1,
029    *         CHARACTER_SET_CODE = 2;
030    */
031   public static int findVariableType(String varValue) {
032     // if the value starts with " it is string set
033     // if the value starts with "[-" it is a character range
034     // if the value starts with "[" it is a character set
035     // otherwise error
036     if (varValue == null) {
037       return Codes.ERROR_CODE;
038     }
039 
040     if (varValue.length() >= 3 && varValue.charAt(0) == '\"'
041         && (varValue.lastIndexOf('\"') == (varValue.length() - 1))) {
042       // for string set it should be greater than 3 because
043       // it requires at least one character to make the string
044       // first and the last character should be "
045       return Codes.STRING_SET_CODE;
046 
047     } else if (varValue.length() >= 6
048         && (((varValue.length() - 3) % 3) == 0)
049         && varValue.substring(0, 2).equals("[-")
050         && varValue.charAt(varValue.length() - 1) == ']') {
051       // for the character range it should be greater than 6 because
052       // three characters as "[-" and "]"
053       // and finally to define the range character-character
054       return Codes.CHARACTER_RANGE_CODE;
055 
056     } else if (varValue.length() >= 3 && varValue.charAt(0) == '['
057         && varValue.charAt(varValue.length() - 1) == ']') {
058       // for the character set it should be greater than 3 characters
059       // because
060       // it requires at least one character
061       // first and the last character should be [ and ] respectively
062       if (varValue.charAt(1) == '-') {
063         return Codes.ERROR_CODE;
064       } else {
065         return Codes.CHARACTER_SET_CODE;
066       }
067 
068     } else {
069       // there are some errors
070       return Codes.ERROR_CODE;
071     }
072

135   public static PatternPart[] getPatternParts(String line) {
136     // the first thing we do is replace all variables with their respective
137     // values
138     ArrayList patterns = new ArrayList();
139     line = line.replaceAll("[\\(]+","(");
140     line = line.replaceAll("[\\)]+",")");
141     line = line.replaceAll("[\\[]+","[");
142     line = line.replaceAll("[\\]]+","]");
143     line = line.replaceAll("(\\()[\\[]+","[");
144     line = line.replaceAll("(\\])[\\)]+","]");
145 
146     while(true) {
147       if(line.trim().length() == 0)
148         break;
149 
150       if(line.charAt(0)!= '(' && line.charAt(0) != '[') {
151         int index = line.indexOf("(");
152         int index1 = line.indexOf("[");
153         if(index < 0 && index1 > 0)
154           index = index1;
155         else if(index > 0 && index1 < 0) {
156           // no need to anything
157         } else if(index > index1)
158           index = index1;
159         
160         if(index < 0) {
161           line = "(" + line + ")";
162         } else {
163           line = '(' + line.substring(0,index) + ")" + line.substring(index, line.length());
164         }
165       }
166       
167       boolean rdBracket = false;
168       boolean rcBracket = false;
169       
170       int index = line.indexOf('(');
171       if(index >= 0)
172         rdBracket = true;
173       
174       int rcindex = line.indexOf('[');
175       if(rcindex >= 0) {
176         // check which one appears first
177         if(rdBracket) {
178           if(index < rcindex)
179             rcBracket = false;
180           else {
181             rcBracket = true;
182             rdBracket = false;
183             index = rcindex;
184           }
185         } else {
186           index = rcindex;
187           rcBracket = true;
188         }
189       }
190       
191       // no bracket found
192       if(!rdBracket && !rcBracket)
193         break;
194       
195       int index1 = -1;
196       if(rdBracket) {
197         index1 = line.indexOf(')');
198         if(index1 < 0)
199           break;
200       }
201       
202       if(rcBracket) {
203         index1 = line.indexOf(']');
204         if(index1 < 0)
205           break;
206       }
207 
208       boolean isPlus = false;
209       boolean isStar = false;
210       
211       if(index1+1 < line.length()) {
212         isPlus = line.charAt(index1+1) == '+';
213         if(!isPlus)
214           isStar = line.charAt(index1+1) == '*';
215       }
216       
217       // we check if the character after closing bracket is
218       String string = line.substring(index+1, index1);
219       // by now there shouldn't be any bracket
220       string = string.replaceAll("\\(","");
221       string = string.replaceAll("\\)","");
222       
223       if(!isPlus && !isStar && rcBracket) {
224         // Style 1
225         // [ABCD]
226         PatternPart pp = new PatternPart(string, OR);
227         patterns.add(pp);
228       } else if(!isPlus && !isStar && rdBracket) { 
229         // Style 2
230         // (ABC)
231         PatternPart pp = new PatternPart(string, AND);
232         patterns.add(pp);
233       } else if(isPlus && rdBracket) {
234         // Style 3
235         // (ABC)+
236         PatternPart pp = new PatternPart(string, AND_PLUS);
237         patterns.add(pp);
238       } else if(isPlus && rcBracket){
239         // Style 4
240         // [ABCD]+
241         PatternPart pp = new PatternPart(string, OR_PLUS);
242         patterns.add(pp);
243       } else if(isStar && rcBracket){
244         // Style 4
245         // [ABCD]*
246         PatternPart pp = new PatternPart(string, OR_STAR);
247         patterns.add(pp);
248       } else {
249         // Style 4
250         // (ABCD)*
251         PatternPart pp = new PatternPart(string, AND_STAR);
252         patterns.add(pp);
253       }
254       
255       if(isPlus || isStar)
256         index1++;
257       
258       if(index1+1 < line.length())
259         line = line.substring(index1+1, line.length());
260       else
261         line = "";
262     }
263 
264     PatternPart[] parts = new PatternPart[patterns.size()];
265     for(int i=0;i<patterns.size();i++) {
266       parts[i] = (PatternPart) patterns.get(i);
267     }
268     return parts;
269   }

438   public static ArrayList andFSMs(String line, HashSet initStates, Interpret owner) {
439     // for the first inital State
440     // we need to find out if any of the parent contains referece to it
441     char ch = line.charAt(0);
442 
443     int nextIndex = getIndex(initStates);
444     FSMState currentState = owner.getState(ch, nextIndex + 1);
445     if(currentState == null) {
446       currentState = new FSMState(nextIndex+1);
447       //System.out.println(ch + " \t "+(nextIndex+1));
448       owner.addState(ch, currentState, nextIndex+1);
449     }
450     
451     // currentState contains the first state
452     // this should be added as a child state to all initStates
453     Iterator iter = initStates.iterator();
454     while(iter.hasNext()) {
455       FSMState state = (FSMState) iter.next();
456       state.put(ch, currentState, FSMState.CHILD_STATE);
457     }
458     
459     // and from current state
460     // do the linking of rest of the characters
461     
462     FSMState nextState = currentState;
463     for (int i = 1; i < line.length(); i++) {
464       ch = line.charAt(i);
465       nextState = currentState.next(ch,  FSMState.CHILD_STATE);
466         if(nextState == null){
467           nextState = owner.getState(ch, currentState.getIndex()+1);
468           if(nextState == null) {
469             nextState = new FSMState(currentState.getIndex()+1);
470             //System.out.println(ch + " \t "+(currentState.getIndex()+1));
471             owner.addState(ch, nextState, currentState.getIndex()+1);
472           }
473         currentState.put(ch, nextState,  FSMState.CHILD_STATE);
474         }      
475       currentState = nextState;
476     }
477     ArrayList nextStates = new ArrayList();
478     HashSet newSet = new HashSet();
479     newSet.add(nextState);
480     nextStates.add(newSet);
481     return nextStates;
482   }

508       nextStates.add(currentState);

512       Iterator iter = initStates.iterator();
513       while(iter.hasNext()) {
514         FSMState state = (FSMState) iter.next();
515         state.put(ch, currentState, FSMState.CHILD_STATE);
516       }
517       
518     }
519     ArrayList newList = new ArrayList();
520     newList.add(nextStates);
521     return newList;
522   }

736   /**
737    * This method convert the expression which has been entered by the user in
738    * the .rul file (i.e. rules defined by the user), into the expression which
739    * are recognized by the regular expression Patterns
740    * 
741    * @param line
742    *            rule defined by the user
743    * @param storage
744    *            this method internally requires values of the used variables
745    *            to replace the them with their values in the expression
746    * @return newly generated regular expression
747    */
748   public static String convertToRegExp(String line, Storage storage) {
749     // replace all OR with |
750     line = line.replaceAll("( OR )", "|");
751     line = line.replaceAll("(\\[\\-)", "[");
752 
753     // we will use the stack concept here
754     // for every occurence of '{', or '(' we will add that into the stack
755     // and for every occurence of '}' or ')' we will remove that element
756     // from
757     // the stack
758     // if the value found between the bracket is an integer value
759     // we won't replace those brackets
760     StringBuffer newExpr = new StringBuffer(line);
761     Stack stack = new Stack();
762     Stack bracketIndexes = new Stack();
763 
764     for (int i = 0; i < newExpr.length(); i++) {
765       if (newExpr.charAt(i) == '{') {
766         // add it to the stack
767         stack.add("{");
768         bracketIndexes.add(new Integer(i));
769 
770       } else if (newExpr.charAt(i) == '(') {
771         // add it to the stack
772         stack.add("(");
773         bracketIndexes.add(new Integer(i));
774 
775       } else if (newExpr.charAt(i) == '[') {
776         // add it to the stack
777         stack.add("[");
778         bracketIndexes.add(new Integer(i));
779 
780       } else if (newExpr.charAt(i) == '\"') {
781         // before adding it to the stack, check if this is the closing
782         // one
783         if (stack.isEmpty()
784             || !(((String) (stack.get(stack.size() - 1)))
785                 .equals("\""))) {
786           // yes this is the opening one
787           // add it to the stack
788           stack.add("\"");
789           bracketIndexes.add(new Integer(i));
790         } else {
791           // this is the closing one
792           stack.pop();
793           int index = ((Integer) (bracketIndexes.pop())).intValue();
794           newExpr.setCharAt(index, '(');
795           newExpr.setCharAt(i, ')');
796         }
797       } else if (newExpr.charAt(i) == '}') {
798         // remove the element from the stack
799         // it must be '{', otherwise generate the error
800         String bracket = (String) (stack.pop());
801         int index = ((Integer) (bracketIndexes.pop())).intValue();
802         if (!bracket.equals("{")) {
803           return null;
804         }
805 
806         // now check if the value between these brackets is integer,
807         // that means
808         // we don't need to change the brackets, otherwise change them
809         // to
810         // '(' and ')'
811         if (isInteger(newExpr.substring(index + 1, i))) {
812           // yes it is an integer
813           // continue
814           continue;
815         } else {
816           // no it is string
817           newExpr.setCharAt(index, '(');
818           newExpr.setCharAt(i, ')');
819         }
820 
821       } else if (newExpr.charAt(i) == ')') {
822         // remove the element from the stack
823         // it must be ')', otherwise generate the error
824         String bracket = (String) (stack.pop());
825         bracketIndexes.pop();
826         if (!bracket.equals("(")) {
827           return null;
828         }
829         continue;
830       } else if (newExpr.charAt(i) == ']') {
831         // remove the element from the stack
832         // it must be '[', otherwise generate the error
833         String bracket = (String) (stack.pop());
834         bracketIndexes.pop();
835         if (!bracket.equals("[")) {
836           return null;
837         }
838       }
839     }
840     // check if all the stacks are empty then and only then the written
841     // expression is true, otherwise it is incorrect
842     if (!stack.empty() || !bracketIndexes.empty()) {
843       return null;
844     }
845     // System.out.println(line+" "+newExpr);
846     // now we need to replace the variables with their values
847     // but how would we know which is the variable
848     // so get the variable list and check if it is available in the
849     // expression
850     String[] varNames = storage.getVarNames();
851     for (int i = 0; i < varNames.length; i++) {
852       // check for the occurance of each varName in the expression
853       int index = -1;
854       String myString = "{[()]} ";
855       while ((index = newExpr.indexOf(varNames[i], index + 1)) != -1) {
856         // System.out.println(index + " "+newExpr.length());
857         // now check for the left and right characters
858         if (index > 0) {
859           if (myString.indexOf(newExpr.charAt(index - 1)) == -1) {
860             index = index + varNames[i].length() - 1;
861             // this is not the varilable
862             continue;
863           }
864         }
865         if ((varNames[i].length() + index) < newExpr.length()) {
866           if (myString.indexOf(newExpr.charAt(varNames[i].length()
867               + index)) == -1) {
868             index = index + varNames[i].length() - 1;
869             // this is not the variable
870             continue;
871           }
872         }
873 
874         // yes it is a variable
875         String replaceWith = "(" + (String) (storage.get(varNames[i]))
876             + ")";
877         newExpr.replace(index, (varNames[i].length() + index),
878             replaceWith);
879         index = index + replaceWith.length();
880       }
881     }
882     return new String(newExpr);
883   }
