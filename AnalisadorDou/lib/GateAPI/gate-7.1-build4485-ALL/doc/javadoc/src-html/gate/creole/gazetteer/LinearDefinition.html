114     try {
115       gazListsByNode = new HashMap();
116       Iterator inodes = nodes.iterator();
117       while (inodes.hasNext()) {
118         LinearNode node = (LinearNode)inodes.next();
119 
120         GazetteerList list = new GazetteerList();
121         list.setSeparator(separator);
122         URL lurl = new URL(url,node.getList());
123         list.setURL(lurl);
124         list.setEncoding(encoding);
125         list.load(isOrdered);
126 
127         gazListsByNode.put(node,list);
128       } // while inodes
129     } catch (Exception ex) {
130       throw new ResourceInstantiationException(ex);
131     }
132     return gazListsByNode;

161 
162       try {

171         }
172 
173         File f = new File(path+listName);
174 
175         if (!f.exists())
176           f.createNewFile();
177 
178         URL lurl = new URL(url,listName);
179         list.setURL(lurl);
180         list.load(isOrdered);
181 
182       }
183 
184 
185 
186     } catch (MalformedURLException murle ) {
187       throw new ResourceInstantiationException(murle);

229   /**
230    * Loads linear definition if url is set
231    */
232   public void load() throws ResourceInstantiationException {
233     if (null == url) {
234       throw new ResourceInstantiationException("URL not set (null).");
235     }
236     try {
237       if("file".equals(url.getProtocol())) {
238         File definitionFile = Files.fileFromURL(url);
239         // create an new definition file only if not existing
240         definitionFile.createNewFile();
241       }
242       BufferedReader defReader =
243       new BomStrippingInputStreamReader((url).openStream(), ENCODING);
244 
245       String line;
246       LinearNode node;
247       while (null != (line = defReader.readLine())) {
248         node = new LinearNode(line);
249         try {
250           this.add(node);
251         } catch (GateRuntimeException ex) {
252           // The add method cannot throw a checked exception because
253           // it implements the List interface. Therefore it throws
254           // a GateRuntimeException which we re-throw as a 
255           // ResourceInstnatiationException here
256           throw new ResourceInstantiationException(ex);
257         }
258       } //while
259 
260       defReader.close();
261       isModified = false;
262     } catch (Exception x){
263       throw new ResourceInstantiationException(x);
264     }
265   } // load();

267   /**
268    * Stores this to a definition file.
269    */
270   public void store() throws ResourceInstantiationException{
271     if (null == url) {
272       throw new ResourceInstantiationException("URL not set.(null)");
273     }
274     try {
275       URL tempUrl = url;
276       if (-1 != url.getProtocol().indexOf("gate")) {
277         tempUrl = gate.util.protocols.gate.Handler.class.getResource(
278                       gate.util.Files.getResourcePath() + url.getPath()
279                     );
280       } // if gate:path url
281 
282       File fileo = Files.fileFromURL(tempUrl);
283       fileo.delete();
284       BufferedWriter defWriter = new BufferedWriter(new FileWriter(fileo));
285       Iterator inodes = nodes.iterator();
286       while (inodes.hasNext()) {
287         defWriter.write(inodes.next().toString());
288         defWriter.newLine();
289       }
290       defWriter.close();
291       isModified = false;
292     } catch(Exception x) {
293       throw new ResourceInstantiationException(x);
294     }
295 
296   } // store();
