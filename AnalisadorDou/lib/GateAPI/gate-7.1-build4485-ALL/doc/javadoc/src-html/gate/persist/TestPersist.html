067   /** Test resource save and restore */
068   public void testSaveRestore() throws Exception {
069     File storageDir = File.createTempFile("TestPersist__", "__StorageDir");
070     storageDir.delete(); // get rid of the temp file
071     storageDir.mkdir(); // create an empty dir of same name
072 
073     SerialDataStore sds =
074             new SerialDataStore(storageDir.toURI().toURL().toString());
075     sds.create();
076     sds.open();
077 
078     // create a document
079     String server = TestDocument.getTestServerName();
080     assertNotNull(server);
081     Document doc = Factory.newDocument(new URL(server + "tests/doc0.html"));
082     assertNotNull(doc);
083     doc.getFeatures().put("hi there", new Integer(23232));
084     doc.getAnnotations().add(new Long(0), new Long(20), "thingymajig",
085             Factory.newFeatureMap());
086 
087     // check that we can't save a resource without adopting it
088     boolean cannotSync = false;
089     try {
090       sds.sync(doc);
091     } catch(PersistenceException e) {
092       cannotSync = true;
093     }
094     if(!cannotSync) assertTrue("doc synced ok before adoption", false);
095 
096     // check that we can't adopt a resource that's stored somewhere else
097     doc.setDataStore(new SerialDataStore(new File("z:\\").toURI().toURL()
098             .toString()));
099     try {
100       sds.adopt(doc, null);
101     } catch(PersistenceException e) {
102       cannotSync = true;
103     }
104     if(!cannotSync)
105       assertTrue("doc adopted but in other datastore already", false);
106     doc.setDataStore(null);
107     doc.setName("Alicia Tonbridge, a Document");
108 
109     // save the document
110     Document persDoc = (Document)sds.adopt(doc, null);
111     sds.sync(persDoc);
112     Object lrPersistenceId = persDoc.getLRPersistenceId();
113 
114     // test the getLrTypes method
115     List lrTypes = sds.getLrTypes();
116     assertTrue("wrong number of types in SDS", lrTypes.size() == 1);
117     assertTrue("wrong type LR in SDS",
118             lrTypes.get(0).equals("gate.corpora.DocumentImpl"));
119 
120     // test the getLrNames method
121     Iterator iter = sds.getLrNames("gate.corpora.DocumentImpl").iterator();
122     String name = (String)iter.next();
123     assertEquals(name, "Alicia Tonbridge, a Document");
124 
125     // read the document back
126     FeatureMap features = Factory.newFeatureMap();
127     features.put(DataStore.LR_ID_FEATURE_NAME, lrPersistenceId);
128     features.put(DataStore.DATASTORE_FEATURE_NAME, sds);
129     Document doc2 =
130             (Document)Factory.createResource("gate.corpora.DocumentImpl",
131                     features);
132     Document doc3 =
133             (Document)sds.getLr("gate.corpora.DocumentImpl", lrPersistenceId);
134 
135     try {
136       boolean value = TestEqual.documentsEqual(doc3, doc2);
137       assertTrue(TestEqual.message, value);
138       value = TestEqual.documentsEqual(persDoc, doc2);
139       assertTrue(TestEqual.message, value);
140     } finally {
141       // delete the datastore
142       sds.delete();
143     }
144   } // testSaveRestore()

193   /** Test multiple LRs */
194   public void testMultipleLrs() throws Exception {
195     // create a temporary directory; because File.createTempFile actually
196     // writes the bloody thing, we need to delete it from disk before calling
197     // DataStore.create
198     File storageDir = File.createTempFile("TestPersist__", "__StorageDir");
199     storageDir.delete();
200 
201     // create and open a serial data store
202     SerialDataStore sds =
203             new SerialDataStore(storageDir.toURI().toURL().toString());
204     sds.create();
205     sds.open();
206 
207     // create a document with some annotations / features on it
208     String server = TestDocument.getTestServerName();
209     Document doc = Factory.newDocument(new URL(server + "tests/doc0.html"));
210     doc.getFeatures().put("hi there", new Integer(23232));
211     doc.getAnnotations().add(new Long(5), new Long(25), "ThingyMaJig",
212             Factory.newFeatureMap());
213 
214     // create another document with some annotations / features on it
215     Document doc2 =
216             Factory.newDocument(new URL(server + "tests/html/test1.htm"));
217     doc.getFeatures().put("hi there again", new Integer(23232));
218     doc.getAnnotations().add(new Long(5), new Long(25), "dog poo irritates",
219             Factory.newFeatureMap());
220 
221     // create a corpus with the documents
222     Corpus corp = Factory.newCorpus("Hamish test corpus");
223     corp.add(doc);
224     corp.add(doc2);
225     LanguageResource persCorpus = sds.adopt(corp, null);
226     sds.sync(persCorpus);
227 
228     // read the documents back
229     ArrayList lrsFromDisk = new ArrayList();
230     List lrIds = sds.getLrIds("gate.corpora.SerialCorpusImpl");
231 
232     Iterator idsIter = lrIds.iterator();
233     while(idsIter.hasNext()) {
234       String lrId = (String)idsIter.next();
235       FeatureMap features = Factory.newFeatureMap();
236       features.put(DataStore.DATASTORE_FEATURE_NAME, sds);
237       features.put(DataStore.LR_ID_FEATURE_NAME, lrId);
238       Resource lr =
239               Factory.createResource("gate.corpora.SerialCorpusImpl", features);
240       lrsFromDisk.add(lr);
241     } // for each LR ID
242 
243     if(DEBUG) System.out.println("LRs on disk" + lrsFromDisk);
244 
245     // check that the versions we read back match the originals
246     Corpus diskCorp = (Corpus)lrsFromDisk.get(0);
247 
248     Document diskDoc = (Document)diskCorp.get(0);
249 
250     if(DEBUG) Out.prln("Documents in corpus: " + corp.getDocumentNames());
251     assertTrue("corp name != mem name",
252             corp.getName().equals(diskCorp.getName()));
253     if(DEBUG) Out.prln("Memory features " + corp.getFeatures());
254     if(DEBUG) Out.prln("Disk features " + diskCorp.getFeatures());
255     assertTrue("corp feat != mem feat",
256             corp.getFeatures().equals(diskCorp.getFeatures()));
257     if(DEBUG) Out.prln("Annotations in doc: " + diskDoc.getAnnotations());
258     assertTrue(
259             "doc annotations from disk not equal to memory version",
260             TestEqual.annotationSetsEqual(doc.getAnnotations(),
261                     diskDoc.getAnnotations()));
262 
263     assertTrue("doc from disk not equal to memory version",
264             TestEqual.documentsEqual(doc, diskDoc));
265 
266     Iterator corpusIter = diskCorp.iterator();
267     while(corpusIter.hasNext()) {
268       if(DEBUG)
269         Out.prln(((Document)corpusIter.next()).getName());
270       else corpusIter.next();
271     }
272 
273     // assertTrue("doc2 from disk not equal to memory version",
274     // doc2.equals(diskDoc2));
275 
276     // delete the datastore
277     sds.delete();
278   } // testMultipleLrs()

334             Factory.createDataStore("gate.persist.SerialDataStore", storageDir
335                     .toURI().toURL().toString());
336 
337     // create a document with some annotations / features on it
338     String server = TestDocument.getTestServerName();

350             dsr.size() == 1);
351

360     assertTrue("DSR has wrong number elements: " + dsr.size(), dsr.size() == 2);

376             dsr.size() == 1);
377     sds2.delete();
378     assertTrue("DSR has wrong number elements (expected 0): " + dsr.size(),
379             dsr.size() == 0);
380 
381   } // testDSR()
