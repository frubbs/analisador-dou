051 /** Some utilities for use with Files and with resources.
052   * <P>
053   * <B>Note</B> that there is a terminology conflict between the use
054   * of "resources" here and <TT>gate.Resource</TT> and its inheritors.
055   * <P>
056   * Java "resources" are files that live on the CLASSPATH or in a Jar
057   * file that are <I>not</I> <TT>.class</TT> files. For example: a
058   * <TT>.gif</TT> file that is used by a GUI, or one of the XML files
059   * used for testing GATE's document format facilities. This class
060   * allows you to access these files in various ways (as streams, as
061   * byte arrays, etc.).
062   * <P>
063   * GATE resources are components (Java Beans) that provide all of the
064   * natural language processing capabilities of a GATE-based system, and
065   * the language data that such systems analsyse and produce. For
066   * example: parsers, lexicons, generators, corpora.
067   * <P>
068   * Where we say "resource" in this class we mean Java resource; elsewhere
069   * in the system we almost always mean GATE resource.

087   /** It returns the last component in a file path.
088     * It takes E.g: d:/tmp/file.txt and returns file.txt
089     */
090   public static String getLastPathComponent(String path){
091     if(path == null || path.length() == 0) return "";
092     //we should look both for "/" and "\" as on windows the file separator is"\"
093     //but a path coming from an URL will be separated by "/"
094     int index = path.lastIndexOf('/');
095     if(index == -1) index = path.lastIndexOf('\\');
096     if(index == -1) return path;
097     else return path.substring(index + 1);
098   }// getLastPathComponent()

421   /**
422    * This method takes a regular expression and a directory name and returns
423    * the set of Files that match the pattern under that directory.
424    *
425    * @param regex regular expression path that begins with <code>pathFile</code>
426    * @param pathFile directory path where to search for files
427    * @return set of file paths under <code>pathFile</code> that matches
428    *  <code>regex</code>
429    */
430   public static Set<String> Find(String regex, String pathFile) {
431     Set<String> regexfinal = new HashSet<String>();
432     String[] tab;
433     File file = null;
434 
435     //open a file
436     try {
437       file = new File(pathFile);
438     } catch(NullPointerException npe) {
439       npe.printStackTrace(Err.getPrintWriter());
440     }
441 
442       Pattern pattern = Pattern.compile("^"+regex);
443 
444       if (file.isDirectory()){
445         tab = file.list();
446         for (int i=0;i<=tab.length-1;i++){
447           String finalPath = pathFile+"/"+tab[i];
448           Matcher matcher = pattern.matcher(finalPath);
449           if (matcher.matches()){
450               regexfinal.add(finalPath);
451           }
452         }
453       }
454       else {
455         if (file.isFile()){
456             Matcher matcher = pattern.matcher(pathFile);
457             if (matcher.matches()){
458                 regexfinal.add(pathFile);
459             }
460         }
461       }
462 
463     return regexfinal;
464   } //find

466   /** Recursively remove a directory <B>even if it contains other files
467     * or directories</B>. Returns true when the directory and all its
468     * contents are successfully removed, else false.
469     */
470   public static boolean rmdir(File dir) {
471     if(dir == null || ! dir.isDirectory()) // only delete directories
472       return false;
473 
474     // list all the members of the dir
475     String[] members = dir.list();
476 
477     // return value indicating success or failure
478     boolean succeeded = true;
479 
480     // for each member, if is dir then recursively delete; if file then delete
481     for(int i = 0; i<members.length; i++) {
482       File member = new File(dir, members[i]);
483 
484       if(member.isFile()) {
485         if(! member.delete())
486           succeeded = false;
487       } else {
488         if(! Files.rmdir(member))
489           succeeded = false;
490       }
491     }
492 
493     // delete the directory itself
494     dir.delete();
495 
496     // return status value
497     return succeeded;
498   } // rmdir(File)

501    * This method updates an XML element with a new set of attributes.
502    * If the element is not found the XML is unchanged. The attributes
503    * keys and values must all be Strings.
504    *
505    * @param xml A stream of the XML data.
506    * @param elementName The name of the element to update.
507    * @param newAttrs The new attributes to place on the element.
508    * @return A string of the whole XML source, with the element updated.
509    */
510   public static String updateXmlElement(
511     BufferedReader xml, String elementName, Map newAttrs
512   ) throws IOException {
513     String line = null;
514     String nl = Strings.getNl();
515     StringBuffer newXml = new StringBuffer();
516 
517     // read the whole source
518     while( ( line = xml.readLine() ) != null ) {
519       newXml.append(line);
520       newXml.append(nl);
521     }
522 
523     // find the location of the element
524     int start = newXml.toString().indexOf("<" + elementName);
525     if(start == -1) return newXml.toString();
526     int end =   newXml.toString().indexOf(">", start);
527     if(end == -1)   return newXml.toString();
528 
529     // check if the old element is empty (ends in "/>") or not
530     boolean isEmpty = false;
531     if(newXml.toString().charAt(end - 1) == '/') isEmpty = true;
532 
533     // create the new element string with the new attributes
534     StringBuffer newElement = new StringBuffer();
535     newElement.append("<");
536     newElement.append(elementName);
537 
538     // add in the new attributes
539     Iterator iter = newAttrs.entrySet().iterator();
540     while(iter.hasNext()) {
541       Map.Entry entry = (Map.Entry) iter.next();
542       String key =   (String) entry.getKey();
543       String value = (String) entry.getValue();
544 
545       newElement.append(" ");
546       newElement.append(DocumentXmlUtils.combinedNormalisation(key));
547       newElement.append("=\"");
548       newElement.append(DocumentXmlUtils.combinedNormalisation(value));
549       newElement.append("\"" + nl);
550     }
551 
552     // terminate the element
553     if(isEmpty) newElement.append("/");
554     newElement.append(">");
555 
556     // replace the old string
557     newXml.replace(start, end + 1, newElement.toString());
558 
559     return newXml.toString();
560   } // updateXmlElement(Reader...)

562   /**
563    * This method updates an XML element in an XML file
564    * with a new set of attributes. If the element is not found the XML
565    * file is unchanged. The attributes keys and values must all be Strings.
566    * We first try to read the file using UTF-8 encoding.  If an error occurs we
567    * fall back to the platform default encoding (for backwards-compatibility
568    * reasons) and try again.  The file is written back in UTF-8, with an
569    * updated encoding declaration.
570    *
571    * @param xmlFile An XML file.
572    * @param elementName The name of the element to update.
573    * @param newAttrs The new attributes to place on the element.
574    * @return A string of the whole XML file, with the element updated (the
575    *   file is also overwritten).
576    */
577   public static String updateXmlElement(
578     File xmlFile, String elementName, Map newAttrs
579   ) throws IOException {
580     String newXml = null;
581     BufferedReader utfFileReader = null;
582     BufferedReader platformFileReader = null;
583     Charset utfCharset = Charset.forName("UTF-8");
584     try {
585       FileInputStream fis = new FileInputStream(xmlFile);
586       // try reading with UTF-8, make sure any errors throw an exception
587       CharsetDecoder decoder = utfCharset.newDecoder()
588         .onUnmappableCharacter(CodingErrorAction.REPORT)
589         .onMalformedInput(CodingErrorAction.REPORT);
590       utfFileReader = new BomStrippingInputStreamReader(fis, decoder);
591       newXml = updateXmlElement(utfFileReader, elementName, newAttrs);
592     }
593     catch(CharacterCodingException cce) {
594       // File not readable as UTF-8, so try the platform default encoding
595       if(utfFileReader != null) {
596         utfFileReader.close();
597         utfFileReader = null;
598       }
599       if(DEBUG) {
600         Err.prln("updateXmlElement: could not read " + xmlFile + " as UTF-8, "
601             + "trying platform default");
602       }
603       platformFileReader = new BufferedReader(new FileReader(xmlFile));
604       newXml = updateXmlElement(platformFileReader, elementName, newAttrs);
605     }
606     finally {
607       if(utfFileReader != null) {
608         utfFileReader.close();
609       }
610       if(platformFileReader != null) {
611         platformFileReader.close();
612       }
613     }
614 
615     // write the updated file in UTF-8, fixing the encoding declaration
616     newXml = newXml.replaceFirst(
617         "\\A<\\?xml (.*)encoding=(?:\"[^\"]*\"|'[^']*')",
618         "<?xml $1encoding=\"UTF-8\"");
619     FileOutputStream fos = new FileOutputStream(xmlFile);
620     OutputStreamWriter fileWriter = new OutputStreamWriter(fos, utfCharset);
621     fileWriter.write(newXml);
622     fileWriter.close();
623 
624     return newXml;
625   } // updateXmlElement(File...)

629    * Convert a file: URL to a <code>java.io.File</code>.  First tries to parse
630    * the URL's toExternalForm as a URI and create the File object from that
631    * URI.  If this fails, just uses the path part of the URL.  This handles
632    * URLs that contain spaces or other unusual characters, both as literals and
633    * when encoded as (e.g.) %20.
634    *
635    * @exception IllegalArgumentException if the URL is not convertable into a
636    * File.
637    */
638   public static File fileFromURL(URL theURL) throws IllegalArgumentException {
639     try {
640       URI uri = new URI(theURL.toExternalForm());
641       return new File(uri);
642     }
643     catch(URISyntaxException use) {
644       try {
645         URI uri = new URI(theURL.getProtocol(), null, theURL.getPath(), null, null);
646         return new File(uri);
647       }
648       catch(URISyntaxException use2) {
649         throw new IllegalArgumentException("Cannot convert " + theURL + " to a file path");
650       }
651     }
652   }

654   /**
655    * Same as {@link java.io.File#listFiles(java.io.FileFilter)}
656    * but recursive on directories.
657    * @param directory file path to start the search, will not be include
658    *   in the results
659    * @param filter filter apply to the search
660    * @return an array of files (including directories) contained inside
661    *   <code>directory</code>. The array will be empty if the directory is
662    *   empty. Returns null if this abstract pathname does not denote a
663    *   directory, or if an I/O error occurs.
664    */
665   public static File[] listFilesRecursively(File directory, FileFilter filter) {
666     List<File> filesList = new ArrayList<File>();
667 
668     File[] filesRootArray = directory.listFiles(filter);
669     if (filesRootArray == null) { return null; }
670 
671     for (File file : filesRootArray) {
672       filesList.add(file);
673       if (file.isDirectory()) {
674         File[] filesDeepArray = listFilesRecursively(file, filter);
675         if (filesDeepArray == null) { return null; }
676         filesList.addAll(Arrays.asList(filesDeepArray));
677       }
678     }
679 
680     return filesList.toArray(new File[filesList.size()]);
681   }
